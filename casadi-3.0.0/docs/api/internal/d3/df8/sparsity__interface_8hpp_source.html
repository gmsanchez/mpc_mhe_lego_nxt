561  std::vector< std::vector< MatType > >
562  SparsityInterface<MatType>::blocksplit(const MatType& x, int vert_incr, int horz_incr) {
563  casadi_assert(horz_incr>=1);
564  casadi_assert(vert_incr>=1);
565  int sz1 = x.size1();
566  std::vector<int> offset1 = range(0, sz1, vert_incr);
567  offset1.push_back(sz1);
568  int sz2 = x.size2();
569  std::vector<int> offset2 = range(0, sz2, horz_incr);
570  offset2.push_back(sz2);
571  return blocksplit(x, offset1, offset2);
572  }
