87  template<>
88  SX GenericMatrix<SX>::sym(const string& name, const Sparsity& sp) {
89  // Create a dense n-by-m matrix
90  vector<SXElem> retv;
91 
92  // Check if individial names have been provided
93  if (name[0]=='[') {
94 
95  // Make a copy of the string and modify it as to remove the special characters
96  string modname = name;
97  for (string::iterator it=modname.begin(); it!=modname.end(); ++it) {
98  switch (*it) {
99  case '(': case ')': case '[': case ']': case '{': case '}': case ',': case ';': *it = ' ';
100  }
101  }
102 
103  istringstream iss(modname);
104  string varname;
105 
106  // Loop over elements
107  while (!iss.fail()) {
108  // Read the name
109  iss >> varname;
110 
111  // Append to the return vector
112  if (!iss.fail())
113  retv.push_back(SXElem::sym(varname));
114  }
115  } else if (sp.is_scalar(true)) {
116  retv.push_back(SXElem::sym(name));
117  } else {
118  // Scalar
119  stringstream ss;
120  for (int k=0; k<sp.nnz(); ++k) {
121  ss.str("");
122  ss << name << "_" << k;
123  retv.push_back(SXElem::sym(ss.str()));
124  }
125  }
126 
127  // Determine dimensions automatically if empty
128  if (sp.is_scalar(true)) {
129  return SX(retv);
130  } else {
131  return SX(sp, retv, false);
132  }
133  }

228  template<>
229  void SX::expand(const SX& ex2, SX& ww, SX& tt) {
230  casadi_assert(ex2.is_scalar());
231  SXElem ex = ex2.scalar();
232 
233  // Terms, weights and indices of the nodes that are already expanded
234  vector<vector<SXNode*> > terms;
235  vector<vector<double> > weights;
236  map<SXNode*, int> indices;
237 
238  // Stack of nodes that are not yet expanded
239  stack<SXNode*> to_be_expanded;
240  to_be_expanded.push(ex.get());
241 
242  while (!to_be_expanded.empty()) { // as long as there are nodes to be expanded
243 
244  // Check if the last element on the stack is already expanded
245  if (indices.find(to_be_expanded.top()) != indices.end()) {
246  // Remove from stack
247  to_be_expanded.pop();
248  continue;
249  }
250 
251  // Weights and terms
252  vector<double> w; // weights
253  vector<SXNode*> f; // terms
254 
255  if (to_be_expanded.top()->is_constant()) { // constant nodes are seen as multiples of one
256  w.push_back(to_be_expanded.top()->to_double());
257  f.push_back(casadi_limits<SXElem>::one.get());
258  } else if (to_be_expanded.top()->is_symbolic()) {
259  // symbolic nodes have weight one and itself as factor
260  w.push_back(1);
261  f.push_back(to_be_expanded.top());
262  } else { // binary node
263 
264  casadi_assert(to_be_expanded.top()->hasDep()); // make sure that the node is binary
265 
266  // Check if addition, subtracton or multiplication
267  SXNode* node = to_be_expanded.top();
268  // If we have a binary node that we can factorize
269  if (node->op() == OP_ADD || node->op() == OP_SUB ||
270  (node->op() == OP_MUL && (node->dep(0)->is_constant() ||
271  node->dep(1)->is_constant()))) {
272  // Make sure that both children are factorized, if not - add to stack
273  if (indices.find(node->dep(0).get()) == indices.end()) {
274  to_be_expanded.push(node->dep(0).get());
275  continue;
276  }
277  if (indices.find(node->dep(1).get()) == indices.end()) {
278  to_be_expanded.push(node->dep(1).get());
279  continue;
280  }
281 
282  // Get indices of children
283  int ind1 = indices[node->dep(0).get()];
284  int ind2 = indices[node->dep(1).get()];
285 
286  // If multiplication
287  if (node->op() == OP_MUL) {
288  double fac;
289  // Multiplication where the first factor is a constant
290  if (node->dep(0)->is_constant()) {
291  fac = node->dep(0)->to_double();
292  f = terms[ind2];
293  w = weights[ind2];
294  } else { // Multiplication where the second factor is a constant
295  fac = node->dep(1)->to_double();
296  f = terms[ind1];
297  w = weights[ind1];
298  }
299  for (int i=0; i<w.size(); ++i) w[i] *= fac;
300 
301  } else { // if addition or subtraction
302  if (node->op() == OP_ADD) { // Addition: join both sums
303  f = terms[ind1]; f.insert(f.end(), terms[ind2].begin(), terms[ind2].end());
304  w = weights[ind1]; w.insert(w.end(), weights[ind2].begin(), weights[ind2].end());
305  } else { // Subtraction: join both sums with negative weights for second term
306  f = terms[ind1]; f.insert(f.end(), terms[ind2].begin(), terms[ind2].end());
307  w = weights[ind1];
308  w.reserve(f.size());
309  for (int i=0; i<weights[ind2].size(); ++i) w.push_back(-weights[ind2][i]);
310  }
311  // Eliminate multiple elements
312  vector<double> w_new; w_new.reserve(w.size()); // weights
313  vector<SXNode*> f_new; f_new.reserve(f.size()); // terms
314  map<SXNode*, int> f_ind; // index in f_new
315 
316  for (int i=0; i<w.size(); i++) {
317  // Try to locate the node
318  map<SXNode*, int>::iterator it = f_ind.find(f[i]);
319  if (it == f_ind.end()) { // if the term wasn't found
320  w_new.push_back(w[i]);
321  f_new.push_back(f[i]);
322  f_ind[f[i]] = f_new.size()-1;
323  } else { // if the term already exists
324  w_new[it->second] += w[i]; // just add the weight
325  }
326  }
327  w = w_new;
328  f = f_new;
329  }
330  } else { // if we have a binary node that we cannot factorize
331  // By default,
332  w.push_back(1);
333  f.push_back(node);
334 
335  }
336  }
337 
338  // Save factorization of the node
339  weights.push_back(w);
340  terms.push_back(f);
341  indices[to_be_expanded.top()] = terms.size()-1;
342 
343  // Remove node from stack
344  to_be_expanded.pop();
345  }
346 
347  // Save expansion to output
348  int thisind = indices[ex.get()];
349  ww = SX(weights[thisind]);
350 
351  vector<SXElem> termsv(terms[thisind].size());
352  for (int i=0; i<termsv.size(); ++i)
353  termsv[i] = SXElem::create(terms[thisind][i]);
354  tt = SX(termsv);
355  }

358  SX SX::pw_const(const SX& t, const SX& tval, const SX& val) {
359  // number of intervals
360  int n = val.numel();
361 
362  casadi_assert_message(t.is_scalar(), "t must be a scalar");
363  casadi_assert_message(tval.numel() == n-1, "dimensions do not match");
364

366  for (int i=0; i<n-1; ++i) {

368  }

373  template<>
374  SX SX::pw_lin(const SX& t, const SX& tval, const SX& val) {
375  // Number of points
376  int N = tval.numel();
377  casadi_assert_message(N>=2, "pw_lin: N>=2");
378 
379  // Gradient for each line segment
380  SX g = SX(1, N-1);
381  for (int i=0; i<N-1; ++i) {
382  g(0, i) = (val(0, i+1)- val(0, i))/(tval(0, i+1)-tval(0, i));
383  }
384 
385  // Line segments
386  SX lseg = SX(1, N-1);
387  for (int i=0; i<N-1; ++i)
388  lseg(0, i) = val(0, i) + g(0, i)*(t-tval(0, i));
389 
390  // interior time points
391  SX tint = tval(0, range(N-2));
392 
393  // Return piecewise linear function
394  return pw_const(t, tint, lseg);
395  }

402  template<>
403  SX SX::gauss_quadrature(const SX& f, const SX& x, const SX& a, const SX& b, int order,
404  const SX& w) {
405  casadi_assert_message(order == 5, "gauss_quadrature: order must be 5");
406  casadi_assert_message(w.is_empty(), "gauss_quadrature: empty weights");
407 
408  // Change variables to [-1, 1]
409  if (!is_equal(a.scalar(), -1) || !is_equal(b.scalar(), 1)) {
410  SX q1 = (b-a)/2;
411  SX q2 = (b+a)/2;
412 
413  Function fcn("gauss_quadrature", {x}, {f});
414 
415  return q1*gauss_quadrature(fcn(q1*x+q2).at(0), x, -1, 1);
416  }
417 
418  // Gauss points
419  vector<double> xi;
420  xi.push_back(-sqrt(5 + 2*sqrt(10.0/7))/3);
421  xi.push_back(-sqrt(5 - 2*sqrt(10.0/7))/3);
422  xi.push_back(0);
423  xi.push_back(sqrt(5 - 2*sqrt(10.0/7))/3);
424  xi.push_back(sqrt(5 + 2*sqrt(10.0/7))/3);
425 
426  // Gauss weights
427  vector<double> wi;
428  wi.push_back((322-13*sqrt(70.0))/900.0);
429  wi.push_back((322+13*sqrt(70.0))/900.0);
430  wi.push_back(128/225.0);
431  wi.push_back((322+13*sqrt(70.0))/900.0);
432  wi.push_back((322-13*sqrt(70.0))/900.0);
433 
434  // Evaluate at the Gauss points
435  Function fcn("gauss_quadrature", {x}, {f});
436  vector<SXElem> f_val(5);
437  for (int i=0; i<5; ++i)
438  f_val[i] = fcn(SX(xi[i])).at(0).scalar();
439 
440  // Weighted sum
441  SXElem sum;
442  for (int i=0; i<5; ++i)
443  sum += wi[i]*f_val[i];
444 
445  return sum;
446  }

467  template<>
468  vector<SX>
469  SX::substitute(const vector<SX>& ex, const vector<SX>& v, const vector<SX>& vdef) {
470 
471  // Assert consistent dimensions
472  casadi_assert_warning(v.size()==vdef.size(), "subtitute: number of symbols to replace ( "
473  << v.size() << ") must match number of expressions (" << vdef.size()
474  << ") to replace them with.");
475 
476  // Quick return if all equal
477  bool all_equal = true;
478  for (int k=0; k<v.size(); ++k) {
479  if (v[k].size()!=vdef[k].size() || !is_equal(v[k], vdef[k])) {
480  all_equal = false;
481  break;
482  }
483  }
484  if (all_equal) return ex;
485 
486  // Check sparsities
487  for (int k=0; k<v.size(); ++k) {
488  if (v[k].sparsity()!=vdef[k].sparsity()) {
489  // Expand vdef to sparsity of v if vdef is scalar
490  if (vdef[k].is_scalar() && vdef[k].nnz()==1) {
491  vector<SX> vdef_mod = vdef;
492  vdef_mod[k] = SX(v[k].sparsity(), vdef[k]->at(0), false);
493  return substitute(ex, v, vdef_mod);
494  } else {
495  casadi_error("subsitute(ex, v, vdef): sparsities of v and vdef must match. Got v: "
496  << v[k].dim() << " and " << "vdef: " << vdef[k].dim() << ".");
497  }
498  }
499  }
500 
501 
502  // Otherwise, evaluate symbolically
503  Function F("tmp", v, ex);
504  return F(vdef);
505  }

507  template<>
508  void SX::substitute_inplace(const vector<SX >& v, vector<SX >& vdef,
509  vector<SX >& ex, bool reverse) {
510  // Assert correctness
511  casadi_assert(v.size()==vdef.size());
512  for (int i=0; i<v.size(); ++i) {
513  casadi_assert_message(v[i].is_symbolic(), "the variable is not symbolic");
514  casadi_assert_message(v[i].sparsity() == vdef[i].sparsity(), "the sparsity patterns of the "
515  "expression and its defining bexpression do not match");
516  }
517 
518  // Quick return if empty or single expression
519  if (v.empty()) return;
520 
521  // Function inputs
522  vector<SX> f_in;
523  if (!reverse) f_in.insert(f_in.end(), v.begin(), v.end());
524 
525  // Function outputs
526  vector<SX> f_out = vdef;
527  f_out.insert(f_out.end(), ex.begin(), ex.end());
528 
529  // Write the mapping function
530  Function f("tmp", f_in, f_out);
531 
532  // Get references to the internal data structures
533  auto *ff = dynamic_cast<SXFunction *>(f.get());
534  const vector<ScalarAtomic>& algorithm = ff->algorithm_;
535  vector<SXElem> work(f.getWorkSize());
536 
537  // Iterator to the binary operations
538  vector<SXElem>::const_iterator b_it=ff->operations_.begin();
539 
540  // Iterator to stack of constants
541  vector<SXElem>::const_iterator c_it = ff->constants_.begin();
542 
543  // Iterator to free variables
544  vector<SXElem>::const_iterator p_it = ff->free_vars_.begin();
545 
546  // Evaluate the algorithm
547  for (vector<ScalarAtomic>::const_iterator it=algorithm.begin(); it<algorithm.end(); ++it) {
548  switch (it->op) {
549  case OP_INPUT:
550  // reverse is false, substitute out
551  work[it->i0] = vdef.at(it->i1)->at(it->i2);
552  break;
553  case OP_OUTPUT:
554  if (it->i0 < v.size()) {
555  vdef.at(it->i0)->at(it->i2) = work[it->i1];
556  if (reverse) {
557  // Use the new variable henceforth, substitute in
558  work[it->i1] = v.at(it->i0)->at(it->i2);
559  }
560  } else {
561  // Auxillary output
562  ex.at(it->i0 - v.size())->at(it->i2) = work[it->i1];
563  }
564  break;
565  case OP_CONST: work[it->i0] = *c_it++; break;
566  case OP_PARAMETER: work[it->i0] = *p_it++; break;
567  default:
568  {
569  switch (it->op) {
570  CASADI_MATH_FUN_BUILTIN(work[it->i1], work[it->i2], work[it->i0])
571  }
572 
573  // Avoid creating duplicates
574  const int depth = 2; // NOTE: a higher depth could possibly give more savings
575  work[it->i0].assignIfDuplicate(*b_it++, depth);
576  }
577  }
578  }
579  }

642  SX SX::taylor(const SX& f, const SX& x,
643  const SX& a, int order) {

645  if (f.nnz()!=f.numel())

652  SX dxa = (x-a);

667  SX mtaylor_recursive(const SX& ex, const SX& x, const SX& a, int order,
668  const vector<int>&order_contributions,
669  const SXElem & current_dx=casadi_limits<SXElem>::one,
670  double current_denom=1, int current_order=1) {
671  SX result = substitute(ex, x, a)*current_dx/current_denom;
672  for (int i=0;i<x.nnz();i++) {
673  if (order_contributions[i]<=order) {
674  result += mtaylor_recursive(jacobian(ex, x->at(i)),
675  x, a,
676  order-order_contributions[i],
677  order_contributions,
678  current_dx*(x->at(i)-a->at(i)),
679  current_denom*current_order, current_order+1);
680  }
681  }
682  return result;
683  }

729  template<>
730  void SX::shared(vector<SX >& ex,
731  vector<SX >& v_sx,
732  vector<SX >& vdef_sx,
733  const string& v_prefix,
734  const string& v_suffix) {
735 
736  // Sort the expression
737  Function f("tmp", vector<SX>(), ex);
738  auto *ff = dynamic_cast<SXFunction *>(f.get());
739 
740  // Get references to the internal data structures
741  const vector<ScalarAtomic>& algorithm = ff->algorithm_;
742  vector<SXElem> work(f.getWorkSize());
743  vector<SXElem> work2 = work;
744 
745  // Iterator to the binary operations
746  vector<SXElem>::const_iterator b_it=ff->operations_.begin();
747 
748  // Iterator to stack of constants
749  vector<SXElem>::const_iterator c_it = ff->constants_.begin();
750 
751  // Iterator to free variables
752  vector<SXElem>::const_iterator p_it = ff->free_vars_.begin();
753 
754  // Count how many times an expression has been used
755  vector<int> usecount(work.size(), 0);
756 
757  // Evaluate the algorithm
758  vector<SXElem> v, vdef;
759  for (vector<ScalarAtomic>::const_iterator it=algorithm.begin(); it<algorithm.end(); ++it) {
760  // Increase usage counters
761  switch (it->op) {
762  case OP_CONST:
763  case OP_PARAMETER:
764  break;
765  CASADI_MATH_BINARY_BUILTIN // Binary operation
766  if (usecount[it->i2]==0) {
767  usecount[it->i2]=1;
768  } else if (usecount[it->i2]==1) {
769  // Get a suitable name
770  vdef.push_back(work[it->i2]);
771  usecount[it->i2]=-1; // Extracted, do not extract again
772  }
773  // fall-through
774  case OP_OUTPUT:
775  default: // Unary operation, binary operation or output
776  if (usecount[it->i1]==0) {
777  usecount[it->i1]=1;
778  } else if (usecount[it->i1]==1) {
779  vdef.push_back(work[it->i1]);
780  usecount[it->i1]=-1; // Extracted, do not extract again
781  }
782  }
783 
784  // Perform the operation
785  switch (it->op) {
786  case OP_OUTPUT:
787  break;
788  case OP_CONST:
789  case OP_PARAMETER:
790  usecount[it->i0] = -1; // Never extract since it is a primitive type
791  break;
792  default:
793  work[it->i0] = *b_it++;
794  usecount[it->i0] = 0; // Not (yet) extracted
795  break;
796  }
797  }
798 
799  // Create intermediate variables
800  stringstream v_name;
801  for (int i=0; i<vdef.size(); ++i) {
802  v_name.str(string());
803  v_name << v_prefix << i << v_suffix;
804  v.push_back(SXElem::sym(v_name.str()));
805  }
806 
807  // Mark the above expressions
808  for (int i=0; i<vdef.size(); ++i) {
809  vdef[i].setTemp(i+1);
810  }
811 
812  // Save the marked nodes for later cleanup
813  vector<SXElem> marked = vdef;
814 
815  // Reset iterator
816  b_it=ff->operations_.begin();
817 
818  // Evaluate the algorithm
819  for (vector<ScalarAtomic>::const_iterator it=algorithm.begin(); it<algorithm.end(); ++it) {
820  switch (it->op) {
821  case OP_OUTPUT: ex.at(it->i0)->at(it->i2) = work[it->i1]; break;
822  case OP_CONST: work2[it->i0] = work[it->i0] = *c_it++; break;
823  case OP_PARAMETER: work2[it->i0] = work[it->i0] = *p_it++; break;
824  default:
825  {
826  switch (it->op) {
827  CASADI_MATH_FUN_BUILTIN(work[it->i1], work[it->i2], work[it->i0])
828  }
829  work2[it->i0] = *b_it++;
830 
831  // Replace with intermediate variables
832  int ind = work2[it->i0].getTemp()-1;
833  if (ind>=0) {
834  vdef.at(ind) = work[it->i0];
835  work[it->i0] = v.at(ind);
836  }
837  }
838  }
839  }
840 
841  // Unmark the expressions
842  for (vector<SXElem>::iterator it=marked.begin(); it!=marked.end(); ++it) {
843  it->setTemp(0);
844  }
845 
846  // Save v, vdef
847  v_sx.resize(v.size());
848  copy(v.begin(), v.end(), v_sx.begin());
849  vdef_sx.resize(vdef.size());
850  copy(vdef.begin(), vdef.end(), vdef_sx.begin());
851  }

882  template<>
883  SX SX::poly_roots(const SX& p) {
884  casadi_assert_message(p.size2()==1,
885  "poly_root(): supplied parameter must be column vector but got "
886  << p.dim() << ".");
887  casadi_assert(p.is_dense());
888  if (p.size1()==2) { // a*x + b
889  SX a = p(0);
890  SX b = p(1);
891  return -b/a;
892  } else if (p.size1()==3) { // a*x^2 + b*x + c
893  SX a = p(0);
894  SX b = p(1);
895  SX c = p(2);
896  SX ds = sqrt(b*b-4*a*c);
897  SX bm = -b;
898  SX a2 = 2*a;
899  SX ret = SX::vertcat({(bm-ds)/a2, (bm+ds)/a2});
900  return ret;
901  } else if (p.size1()==4) {
902  // www.cs.iastate.edu/~cs577/handouts/polyroots.pdf
903  SX ai = 1/p(0);
904 
905  SX p_ = p(1)*ai;
906  SX q = p(2)*ai;
907  SX r = p(3)*ai;
908 
909  SX pp = p_*p_;
910 
911  SX a = q - pp/3;
912  SX b = r + 2.0/27*pp*p_-p_*q/3;
913 
914  SX a3 = a/3;
915 
916  SX phi = acos(-b/2/sqrt(-a3*a3*a3));
917 
918  SX ret = SX::vertcat({cos(phi/3), cos((phi+2*pi)/3), cos((phi+4*pi)/3)});
919  ret*= 2*sqrt(-a3);
920 
921  ret-= p_/3;
922  return ret;
923  } else if (p.size1()==5) {
924  SX ai = 1/p(0);
925  SX b = p(1)*ai;
926  SX c = p(2)*ai;
927  SX d = p(3)*ai;
928  SX e = p(4)*ai;
929 
930  SX bb= b*b;
931  SX f = c - (3*bb/8);
932  SX g = d + (bb*b / 8) - b*c/2;
933  SX h = e - (3*bb*bb/256) + (bb * c/16) - (b*d/4);
934  SX poly = SX::vertcat({1, f/2, ((f*f -4*h)/16), -g*g/64});
935  SX y = poly_roots(poly);
936 
937  SX r0 = y(0);
938  SX r1 = y(2);
939 
940  SX p = sqrt(r0); // two non-zero-roots
941  SX q = sqrt(r1);
942 
943  SX r = -g/(8*p*q);
944 
945  SX s = b/4;
946 
947  SX ret = SX::vertcat({
948  p + q + r -s,
949  p - q - r -s,
950  -p + q - r -s,
951  -p - q + r -s});
952  return ret;
953  } else if (is_equal(p(p.nnz()-1)->at(0), 0)) {
954  SX ret = SX::vertcat({poly_roots(p(range(p.nnz()-1))), 0});
955  return ret;
956  } else {
957  casadi_error("poly_root(): can only solve cases for first or second order polynomial. "
958  "Got order " << p.size1()-1 << ".");
959  }
960 
961  }
