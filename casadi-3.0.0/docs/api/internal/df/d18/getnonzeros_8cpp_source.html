183  void GetNonzeros::eval_mx(const std::vector<MX>& arg, std::vector<MX>& res) {
184  // Get all the nonzeros
185  vector<int> nz = all();
186 
187  // Output sparsity
188  const Sparsity& osp = sparsity();
189  const int* orow = osp.row();
190  vector<int> ocol = osp.get_col();
191 
192  // Input sparsity
193  const Sparsity& isp = dep().sparsity();
194  //const vector<int>& irow = isp.row();
195  vector<int> icol = isp.get_col();
196 
197  // Get all input elements
198  vector<int> el_input;
199  isp.find(el_input);
200 
201  // Sparsity pattern being formed and corresponding nonzero mapping
202  vector<int> r_colind, r_row, r_nz, r_ind;
203 
204  // Get the matching nonzeros
205  r_ind.resize(el_input.size());
206  copy(el_input.begin(), el_input.end(), r_ind.begin());
207  arg[0].sparsity().get_nz(r_ind);
208 
209  // Sparsity pattern for the result
210  r_colind.resize(osp.size2()+1); // Col count
211  fill(r_colind.begin(), r_colind.end(), 0);
212  r_row.clear();
213 
214  // Perform the assignments
215  r_nz.clear();
216  for (int k=0; k<nz.size(); ++k) {
217 
218  // Get the corresponding nonzero for the input
219  int el = nz[k];
220 
221  // Skip if zero assignment
222  if (el==-1) continue;
223 
224  // Get the corresponding nonzero in the argument
225  int el_arg = r_ind[el];
226 
227  // Skip if no argument
228  if (el_arg==-1) continue;
229 
230  // Save the assignment
231  r_nz.push_back(el_arg);
232 
233  // Get the corresponding element
234  int i=ocol[k], j=orow[k];
235 
236  // Add to sparsity pattern
237  r_row.push_back(j);
238  r_colind[1+i]++;
239  }
240 
241  // col count -> col offset
242  for (int i=1; i<r_colind.size(); ++i) r_colind[i] += r_colind[i-1];
243 
244  // Create a sparsity pattern from vectors
245  if (r_nz.size()==0) {
246  res[0] = MX(osp.size());
247  } else {
248  Sparsity f_sp(osp.size1(), osp.size2(), r_colind, r_row);
249  res[0] = arg[0]->getGetNonzeros(f_sp, r_nz);
250  }
251  }

335  void GetNonzeros::evalAdj(const std::vector<std::vector<MX> >& aseed,
336  std::vector<std::vector<MX> >& asens) {
337  // Get all the nonzeros
338  vector<int> nz = all();
339 
340  // Number of derivative directions
341  int nadj = aseed.size();
342 
343  // Output sparsity
344  const Sparsity& osp = sparsity();
345  vector<int> ocol = osp.get_col();
346 
347  // Input sparsity
348  const Sparsity& isp = dep().sparsity();
349  //const vector<int>& irow = isp.row();
350  vector<int> icol = isp.get_col();
351 
352  // Get all input elements
353  vector<int> el_input;
354  isp.find(el_input);
355 
356  // Sparsity pattern being formed and corresponding nonzero mapping
357  vector<int> r_colind, r_row, r_nz, r_ind;
358 
359  // Adjoint sensitivities
360  for (int d=0; d<nadj; ++d) {
361 
362  // Get an owning references to the seeds and sensitivities
363  // and clear the seeds for the next run
364  MX aseed0 = aseed[d][0];
365  MX asens0 = asens[d][0]; // Sensitivity before addition
366 
367  // Get the corresponding nz locations in the output sparsity pattern
368  aseed0.sparsity().find(r_nz);
369  osp.get_nz(r_nz);
370 
371  // Filter out ignored entries and check if there is anything to add at all
372  bool elements_to_add = false;
373  for (vector<int>::iterator k=r_nz.begin(); k!=r_nz.end(); ++k) {
374  if (*k>=0) {
375  if (nz[*k]>=0) {
376  elements_to_add = true;
377  } else {
378  *k = -1;
379  }
380  }
381  }
382 
383  // Quick continue of no elements to add
384  if (!elements_to_add) continue;
385 
386  // Get the nz locations in the adjoint sensitivity corresponding to the inputs
387  r_ind.resize(el_input.size());
388  copy(el_input.begin(), el_input.end(), r_ind.begin());
389  asens0.sparsity().get_nz(r_ind);
390 
391  // Enlarge the sparsity pattern of the sensitivity if not all additions fit
392  for (vector<int>::iterator k=r_nz.begin(); k!=r_nz.end(); ++k) {
393  if (*k>=0 && r_ind[nz[*k]]<0) {
394 
395  // Create a new pattern which includes both the the previous seed and the addition
396  Sparsity sp = asens0.sparsity().unite(dep().sparsity());
397  asens0 = asens0->getProject(sp);
398 
399  // Recalculate the nz locations in the adjoint sensitivity corresponding to the inputs
400  copy(el_input.begin(), el_input.end(), r_ind.begin());
401  asens0.sparsity().get_nz(r_ind);
402 
403  break;
404  }
405  }
406 
407  // Have r_nz point to locations in the sensitivity instead of the output
408  for (vector<int>::iterator k=r_nz.begin(); k!=r_nz.end(); ++k) {
409  if (*k>=0) {
410  *k = r_ind[nz[*k]];
411  }
412  }
413 
414  // Add to the element to the sensitivity
415  asens[d][0] = aseed0->getAddNonzeros(asens0, r_nz);
416  }
417  }
