31  std::vector<int> range(int start, int stop, int step, int len) {
32  start = std::min(start, len);
33  stop = std::min(stop, len);
34  int nret = (stop-start)/step + ((stop-start)%step!=0);
35  std::vector<int> ret(nret);
36  int ind = start;
37  for (std::vector<int>::iterator it=ret.begin(); it!=ret.end(); ++it) {
38  *it = ind;
39  ind += step;
40  }
41  return ret;
42  }

51  std::vector<int> lookup(size, 0);
52  std::vector<int> ret;

55  lookup[v[i]] = 1;
