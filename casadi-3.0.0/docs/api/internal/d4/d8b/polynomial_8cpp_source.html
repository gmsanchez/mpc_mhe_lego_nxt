85  Polynomial Polynomial::operator*(const Polynomial& a) const {
86  vector<real_t> p_ret(p_.size() + a.p_.size() - 1, 0);
87  for (int d=0; d<p_.size(); ++d) {
88  for (int d_a=0; d_a<a.p_.size(); ++d_a) {
89  p_ret[d+d_a] += p_[d] * a.p_[d_a];
90  }
91  }
92  return Polynomial(p_ret);
93  }
