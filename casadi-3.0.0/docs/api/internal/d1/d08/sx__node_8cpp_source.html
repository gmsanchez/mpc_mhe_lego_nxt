146  void SXNode::can_inline(std::map<const SXNode*, int>& nodeind) const {
147  // Add or mark node in map
148  std::map<const SXNode*, int>::iterator it=nodeind.find(this);
149  if (it==nodeind.end()) {
150  // First time encountered, mark inlined
151  nodeind.insert(it, make_pair(this, 0));
152 
153  // Handle dependencies with recursion
154  for (int i=0; i<ndep(); ++i) {
155  dep(i)->can_inline(nodeind);
156  }
157  } else if (it->second==0 && op()!=OP_PARAMETER) {
158  // Node encountered before, do not inline (except if symbolic primitive)
159  it->second = -1;
160  }
161  }

163  std::string SXNode::print_compact(std::map<const SXNode*, int>& nodeind,
164  std::vector<std::string>& intermed) const {
165  // Get reference to node index
166  int& ind = nodeind[this];
167 
168  // If positive, already in intermediate expressions
169  if (ind>0) {
170  stringstream ss;
171  ss << "@" << ind;
172  return ss.str();
173  }
174 
175  // Get expressions for dependencies
176  std::string arg[2];
177  for (int i=0; i<ndep(); ++i) {
178  arg[i] = dep(i)->print_compact(nodeind, intermed);
179  }
180 
181  // Get expression for this
182  string s = print(arg[0], arg[1]);
183 
184  // Decide what to do with the expression
185  if (ind==0) {
186  // Inline expression
187  return s;
188  } else {
189  // Add to list of intermediate expressions and return reference
190  intermed.push_back(s);
191  ind = intermed.size(); // For subsequent references
192  stringstream ss;
193  ss << "@" << ind;
194  return ss.str();
195  }
196  }
