94  MX Concat::getGetNonzeros(const Sparsity& sp, const std::vector<int>& nz) const {
95  // Get the first nonnegative nz
96  int nz_test = -1;
97  for (vector<int>::const_iterator i=nz.begin(); i!=nz.end(); ++i) {
98  if (*i>=0) {
99  nz_test = *i;
100  break;
101  }
102  }
103 
104  // Quick return if none
105  if (nz_test<0) return MX::zeros(sp);
106 
107  // Find out to which dependency it might depend
108  int begin=0, end=0;
109  int i;
110  for (i=0; i<ndep(); ++i) {
111  begin = end;
112  end += dep(i).nnz();
113  if (nz_test < end) break;
114  }
115 
116  // Check if any nz refer to a different nonzero
117  for (vector<int>::const_iterator j=nz.begin(); j!=nz.end(); ++j) {
118  if (*j>=0 && (*j < begin || *j >= end)) {
119 
120  // Fallback to the base class
121  return MXNode::getGetNonzeros(sp, nz);
122  }
123  }
124 
125  // All nz refer to the same dependency, update the nonzero indices
126  if (begin==0) {
127  return dep(i)->getGetNonzeros(sp, nz);
128  } else {
129  vector<int> nz_new(nz);
130  for (vector<int>::iterator j=nz_new.begin(); j!=nz_new.end(); ++j) {
131  if (*j>=0) *j -= begin;
132  }
133  return dep(i)->getGetNonzeros(sp, nz_new);
134  }
135  }

167  std::pair<std::vector<int>, std::vector<int> > Diagcat::offset() const {
168  vector<int> offset1(ndep()+1, 0);
169  vector<int> offset2(ndep()+1, 0);
170  for (int i=0; i<ndep(); ++i) {
171  int ncol = dep(i).sparsity().size2();
172  int nrow = dep(i).sparsity().size1();
173  offset2[i+1] = offset2[i] + ncol;
174  offset1[i+1] = offset1[i] + nrow;
175  }
176  return make_pair(offset1, offset2);

179  void Diagcat::evalAdj(const std::vector<std::vector<MX> >& aseed,
180  std::vector<std::vector<MX> >& asens) {
181  // Get offsets for each row and column
182  std::pair<std::vector<int>, std::vector<int> > off = offset();
183 
184  // Adjoint sensitivities
185  int nadj = aseed.size();
186  for (int d=0; d<nadj; ++d) {
187  vector<MX> s = diagsplit(aseed[d][0], off.first, off.second);
188  for (int i=0; i<ndep(); ++i) {
189  asens[d][i] += s[i];
