43  Function jit(const ParsedFile& file) {
44  // Function name
45  string name = file.to_string("NAME");
46 
47  // Number of inputs (default 1)
48  int n_in = file.has("N_IN") ? file.to_int("N_IN") : 1;
49 
50  // Number of outputs (default 1)
51  int n_out = file.has("N_OUT") ? file.to_int("N_OUT") : 1;
52 
53  // Function body
54  string body = file.to_text("BODY");
55 
56  // Read options
57  Dict opts;
58  if (file.has("JAC_BODY")) opts["jac"] = file.to_text("JAC_BODY");
59  if (file.has("HESS_BODY")) opts["hess"] = file.to_text("HESS_BODY");
60  if (file.has("JIT")) opts["jit"] = file.to_int("JIT");
61 
62  // Create function object
63  return jit(name, n_in, n_out, body, opts);
64  }

113  Function Jit::getFullJacobian(const std::string& name, const Dict& opts) {
114  // Create a JIT-function for the Jacobian
115  Dict jit_opts;
116  if (!hess_body_.empty()) jit_opts["jac"] = hess_body_;
117  Function fcn = jit(name + "_jit", n_in_, n_in_*n_out_, jac_body_, jit_opts);
118 
119  // Wrap in an MX function
120  std::vector<MX> arg = fcn.mx_in();
121  std::vector<MX> res = fcn(arg);
122  MX J = reshape(vertcat(res), n_in_, n_out_);
123  return Function(name, arg, {J}, opts);
124  }
