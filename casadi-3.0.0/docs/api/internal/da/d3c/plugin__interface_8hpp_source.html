116  // Quick return if available

146  template<class Derived>
147  typename PluginInterface<Derived>::Plugin
148  PluginInterface<Derived>::loadPlugin(const std::string& pname, bool register_plugin) {
149  // Issue warning and quick return if already loaded
150  if (Derived::solvers_.find(pname) != Derived::solvers_.end()) {
151  casadi_warning("PluginInterface: Solver " + pname + " is already in use. Ignored.");
152  return Plugin();
153  }
154 
155 
156 #ifndef WITH_DL
157  casadi_error("WITH_DL option needed for dynamic loading");
158 #else // WITH_DL
159  // Retrieve the registration function
160  RegFcn reg;
161 
162  // Get the name of the shared library
163  std::string lib = SHARED_LIBRARY_PREFIX "casadi_"
164  + Derived::infix_ + "_" + pname + SHARED_LIBRARY_SUFFIX;
165 
166  // Load the dll
167  std::string regName = "casadi_register_" + Derived::infix_ + "_" + pname;
168 
169  // Build up search paths;
170  std::vector<std::string> search_paths;
171 
172  #ifdef _WIN32
173  char pathsep = ';';
174  const std::string filesep("\\");
175  #else
176  char pathsep = ':';
177  const std::string filesep("/");
178  #endif
179 
180  // Search path: global casadipath option
181  std::stringstream casadipaths(GlobalOptions::getCasadiPath());
182  std::string casadipath;
183  while (std::getline(casadipaths, casadipath, pathsep)) {
184  search_paths.push_back(casadipath);
185  }
186 
187  // Search path: CASADIPATH env variable
188  char* pLIBDIR;
189  pLIBDIR = getenv("CASADIPATH");
190 
191  if (pLIBDIR!=0) {
192  std::stringstream casadipaths(pLIBDIR);
193  std::string casadipath;
194  while (std::getline(casadipaths, casadipath, pathsep)) {
195  search_paths.push_back(casadipath);
196  }
197  }
198 
199  // Search path: bare
200  search_paths.push_back("");
201 
202  // Search path : PLUGIN_EXTRA_SEARCH_PATH
203  #ifdef PLUGIN_EXTRA_SEARCH_PATH
204  search_paths.push_back(
205  std::string("") + PLUGIN_EXTRA_SEARCH_PATH);
206  #endif // PLUGIN_EXTRA_SEARCH_PATH
207 
208  // Search path : current directory
209  search_paths.push_back(".");
210 
211  // Prepare error string
212  std::stringstream errors;
213  errors << "PluginInterface::loadPlugin: Cannot load shared library '"
214  << lib << "': " << std::endl;
215  errors << " (\n"
216  << " Searched directories: 1. casadipath from GlobalOptions\n"
217  << " 2. CASADIPATH env var\n"
218  << " 3. PATH env var (Windows)\n"
219  << " 4. LD_LIBRARY_PATH env var (Linux)\n"
220  << " A library may be 'not found' even if the file exists:\n"
221  << " * library is not compatible (different compiler/bitness)\n"
222  << " * the dependencies are not found\n"
223  << " )";
224 
225  // Alocate a handle pointer
226 #ifdef _WIN32
227  HINSTANCE handle = 0;
228 #else // _WIN32
229  void * handle = 0;
230 #endif
231 
232  // Alocate a handle pointer
233 #ifndef _WIN32
234  int flag = RTLD_LAZY | RTLD_LOCAL;
235 #ifdef WITH_DEEPBIND
236  flag |= RTLD_DEEPBIND;
237 #endif
238 #endif
239 
240  std::string searchpath;
241 
242  // Try getting a handle
243  for (int i=0;i<search_paths.size();++i) {
244  searchpath = search_paths[i];
245 #ifdef _WIN32
246  SetDllDirectory(TEXT(searchpath.c_str()));
247  handle = LoadLibrary(TEXT(lib.c_str()));
248  SetDllDirectory(NULL);
249 #else // _WIN32
250  std::string pname = searchpath.size()==0 ? lib : searchpath + filesep + lib;
251  handle = dlopen(pname.c_str(), flag);
252 #endif // _WIN32
253  if (handle) {
254  break;
255  } else {
256  errors << std::endl << " Tried '" << searchpath << "' :";
257 #ifdef _WIN32
258  errors << std::endl << " Error code (WIN32): " << STRING(GetLastError());
259 #else // _WIN32
260  errors << std::endl << " Error code: " << dlerror();
261 #endif // _WIN32
262  }
263  }
264 
265  casadi_assert_message(handle!=0, errors.str());
266 
267 #ifdef _WIN32
268  reg = (RegFcn)GetProcAddress(handle, TEXT(regName.c_str()));
269 #else // _WIN32
270  // Reset error
271  dlerror();
272 
273  // Load creator
274  reg = (RegFcn)dlsym(handle, regName.c_str());
275 #endif // _WIN32
276  casadi_assert_message(reg!=0,
277  "PluginInterface::loadPlugin: no \"" + regName + "\" found in " + searchpath + ".");
278 
279  // Create a temporary struct
280  Plugin plugin = pluginFromRegFcn(reg);
281  // Register the plugin
282  if (register_plugin) {
283  registerPlugin(plugin);
284  }
285 
286  return plugin;
287 
288 #endif // WITH_DL
289  }

310  PluginInterface<Derived>::getPlugin(const std::string& pname) {
311 
312  // Check if the solver has been loaded
313  auto it=Derived::solvers_.find(pname);
314 
315  // Load the solver if needed
316  if (it==Derived::solvers_.end()) {
317  loadPlugin(pname);
