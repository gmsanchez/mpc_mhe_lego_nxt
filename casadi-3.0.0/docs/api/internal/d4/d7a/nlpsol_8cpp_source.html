237  void Nlpsol::init(const Dict& opts) {
238  // Call the initialization method of the base class
239  FunctionInternal::init(opts);
240 
241  // Default options
242  bool expand = false;
243 
244  // Read options
245  for (auto&& op : opts) {
246  if (op.first=="expand") {
247  expand = op.second;
248  } else if (op.first=="iteration_callback") {
249  fcallback_ = op.second;
250  } else if (op.first=="iteration_callback_step") {
251  callback_step_ = op.second;
252  } else if (op.first=="eval_errors_fatal") {
253  eval_errors_fatal_ = op.second;
254  } else if (op.first=="warn_initial_bounds") {
255  warn_initial_bounds_ = op.second;
256  } else if (op.first=="iteration_callback_ignore_errors") {
257  iteration_callback_ignore_errors_ = op.second;
258  } else if (op.first=="print_time") {
259  print_time_ = op.second;
260  }
261  }
262 
263  // Replace MX oracle with SX oracle?
264  if (expand && nlp_) {
265  Oracle* nlp_new = nlp_->expand();
266  delete nlp_;
267  nlp_ = nlp_new;
268  }
269 
270  // Get dimensions
271  nx_ = nnz_out(NLPSOL_X);
272  np_ = nnz_in(NLPSOL_P);
273  ng_ = nnz_out(NLPSOL_G);
274 
275  if (!fcallback_.is_null()) {
276  // Consistency checks
277  casadi_assert(!fcallback_.is_null());
278  casadi_assert_message(fcallback_.n_out()==1 && fcallback_.numel_out()==1,
279  "Callback function must return a scalar");
280  casadi_assert_message(fcallback_.n_in()==n_out(),
281  "Callback input signature must match the NLP solver output signature");
282  for (int i=0; i<n_out(); ++i) {
283  casadi_assert_message(fcallback_.size_in(i)==size_out(i),
284  "Callback function input size mismatch");
285  // TODO(@jaeandersson): Wrap fcallback_ in a function with correct sparsity
286  casadi_assert_message(fcallback_.sparsity_in(i)==sparsity_out(i),
287  "Not implemented");
288  }
289 
290  // Allocate temporary memory
291  alloc(fcallback_);
292  }
293  }

308  void Nlpsol::checkInputs(void* mem) const {
309  auto m = static_cast<NlpsolMemory*>(mem);
310 
311  // Skip check?
312  if (!inputs_check_) return;
313 
314  const double inf = std::numeric_limits<double>::infinity();
315 
316  // Number of equality constraints
317  int n_eq = 0;
318 
319  // Detect ill-posed problems (simple bounds)
320  for (int i=0; i<nx_; ++i) {
321  double lbx = m->lbx ? m->lbx[i] : 0;
322  double ubx = m->ubx ? m->ubx[i] : 0;
323  double x0 = m->x0 ? m->x0[i] : 0;
324  casadi_assert_message(!(lbx==inf || lbx>ubx || ubx==-inf),
325  "Ill-posed problem detected (x bounds)");
326  if (warn_initial_bounds_ && (x0>ubx || x0<lbx)) {
327  casadi_warning("Nlpsol: The initial guess does not satisfy LBX and UBX. "
328  "Option 'warn_initial_bounds' controls this warning.");
329  break;
330  }
331  if (lbx==ubx) n_eq++;
332  }
333 
334  // Detect ill-posed problems (nonlinear bounds)
335  for (int i=0; i<ng_; ++i) {
336  double lbg = m->lbg ? m->lbg[i] : 0;
337  double ubg = m->ubg ? m->ubg[i] : 0;
338  casadi_assert_message(!(lbg==inf || lbg>ubg || ubg==-inf),
339  "Ill-posed problem detected (g bounds)");
340  if (lbg==ubg) n_eq++;
341  }
342 
343  // Make sure enough degrees of freedom
344  using casadi::to_string; // Workaround, MingGW bug, cf. CasADi issue #890
345  casadi_assert_message(n_eq <= nx_, "NLP is overconstrained: There are " + to_string(n_eq)
346  + " equality constraints but only " + to_string(nx_) + " variables.");
347  }

406  // Convert a float to a string of an exact length.
407  // First it tries fixed precision, then falls back to exponential notation.
408  //
409  // todo(jaeandersson,jgillis): needs either review or unit tests

411  std::string formatFloat(double x, int totalWidth, int maxPrecision, int fallbackPrecision) {
412  std::ostringstream out0;
413  out0 << fixed << setw(totalWidth) << setprecision(maxPrecision) << x;
414  std::string ret0 = out0.str();
415  if (ret0.length() == totalWidth) {
416  return ret0;
417  } else if (ret0.length() > totalWidth) {
418  std::ostringstream out1;
419  out1 << setw(totalWidth) << setprecision(fallbackPrecision) << x;
420  std::string ret1 = out1.str();
421  if (ret1.length() != totalWidth)
422  casadi_error(
423  "ipopt timing formatting fallback is bugged, sorry about that."
424  << "expected " << totalWidth << " digits, but got " << ret1.length()
425  << ", string: \"" << ret1 << "\", number: " << x);
426  return ret1;
427  } else {
428  casadi_error("ipopt timing formatting is bugged, sorry about that.");
429  }

432  void print_stats_line(int maxNameLen, std::string label,
433  double n_call, double t_proc, double t_wall) {
434  // Skip when not called
435  if (n_call == 0) return;
436 
437  std::stringstream s;

439  s

443  if (n_call == -1) {
444  // things like main loop don't have # evals
445  s << endl;
446  } else {
447  s
448  << " "

450  if (n_call < 2) {
451  s << endl;
452  } else {
453  // only print averages if there is more than 1 eval
454  s
455  << " "
456  << formatFloat(1000.0*t_proc/n_call, 10, 2, 3) << " [ms] "
457  << formatFloat(1000.0*t_wall/n_call, 10, 2, 3) << " [ms]"
458  << endl;
459  }
460  }
461  userOut() << s.str();
462  }

464  void Nlpsol::print_fstats(const NlpsolMemory* m) const {
465 
466  size_t maxNameLen=0;
467 
468  // Retrieve all nlp keys
469  std::vector<std::string> keys;
470  std::vector<std::string> keys_other;
471  for (auto &&s : m->fstats) {
472  maxNameLen = max(s.first.size(), maxNameLen);
473  if (s.first.find("nlp")!=std::string::npos) {
474  keys.push_back(s.first);
475  } else if (s.first.find("mainloop")==std::string::npos) {
476  keys_other.push_back(s.first);
477  } else {
478  continue;
479  }
480  }
481 
482  maxNameLen = max(std::string("all previous").size(), maxNameLen);
483  maxNameLen = max(std::string("solver").size(), maxNameLen);
484 
485  // Print header
486  std::stringstream s;
487  std::string blankName(maxNameLen, ' ');
488  s
489  << blankName
490  << " proc wall num mean mean"
491  << endl << blankName
492  << " time time evals proc time wall time";
493  userOut() << s.str() << endl;
494 
495  // Sort the keys according to order
496  std::vector<std::string> keys_order0;
497  std::vector<std::string> keys_order1;
498  std::vector<std::string> keys_order2;
499  for (auto k : keys) {
500  if (k.find("hess")!=std::string::npos) {
501  keys_order2.push_back(k);
502  continue;
503  }
504  if (k.find("grad")!=std::string::npos ||
505  k.find("jac")!=std::string::npos) {
506  keys_order1.push_back(k);
507  continue;
508  }
509  keys_order0.push_back(k);
510  }
511 
512  // Print all NLP stats
513  for (auto keys : {&keys_order0, &keys_order1, &keys_order2}) {
514  std::sort(keys->begin(), keys->end());
515  for (auto k : *keys) {
516  const FStats& fs = m->fstats.at(k);
517  print_stats_line(maxNameLen, k, fs.n_call, fs.t_proc, fs.t_wall);
518  }
519  }
520 
521  // Sum the previously printed stats
522  double t_wall_all_previous = 0;
523  double t_proc_all_previous = 0;
524  for (auto k : keys) {
525  const FStats& fs = m->fstats.at(k);
526  t_proc_all_previous += fs.t_proc;
527  t_wall_all_previous += fs.t_wall;
528  }
529  print_stats_line(maxNameLen, "all previous", -1, t_proc_all_previous, t_wall_all_previous);
530 
531  // Sort and show the remainder of keys
532  std::sort(keys_other.begin(), keys_other.end());
533  for (std::string k : keys_other) {
534  const FStats& fs = m->fstats.at(k);
535  print_stats_line(maxNameLen, k, fs.n_call, fs.t_proc, fs.t_wall);
536  t_proc_all_previous += fs.t_proc;
537  t_wall_all_previous += fs.t_wall;
538  }
539 
540  // Show the mainloop stats
541  const FStats& fs_mainloop = m->fstats.at("mainloop");
542  if (fs_mainloop.n_call>0) {
543  print_stats_line(maxNameLen, "solver", -1,
544  fs_mainloop.t_proc-t_proc_all_previous, fs_mainloop.t_wall-t_wall_all_previous);
545  print_stats_line(maxNameLen, "mainloop", -1, fs_mainloop.t_proc, fs_mainloop.t_wall);
546  }
547 
548  }

563  int Nlpsol::calc_function(NlpsolMemory* m, const Function& fcn,
564  std::initializer_list<const double*> arg,
565  std::initializer_list<double*> res) const {
566  // Respond to a possible Crl+C signals
567  InterruptHandler::check();
568 
569  // Get statistics structure
570  FStats& fstats = m->fstats.at(fcn.name());
571 
572  // Prepare stats, start timer
573  fstats.tic();
574 
575  // Number of inputs and outputs
576  int n_in = fcn.n_in(), n_out = fcn.n_out();
577 
578  // Input buffers
579  fill_n(m->arg, n_in, nullptr);
580  auto arg_it = arg.begin();
581  for (int i=0; i<n_in; ++i) m->arg[i] = *arg_it++;
582  casadi_assert(arg_it==arg.end());
583 
584  // Output buffers
585  fill_n(m->res, n_out, nullptr);
586  auto res_it = res.begin();
587  for (int i=0; i<n_out; ++i) m->res[i] = *res_it++;
588  casadi_assert(res_it==res.end());
589 
590  // Evaluate memory-less
591  try {
592  fcn(m->arg, m->res, m->iw, m->w, 0);
593  } catch(exception& ex) {
594  // Fatal error
595  userOut<true, PL_WARN>()
596  << name() << ":" << fcn.name() << " failed:" << ex.what() << endl;
597  return 1;
598  }
599 
600  // Make sure not NaN or Inf
601  for (int i=0; i<n_out; ++i) {
602  if (!m->res[i]) continue;
603  if (!all_of(m->res[i], m->res[i]+fcn.nnz_out(i), [](double v) { return isfinite(v);})) {
604  userOut<true, PL_WARN>()
605  << name() << ":" << fcn.name() << " failed: NaN or Inf detected for output "
606  << fcn.name_out(i) << endl;
607  return -1;
608  }
609  }
610 
611  // Update stats
612  fstats.toc();
613 
614  // Success
615  return 0;
616  }
