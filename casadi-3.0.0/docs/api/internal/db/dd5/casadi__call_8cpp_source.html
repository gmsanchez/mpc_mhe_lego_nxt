34  MX GenericCall::projectArg(const MX& x, const Sparsity& sp, int i) {
35  if (x.size()==sp.size()) {
36  // Insert sparsity projection nodes if needed
37  return project(x, sp);
38  } else {
39  // Different dimensions
40  if (x.is_empty() || sp.is_empty()) { // NOTE: To permissive?
41  // Replace nulls with zeros of the right dimension
42  return MX::zeros(sp);
43  } else if (x.is_scalar()) {
44  // Scalar argument means set all
45  return MX(sp, x);
46  } else if (x.size1()==sp.size2() && x.size2()==sp.size1() && sp.is_vector()) {
47  // Transposed vector
48  return projectArg(x.T(), sp, i);
49  } else {
50  // Mismatching dimensions
51  casadi_error("Cannot create function call node: Dimension mismatch for argument "
52  << i << ". Argument has shape " << x.size()
53  << " but function input has shape " << sp.size());
54  }
55  }
56  }

58  Call::Call(const Function& fcn, const vector<MX>& arg) : fcn_(fcn) {
59 
60  // Number inputs and outputs
61  int num_in = fcn.n_in();
62  casadi_assert_message(arg.size()==num_in, "Argument list length (" << arg.size()
63  << ") does not match number of inputs (" << num_in << ")"
64  << " for function " << fcn.name());
65 
66  // Create arguments of the right dimensions and sparsity
67  vector<MX> arg1(num_in);
68  for (int i=0; i<num_in; ++i) {
69  arg1[i] = projectArg(arg[i], fcn_.sparsity_in(i), i);
70  }
71  setDependencies(arg1);
72  setSparsity(Sparsity::scalar());
73  }

106  void Call::evalFwd(const vector<vector<MX> >& fseed,
107  vector<vector<MX> >& fsens) {
108  // Nondifferentiated inputs and outputs
109  vector<MX> arg(ndep());
110  for (int i=0; i<arg.size(); ++i) arg[i] = dep(i);
111  vector<MX> res(nout());
112  for (int i=0; i<res.size(); ++i) res[i] = getOutput(i);
113 
114  // Call the cached functions
115  fcn_.forward(arg, res, fseed, fsens);
116  }

118  void Call::evalAdj(const vector<vector<MX> >& aseed,
119  vector<vector<MX> >& asens) {
120  // Nondifferentiated inputs and outputs
121  vector<MX> arg(ndep());
122  for (int i=0; i<arg.size(); ++i) arg[i] = dep(i);
123  vector<MX> res(nout());
124  for (int i=0; i<res.size(); ++i) res[i] = getOutput(i);
125 
126  // Call the cached functions
127  vector<vector<MX> > v;
128  fcn_.reverse(arg, res, aseed, v);
129  for (int i=0; i<v.size(); ++i) {
130  for (int j=0; j<v[i].size(); ++j) {
131  if (!v[i][j].is_empty()) { // TODO(@jaeandersson): Hack
132  asens[i][j] += v[i][j];
133  }
134  }
135  }

154  void Call::generate(CodeGenerator& g, const std::string& mem,
155  const vector<int>& arg, const vector<int>& res) const {
156  if (fcn_->simplifiedCall()) {
157 
158  // Collect input arguments
159  for (int i=0; i<arg.size(); ++i) {
160  g.body << " w[" << i << "]=" << g.workel(arg[i]) << ";" << endl;
161  }
162 
163  // Call function
164  g.body << " " << g(fcn_, "w", "w+"+g.to_string(arg.size())) << ";" << endl;
165 
166  // Collect output arguments
167  for (int i=0; i<res.size(); ++i) {
168  if (res[i]>=0) {
169  g.body << " " << g.workel(res[i]) << "=w[" << (arg.size()+i) << "];" << endl;
170  }
171  }
172  } else {
173 
174  // Collect input arguments
175  for (int i=0; i<arg.size(); ++i) {
176  g.body << " arg1[" << i << "]=" << g.work(arg[i], fcn_.nnz_in(i)) << ";" << endl;
177  }
178 
179  // Collect output arguments
180  for (int i=0; i<res.size(); ++i) {
181  g.body << " res1[" << i << "]=" << g.work(res[i], fcn_.nnz_out(i)) << ";" << endl;
182  }
183 
184  // Call function
185  g.body << " if (" << g(fcn_, "arg1", "res1", "iw", "w") << ") return 1;" << endl;
186  }
