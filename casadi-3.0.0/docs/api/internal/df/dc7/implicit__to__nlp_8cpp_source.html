66  void ImplicitToNlp::init(const Dict& opts) {
67  // Call the base class initializer
68  Rootfinder::init(opts);
69 
70  // Default options
71  string nlpsol_plugin;
72  Dict nlpsol_options;
73 
74  // Read user options
75  for (auto&& op : opts) {
76  if (op.first=="nlpsol") {
77  nlpsol_plugin = op.second.to_string();
78  } else if (op.first=="nlpsol_options") {
79  nlpsol_options = op.second;
80  }
81  }
82 
83  // Free variable in the NLP
84  MX u = MX::sym("u", sparsity_in(iin_));
85 
86  // So that we can pass it on to createParent
87  std::vector<MX> inputs;
88  for (int i=0; i<n_in(); ++i) {
89  if (i!=iin_) {
90  stringstream ss;
91  ss << "p" << i;
92  inputs.push_back(MX::sym(ss.str(), sparsity_in(i)));
93  }
94  }
95  MX p = veccat(inputs);
96 
97  // Dummy NLP objective
98  MX nlp_f = 0;
99 
100  // NLP constraints
101  std::vector< MX > args_call(n_in());
102  args_call[iin_] = u;
103  for (int i=0, i2=0; i<n_in(); ++i)
104  if (i!=iin_) args_call[i] = inputs[i2++];
105  MX nlp_g = f_(args_call).at(iout_);
106 
107  // We're going to use two-argument objective and constraints to allow the use of parameters
108  MXDict nlp = {{"x", u}, {"p", p}, {"f", nlp_f}, {"g", nlp_g}};
109 
110  // Create an Nlpsol instance
111  casadi_assert_message(!nlpsol_plugin.empty(), "'nlpsol' option has not been set");
112  solver_ = nlpsol("nlpsol", nlpsol_plugin, nlp, nlpsol_options);
113  alloc(solver_);
114 
115  // Allocate storage for variable bounds
116  alloc_w(n_, true); // lbx
117  alloc_w(n_, true); // ubx
118 
119  // Allocate storage for NLP solver parameters
120  alloc_w(f_.nnz_in() - nnz_in(iin_), true);
121 
122  // Allocate storage for NLP primal solution
123  alloc_w(n_, true);
124  }

126  void ImplicitToNlp::eval(void* mem, const double** arg, double** res,
127  int* iw, double* w) const {
128  auto m = static_cast<ImplicitToNlpMemory*>(mem);
129 
130  // Buffers for calling the NLP solver
131  const double** arg1 = arg + n_in();
132  double** res1 = res + n_out();
133  fill(arg1, arg1+NLPSOL_NUM_IN, nullptr);
134  fill(res1, res1+NLPSOL_NUM_OUT, nullptr);
135 
136  // Initial guess
137  arg1[NLPSOL_X] = arg[iin_];
138 
139  // Nonlinear bounds
140  arg1[NLPSOL_LBG] = 0;
141  arg1[NLPSOL_UBG] = 0;
142 
143  // Variable bounds
144  double* lbx = w; w += n_;
145  fill_n(lbx, n_, -std::numeric_limits<double>::infinity());
146  arg1[NLPSOL_LBX] = lbx;
147  double* ubx = w; w += n_;
148  fill_n(ubx, n_, std::numeric_limits<double>::infinity());
149  arg1[NLPSOL_UBX] = ubx;
150  for (int k=0; k<u_c_.size(); ++k) {
151  if (u_c_[k] > 0) lbx[k] = 0;
152  if (u_c_[k] < 0) ubx[k] = 0;
153  }
154 
155  // NLP parameters
156  arg1[NLPSOL_P] = w;
157  for (int i=0; i<n_in(); ++i) {
158  if (i!=iin_) {
159  int n = f_.nnz_in(i);
160  if (arg[i]) {
161  copy_n(arg[i], n, w);
162  } else {
163  fill_n(w, n, 0.);
164  }
165  w += n;
166  }
167  }
168 
169  // Primal solution
170  double* x = w; w += n_;
171  res1[NLPSOL_X] = x;
172 
173  // Solve the NLP
174  solver_(arg1, res1, iw, w, 0);
175  m->solver_stats = solver_.stats();
176 
177  // Get the implicit variable
178  if (res[iout_]) {
179  copy_n(x, n_, res[iout_]);
180  }
181 
182  // Check if any auxilary outputs to evaluate
183  bool has_aux = false;
184  for (int i=0; i<n_out(); ++i) {
185  if (i!=iout_ && res[i]) {
186  has_aux = true;
187  break;
188  }
189  }
190 
191  // Evaluate auxilary outputs, if necessary
192  if (has_aux) {
193  copy_n(arg, n_in(), arg1);
194  arg1[iin_] = x;
195  copy_n(res, n_out(), res1);
196  res1[iout_] = 0;
197  f_(arg1, res1, iw, w, 0);
198  }
199  }
