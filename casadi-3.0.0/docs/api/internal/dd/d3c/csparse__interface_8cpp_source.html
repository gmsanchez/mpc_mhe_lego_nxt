66  void CsparseInterface::init_memory(void* mem) const {
67  auto m = static_cast<CsparseMemory*>(mem);
68  m->N = 0;
69  m->S = 0;
70  m->A.nzmax = nnz_in(0); // maximum number of entries
71  m->A.m = size1_in(0); // number of rows
72  m->A.n = size2_in(0); // number of columns
73  m->A.p = const_cast<int*>(sparsity_in(0).colind()); // column pointers (size n+1)
74  // or column indices (size nzmax)
75  m->A.i = const_cast<int*>(sparsity_in(0).row()); // row indices, size nzmax
76  m->A.x = 0; // numerical values, size nzmax
77  m->A.nz = -1; // of entries in triplet matrix, -1 for compressed-col
78 
79  // Temporary
80  m->temp_.resize(m->A.n);
81 
82  // Has the routine been called once
83  m->called_once_ = false;
84  }

86  void CsparseInterface::linsol_factorize(void* mem, const double* A) const {
87  auto m = static_cast<CsparseMemory*>(mem);
88  casadi_assert(A!=0);
89 
90  // Set the nonzeros of the matrix
91  m->A.x = const_cast<double*>(A);
92 
93  if (!m->called_once_) {
94  if (verbose()) {
95  userOut() << "CsparseInterface::prepare: symbolic factorization" << endl;
96  }
97 
98  // ordering and symbolic analysis
99  int order = 0; // ordering?
100  if (m->S) cs_sfree(m->S);
101  m->S = cs_sqr(order, &m->A, 0) ;
102  }
103 
104  m->called_once_ = true;
105 
106  // Make sure that all entries of the linear system are valid
107  for (int k=0; k<sparsity_.nnz(); ++k) {
108  casadi_assert_message(!isnan(A[k]), "Nonzero " << k << " is not-a-number");
109  casadi_assert_message(!isinf(A[k]), "Nonzero " << k << " is infinite");
110  }
111 
112  if (verbose()) {
113  userOut() << "CsparseInterface::prepare: numeric factorization" << endl;
114  userOut() << "linear system to be factorized = " << endl;
115  DM(sparsity_, vector<double>(A, A+sparsity_.nnz())).print_sparse();
116  }
117 
118  double tol = 1e-8;
119 
120  if (m->N) cs_nfree(m->N);
121  m->N = cs_lu(&m->A, m->S, tol) ; // numeric LU factorization
122  if (m->N==0) {
123  DM temp(sparsity_, vector<double>(A, A+sparsity_.nnz()));
124  temp = sparsify(temp);
125  if (temp.sparsity().is_singular()) {
126  stringstream ss;
127  ss << "CsparseInterface::prepare: factorization failed due to matrix"
128  " being singular. Matrix contains numerical zeros which are "
129  "structurally non-zero. Promoting these zeros to be structural "
130  "zeros, the matrix was found to be structurally rank deficient."
131  " sprank: " << sprank(temp.sparsity()) << " <-> " << temp.size2() << endl;
132  if (verbose()) {
133  ss << "Sparsity of the linear system: " << endl;
134  sparsity_.print(ss); // print detailed
135  }
136  throw CasadiException(ss.str());
137  } else {
138  stringstream ss;
139  ss << "CsparseInterface::prepare: factorization failed, check if Jacobian is singular"
140  << endl;
141  if (verbose()) {
142  ss << "Sparsity of the linear system: " << endl;
143  sparsity_.print(ss); // print detailed
144  }
145  throw CasadiException(ss.str());
146  }
147  }
148  casadi_assert(m->N!=0);
149  }

151  void CsparseInterface::linsol_solve(void* mem, double* x, int nrhs, bool tr) const {
152  auto m = static_cast<CsparseMemory*>(mem);
153  casadi_assert(m->N!=0);
154 
155  double *t = &m->temp_.front();
156 
157  for (int k=0; k<nrhs; ++k) {
158  if (tr) {
159  cs_pvec(m->S->q, x, t, m->A.n) ; // t = P2*b
160  casadi_assert(m->N->U!=0);
161  cs_utsolve(m->N->U, t) ; // t = U'\t
162  cs_ltsolve(m->N->L, t) ; // t = L'\t
163  cs_pvec(m->N->pinv, t, x, m->A.n) ; // x = P1*t
164  } else {
165  cs_ipvec(m->N->pinv, x, t, m->A.n) ; // t = P1\b
166  cs_lsolve(m->N->L, t) ; // t = L\t
167  cs_usolve(m->N->U, t) ; // t = U\t
168  cs_ipvec(m->S->q, t, x, m->A.n) ; // x = P2\t
169  }
170  x += ncol();
171  }
172  }
