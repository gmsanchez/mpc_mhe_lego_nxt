558  template<typename T, typename F, typename L>
559  void linspace(std::vector<T> &v, const F& first, const L& last) {
560  if (v.size()<2)
561  throw CasadiException("std::linspace: vector must contain at least two elements");
562 
563  // Increment
564  T increment = (last-first)/T(v.size()-1);
565 
566  v[0] = first;
567  for (unsigned i=1; i<v.size()-1; ++i)
568  v[i] = v[i-1] + increment;
569  v[v.size()-1] = last;
570  }

596  template<typename T>
597  void sort(const std::vector<T> &values, std::vector<T> &sorted_values,
598  std::vector<int> &indices, bool invert_indices) {
599  // Call recursively if indices need to be inverted
600  if (invert_indices) {
601  std::vector<int> inverted;
602  sort(values, sorted_values, inverted, false);
603  indices.resize(inverted.size());
604  for (size_t i=0; i<inverted.size(); ++i) {
605  indices[inverted[i]] = i;
606  }
607  return;
608  }
609 
610  // Create list of indices
611  indices.resize(values.size());
612  for (size_t i=0; i<indices.size(); ++i) indices[i] = i;
613 
614  // Sort this list by the values
615  std::sort(indices.begin(), indices.end(), sortCompare<T>(values));
616 
617  // Sort the values accordingly
618  sorted_values.resize(values.size());
619  for (size_t i=0; i<values.size(); ++i) {
620  sorted_values[i] = values[indices[i]];
621  }
622  }

761  template<typename T>
762  std::vector< std::vector<T> > swapIndices(const std::vector< std::vector<T> > &mat) {
763 
764  // Get the matrix dimensions
765  int n = mat.size();
766  int m = -1;
767  for (int i=0;i<n;++i) {
768  casadi_assert_message(m==-1 || m==mat[i].size(),
769  "swapIndices(vector<vector>) dimension mismatch.");
770  if (m==-1) m = mat[i].size();
771  }
772 
773  // Allocate the result
774  std::vector< std::vector<T> > ret(m);
775  for (int i=0;i<m;++i) {
776  ret[i].resize(n);
777  }
778 
779  // Assign the result
780  for (int i=0;i<n;++i) {
