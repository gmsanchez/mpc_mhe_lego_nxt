70  void KnitroInterface::init(const Dict& opts) {
71  // Call the init method of the base class
72  Nlpsol::init(opts);
73 
74  // Read user options
75  for (auto&& op : opts) {
76  if (op.first=="knitro") {
77  opts_ = op.second;
78  } else if (op.first=="contype") {
79  contype_ = op.second;
80  }
81  }
82 
83  // Type of constraints, general by default
84  if (contype_.empty()) {
85  contype_.resize(ng_, KTR_CONTYPE_GENERAL);
86  } else {
87  casadi_assert(contype_.size()==ng_);
88  }
89 
90  // Setup NLP functions
91  fg_fcn_ = create_function("nlp_fg", {"x", "p"}, {"f", "g"});
92  gf_jg_fcn_ = create_function("nlp_gf_jg", {"x", "p"}, {"grad_f_x", "jac_g_x"});
93  jacg_sp_ = gf_jg_fcn_.sparsity_out(1);
94  hess_l_fcn_ = create_function("nlp_jac_f", {"x", "p", "lam_f", "lam_g"},
95  {"hess_gamma_x_x"},
96  {{"gamma", {"f", "g"}}});
97  hesslag_sp_ = hess_l_fcn_.sparsity_out(0);
98 
99  // Allocate persistent memory
100  alloc_w(nx_, true); // wx_
101  alloc_w(nx_, true); // wlbx_
102  alloc_w(nx_, true); // wubx_
103  alloc_w(ng_, true); // wlbg_
104  alloc_w(ng_, true); // wubg_
105  }

131  void KnitroInterface::solve(void* mem) const {
132  auto m = static_cast<KnitroMemory*>(mem);
133 
134  // Allocate KNITRO memory block (move back to init!)
135  casadi_assert(m->kc_handle==0);
136  m->kc_handle = KTR_new();
137  casadi_assert(m->kc_handle!=0);
138  int status;
139 
140  // Jacobian sparsity
141  vector<int> Jcol, Jrow;
142  if (!jacg_sp_.is_null()) {
143  Jcol = jacg_sp_.get_col();
144  Jrow = jacg_sp_.get_row();
145  }
146 
147  // Hessian sparsity
148  int nnzH = hesslag_sp_.is_null() ? 0 : hesslag_sp_.nnz();
149  vector<int> Hcol, Hrow;
150  if (nnzH>0) {
151  Hcol = hesslag_sp_.get_col();
152  Hrow = hesslag_sp_.get_row();
153  status = KTR_set_int_param_by_name(m->kc_handle, "hessopt", KTR_HESSOPT_EXACT);
154  casadi_assert_message(status==0, "KTR_set_int_param failed");
155  } else {
156  status = KTR_set_int_param_by_name(m->kc_handle, "hessopt", KTR_HESSOPT_LBFGS);
157  casadi_assert_message(status==0, "KTR_set_int_param failed");
158  }
159 
160  // Pass user set options
161  for (auto&& op : opts_) {
162  // Try double
163  if (op.second.can_cast_to(OT_DOUBLE)) {
164  status = KTR_set_double_param_by_name(m->kc_handle, op.first.c_str(), op.second);
165  if (status==0) continue;
166  }
167 
168  // Try integer
169  if (op.second.can_cast_to(OT_INT)) {
170  status = KTR_set_int_param_by_name(m->kc_handle, op.first.c_str(), op.second);
171  if (status==0) continue;
172  }
173 
174  // try string
175  if (op.second.can_cast_to(OT_STRING)) {
176  string str = op.second.to_string();
177  status = KTR_set_char_param_by_name(m->kc_handle, op.first.c_str(), str.c_str());
178  if (status==0) continue;
179  }
180 
181  // Error if reached this point
182  casadi_error("KNITRO error setting option \"" + op.first + "\"");
183  }
184 
185  // "Correct" upper and lower bounds
186  casadi_copy(m->x0, nx_, m->wx);
187  casadi_copy(m->lbx, nx_, m->wlbx);
188  casadi_copy(m->ubx, nx_, m->wubx);
189  casadi_copy(m->lbg, ng_, m->wlbg);
190  casadi_copy(m->ubg, ng_, m->wubg);
191  for (int i=0; i<nx_; ++i) if (isinf(m->wlbx[i])) m->wlbx[i] = -KTR_INFBOUND;
192  for (int i=0; i<nx_; ++i) if (isinf(m->wubx[i])) m->wubx[i] = KTR_INFBOUND;
193  for (int i=0; i<ng_; ++i) if (isinf(m->wlbg[i])) m->wlbg[i] = -KTR_INFBOUND;
194  for (int i=0; i<ng_; ++i) if (isinf(m->wubg[i])) m->wubg[i] = KTR_INFBOUND;
195 
196  // Initialize KNITRO
197  status = KTR_init_problem(m->kc_handle, nx_, KTR_OBJGOAL_MINIMIZE,
198  KTR_OBJTYPE_GENERAL, m->wlbx, m->wubx, ng_, get_ptr(contype_),
199  m->wlbg, m->wubg, Jcol.size(), get_ptr(Jcol), get_ptr(Jrow),
200  nnzH, get_ptr(Hrow), get_ptr(Hcol), m->wx, 0); // initial lambda
201  casadi_assert_message(status==0, "KTR_init_problem failed");
202 
203  // Register callback functions
204  status = KTR_set_func_callback(m->kc_handle, &callback);
205  casadi_assert_message(status==0, "KTR_set_func_callback failed");
206 
207  status = KTR_set_grad_callback(m->kc_handle, &callback);
208  casadi_assert_message(status==0, "KTR_set_grad_callbackfailed");
209 
210  if (nnzH>0) {
211  status = KTR_set_hess_callback(m->kc_handle, &callback);
212  casadi_assert_message(status==0, "KTR_set_hess_callbackfailed");
213  }
214 
215  // Lagrange multipliers
216  vector<double> lambda(nx_+ng_);
217 
218  // Solve NLP
219  double f;
220  status = KTR_solve(m->kc_handle, m->wx, get_ptr(lambda), 0, &f,
221  0, 0, 0, 0, 0, static_cast<void*>(&m));
222  m->return_status = return_codes(status);
223 
224  // Output primal solution
225  casadi_copy(m->wx, nx_, m->x);
226 
227  // Output dual solution
228  casadi_copy(get_ptr(lambda), ng_, m->lam_g);
229  casadi_copy(get_ptr(lambda)+ng_, nx_, m->lam_x);
230 
231  // Output optimal cost
232  if (m->f) *m->f = f;
233 
234  // Calculate constraints
235  if (m->g) calc_function(m, fg_fcn_, {m->wx, m->p}, {0, m->g});
236 
237  // Free memory (move to destructor!)
238  KTR_free(&m->kc_handle);
239  m->kc_handle = 0;
240  }

242  int KnitroInterface::callback(const int evalRequestCode, const int n, const int m, const int nnzJ,
243  const int nnzH, const double* const x, const double* const lambda,
244  double* const obj, double* const c, double* const objGrad,
245  double* const jac, double* const hessian, double* const hessVector,
246  void *userParams) {
247  try {
248  // Get a pointer to the calling object
249  auto m = static_cast<KnitroMemory*>(userParams);
250 
251  // Direct to the correct function
252  switch (evalRequestCode) {
253  case KTR_RC_EVALFC:
254  m->self.calc_function(m, m->self.fg_fcn_, {x, m->p}, {obj, c});
255  break;
256  case KTR_RC_EVALGA:
257  m->self.calc_function(m, m->self.gf_jg_fcn_, {x, m->p}, {objGrad, jac});
258  break;
259  case KTR_RC_EVALH:
260  {
261  double sigma = 1.;
262  if (m->self.calc_function(m, m->self.hess_l_fcn_, {x, m->p, &sigma, lambda}, {hessian})) {
263  casadi_error("calc_hess_l failed");
264  }
265  }
266  break;
267  default:
268  casadi_error("KnitroInterface::callback: unknown method");
269  }
270 
271  return 0;
272  } catch(exception& ex) {
273  userOut<true, PL_WARN>() << "KnitroInterface::callback caugth exception: "
274  << ex.what() << endl;
275  return -1;
276  }
277  }
