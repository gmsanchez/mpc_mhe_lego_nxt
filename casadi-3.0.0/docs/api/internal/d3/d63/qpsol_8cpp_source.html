113  template<typename M>
114  Function qpsol_nlp(const std::string& name, const std::string& solver,
115  const std::map<std::string, M>& qp, const Dict& opts) {
116  // We have: minimize f(x) = 1/2 * x' H x + c'x
117  // subject to lbx <= x <= ubx
118  // lbg <= g(x) = A x + b <= ubg
119  M x, p, f, g;
120  for (auto&& i : qp) {
121  if (i.first=="x") {
122  x = i.second;
123  } else if (i.first=="p") {
124  p = i.second;
125  } else if (i.first=="f") {
126  f = i.second;
127  } else if (i.first=="g") {
128  g = i.second;
129  } else {
130  casadi_error("No such field: " + i.first);
131  }
132  }
133  if (g.is_empty(true)) g = M(0, 1); // workaround
134 
135  // Gradient of the objective: gf == Hx + g
136  M gf = M::gradient(f, x);
137 
138  // Identify the linear term in the objective
139  M c = substitute(gf, x, M::zeros(x.sparsity()));
140 
141  // Identify the quadratic term in the objective
142  M H = M::jacobian(gf, x, true);
143 
144  // Identify the constant term in the constraints
145  M b = substitute(g, x, M::zeros(x.sparsity()));
146 
147  // Identify the linear term in the constraints
148  M A = M::jacobian(g, x);
149 
150  // Create a function for calculating the required matrices vectors
151  Function prob(name + "_qp", {x, p}, {H, c, A, b});
152 
153  // Create the QP solver
154  Function qpsol_f = qpsol(name + "_qpsol", solver,
155  {{"h", H.sparsity()}, {"a", A.sparsity()}}, opts);
156 
157  // Create an MXFunction with the right signature
158  vector<MX> ret_in(NLPSOL_NUM_IN);
159  ret_in[NLPSOL_X0] = MX::sym("x0", x.sparsity());
160  ret_in[NLPSOL_P] = MX::sym("p", p.sparsity());
161  ret_in[NLPSOL_LBX] = MX::sym("lbx", x.sparsity());
162  ret_in[NLPSOL_UBX] = MX::sym("ubx", x.sparsity());
163  ret_in[NLPSOL_LBG] = MX::sym("lbg", g.sparsity());
164  ret_in[NLPSOL_UBG] = MX::sym("ubg", g.sparsity());
165  ret_in[NLPSOL_LAM_X0] = MX::sym("lam_x0", x.sparsity());
166  ret_in[NLPSOL_LAM_G0] = MX::sym("lam_g0", g.sparsity());
167  vector<MX> ret_out(NLPSOL_NUM_OUT);
168 
169  // Get expressions for the QP matrices and vectors
170  vector<MX> v(NL_NUM_IN);
171  v[NL_X] = ret_in[NLPSOL_X0];
172  v[NL_P] = ret_in[NLPSOL_P];
173  v = prob(v);
174 
175  // Call the QP solver
176  vector<MX> w(QPSOL_NUM_IN);
177  w[QPSOL_H] = v.at(0);
178  w[QPSOL_G] = v.at(1);
179  w[QPSOL_A] = v.at(2);
180  w[QPSOL_LBX] = ret_in[NLPSOL_LBX];
181  w[QPSOL_UBX] = ret_in[NLPSOL_UBX];
182  w[QPSOL_LBA] = ret_in[NLPSOL_LBG] - v.at(3);
183  w[QPSOL_UBA] = ret_in[NLPSOL_UBG] - v.at(3);
184  w[QPSOL_X0] = ret_in[NLPSOL_X0];
185  w[QPSOL_LAM_X0] = ret_in[NLPSOL_LAM_X0];
186  w = qpsol_f(w);
187 
188  // Get expressions for the solution
189  ret_out[NLPSOL_X] = w[QPSOL_X];
190  ret_out[NLPSOL_F] = w[QPSOL_COST];
191  ret_out[NLPSOL_G] = mtimes(v.at(2), w[QPSOL_X]) + v.at(3);
192  ret_out[NLPSOL_LAM_X] = w[QPSOL_LAM_X];
193  ret_out[NLPSOL_LAM_G] = w[QPSOL_LAM_A];
194  ret_out[NLPSOL_LAM_P] = MX::nan(p.sparsity());
195  return Function(name, ret_in, ret_out, nlpsol_in(), nlpsol_out(),
196  {{"default_in", nlpsol_default_in()}});
197  }

209  // Constructor
210  Qpsol::Qpsol(const std::string& name, const std::map<std::string, Sparsity> &st)
211  : FunctionInternal(name) {
212  for (auto i=st.begin(); i!=st.end(); ++i) {
213  if (i->first=="a") {
214  A_ = i->second;
215  } else if (i->first=="h") {
216  H_ = i->second;
217  } else {
218  casadi_error("Unrecognized field in QP structure: " << i->first);
219  }
220  }
221 
222  n_ = H_.size2();
223  nc_ = A_.is_null() ? 0 : A_.size1();
224 
225  if (!A_.is_null()) {
226  casadi_assert_message(A_.size2()==n_,
227  "Got incompatible dimensions. min x'Hx + G'x s.t. LBA <= Ax <= UBA :"
228  << std::endl <<
229  "H: " << H_.dim() << " - A: " << A_.dim() << std::endl <<
230  "We need: H_.size2()==A_.size2()" << std::endl);
231  }
232 
233  casadi_assert_message(H_.is_symmetric(),
234  "Got incompatible dimensions. min x'Hx + G'x" << std::endl <<
235  "H: " << H_.dim() <<
236  "We need H square & symmetric" << std::endl);
237 
238  // Sparsity
239  Sparsity x_sparsity = Sparsity::dense(n_, 1);
240  Sparsity bounds_sparsity = Sparsity::dense(nc_, 1);
241  }

309  void Qpsol::checkInputs(const double* lbx, const double* ubx,
310  const double* lba, const double* uba) const {
311  for (int i=0; i<n_; ++i) {
312  double lb = lbx ? lbx[i] : 0., ub = ubx ? ubx[i] : 0.;
313  casadi_assert_message(lb <= ub,
314  "LBX[" << i << "] <= UBX[" << i << "] was violated. "
315  << "Got LBX[" << i << "]=" << lb <<
316  " and UBX[" << i << "] = " << ub << ".");
317  }
318  for (int i=0; i<nc_; ++i) {
319  double lb = lba ? lba[i] : 0., ub = uba ? uba[i] : 0.;
320  casadi_assert_message(lb <= ub,
321  "LBA[" << i << "] <= UBA[" << i << "] was violated. "
322  << "Got LBA[" << i << "] = " << lb <<
323  " and UBA[" << i << "] = " << ub << ".");
324  }
325  }
