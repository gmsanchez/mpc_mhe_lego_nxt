70  void BinaryMX<ScX, ScY>::evalFwd(const std::vector<std::vector<MX> >& fseed,
71  std::vector<std::vector<MX> >& fsens) {
72  // Get partial derivatives
73  MX pd[2];
74  casadi_math<MX>::der(op_, dep(0), dep(1), shared_from_this<MX>(), pd);
75 
76  // Propagate forward seeds
77  for (int d=0; d<fsens.size(); ++d) {
78  fsens[d][0] = pd[0]*fseed[d][0] + pd[1]*fseed[d][1];
79  }
80  }

82  template<bool ScX, bool ScY>
83  void BinaryMX<ScX, ScY>::evalAdj(const std::vector<std::vector<MX> >& aseed,
84  std::vector<std::vector<MX> >& asens) {
85  // Get partial derivatives
86  MX pd[2];
87  casadi_math<MX>::der(op_, dep(0), dep(1), shared_from_this<MX>(), pd);
88 
89  // Propagate adjoint seeds
90  for (int d=0; d<aseed.size(); ++d) {
91  MX s = aseed[d][0];
92  for (int c=0; c<2; ++c) {
93  // Get increment of sensitivity c
94  MX t = pd[c]*s;
95 
96  // If dimension mismatch (i.e. one argument is scalar), then sum all the entries
97  if (!t.is_scalar() && t.size() != dep(c).size()) {
98  if (pd[c].size()!=s.size()) pd[c] = MX(s.sparsity(), pd[c]);
99  t = dot(pd[c], s);
100  }
101 
102  // Propagate the seeds
103  asens[d][c] += t;
104  }
105  }
106  }

108  template<bool ScX, bool ScY>
109  void BinaryMX<ScX, ScY>::
110  generate(CodeGenerator& g, const std::string& mem,
111  const std::vector<int>& arg, const std::vector<int>& res) const {
112  // Quick return if nothing to do
113  if (nnz()==0) return;
114 
115  // Check if inplace
116  bool inplace;
117  switch (op_) {
118  case OP_ADD:
119  case OP_SUB:
120  case OP_MUL:
121  case OP_DIV:
122  inplace = res[0]==arg[0];
123  break;
124  default:
125  inplace = false;
126  break;
127  }
128 
129  // Print indent
130  g.body << " ";
131 
132  // Scalar names of arguments (start assuming all scalars)
133  string r = g.workel(res[0]);
134  string x = g.workel(arg[0]);
135  string y = g.workel(arg[1]);
136 
137  // Codegen loop, if needed
138  if (nnz()>1) {
139  // Iterate over result
140  g.body << "for (i=0, " << "rr=" << g.work(res[0], nnz());
141  r = "(*rr++)";
142 
143  // Iterate over first argument?
144  if (!ScX && !inplace) {
145  g.body << ", cr=" << g.work(arg[0], dep(0).nnz());
146  x = "(*cr++)";
147  }
148 
149  // Iterate over second argument?
150  if (!ScY) {
151  g.body << ", cs=" << g.work(arg[1], dep(1).nnz());
152  y = "(*cs++)";
153  }
154 
155  // Close loop
156  g.body << "; i<" << nnz() << "; ++i) ";
157  }
158 
159  // Perform operation
160  g.body << r << " ";
161  if (inplace) {
162  casadi_math<double>::printSep(op_, g.body);
163  g.body << "= " << y;
164  } else {
165  g.body << " = ";
166  casadi_math<double>::printPre(op_, g.body);
167  g.body << x;
168  casadi_math<double>::printSep(op_, g.body);
169  g.body << y;
170  casadi_math<double>::printPost(op_, g.body);
171  }
172  g.body << ";" << endl;
173  }

205  template<bool ScX, bool ScY>
206  void BinaryMX<ScX, ScY>::
207  spFwd(const bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem) {
208  const bvec_t *a0=arg[0], *a1=arg[1];
209  bvec_t *r=res[0];
210  int n=nnz();
211  for (int i=0; i<n; ++i) {
212  if (ScX && ScY)
213  *r++ = *a0 | *a1;
214  else if (ScX && !ScY)
215  *r++ = *a0 | *a1++;
216  else if (!ScX && ScY)
217  *r++ = *a0++ | *a1;
218  else
219  *r++ = *a0++ | *a1++;
220  }
221  }

252  template<bool ScX, bool ScY>
253  MX BinaryMX<ScX, ScY>::getBinary(int op, const MX& y, bool scX, bool scY) const {
254  if (!GlobalOptions::simplification_on_the_fly) return MXNode::getBinary(op, y, scX, scY);
255 
256  switch (op_) {
257  case OP_ADD:
258  if (op==OP_SUB && MX::is_equal(y, dep(0), maxDepth())) return dep(1);
259  if (op==OP_SUB && MX::is_equal(y, dep(1), maxDepth())) return dep(0);
260  break;
261  case OP_SUB:
262  if (op==OP_SUB && MX::is_equal(y, dep(0), maxDepth())) return -dep(1);
263  if (op==OP_ADD && MX::is_equal(y, dep(1), maxDepth())) return dep(0);
264  break;
265  default: break; // no rule
266  }
267 
268  // Fallback to default implementation
269  return MXNode::getBinary(op, y, scX, scY);
