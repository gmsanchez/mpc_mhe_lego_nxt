103  void LapackLu::linsol_factorize(void* mem, const double* A) const {
104  auto m = static_cast<LapackLuMemory*>(mem);
105 
106  // Dimensions
107  int nrow = this->nrow();
108  int ncol = this->ncol();
109 
110  // Get the elements of the matrix, dense format
111  casadi_densify(A, sparsity_, get_ptr(m->mat), false);
112 
113  if (equilibriate_) {
114  // Calculate the col and row scaling factors
115  double colcnd, rowcnd; // ratio of the smallest to the largest col/row scaling factor
116  double amax; // absolute value of the largest matrix element
117  int info = -100;
118  dgeequ_(&ncol, &nrow, get_ptr(m->mat), &ncol, get_ptr(m->r),
119  get_ptr(m->c), &colcnd, &rowcnd, &amax, &info);
120  if (info < 0)
121  throw CasadiException("LapackQrDense::prepare: "
122  "dgeequ_ failed to calculate the scaling factors");
123  if (info>0) {
124  stringstream ss;
125  ss << "LapackLu::prepare: ";
126  if (info<=ncol) ss << (info-1) << "-th row (zero-based) is exactly zero";
127  else ss << (info-1-ncol) << "-th col (zero-based) is exactly zero";
128 
129  userOut() << "Warning: " << ss.str() << endl;
130 
131 
132 
133  if (allow_equilibration_failure_) userOut() << "Warning: " << ss.str() << endl;
134  else casadi_error(ss.str());
135  }
136 
137  // Equilibrate the matrix if scaling was successful
138  if (info!=0)
139  dlaqge_(&ncol, &nrow, get_ptr(m->mat), &ncol, get_ptr(m->r), get_ptr(m->c),
140  &colcnd, &rowcnd, &amax, &m->equed);
141  else
142  m->equed = 'N';
143  }
144 
145  // Factorize the matrix
146  int info = -100;
147  dgetrf_(&ncol, &ncol, get_ptr(m->mat), &ncol, get_ptr(m->ipiv), &info);
148  casadi_assert_message(info==0, "LapackLu::prepare: "
149  "dgetrf_ failed to factorize the Jacobian");
150  }

152  void LapackLu::linsol_solve(void* mem, double* x, int nrhs, bool tr) const {
153  auto m = static_cast<LapackLuMemory*>(mem);
154 
155  // Dimensions
156  int nrow = this->nrow();
157  int ncol = this->ncol();
158 
159  // Scale the right hand side
160  if (tr) {
161  if (m->equed=='C' || m->equed=='B')
162  for (int rhs=0; rhs<nrhs; ++rhs)
163  for (int i=0; i<nrow; ++i)
164  x[i+rhs*nrow] *= m->c[i];
165  } else {
166  if (m->equed=='R' || m->equed=='B')
167  for (int rhs=0; rhs<nrhs; ++rhs)
168  for (int i=0; i<ncol; ++i)
169  x[i+rhs*nrow] *= m->r[i];
170  }
171 
172  // Solve the system of equations
173  int info = 100;
174  char trans = tr ? 'T' : 'N';
175  dgetrs_(&trans, &ncol, &nrhs, get_ptr(m->mat), &ncol, get_ptr(m->ipiv), x, &ncol, &info);
176  if (info != 0) throw CasadiException("LapackLu::solve: "
177  "failed to solve the linear system");
178 
179  // Scale the solution
180  if (tr) {
181  if (m->equed=='R' || m->equed=='B')
182  for (int rhs=0; rhs<nrhs; ++rhs)
183  for (int i=0; i<ncol; ++i)
184  x[i+rhs*nrow] *= m->r[i];
185  } else {
186  if (m->equed=='C' || m->equed=='B')
187  for (int rhs=0; rhs<nrhs; ++rhs)
188  for (int i=0; i<nrow; ++i)
189  x[i+rhs*nrow] *= m->c[i];
190  }
191  }
