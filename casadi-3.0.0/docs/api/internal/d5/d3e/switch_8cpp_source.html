87  void Switch::init(const Dict& opts) {
88  // Initialize the functions, get input and output sparsities
89  // Input and output sparsities
90  std::vector<Sparsity> sp_in, sp_out;
91  int num_in = -1, num_out=-1;
92  for (int k=0; k<=f_.size(); ++k) {
93  Function& fk = k<f_.size() ? f_[k] : f_def_;
94  if (fk.is_null()) continue;
95  if (num_in<0) {
96  // Number of inputs and outputs
97  num_in=fk.n_in();
98  num_out=fk.n_out();
99  // Output sparsity
100  sp_out.resize(num_out);
101  for (int i=0; i<num_out; ++i) sp_out[i] = fk.sparsity_out(i);
102  // Input sparsity
103  sp_in.resize(num_in);
104  for (int i=0; i<num_in; ++i) sp_in[i] = fk.sparsity_in(i);
105  } else {
106  // Assert matching number of inputs and outputs
107  casadi_assert(num_in==fk.n_in());
108  casadi_assert(num_out==fk.n_out());
109  // Intersect with output sparsity
110  for (int i=0; i<num_out; ++i) {
111  sp_out[i] = sp_out[i].intersect(fk.sparsity_out(i));
112  }
113  // Intersect with input sparsity
114  for (int i=0; i<num_in; ++i) {
115  sp_in[i] = sp_in[i].intersect(fk.sparsity_in(i));
116  }
117  }
118  }
119 
120  // Illegal to pass only "null" functions
121  casadi_assert_message(num_in>=0, "All functions are null");
122 
123  // Call the initialization method of the base class
124  FunctionInternal::init(opts);
125 
126  // Get required work
127  for (int k=0; k<=f_.size(); ++k) {
128  const Function& fk = k<f_.size() ? f_[k] : f_def_;
129  if (fk.is_null()) continue;
130 
131  // Get local work vector sizes
132  alloc(fk);
133  size_t sz_w = fk.sz_w();
134 
135  // Add size for input buffers
136  for (int i=1; i<n_in(); ++i) {
137  const Sparsity& s = fk.sparsity_in(i-1);
138  if (s!=sparsity_in(i)) sz_w += s.nnz();
139  }
140 
141  // Add size for output buffers
142  for (int i=0; i<n_out(); ++i) {
143  const Sparsity& s = fk.sparsity_out(i);
144  if (s!=sparsity_out(i)) sz_w += s.nnz();
145  }
146 
147  // Make sure enough work for this
148  alloc_w(sz_w);
149  }
150  }

152  void Switch::eval(void* mem, const double** arg, double** res, int* iw, double* w) const {
153  // Get conditional
154  int k = static_cast<int>(*arg[0]);
155 
156  // Get the function to be evaluated
157  const Function& fk = k<f_.size() ? f_[k] : f_def_;
158 
159  // Input buffers
160  for (int i=1; i<n_in(); ++i) {
161  const Sparsity& s = fk.sparsity_in(i-1);
162  casadi_assert_message(s==sparsity_in(i), "Not implemented");
163  }
164 
165  // Output buffers
166  for (int i=0; i<n_out(); ++i) {
167  const Sparsity& s = fk.sparsity_out(i);

169  }
170 
171  // Evaluate the corresponding function
172  fk(arg+1, res, iw, w, 0);

175  Function Switch
176  ::get_forward(const std::string& name, int nfwd, Dict& opts) {
177  // Derivative of each case
178  vector<Function> der(f_.size());
179  for (int k=0; k<f_.size(); ++k) {
180  if (!f_[k].is_null()) der[k] = f_[k].forward(nfwd);
181  }
182 
183  // Default case
184  Function der_def;
185  if (!f_def_.is_null()) der_def = f_def_.forward(nfwd);
186 
187  // New Switch for derivatives
188  stringstream ss;
189  ss << "fwd" << nfwd << "_" << name_;
190  Function sw = Function::conditional(ss.str(), der, der_def);
191 
192  // Construct wrapper inputs and arguments for calling sw
193  vector<MX> arg = mx_in();
194  vector<MX> res = mx_out();
195  vector<vector<MX> > seed = symbolicFwdSeed(nfwd, arg);
196  // Wrapper input being constructed
197  vector<MX> w_in = arg;
198  w_in.insert(w_in.end(), res.begin(), res.end());
199  // Arguments for calling sw being constructed
200  vector<MX> v;
201  v.insert(v.end(), arg.begin(), arg.end());
202  v.insert(v.end(), res.begin(), res.end());
203  for (int d=0; d<nfwd; ++d) {
204  // ignore seed for ind
205  seed[d][0] = MX::sym(seed[d][0].name(), Sparsity::scalar(false));
206  // Add to wrapper input
207  w_in.insert(w_in.end(), seed[d].begin(), seed[d].end());
208  // Add to sw argument vector
209  v.insert(v.end(), seed[d].begin()+1, seed[d].end());
210  }
211 
212  // Create wrapper
213  casadi_assert(v.size()==sw.n_in());
214  return Function(name, w_in, sw(v), opts);
215  }

217  Function Switch
218  ::get_reverse(const std::string& name, int nadj, Dict& opts) {
219  // Derivative of each case
220  vector<Function> der(f_.size());
221  for (int k=0; k<f_.size(); ++k) {
222  if (!f_[k].is_null()) der[k] = f_[k].reverse(nadj);
223  }
224 
225  // Default case
226  Function der_def;
227  if (!f_def_.is_null()) der_def = f_def_.reverse(nadj);
228 
229  // New Switch for derivatives
230  stringstream ss;
231  ss << "adj" << nadj << "_" << name_;
232  Function sw = Function::conditional(ss.str(), der, der_def);
233 
234  // Construct wrapper inputs and arguments for calling sw
235  vector<MX> arg = mx_in();
236  vector<MX> res = mx_out();
237  vector<vector<MX> > seed = symbolicAdjSeed(nadj, res);
238  vector<MX> w_in = arg;
239  w_in.insert(w_in.end(), res.begin(), res.end());
240  // Arguments for calling sw being constructed
241  vector<MX> v;
242  v.push_back(arg.at(0)); // index
243  for (int d=0; d<nadj; ++d) {
244  // Add to wrapper input
245  w_in.insert(w_in.end(), seed[d].begin(), seed[d].end());
246  // Add to sw argument vector
247  v.insert(v.end(), arg.begin()+1, arg.end());
248  v.insert(v.end(), res.begin(), res.end());
249  v.insert(v.end(), seed[d].begin(), seed[d].end());
250  }
251 
252  // Construct wrapper outputs
253  casadi_assert(v.size()==sw.n_in());
254  v = sw(v);
255  vector<MX> w_out;
256  MX ind_sens = MX(1, 1); // no dependency on index
257  vector<MX>::const_iterator v_it = v.begin(), v_it_next;
258  for (int d=0; d<nadj; ++d) {
259  w_out.push_back(ind_sens);
260  v_it_next = v_it + (n_in()-1);
261  w_out.insert(w_out.end(), v_it, v_it_next);
262  v_it = v_it_next;
263  }
264 
265  // Create wrapper
266  return Function(name, w_in, w_out, opts);
267  }

291  void Switch::generateBody(CodeGenerator& g) const {
292  // Codegen as if-else
293  bool if_else = f_.size()==1;
294 
295  // Codegen condition
296  g.body << " " << (if_else ? "if" : "switch") << " (to_int(arg[0][0])) {" << endl;
297 
298  // Loop over cases/functions
299  for (int k=0; k<=f_.size(); ++k) {
300 
301  // For if, reverse order
302  int k1 = if_else ? 1-k : k;
303 
304  if (!if_else) {
305  // Codegen cases
306  if (k1<f_.size()) {
307  g.body << " case " << k1 << ":" << endl;
308  } else {
309  g.body << " default:" << endl;
310  }
311  } else if (k1==0) {
312  // Else
313  g.body << " } else {" << endl;
314  }
315 
316  // Get the function:
317  const Function& fk = k1<f_.size() ? f_[k1] : f_def_;
318  if (fk.is_null()) {
319  g.body << " return 1;" << endl;
320  } else {
321  // Call function
322  if (g.simplifiedCall(fk)) {
323  casadi_error("Not implemented.");
324  } else {
325  g.body << " if (" << g(fk, "arg+1", "res", "iw", "w") << ") return 1;" << endl;
326  if (!if_else)
327  g.body << " break;" << endl;
328  }
329  }
330  }
331 
332  // End switch/else
333  g.body << " }" << endl;
334  }
