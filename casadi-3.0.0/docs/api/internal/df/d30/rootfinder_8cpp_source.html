112  void Rootfinder::init(const Dict& opts) {
113 
114  // Default (temporary) options
115  Dict linear_solver_options;
116  string linear_solver = "csparse";
117 
118  // Read options
119  for (auto&& op : opts) {
120  if (op.first=="implicit_input") {
121  iin_ = op.second;
122  } else if (op.first=="implicit_output") {
123  iout_ = op.second;
124  } else if (op.first=="jacobian_function") {
125  jac_ = op.second;
126  } else if (op.first=="linear_solver_options") {
127  linear_solver_options = op.second;
128  } else if (op.first=="linear_solver") {
129  linear_solver = op.second.to_string();
130  } else if (op.first=="constraints") {
131  u_c_ = op.second;
132  }
133  }
134 
135  // Get the number of equations and check consistency
136  casadi_assert_message(iin_>=0 && iin_<f_.n_in() && f_.n_in()>0,
137  "Implicit input not in range");
138  casadi_assert_message(iout_>=0 && iout_<f_.n_out() && f_.n_out()>0,
139  "Implicit output not in range");
140  casadi_assert_message(f_.sparsity_out(iout_).is_dense()
141  && f_.sparsity_out(iout_).is_column(),
142  "Residual must be a dense vector");
143  casadi_assert_message(f_.sparsity_in(iin_).is_dense()
144  && f_.sparsity_in(iin_).is_column(),
145  "Unknown must be a dense vector");
146  n_ = f_.nnz_out(iout_);
147  casadi_assert_message(n_ == f_.nnz_in(iin_),
148  "Dimension mismatch. Input size is "
149  << f_.nnz_in(iin_)
150  << ", while output size is "
151  << f_.nnz_out(iout_));
152 
153  // Call the base class initializer
154  FunctionInternal::init(opts);
155 
156 
157  // Generate Jacobian if not provided
158  if (jac_.is_null()) jac_ = f_.jacobian(iin_, iout_);
159 
160  // Check for structural singularity in the Jacobian
161  casadi_assert_message(
162  !jac_.sparsity_out(0).is_singular(),
163  "Rootfinder::init: singularity - the jacobian is structurally rank-deficient. "
164  "sprank(J)=" << sprank(jac_.sparsity_out(0)) << " (instead of "<< jac_.size1_out(0) << ")");
165 
166  // Get the linear solver creator function
167  if (linsol_.is_null()) {
168  linsol_ = linsol("linsol", linear_solver,
169  jac_.sparsity_out(0), 1, linear_solver_options);
170  } else {
171  casadi_assert(linsol_.sparsity_in(0)==jac_.sparsity_out(0));
172  }
173  alloc(linsol_);
174 
175  // Constraints
176  casadi_assert_message(u_c_.size()==n_ || u_c_.empty(),
177  "Constraint vector if supplied, must be of length n, but got "
178  << u_c_.size() << " and n = " << n_);
179 
180  // Allocate sufficiently large work vectors
181  alloc(f_);
182  size_t sz_w = f_.sz_w();
183  if (!jac_.is_null()) {
184  alloc(jac_);
185  sz_w = max(sz_w, jac_.sz_w());
186  }
187  alloc_w(sz_w + 2*static_cast<size_t>(n_));
188  }

190  Function Rootfinder
191  ::get_forward(const std::string& name, int nfwd, Dict& opts) {
192  // Symbolic expression for the input
193  vector<MX> arg = mx_in();
194  arg[iin_] = MX::sym(arg[iin_].name() + "_guess",
195  Sparsity(arg[iin_].size()));
196  vector<MX> res = mx_out();
197  vector<vector<MX> > fseed = symbolicFwdSeed(nfwd, arg), fsens;
198  forward(arg, res, fseed, fsens, false, false);
199 
200  // Construct return function
201  arg.insert(arg.end(), res.begin(), res.end());
202  for (int d=0; d<nfwd; ++d) arg.insert(arg.end(), fseed[d].begin(), fseed[d].end());
203  res.clear();
204  for (int d=0; d<nfwd; ++d) res.insert(res.end(), fsens[d].begin(), fsens[d].end());
205  return Function(name, arg, res, opts);
206  }

226  void Rootfinder::spFwd(const bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem) {
227  int num_out = n_out();
228  int num_in = n_in();
229  bvec_t* tmp1 = w; w += n_;
230  bvec_t* tmp2 = w; w += n_;
231 
232  // Propagate dependencies through the function
233  const bvec_t** arg1 = arg+n_in();
234  copy(arg, arg+num_in, arg1);
235  arg1[iin_] = 0;
236  bvec_t** res1 = res+n_out();
237  fill_n(res1, num_out, static_cast<bvec_t*>(0));
238  res1[iout_] = tmp1;
239  f_(arg1, res1, iw, w, 0);
240 
241  // "Solve" in order to propagate to z
242  fill_n(tmp2, n_, 0);
243  linsol_.linsol_spsolve(tmp2, tmp1, false);
244  if (res[iout_]) copy(tmp2, tmp2+n_, res[iout_]);
245 
246  // Propagate to auxiliary outputs
247  if (num_out>1) {
248  arg1[iin_] = tmp2;
249  copy(res, res+num_out, res1);
250  res1[iout_] = 0;
251  f_(arg1, res1, iw, w, 0);
252  }
253  }

255  void Rootfinder::spAdj(bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem) {
256  int num_out = n_out();
257  int num_in = n_in();
258  bvec_t* tmp1 = w; w += n_;
259  bvec_t* tmp2 = w; w += n_;
260 
261  // Get & clear seed corresponding to implicitly defined variable
262  if (res[iout_]) {
263  copy(res[iout_], res[iout_]+n_, tmp1);
264  fill_n(res[iout_], n_, 0);
265  } else {
266  fill_n(tmp1, n_, 0);
267  }
268 
269  // Propagate dependencies from auxiliary outputs to z
270  bvec_t** res1 = res+num_out;
271  copy(res, res+num_out, res1);
272  res1[iout_] = 0;
273  bvec_t** arg1 = arg+num_in;
274  copy(arg, arg+num_in, arg1);
275  arg1[iin_] = tmp1;
276  if (num_out>1) {
277  f_.rev(arg1, res1, iw, w, 0);
278  }
279 
280  // "Solve" in order to get seed
281  fill_n(tmp2, n_, 0);
282  linsol_.linsol_spsolve(tmp2, tmp1, true);
283 
284  // Propagate dependencies through the function
285  for (int i=0; i<num_out; ++i) res1[i] = 0;
286  res1[iout_] = tmp2;
287  arg1[iin_] = 0; // just a guess
288  f_.rev(arg1, res1, iw, w, 0);
289  }

295  void Rootfinder::
296  forward(const std::vector<MX>& arg, const std::vector<MX>& res,
297  const std::vector<std::vector<MX> >& fseed,
298  std::vector<std::vector<MX> >& fsens,
299  bool always_inline, bool never_inline) {
300  // Number of directional derivatives
301  int nfwd = fseed.size();
302  fsens.resize(nfwd);
303 
304  // Quick return if no seeds
305  if (nfwd==0) return;
306 
307  // Propagate through f_
308  vector<MX> f_arg(arg);
309  f_arg.at(iin_) = res.at(iout_);
310  vector<MX> f_res(res);
311  f_res.at(iout_) = MX(size_in(iin_)); // zero residual
312  std::vector<std::vector<MX> > f_fseed(fseed);
313  for (int d=0; d<nfwd; ++d) {
314  f_fseed[d].at(iin_) = MX(size_in(iin_)); // ignore seeds for guess
315  }
316  f_.forward(f_arg, f_res, f_fseed, fsens, always_inline, never_inline);
317 
318  // Get expression of Jacobian
319  MX J = jac_(f_arg).front();
320 
321  // Solve for all the forward derivatives at once
322  vector<MX> rhs(nfwd);
323  for (int d=0; d<nfwd; ++d) rhs[d] = vec(fsens[d][iout_]);
324  rhs = horzsplit(J->getSolve(-horzcat(rhs), false, linsol_));
325  for (int d=0; d<nfwd; ++d) fsens[d][iout_] = reshape(rhs[d], size_in(iin_));
326 
327  // Propagate to auxiliary outputs
328  int num_out = n_out();
329  if (num_out>1) {
330  for (int d=0; d<nfwd; ++d) f_fseed[d][iin_] = fsens[d][iout_];
331  f_.forward(f_arg, f_res, f_fseed, fsens, always_inline, never_inline);
332  for (int d=0; d<nfwd; ++d) fsens[d][iout_] = f_fseed[d][iin_]; // Otherwise overwritten
333  }
334  }

336  void Rootfinder::
337  reverse(const std::vector<MX>& arg, const std::vector<MX>& res,
338  const std::vector<std::vector<MX> >& aseed,
339  std::vector<std::vector<MX> >& asens,
340  bool always_inline, bool never_inline) {
341 
342  // Number of directional derivatives
343  int nadj = aseed.size();
344  asens.resize(nadj);
345 
346  // Quick return if no seeds
347  if (nadj==0) return;
348 
349  // Get expression of Jacobian
350  vector<MX> f_arg(arg);
351  f_arg[iin_] = res.at(iout_);
352  MX J = jac_(f_arg).front();
353 
354  // Get adjoint seeds for calling f
355  int num_out = n_out();
356  int num_in = n_in();
357  vector<MX> f_res(res);
358  f_res[iout_] = MX(size_in(iin_)); // zero residual
359  vector<vector<MX> > f_aseed(nadj);
360  for (int d=0; d<nadj; ++d) {
361  f_aseed[d].resize(num_out);
362  for (int i=0; i<num_out; ++i) f_aseed[d][i] = i==iout_ ? f_res[iout_] : aseed[d][i];
363  }
364 
365  // Propagate dependencies from auxiliary outputs
366  vector<MX> rhs(nadj);
367  vector<vector<MX> > asens_aux;
368  if (num_out>1) {
369  f_.reverse(f_arg, f_res, f_aseed, asens_aux, always_inline, never_inline);
370  for (int d=0; d<nadj; ++d) rhs[d] = vec(asens_aux[d][iin_] + aseed[d][iout_]);
371  } else {
372  for (int d=0; d<nadj; ++d) rhs[d] = vec(aseed[d][iout_]);
373  }
374 
375  // Solve for all the adjoint seeds at once
376  rhs = horzsplit(J->getSolve(-horzcat(rhs), true, linsol_));
377  for (int d=0; d<nadj; ++d) {
378  for (int i=0; i<num_out; ++i) {
379  if (i==iout_) {
380  f_aseed[d][i] = reshape(rhs[d], size_out(i));
381  } else {
382  // Avoid counting the auxiliary seeds twice
383  f_aseed[d][i] = MX(size_out(i));
384  }
385  }
386  }
387 
388  // No dependency on guess (1)
389  vector<MX> tmp(nadj);
390  for (int d=0; d<nadj; ++d) {
391  asens[d].resize(num_in);
392  tmp[d] = asens[d][iin_].is_empty(true) ? MX(size_in(iin_)) : asens[d][iin_];
393  }
394 
395  // Propagate through f_
396  f_.reverse(f_arg, f_res, f_aseed, asens, always_inline, never_inline);
397 
398  // No dependency on guess (2)
399  for (int d=0; d<nadj; ++d) {
400  asens[d][iin_] = tmp[d];
401  }
402 
403  // Add contribution from auxiliary outputs
404  if (num_out>1) {
405  for (int d=0; d<nadj; ++d) {
406  for (int i=0; i<num_in; ++i) if (i!=iin_) asens[d][i] += asens_aux[d][i];
407  }
408  }
409  }
