69  void SymbolicQr::init(const Dict& opts) {
70  // Call the base class initializer
71  Linsol::init(opts);
72 
73  // Default options
74  bool codegen = false;
75  string compiler = "gcc -fPIC -O2";
76 
77  // Read options
78  for (auto&& op : opts) {
79  if (op.first=="codegen") {
80  codegen = op.second;
81  } else if (op.first=="compiler") {
82  compiler = op.second.to_string();
83  }
84  }
85 
86  // Make sure that command processor is available
87  if (codegen) {
88 #ifdef WITH_DL
89  int flag = system(static_cast<const char*>(0));
90  casadi_assert_message(flag!=0, "No command procesor available");
91 #else // WITH_DL
92  casadi_error("Codegen requires CasADi to be compiled with option \"WITH_DL\" enabled");
93 #endif // WITH_DL
94  }
95 
96  // Symbolic expression for A
97  SX A = SX::sym("A", sparsity_in(LINSOL_A));
98 
99  // Get the inverted column permutation
100  std::vector<int> inv_colperm(colperm_.size());
101  for (int k=0; k<colperm_.size(); ++k)
102  inv_colperm[colperm_[k]] = k;
103 
104  // Get the inverted row permutation
105  std::vector<int> inv_rowperm(rowperm_.size());
106  for (int k=0; k<rowperm_.size(); ++k)
107  inv_rowperm[rowperm_[k]] = k;
108 
109  // Permute the linear system
110  SX Aperm = A(rowperm_, colperm_);
111 
112  // Generate the QR factorization function
113  SX Q1, R1;
114  qr(Aperm, Q1, R1);
115  Function fact_fcn("QR_fact", {A}, {Q1, R1});
116 
117  // Optionally generate c code and load as DLL
118  if (codegen) {
119  stringstream ss;
120  ss << "symbolic_qr_fact_fcn_" << this;
121  fact_fcn_ = dynamicCompilation(fact_fcn, ss.str(),
122  "Symbolic QR factorization function", compiler);
123  } else {
124  fact_fcn_ = fact_fcn;
125  }
126  alloc(fact_fcn_);
127 
128  // Symbolic expressions for solve function
129  SX Q = SX::sym("Q", Q1.sparsity());
130  SX R = SX::sym("R", R1.sparsity());
131  SX b = SX::sym("b", size1_in(LINSOL_B), 1);
132 
133  // Solve non-transposed
134  // We have Pb' * Q * R * Px * x = b <=> x = Px' * inv(R) * Q' * Pb * b
135 
136  // Permute the right hand sides
137  SX bperm = b(rowperm_, Slice());
138 
139  // Solve the factorized system
140  SX xperm = SX::solve(R, mtimes(Q.T(), bperm));
141 
142  // Permute back the solution
143  SX x = xperm(inv_colperm, Slice());
144 
145  // Generate the QR solve function
146  vector<SX> solv_in = {Q, R, b};
147  Function solv_fcn("QR_solv", solv_in, {x});
148 
149  // Optionally generate c code and load as DLL
150  if (codegen) {
151  stringstream ss;
152  ss << "symbolic_qr_solv_fcn_N_" << this;
153  solv_fcn_N_ = dynamicCompilation(solv_fcn, ss.str(), "QR_solv_N", compiler);
154  } else {
155  solv_fcn_N_ = solv_fcn;
156  }
157  alloc(solv_fcn_N_);
158 
159  // Solve transposed
160  // We have (Pb' * Q * R * Px)' * x = b
161  // <=> Px' * R' * Q' * Pb * x = b
162  // <=> x = Pb' * Q * inv(R') * Px * b
163 
164  // Permute the right hand side
165  bperm = b(colperm_, Slice());
166 
167  // Solve the factorized system
168  xperm = mtimes(Q, SX::solve(R.T(), bperm));
169 
170  // Permute back the solution
171  x = xperm(inv_rowperm, Slice());
172 
173  // Mofify the QR solve function
174  solv_fcn = Function("QR_solv_T", solv_in, {x});
175 
176  // Optionally generate c code and load as DLL
177  if (codegen) {
178  stringstream ss;
179  ss << "symbolic_qr_solv_fcn_T_" << this;
180  solv_fcn_T_ = dynamicCompilation(solv_fcn, ss.str(), "QR_solv_T", compiler);
181  } else {
182  solv_fcn_T_ = solv_fcn;
183  }
184  alloc(solv_fcn_T_);
185 
186  // Temporary storage
187  alloc_w(neq_, true);
188  }

206  void SymbolicQr::linsol_factorize(void* mem, const double* A) const {
207  auto m = static_cast<SymbolicQrMemory*>(mem);
208 
209  // Factorize
210  fill_n(m->arg, fact_fcn_.n_in(), nullptr);
211  m->arg[0] = A;
212  fill_n(m->res, fact_fcn_.n_out(), nullptr);
213  m->res[0] = get_ptr(m->q);
214  m->res[1] = get_ptr(m->r);
215  fact_fcn_(m->arg, m->res, m->iw, m->w);
216  }

218  void SymbolicQr::linsol_solve(void* mem, double* x, int nrhs, bool tr) const {
219  auto m = static_cast<SymbolicQrMemory*>(mem);
220 
221  // Select solve function
222  const Function& solv = tr ? solv_fcn_T_ : solv_fcn_N_;
223 
224  // Solve for all right hand sides
225  fill_n(m->arg, solv.n_in(), nullptr);
226  m->arg[0] = get_ptr(m->q);
227  m->arg[1] = get_ptr(m->r);
228  fill_n(m->res, solv.n_out(), nullptr);
229  for (int i=0; i<nrhs; ++i) {
230  copy_n(x, neq_, m->w); // Copy x to a temporary
231  m->arg[2] = m->w;
232  m->res[0] = x;
233  solv(m->arg, m->res, m->iw, m->w+neq_, 0);
234  x += neq_;
235  }
236  }

238  void SymbolicQr::linsol_eval_sx(const SXElem** arg, SXElem** res, int* iw, SXElem* w, int mem,
239  bool tr, int nrhs) {
240  casadi_assert(arg[0]!=0);
241  casadi_assert(arg[1]!=0);
242  casadi_assert(res[0]!=0);
243 
244  // Get A and factorize it
245  SX A = SX::zeros(sparsity_in(LINSOL_A));
246  copy(arg[1], arg[1]+A.nnz(), A->begin());
247  vector<SX> v = fact_fcn_(A);
248 
249  // Select solve function
250  Function& solv = tr ? solv_fcn_T_ : solv_fcn_N_;
251 
252  // Solve for every right hand side
253  v.push_back(SX::zeros(A.size1()));
254  const SXElem* a=arg[0];
255  SXElem* r=res[0];
256  for (int i=0; i<nrhs; ++i) {
257  copy(a, a+v[2].nnz(), v[2]->begin());
258  SX rr = solv(v).at(0);
259  copy(rr->begin(), rr->end(), r);
260  r += rr.nnz();
261  }
262  }
