85  void CplexInterface::init(const Dict& opts) {
86  // Call the init method of the base class
87  Qpsol::init(opts);
88 
89  // Default options
90  qp_method_ = 0;
91  dump_to_file_ = false;
92  dump_filename_ = "qp.dat";
93  tol_ = 1e-6;
94  dep_check_ = 0;
95  warm_start_ = false;
96 
97  // Read options
98  for (auto&& op : opts) {
99  if (op.first=="cplex") {
100  opts_ = op.second;
101  } else if (op.first=="qp_method") {
102  qp_method_ = op.second;
103  } else if (op.first=="dump_to_file") {
104  dump_to_file_ = op.second;
105  } else if (op.first=="dump_filename") {
106  dump_filename_ = op.second.to_string();
107  } else if (op.first=="tol") {
108  tol_ = op.second;
109  } else if (op.first=="dep_check") {
110  dep_check_ = op.second;
111  } else if (op.first=="warm_start") {
112  warm_start_ = op.second;
113  }
114  }
115 
116  // Are we solving a mixed-integer problem?
117  mip_ = !discrete_.empty()
118  && find(discrete_.begin(), discrete_.end(), true)!=discrete_.end();
119 
120  // Type of variable
121  if (mip_) {
122  ctype_.resize(n_);
123  for (int i=0; i<n_; ++i) {
124  ctype_[i] = discrete_[i] ? 'I' : 'C';
125  }
126  }
127 
128  // Allocate work vectors
129  alloc_w(n_, true); // g
130  alloc_w(n_, true); // lbx
131  alloc_w(n_, true); // ubx
132  alloc_w(nc_, true); // lba
133  alloc_w(nc_, true); // uba
134  alloc_w(nnz_in(QPSOL_H), true); // H
135  alloc_w(nnz_in(QPSOL_A), true); // A
136  alloc_w(n_, true); // x
137  alloc_w(n_, true); // lam_x
138  alloc_w(nc_, true); // lam_a
139  }

141  void CplexInterface::init_memory(void* mem) const {
142  auto m = static_cast<CplexMemory*>(mem);
143 
144  // Start CPLEX
145  int status;
146  casadi_assert(m->env==0);
147  m->env = CPXopenCPLEX(&status);
148  if (m->env==0) {
149  char errmsg[CPXMESSAGEBUFSIZE];
150  CPXgeterrorstring(m->env, status, errmsg);
151  casadi_error(string("Cannot initialize CPLEX environment: ") + errmsg);
152  }
153 
154  // Set parameters to their default values
155  if (CPXsetdefaults(m->env)) {
156  casadi_error("CPXsetdefaults failed");
157  }
158 
159  // Enable output by default
160  if (CPXsetintparam(m->env, CPX_PARAM_SCRIND, CPX_ON)) {
161  casadi_error("Failure setting CPX_PARAM_SCRIND");
162  }
163 
164  // Optimality tolerance
165  if (CPXsetdblparam(m->env, CPX_PARAM_EPOPT, tol_)) {
166  casadi_error("Failure setting CPX_PARAM_EPOPT");
167  }
168 
169  // Feasibility tolerance
170  if (CPXsetdblparam(m->env, CPX_PARAM_EPRHS, tol_)) {
171  casadi_error("Failure setting CPX_PARAM_EPRHS");
172  }
173 
174  // We start with barrier if crossover was chosen.
175  if (CPXsetintparam(m->env, CPX_PARAM_QPMETHOD, qp_method_ == 7 ? 4 : qp_method_)) {
176  casadi_error("Failure setting CPX_PARAM_QPMETHOD");
177  }
178 
179  // Setting dependency check option
180  if (CPXsetintparam(m->env, CPX_PARAM_DEPIND, dep_check_)) {
181  casadi_error("Failure setting CPX_PARAM_DEPIND");
182  }
183 
184  // Setting crossover algorithm
185  if (qp_method_ == 7) {
186  if (CPXsetintparam(m->env, CPX_PARAM_BARCROSSALG, 1)) {
187  casadi_error("Failure setting CPX_PARAM_BARCROSSALG");
188  }
189  }
190 
191  // Set parameters
192  for (auto&& op : opts_) {
193  // Get parameter index
194  int whichparam;
195  if (CPXgetparamnum(m->env, op.first.c_str(), &whichparam)) {
196  casadi_error("No such CPLEX parameter: " + op.first);
197  }
198 
199  // Get type of parameter
200  int paramtype;
201  if (CPXgetparamtype(m->env, whichparam, &paramtype)) {
202  casadi_error("CPXgetparamtype failed");
203  }
204 
205  // Pass to CPLEX
206  switch (paramtype) {
207  case CPX_PARAMTYPE_NONE:
208  casadi_error("CPX_PARAMTYPE_NONE unsupported");
209  break;
210  case CPX_PARAMTYPE_INT:
211  status = CPXsetintparam(m->env, whichparam, op.second);
212  break;
213  case CPX_PARAMTYPE_DOUBLE:
214  status = CPXsetdblparam(m->env, whichparam, op.second);
215  break;
216  case CPX_PARAMTYPE_STRING:
217  status = CPXsetstrparam(m->env, whichparam,
218  static_cast<string>(op.second).c_str());
219  break;
220  case CPX_PARAMTYPE_LONG:
221  status = CPXsetlongparam(m->env, whichparam,
222  static_cast<CPXLONG>(static_cast<int>(op.second)));
223  break;
224  default:
225  casadi_error("Unknown CPLEX parameter type (" << paramtype << ") for " + op.first);
226  }
227  // Error handling
228  if (status) {
229  casadi_error("Failure setting option " + op.first);
230  }
231  }
232 
233  // Doing allocation of CPLEX data
234  // Objective is to be minimized
235  m->objsen = CPX_MIN;
236 
237  // Allocation of data
238  // Type of constraint
239  m->sense.resize(nc_);
240  // Right-hand side of constraints
241  m->rhs.resize(nc_);
242  // Range value for lower AND upper bounded constraints
243  m->rngval.resize(nc_);
244  // Basis for primal variables
245  m->cstat.resize(n_);
246  m->rstat.resize(nc_);
247 
248  // Matrix A, count the number of elements per column
249  const Sparsity& A_sp = sparsity_in(QPSOL_A);
250  m->matcnt.resize(A_sp.size2());
251  transform(A_sp.colind()+1, A_sp.colind() + A_sp.size2()+1, A_sp.colind(), m->matcnt.begin(),
252  minus<int>());
253 
254  // Matrix H, count the number of elements per column
255  const Sparsity& H_sp = sparsity_in(QPSOL_H);
256  m->qmatcnt.resize(H_sp.size2());
257  transform(H_sp.colind()+1, H_sp.colind() + H_sp.size2()+1, H_sp.colind(), m->qmatcnt.begin(),
258  minus<int>());
259 
260  // Create problem object
261  casadi_assert(m->lp==0);
262  m->lp = CPXcreateprob(m->env, &status, "QP from CasADi");
263  casadi_assert_message(m->lp!=0, "CPXcreateprob failed");
264  }

266  void CplexInterface::
267  eval(void* mem, const double** arg, double** res, int* iw, double* w) const {
268  auto m = static_cast<CplexMemory*>(mem);
269 
270  if (inputs_check_) {
271  checkInputs(arg[QPSOL_LBX], arg[QPSOL_UBX], arg[QPSOL_LBA], arg[QPSOL_UBA]);
272  }
273 
274  // Get inputs
275  double* g=w; w += n_;
276  casadi_copy(arg[QPSOL_G], n_, g);
277  double* lbx=w; w += n_;
278  casadi_copy(arg[QPSOL_LBX], n_, lbx);
279  double* ubx=w; w += n_;
280  casadi_copy(arg[QPSOL_UBX], n_, ubx);
281  double* lba=w; w += nc_;
282  casadi_copy(arg[QPSOL_LBA], nc_, lba);
283  double* uba=w; w += nc_;
284  casadi_copy(arg[QPSOL_UBA], nc_, uba);
285  double* H=w; w += nnz_in(QPSOL_H);
286  casadi_copy(arg[QPSOL_H], nnz_in(QPSOL_H), H);
287  double* A=w; w += nnz_in(QPSOL_A);
288  casadi_copy(arg[QPSOL_A], nnz_in(QPSOL_A), A);
289  double* x=w; w += n_;
290  casadi_copy(arg[QPSOL_X0], n_, x);
291  double* lam_x=w; w += n_;
292  casadi_copy(arg[QPSOL_LAM_X0], n_, lam_x);
293 
294  // Temporaries
295  double* lam_a=w; w += nc_;
296 
297  int status;
298 
299  // We change method in crossover
300  if (m->is_warm && qp_method_ == 7) {
301  status = CPXsetintparam(m->env, CPX_PARAM_QPMETHOD, 1);
302  }
303 
304  for (int i = 0; i < nc_; ++i) {
305  // CPX_INFBOUND
306 
307  // Equality
308  if (uba[i] - lba[i] < 1e-20) {
309  m->sense[i] = 'E';
310  m->rhs[i] = lba[i];
311  m->rngval[i] = 0.;
312  } else if (lba[i] < -CPX_INFBOUND) {
313  // Ineq - no lower bound
314  m->sense[i] = 'L';
315  m->rhs[i] = uba[i];
316  m->rngval[i] = 0.;
317  } else if (uba[i] > CPX_INFBOUND) {
318  // Ineq - no upper bound
319  m->sense[i] = 'G';
320  m->rhs[i] = lba[i];
321  m->rngval[i] = 0.;
322  } else { // Inew both upper and lower bounds
323  m->sense[i] = 'R';
324  m->rhs[i] = lba[i];
325  m->rngval[i] = uba[i] - lba[i];
326  }
327  }
328 
329  // Copying objective, constraints, and bounds.
330  const Sparsity& A_sp = sparsity_in(QPSOL_A);
331  const int* matbeg = A_sp.colind();
332  const int* matind = A_sp.row();
333  const double* matval = A;
334  const double* obj = g;
335  const double* lb = lbx;
336  const double* ub = ubx;
337  if (CPXcopylp(m->env, m->lp, n_, nc_, m->objsen, obj, get_ptr(m->rhs), get_ptr(m->sense),
338  matbeg, get_ptr(m->matcnt), matind, matval, lb, ub, get_ptr(m->rngval))) {
339  casadi_error("CPXcopylp failed");
340  }
341 
342  // Preparing coefficient matrix Q
343  const Sparsity& H_sp = sparsity_in(QPSOL_H);
344  const int* qmatbeg = H_sp.colind();
345  const int* qmatind = H_sp.row();
346  const double* qmatval = H;
347  if (CPXcopyquad(m->env, m->lp, qmatbeg, get_ptr(m->qmatcnt), qmatind, qmatval)) {
348  }
349 
350  if (dump_to_file_) {
351  CPXwriteprob(m->env, m->lp, dump_filename_.c_str(), "LP");
352  casadi_error("CPXwriteprob failed");
353  }
354 
355  // Warm-starting if possible
356  if (qp_method_ != 0 && qp_method_ != 4 && m->is_warm) {
357  // TODO(Joel): Initialize slacks and dual variables of bound constraints
358  if (CPXcopystart(m->env, m->lp, get_ptr(m->cstat), get_ptr(m->rstat), x, 0, 0, lam_x)) {
359  casadi_error("CPXcopystart failed");
360  }
361  } else {
362  if (CPXcopystart(m->env, m->lp, 0, 0, x, 0, 0, lam_x)) {
363  casadi_error("CPXcopystart failed");
364  }
365  }
366 
367  // Solution
368  double f;
369  std::vector<double> slack(nc_);
370  int solstat;
371 
372  if (mip_) {
373  // Pass type of variables
374  if (CPXcopyctype(m->env, m->lp, &ctype_[0])) {
375  casadi_error("CPXcopyctype failed");
376  }
377 
378  // Optimize
379  if (CPXmipopt(m->env, m->lp)) {
380  casadi_error("CPXmipopt failed");
381  }
382 
383  // Get objective value
384  if (CPXgetobjval(m->env, m->lp, &f)) {
385  casadi_error("CPXgetobjval failed");
386  }
387 
388  // Get primal solution
389  int cur_numcols = CPXgetnumcols(m->env, m->lp);
390  if (CPXgetx(m->env, m->lp, x, 0, cur_numcols-1)) {
391  casadi_error("CPXgetx failed");
392  }
393 
394  // Get slacks
395  int cur_numrows = CPXgetnumrows(m->env, m->lp);
396  if (CPXgetslack(m->env, m->lp, get_ptr(slack), 0, cur_numrows-1)) {
397  casadi_error("CPXgetslack failed");
398  }
399 
400  // Not a number as dual variables (not calculated with MIQP algorithm)
401  casadi_fill(lam_a, nc_, nan);
402  casadi_fill(lam_x, n_, nan);
403 
404  } else {
405  // Optimize
406  if (CPXqpopt(m->env, m->lp)) {
407  casadi_error("CPXqpopt failed");
408  }
409 
410  // Retrieving solution
411  if (CPXsolution(m->env, m->lp, &solstat, &f, x, lam_a, get_ptr(slack), lam_x)) {
412  casadi_error("CPXsolution failed");
413  }
414  }
415 
416  // Retrieving the basis
417  if (qp_method_ != 0 && qp_method_ != 4) {
418  status = CPXgetbase(m->env, m->lp, get_ptr(m->cstat), get_ptr(m->rstat));
419  }
420 
421  // Flip the sign of the multipliers
422  casadi_scal(nc_, -1., lam_a);
423  casadi_scal(n_, -1., lam_x);
424 
425  int solnstat = CPXgetstat(m->env, m->lp);
426  stringstream errormsg;
427  // NOTE: Why not print directly to userOut() and userOut<true, PL_WARN>()?
428  if (verbose()) {
429  if (solnstat == CPX_STAT_OPTIMAL) {
430  errormsg << "CPLEX: solution status: Optimal solution found.\n";
431  } else if (solnstat == CPX_STAT_UNBOUNDED) {
432  errormsg << "CPLEX: solution status: Model is unbounded\n";
433  } else if (solnstat == CPX_STAT_INFEASIBLE) {
434  errormsg << "CPLEX: solution status: Model is infeasible\n";
435  } else if (solnstat == CPX_STAT_INForUNBD) {
436  errormsg << "CPLEX: solution status: Model is infeasible or unbounded\n";
437  } else if (solnstat == CPX_STAT_OPTIMAL_INFEAS) {
438  errormsg << "CPLEX: solution status: Optimal solution "
439  "is available but with infeasibilities\n";
440  } else if (solnstat == CPX_STAT_NUM_BEST) {
441  errormsg << "CPLEX: solution status: Solution available, but not "
442  "proved optimal due to numeric difficulties.\n";
443  } else if (solnstat == CPX_STAT_FIRSTORDER) {
444  errormsg << "CPLEX: solution status: Solution satisfies first-order optimality "
445  "conditions, but is not necessarily globally optimal.\n";
446  } else {
447  errormsg << "CPLEX: solution status: " << solnstat << "\n";
448  }
449  userOut() << errormsg.str();
450 
451  // Printing basis condition number
452  //double cn;
453  //status = CPXgetdblquality(m->env, m->lp, &cn, CPX_KAPPA);
454  //userOut() << "CPLEX: Basis condition number: " << cn << endl;
455  }
456  if (solnstat != CPX_STAT_OPTIMAL) {
457  // throw CasadiException(errormsg.c_str());
458  }
459 
460  // Next time we warm start
461  if (warm_start_) {
462  m->is_warm = true;
463  }
464 
465  // Get the outputs
466  if (res[QPSOL_COST]) *res[QPSOL_COST] = f;
467  casadi_copy(lam_a, nc_, res[QPSOL_LAM_A]);
468  casadi_copy(lam_x, n_, res[QPSOL_LAM_X]);
469  casadi_copy(x, n_, res[QPSOL_X]);
470  }
