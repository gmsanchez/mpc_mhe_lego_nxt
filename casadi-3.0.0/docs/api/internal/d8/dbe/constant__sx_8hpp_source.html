97  inline static RealtypeSX* create(double value) {
98  // Try to find the constant
99  CACHING_MAP<double, RealtypeSX*>::iterator it = cached_constants_.find(value);
100 
101  // If not found, add it,
102  if (it==cached_constants_.end()) {
103  // Allocate a new object
104  RealtypeSX* n = new RealtypeSX(value);
105 
106  // Add to hash_table
107  cached_constants_.insert(it, std::make_pair(value, n));
108 
109  // Return it to caller
110  return n;
111  } else { // Else, returned the object
112  return it->second;
113  }
114  }

119  virtual int to_int() const { return static_cast<int>(value);}
