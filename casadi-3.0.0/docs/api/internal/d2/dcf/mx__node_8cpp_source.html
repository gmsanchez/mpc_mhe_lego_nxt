66  MXNode::~MXNode() {
67 
68  // Start destruction method if any of the dependencies has dependencies
69  for (vector<MX>::iterator cc=dep_.begin(); cc!=dep_.end(); ++cc) {
70  // Skip if constant
71  if (cc->is_constant()) continue;
72 
73  // Check if there are other "owners" of the node
74  if (cc->getCount()!= 1) {
75 
76  // Replace with a 0-by-0 matrix
77  *cc = MX();
78 
79  } else {
80  // Stack of expressions to be deleted
81  std::stack<MX> deletion_stack;
82 
83  // Move the child to the deletion stack
84  deletion_stack.push(*cc);
85  *cc = MX();
86 
87  // Process stack
88  while (!deletion_stack.empty()) {
89 
90  // Top element
91  MX t = deletion_stack.top();
92 
93  // Check if the top element has dependencies with dependencies
94  bool found_dep = false;
95 
96  // Start destruction method if any of the dependencies has dependencies
97  for (vector<MX>::iterator ii=t->dep_.begin(); ii!=t->dep_.end(); ++ii) {
98 
99  // Skip if constant
100  if (ii->is_constant()) continue;
101 
102  // Check if this is the only reference to the element
103  if (ii->getCount()==1) {
104 
105  // Remove and add to stack
106  deletion_stack.push(*ii);
107  *ii = MX();
108  found_dep = true;
109  break;
110  } else {
111  // Replace with an element without dependencies
112  *ii = MX();
113  }
114  }
115 
116  // Pop from stack if no dependencies found
117  if (!found_dep) {
118  deletion_stack.pop();
119  }
120  }
121  }
122  }
123  }

228  void MXNode::print(std::ostream &stream) const {
229  // Find out which noded can be inlined
230  std::map<const MXNode*, int> nodeind;
231  can_inline(nodeind);
232 
233  // Print expression
234  vector<string> intermed;
235  string s = print_compact(nodeind, intermed);
236 
237  // Print intermediate expressions
238  for (int i=0; i<intermed.size(); ++i)
239  stream << "@" << (i+1) << "=" << intermed[i] << ", ";
240 
241  // Print this
242  stream << s;

245  void MXNode::can_inline(std::map<const MXNode*, int>& nodeind) const {
246  // Add or mark node in map
247  std::map<const MXNode*, int>::iterator it=nodeind.find(this);
248  if (it==nodeind.end()) {
249  // First time encountered, mark inlined
250  nodeind.insert(it, make_pair(this, 0));
251 
252  // Handle dependencies with recursion
253  for (int i=0; i<ndep(); ++i) {
254  dep(i)->can_inline(nodeind);
255  }
256  } else if (it->second==0 && op()!=OP_PARAMETER) {
257  // Node encountered before, do not inline (except if symbolic primitive)
258  it->second = -1;
259  }
260  }

262  std::string MXNode::print_compact(std::map<const MXNode*, int>& nodeind,
263  vector<std::string>& intermed) const {
264  // Get reference to node index
265  int& ind = nodeind[this];
266 
267  // If positive, already in intermediate expressions
268  if (ind>0) return "@" + CodeGenerator::to_string(ind);
269 
270  // Get expressions for dependencies
271  vector<string> arg(ndep());
272  for (int i=0; i<arg.size(); ++i) {
273  arg[i] = dep(i)->print_compact(nodeind, intermed);
274  }
275 
276  // Get expression for this
277  string s = print(arg);
278 
279  // Decide what to do with the expression
280  if (ind==0) {
281  // Inline expression
282  return s;
283  } else {
284  // Add to list of intermediate expressions and return reference
285  intermed.push_back(s);
286  ind = intermed.size(); // For subsequent references
287  return "@" + CodeGenerator::to_string(ind);
288  }
289  }

333  void MXNode::spFwd(const bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem) {
334  // By default, everything depends on everything
335  bvec_t all_depend(0);
336 
337  // Get dependencies of all inputs
338  for (int k=0; k<ndep(); ++k) {
339  const bvec_t* v = arg[k];
340  for (int i=0; i<dep(k).nnz(); ++i) {
341  all_depend |= v[i];
342  }
343  }
344 
345  // Propagate to all outputs
346  for (int k=0; k<nout(); ++k) {
347  bvec_t* v = res[k];
348  for (int i=0; i<sparsity(k).nnz(); ++i) {
349  v[i] = all_depend;
350  }
351  }
352  }

540  MX MXNode::getBinarySwitch(int op, const MX& y) const {
541  // Make sure that dimensions match
542  casadi_assert_message(sparsity().is_scalar() || y.is_scalar() || sparsity().size()==y.size(),
543  "Dimension mismatch." << "lhs is " << sparsity().dim()
544  << ", while rhs is " << y.dim());
545 
546  // Create binary node
547  if (sparsity().is_scalar(false)) {
548  if (nnz()==0) {
549  return toMatrix(MX(0)->getBinary(op, y, true, false), y.sparsity());
550  } else {
551  return toMatrix(getBinary(op, y, true, false), y.sparsity());
552  }
553  } else if (y.is_scalar()) {
554  if (y.nnz()==0) {
555  return toMatrix(getBinary(op, MX(0), false, true), sparsity());
556  } else {
557  return toMatrix(getBinary(op, y, false, true), sparsity());
558  }
559  } else {
560  casadi_assert_message(sparsity().size() == y.sparsity().size(), "Dimension mismatch.");
561  if (sparsity()==y.sparsity()) {
562  // Matching sparsities
563  return getBinary(op, y, false, false);
564  } else {
565  // Get the sparsity pattern of the result
566  // (ignoring structural zeros giving rise to nonzero result)
567  const Sparsity& x_sp = sparsity();
568  const Sparsity& y_sp = y.sparsity();
569  Sparsity r_sp = x_sp.combine(y_sp, operation_checker<F0XChecker>(op),
570  operation_checker<FX0Checker>(op));
571 
572  // Project the arguments to this sparsity
573  MX xx = project(shared_from_this<MX>(), r_sp);
574  MX yy = project(y, r_sp);
575  return xx->getBinary(op, yy, false, false);
576  }
577  }
578  }

580  MX MXNode::getBinary(int op, const MX& y, bool scX, bool scY) const {
581  casadi_assert(sparsity()==y.sparsity() || scX || scY);
582 
583  if (GlobalOptions::simplification_on_the_fly) {
584 
585  // If identically zero due to one argumebt being zero
586  if ((operation_checker<F0XChecker>(op) && is_zero()) ||
587  (operation_checker<FX0Checker>(op) && y->is_zero())) {
588  return MX::zeros(sparsity());
589  }
590 
591  // Handle special operations (independent of type)
592  switch (op) {
593  case OP_ADD:
594  if (MXNode::is_equal(y.get(), this, maxDepth())) return getUnary(OP_TWICE);
595  break;
596  case OP_SUB:
597  case OP_NE:
598  case OP_LT:
599  if (MXNode::is_equal(y.get(), this, maxDepth())) return MX::zeros(sparsity());
600  break;
601  case OP_DIV:
602  if (y->is_zero()) return MX::nan(sparsity());
603  // fall-through
604  case OP_EQ:
605  case OP_LE:
606  if (MXNode::is_equal(y.get(), this, maxDepth())) return MX::ones(sparsity());
607  break;
608  case OP_MUL:
609  if (MXNode::is_equal(y.get(), this, maxDepth())) return getUnary(OP_SQ);
610  break;
611  default: break; // no rule
612  }
613 
614  // Handle special cases for the second argument
615  switch (y->op()) {
616  case OP_CONST:
617  // Make the constant the first argument, if possible
618  if (this->op()!=OP_CONST && operation_checker<CommChecker>(op)) {
619  return y->getBinary(op, shared_from_this<MX>(), scY, scX);
620  } else {
621  switch (op) {
622  case OP_POW:
623  return getBinary(OP_CONSTPOW, y, scX, scY);
624  case OP_CONSTPOW:
625  if (y->isValue(-1)) return getUnary(OP_INV);
626  else if (y->isValue(0)) return MX::ones(sparsity());
627  else if (y->isValue(1)) return shared_from_this<MX>();
628  else if (y->isValue(2)) return getUnary(OP_SQ);
629  break;
630  case OP_ADD:
631  case OP_SUB:
632  if (y->is_zero())
633  return scX ? repmat(shared_from_this<MX>(), y.size()) : shared_from_this<MX>();
634  break;
635  case OP_MUL:
636  if (y->isValue(1)) return shared_from_this<MX>();
637  break;
638  case OP_DIV:
639  if (y->isValue(1)) return shared_from_this<MX>();
640  else if (y->isValue(0.5)) return getUnary(OP_TWICE);
641  break;
642  default: break; // no rule
643  }
644  }
645  break;
646  case OP_NEG:
647  if (op==OP_ADD) {
648  return getBinary(OP_SUB, y->dep(), scX, scY);
649  } else if (op==OP_SUB) {
650  return getBinary(OP_ADD, y->dep(), scX, scY);
651  } else if (op==OP_MUL) {
652  return -getBinary(OP_MUL, y->dep(), scX, scY);
653  } else if (op==OP_DIV) {
654  return -getBinary(OP_DIV, y->dep(), scX, scY);
655  }
656  break;
657  case OP_INV:
658  if (op==OP_MUL) {
659  return getBinary(OP_DIV, y->dep(), scX, scY);
660  } else if (op==OP_DIV) {
661  return getBinary(OP_MUL, y->dep(), scX, scY);
662  }
663  break;
664  default: break; // no rule
665  }
666 
667  }
668 
669  if (scX) {
670  // Check if it is ok to loop over nonzeros only
671  if (y.is_dense() || operation_checker<FX0Checker>(op)) {
672  // Loop over nonzeros
673  return MX::create(new BinaryMX<true, false>(Operation(op), shared_from_this<MX>(), y));
674  } else {
675  // Put a densification node in between
676  return getBinary(op, densify(y), true, false);
677  }
678  } else if (scY) {
679  // Check if it is ok to loop over nonzeros only
680  if (sparsity().is_dense() || operation_checker<F0XChecker>(op)) {
681  // Loop over nonzeros
682  return MX::create(new BinaryMX<false, true>(Operation(op), shared_from_this<MX>(), y));
683  } else {
684  // Put a densification node in between
685  return densify(shared_from_this<MX>())->getBinary(op, y, false, true);
686  }
687  } else {
688  // Loop over nonzeros only
689  MX rr = MX::create(new BinaryMX<false, false>(Operation(op), shared_from_this<MX>(), y));
690 
691  // Handle structural zeros giving rise to nonzero result, e.g. cos(0) == 1
692  if (!rr.is_dense() && !operation_checker<F00Checker>(op)) {
693  // Get the value for the structural zeros
694  double fcn_0(0);
695  casadi_math<double>::fun(op, 0, 0, fcn_0);
696  rr = densify(rr, fcn_0);
697  }
698  return rr;
699  }
