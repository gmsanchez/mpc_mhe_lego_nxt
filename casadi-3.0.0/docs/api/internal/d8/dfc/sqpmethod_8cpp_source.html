111  void Sqpmethod::init(const Dict& opts) {
112  // Call the init method of the base class
113  Nlpsol::init(opts);
114 
115  // Default options
116  max_iter_ = 50;
117  max_iter_ls_ = 3;
118  c1_ = 1e-4;
119  beta_ = 0.8;
120  merit_memsize_ = 4;
121  lbfgs_memory_ = 10;
122  tol_pr_ = 1e-6;
123  tol_du_ = 1e-6;
124  regularize_ = false;
125  string hessian_approximation = "exact";
126  min_step_size_ = 1e-10;
127  string qpsol_plugin;
128  Dict qpsol_options;
129  print_header_ = true;
130 
131  // Read user options
132  for (auto&& op : opts) {
133  if (op.first=="max_iter") {
134  max_iter_ = op.second;
135  } else if (op.first=="max_iter_ls") {
136  max_iter_ls_ = op.second;
137  } else if (op.first=="c1") {
138  c1_ = op.second;
139  } else if (op.first=="beta") {
140  beta_ = op.second;
141  } else if (op.first=="merit_memory") {
142  merit_memsize_ = op.second;
143  } else if (op.first=="lbfgs_memory") {
144  lbfgs_memory_ = op.second;
145  } else if (op.first=="tol_pr") {
146  tol_pr_ = op.second;
147  } else if (op.first=="tol_du") {
148  tol_du_ = op.second;
149  } else if (op.first=="hessian_approximation") {
150  hessian_approximation = op.second.to_string();
151  } else if (op.first=="min_step_size") {
152  min_step_size_ = op.second;
153  } else if (op.first=="qpsol") {
154  qpsol_plugin = op.second.to_string();
155  } else if (op.first=="qpsol_options") {
156  qpsol_options = op.second;
157  } else if (op.first=="print_header") {
158  print_header_ = op.second;
159  }
160  }
161 
162  // Use exact Hessian?
163  exact_hessian_ = hessian_approximation =="exact";
164 
165  // Get/generate required functions
166  f_fcn_ = create_function("nlp_f", {"x", "p"}, {"f"});
167  g_fcn_ = create_function("nlp_g", {"x", "p"}, {"g"});
168  grad_f_fcn_ = create_function("nlp_grad_f", {"x", "p"}, {"f", "grad_f_x"});
169  jac_g_fcn_ = create_function("nlp_jac_g", {"x", "p"}, {"g", "jac_g_x"});
170  if (exact_hessian_) {
171  hess_l_fcn_ = create_function("nlp_jac_f", {"x", "p", "lam_f", "lam_g"},
172  {"sym_hess_gamma_x_x"},
173  {{"gamma", {"f", "g"}}});
174  }
175 
176  // Allocate a QP solver
177  Hsp_ = exact_hessian_ ? hess_l_fcn_.sparsity_out(0) : Sparsity::dense(nx_, nx_);
178  Asp_ = jac_g_fcn_.is_null() ? Sparsity(0, nx_) : jac_g_fcn_.sparsity_out(1);
179 
180  // Allocate a QP solver
181  casadi_assert_message(!qpsol_plugin.empty(), "'qpsol' option has not been set");
182  qpsol_ = qpsol("qpsol", qpsol_plugin, {{"h", Hsp_}, {"a", Asp_}},
183  qpsol_options);
184  alloc(qpsol_);
185 
186  // Create Hessian update function
187  if (!exact_hessian_) {
188  // Create expressions corresponding to Bk, x, x_old, gLag and gLag_old
189  SX Bk = SX::sym("Bk", Hsp_);
190  SX x = SX::sym("x", sparsity_in(NLPSOL_X0));
191  SX x_old = SX::sym("x", x.sparsity());
192  SX gLag = SX::sym("gLag", x.sparsity());
193  SX gLag_old = SX::sym("gLag_old", x.sparsity());
194 
195  SX sk = x - x_old;
196  SX yk = gLag - gLag_old;
197  SX qk = mtimes(Bk, sk);
198 
199  // Calculating theta
200  SX skBksk = dot(sk, qk);
201  SX omega = if_else(dot(yk, sk) < 0.2 * dot(sk, qk),
202  0.8 * skBksk / (skBksk - dot(sk, yk)),
203  1);
204  yk = omega * yk + (1 - omega) * qk;
205  SX theta = 1. / dot(sk, yk);
206  SX phi = 1. / dot(qk, sk);
207  SX Bk_new = Bk + theta * mtimes(yk, yk.T()) - phi * mtimes(qk, qk.T());
208 
209  // Inputs of the BFGS update function
210  vector<SX> bfgs_in(BFGS_NUM_IN);
211  bfgs_in[BFGS_BK] = Bk;
212  bfgs_in[BFGS_X] = x;
213  bfgs_in[BFGS_X_OLD] = x_old;
214  bfgs_in[BFGS_GLAG] = gLag;
215  bfgs_in[BFGS_GLAG_OLD] = gLag_old;
216  bfgs_ = Function("bfgs", bfgs_in, {Bk_new});
217  alloc(bfgs_);
218 
219  // Initial Hessian approximation
220  B_init_ = project(DM::eye(nx_), Hsp_);
221  }
222 
223  // Header
224  if (print_header_) {
225  userOut()
226  << "-------------------------------------------" << endl
227  << "This is casadi::SQPMethod." << endl;
228  if (exact_hessian_) {
229  userOut() << "Using exact Hessian" << endl;
230  } else {
231  userOut() << "Using limited memory BFGS Hessian approximation" << endl;
232  }
233  userOut()
234  << endl
235  << "Number of variables: " << setw(9) << nx_ << endl
236  << "Number of constraints: " << setw(9) << ng_ << endl
237  << "Number of nonzeros in constraint Jacobian: " << setw(9) << Asp_.nnz() << endl
238  << "Number of nonzeros in Lagrangian Hessian: " << setw(9) << Hsp_.nnz() << endl
239  << endl;
240  }
241 
242  // Lagrange multipliers of the NLP
243  alloc_w(ng_, true); // mu_
244  alloc_w(nx_, true); // mu_x_
245 
246  // Current linearization point
247  alloc_w(nx_, true); // x_
248  alloc_w(nx_, true); // x_cand_
249  alloc_w(nx_, true); // x_old_
250 
251  // Lagrange gradient in the next iterate
252  alloc_w(nx_, true); // gLag_
253  alloc_w(nx_, true); // gLag_old_
254 
255  // Constraint function value
256  alloc_w(ng_, true); // gk_
257  alloc_w(ng_, true); // gk_cand_
258 
259  // Gradient of the objective
260  alloc_w(nx_, true); // gf_
261 
262  // Bounds of the QP
263  alloc_w(ng_, true); // qp_LBA
264  alloc_w(ng_, true); // qp_UBA
265  alloc_w(nx_, true); // qp_LBX
266  alloc_w(nx_, true); // qp_UBX
267 
268  // QP solution
269  alloc_w(nx_, true); // dx_
270  alloc_w(nx_, true); // qp_DUAL_X_
271  alloc_w(ng_, true); // qp_DUAL_A_
272 
273  // Hessian approximation
274  alloc_w(Hsp_.nnz(), true); // Bk_
275 
276  // Jacobian
277  alloc_w(Asp_.nnz(), true); // Jk_
278  }

280  void Sqpmethod::set_work(void* mem, const double**& arg, double**& res,
281  int*& iw, double*& w) const {
282  auto m = static_cast<SqpmethodMemory*>(mem);
283 
284  // Set work in base classes
285  Nlpsol::set_work(mem, arg, res, iw, w);
286 
287  // Lagrange multipliers of the NLP
288  m->mu = w; w += ng_;
289  m->mu_x = w; w += nx_;
290 
291  // Current linearization point
292  m->xk = w; w += nx_;
293  m->x_cand = w; w += nx_;
294  m->x_old = w; w += nx_;
295 
296  // Lagrange gradient in the next iterate
297  m->gLag = w; w += nx_;
298  m->gLag_old = w; w += nx_;
299 
300  // Constraint function value
301  m->gk = w; w += ng_;
302  m->gk_cand = w; w += ng_;
303 
304  // Gradient of the objective
305  m->gf = w; w += nx_;
306 
307  // Bounds of the QP
308  m->qp_LBA = w; w += ng_;
309  m->qp_UBA = w; w += ng_;
310  m->qp_LBX = w; w += nx_;
311  m->qp_UBX = w; w += nx_;
312 
313  // QP solution
314  m->dx = w; w += nx_;
315  m->qp_DUAL_X = w; w += nx_;
316  m->qp_DUAL_A = w; w += ng_;
317 
318  // Hessian approximation
319  m->Bk = w; w += Hsp_.nnz();
320 
321  // Jacobian
322  m->Jk = w; w += Asp_.nnz();
323  }

325  void Sqpmethod::solve(void* mem) const {
326  auto m = static_cast<SqpmethodMemory*>(mem);
327 
328  // Check the provided inputs
329  checkInputs(mem);
330 
331  // Statistics
332  for (auto&& s : m->fstats) s.second.reset();
333 
334  m->fstats.at("mainloop").tic();
335 
336  // Set linearization point to initial guess
337  casadi_copy(m->x0, nx_, m->xk);
338 
339  // Initialize Lagrange multipliers of the NLP
340  casadi_copy(m->lam_g0, ng_, m->mu);
341  casadi_copy(m->lam_x0, nx_, m->mu_x);
342 
343  // Initial constraint Jacobian
344  eval_jac_g(m, m->xk, m->gk, m->Jk);
345 
346  // Initial objective gradient
347  eval_grad_f(m, m->xk, &m->fk, m->gf);
348 
349  // Initialize or reset the Hessian or Hessian approximation
350  m->reg = 0;
351  if (exact_hessian_) {
352  eval_h(m, m->xk, m->mu, 1.0, m->Bk);
353  } else {
354  reset_h(m);
355  }
356 
357  // Evaluate the initial gradient of the Lagrangian
358  casadi_copy(m->gf, nx_, m->gLag);
359  if (ng_>0) casadi_mv(m->Jk, Asp_, m->mu, m->gLag, true);
360  // gLag += mu_x_;
361  transform(m->gLag, m->gLag+nx_, m->mu_x, m->gLag, plus<double>());
362 
363  // Number of SQP iterations
364  int iter = 0;
365 
366  // Number of line-search iterations
367  int ls_iter = 0;
368 
369  // Last linesearch successfull
370  bool ls_success = true;
371 
372  // Reset
373  m->merit_mem.clear();
374  m->sigma = 0.; // NOTE: Move this into the main optimization loop
375 
376  // Default stepsize
377  double t = 0;
378 
379  // MAIN OPTIMIZATION LOOP
380  while (true) {
381 
382  // Primal infeasability
383  double pr_inf = primalInfeasibility(m->xk, m->lbx, m->ubx, m->gk, m->lbg, m->ubg);
384 
385  // inf-norm of lagrange gradient
386  double gLag_norminf = casadi_norm_inf(nx_, m->gLag);
387 
388  // inf-norm of step
389  double dx_norminf = casadi_norm_inf(nx_, m->dx);
390 
391  // Print header occasionally
392  if (iter % 10 == 0) printIteration(userOut());
393 
394  // Printing information about the actual iterate
395  printIteration(userOut(), iter, m->fk, pr_inf, gLag_norminf, dx_norminf,
396  m->reg, ls_iter, ls_success);
397 
398  // Call callback function if present
399  if (!fcallback_.is_null()) {
400  m->fstats.at("callback_prep").tic();
401 
402  // Callback inputs
403  fill_n(m->arg, fcallback_.n_in(), nullptr);
404  m->arg[NLPSOL_F] = &m->fk;
405  m->arg[NLPSOL_X] = m->x;
406  m->arg[NLPSOL_LAM_G] = m->lam_g;
407  m->arg[NLPSOL_LAM_X] = m->lam_x;
408  m->arg[NLPSOL_G] = m->g;
409 
410  // Callback outputs
411  fill_n(m->res, fcallback_.n_out(), nullptr);
412  double ret;
413  m->arg[0] = &ret;
414 
415  m->fstats.at("callback_prep").toc();
416  m->fstats.at("callback_fun").tic();
417  // Evaluate
418  fcallback_(m->arg, m->res, m->iw, m->w, 0);
419 
420  m->fstats.at("callback_fun").toc();
421 
422  if (static_cast<int>(ret)) {
423  userOut() << endl;
424  userOut() << "casadi::SQPMethod: aborted by callback..." << endl;
425  m->return_status = "User_Requested_Stop";
426  break;
427  }
428  }
429 
430  // Checking convergence criteria
431  if (pr_inf < tol_pr_ && gLag_norminf < tol_du_) {
432  userOut() << endl;
433  userOut() << "casadi::SQPMethod: Convergence achieved after "
434  << iter << " iterations." << endl;
435  m->return_status = "Solve_Succeeded";
436  break;
437  }
438 
439  if (iter >= max_iter_) {
440  userOut() << endl;
441  userOut() << "casadi::SQPMethod: Maximum number of iterations reached." << endl;
442  m->return_status = "Maximum_Iterations_Exceeded";
443  break;
444  }
445 
446  if (iter > 0 && dx_norminf <= min_step_size_) {
447  userOut() << endl;
448  userOut() << "casadi::SQPMethod: Search direction becomes too small without "
449  "convergence criteria being met." << endl;
450  m->return_status = "Search_Direction_Becomes_Too_Small";
451  break;
452  }
453 
454  // Start a new iteration
455  iter++;
456 
457  log("Formulating QP");
458  // Formulate the QP
459  transform(m->lbx, m->lbx + nx_, m->xk, m->qp_LBX, minus<double>());
460  transform(m->ubx, m->ubx + nx_, m->xk, m->qp_UBX, minus<double>());
461  transform(m->lbg, m->lbg + ng_, m->gk, m->qp_LBA, minus<double>());
462  transform(m->ubg, m->ubg + ng_, m->gk, m->qp_UBA, minus<double>());
463 
464  // Solve the QP
465  solve_QP(m, m->Bk, m->gf, m->qp_LBX, m->qp_UBX, m->Jk, m->qp_LBA,
466  m->qp_UBA, m->dx, m->qp_DUAL_X, m->qp_DUAL_A);
467  log("QP solved");
468 
469  // Detecting indefiniteness
470  double gain = casadi_bilin(m->Bk, Hsp_, m->dx, m->dx);
471  if (gain < 0) {
472  casadi_warning("Indefinite Hessian detected...");
473  }
474 
475  // Calculate penalty parameter of merit function
476  m->sigma = std::max(m->sigma, 1.01*casadi_norm_inf(nx_, m->qp_DUAL_X));
477  m->sigma = std::max(m->sigma, 1.01*casadi_norm_inf(ng_, m->qp_DUAL_A));
478 
479  // Calculate L1-merit function in the actual iterate
480  double l1_infeas = primalInfeasibility(m->xk, m->lbx, m->ubx, m->gk, m->lbg, m->ubg);
481 
482  // Right-hand side of Armijo condition
483  double F_sens = casadi_dot(nx_, m->dx, m->gf);
484  double L1dir = F_sens - m->sigma * l1_infeas;
485  double L1merit = m->fk + m->sigma * l1_infeas;
486 
487  // Storing the actual merit function value in a list
488  m->merit_mem.push_back(L1merit);
489  if (m->merit_mem.size() > merit_memsize_) {
490  m->merit_mem.pop_front();
491  }
492  // Stepsize
493  t = 1.0;
494  double fk_cand;
495  // Merit function value in candidate
496  double L1merit_cand = 0;
497 
498  // Reset line-search counter, success marker
499  ls_iter = 0;
500  ls_success = true;
501 
502  // Line-search
503  log("Starting line-search");
504  if (max_iter_ls_>0) { // max_iter_ls_== 0 disables line-search
505 
506  // Line-search loop
507  while (true) {
508  for (int i=0; i<nx_; ++i) m->x_cand[i] = m->xk[i] + t * m->dx[i];
509 
510  try {
511  // Evaluating objective and constraints
512  fk_cand = eval_f(m, m->x_cand);
513  eval_g(m, m->x_cand, m->gk_cand);
514  } catch(const CasadiException& ex) {
515  // Silent ignore; line-search failed
516  ls_iter++;
517  // Backtracking
518  t = beta_ * t;
519  continue;
520  }
521 
522  ls_iter++;
523 
524  // Calculating merit-function in candidate
525  l1_infeas = primalInfeasibility(m->x_cand, m->lbx, m->ubx, m->gk_cand, m->lbg, m->ubg);
526 
527  L1merit_cand = fk_cand + m->sigma * l1_infeas;
528  // Calculating maximal merit function value so far
529  double meritmax = *max_element(m->merit_mem.begin(), m->merit_mem.end());
530  if (L1merit_cand <= meritmax + t * c1_ * L1dir) {
531  // Accepting candidate
532  log("Line-search completed, candidate accepted");
533  break;
534  }
535 
536  // Line-search not successful, but we accept it.
537  if (ls_iter == max_iter_ls_) {
538  ls_success = false;
539  log("Line-search completed, maximum number of iterations");
540  break;
541  }
542 
543  // Backtracking
544  t = beta_ * t;
545  }
546 
547  // Candidate accepted, update dual variables
548  for (int i=0; i<ng_; ++i) m->mu[i] = t * m->qp_DUAL_A[i] + (1 - t) * m->mu[i];
549  for (int i=0; i<nx_; ++i) m->mu_x[i] = t * m->qp_DUAL_X[i] + (1 - t) * m->mu_x[i];
550 
551  // Candidate accepted, update the primal variable
552  casadi_copy(m->xk, nx_, m->x_old);
553  casadi_copy(m->x_cand, nx_, m->xk);
554 
555  } else {
556  // Full step
557  casadi_copy(m->qp_DUAL_A, ng_, m->mu);
558  casadi_copy(m->qp_DUAL_X, nx_, m->mu_x);
559  casadi_copy(m->xk, nx_, m->x_old);
560  // x+=dx
561  transform(m->xk, m->xk+nx_, m->dx, m->xk, plus<double>());
562  }
563 
564  if (!exact_hessian_) {
565  // Evaluate the gradient of the Lagrangian with the old x but new mu (for BFGS)
566  casadi_copy(m->gf, nx_, m->gLag_old);
567  if (ng_>0) casadi_mv(m->Jk, Asp_, m->mu, m->gLag_old, true);
568  // gLag_old += mu_x_;
569  transform(m->gLag_old, m->gLag_old+nx_, m->mu_x, m->gLag_old, plus<double>());
570  }
571 
572  // Evaluate the constraint Jacobian
573  log("Evaluating jac_g");
574  eval_jac_g(m, m->xk, m->gk, m->Jk);
575 
576  // Evaluate the gradient of the objective function
577  log("Evaluating grad_f");
578  eval_grad_f(m, m->xk, &m->fk, m->gf);
579 
580  // Evaluate the gradient of the Lagrangian with the new x and new mu
581  casadi_copy(m->gf, nx_, m->gLag);
582  if (ng_>0) casadi_mv(m->Jk, Asp_, m->mu, m->gLag, true);
583 
584  // gLag += mu_x_;
585  transform(m->gLag, m->gLag+nx_, m->mu_x, m->gLag, plus<double>());
586 
587  // Updating Lagrange Hessian
588  if (!exact_hessian_) {
589  log("Updating Hessian (BFGS)");
590  // BFGS with careful updates and restarts
591  if (iter % lbfgs_memory_ == 0) {
592  // Reset Hessian approximation by dropping all off-diagonal entries
593  const int* colind = Hsp_.colind(); // Access sparsity (column offset)
594  int ncol = Hsp_.size2();
595  const int* row = Hsp_.row(); // Access sparsity (row)
596  for (int cc=0; cc<ncol; ++cc) { // Loop over the columns of the Hessian
597  for (int el=colind[cc]; el<colind[cc+1]; ++el) {
598  // Loop over the nonzero elements of the column
599  if (cc!=row[el]) m->Bk[el] = 0; // Remove if off-diagonal entries
600  }
601  }
602  }
603 
604  // Update the Hessian approximation
605  fill_n(m->arg, bfgs_.n_in(), nullptr);
606  m->arg[BFGS_BK] = m->Bk;
607  m->arg[BFGS_X] = m->xk;
608  m->arg[BFGS_X_OLD] = m->x_old;
609  m->arg[BFGS_GLAG] = m->gLag;
610  m->arg[BFGS_GLAG_OLD] = m->gLag_old;
611  fill_n(m->res, bfgs_.n_out(), nullptr);
612  m->res[0] = m->Bk;
613  bfgs_(m->arg, m->res, m->iw, m->w, 0);
614 
615  } else {
616  // Exact Hessian
617  log("Evaluating hessian");
618  eval_h(m, m->xk, m->mu, 1.0, m->Bk);
619  }
620  }
621 
622  m->fstats.at("mainloop").toc();
623 
624  // Save results to outputs
625  if (m->f) *m->f = m->fk;
626  if (m->x) casadi_copy(m->xk, nx_, m->x);
627  if (m->lam_g) casadi_copy(m->mu, ng_, m->lam_g);
628  if (m->lam_x) casadi_copy(m->mu_x, nx_, m->lam_x);
629  if (m->g) casadi_copy(m->gk, ng_, m->g);
630 
631  // Show statistics
632  if (print_time_) print_fstats(m);
633  }

674  double Sqpmethod::getRegularization(const double* H) const {
675  const int* colind = Hsp_.colind();
676  int ncol = Hsp_.size2();
677  const int* row = Hsp_.row();
678  double reg_param = 0;
679  for (int cc=0; cc<ncol; ++cc) {
680  double mineig = 0;
681  for (int el=colind[cc]; el<colind[cc+1]; ++el) {
682  int rr = row[el];
683  if (rr == cc) {
684  mineig += H[el];
685  } else {
686  mineig -= fabs(H[el]);
687  }
688  }
689  reg_param = fmin(reg_param, mineig);
690  }
691  return -reg_param;
692  }

809  double Sqpmethod::
810  primalInfeasibility(const double* x, const double* lbx, const double* ubx,
811  const double* g, const double* lbg, const double* ubg) const {
812  // Linf-norm of the primal infeasibility
813  double pr_inf = 0;
814 
815  // Bound constraints
816  for (int j=0; j<nx_; ++j) {
817  pr_inf = fmax(pr_inf, lbx[j] - x[j]);
818  pr_inf = fmax(pr_inf, x[j] - ubx[j]);
819  }
820 
821  // Nonlinear constraints
822  for (int j=0; j<ng_; ++j) {
823  pr_inf = fmax(pr_inf, lbg[j] - g[j]);
824  pr_inf = fmax(pr_inf, g[j] - ubg[j]);
825  }
826 
827  return pr_inf;
828  }
