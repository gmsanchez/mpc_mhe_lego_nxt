151  const std::vector<MatType>& outputv)
152  : FunctionInternal(name), inputv_(inputv), outputv_(outputv) {
153 
154  // Make sure that inputs are symbolic
155  for (int i=0; i<inputv.size(); ++i) {
156  if (inputv[i].nnz()==0) continue; // That's okay
157  if (!inputv[i].is_valid_input()) {
158  casadi_error("XFunction::XFunction: Xfunction input arguments must be"

165  bool has_duplicates = false;
166  for (auto it = inputv_.begin(); it != inputv_.end(); ++it) {
167  has_duplicates = it->has_duplicates() || has_duplicates;
168  }
169 
170  // Reset temporaries

172  it->resetInput();

185  template<typename DerivedType, typename MatType, typename NodeType>
186  void XFunction<DerivedType, MatType, NodeType>::sort_depth_first(
187  std::stack<NodeType*>& s, std::vector<NodeType*>& nodes) {
188  while (!s.empty()) {
189 
190  // Get the topmost element
191  NodeType* t = s.top();
192 
193  // If the last element on the stack has not yet been added
194  if (t && !t->temp) {
195 
196  // Find out which not yet added dependency has most number of dependencies
197  int max_deps = -1, dep_with_max_deps = -1;
198  for (int i=0; i<t->ndep(); ++i) {
199  if (t->dep(i).get() !=0 && static_cast<NodeType*>(t->dep(i).get())->temp == 0) {
200  int ndep_i = t->dep(i)->ndep();
201  if (ndep_i>max_deps) {
202  max_deps = ndep_i;
203  dep_with_max_deps = i;
204  }
205  }
206  }
207 
208  // If there is any dependency which has not yet been added
209  if (dep_with_max_deps>=0) {
210 
211  // Add to the stack the dependency with the most number of dependencies
212  // (so that constants, inputs etc are added last)
213  s.push(static_cast<NodeType*>(t->dep(dep_with_max_deps).get()));
214 
215  } else {
216 
217  // if no dependencies need to be added, we can add the node to the algorithm
218  nodes.push_back(t);
219 
220  // Mark the node as found
221  t->temp = 1;
222 
223  // Remove from stack
224  s.pop();
225  }
226  } else {
227  // If the last element on the stack has already been added
228  s.pop();
229  }
230  }
231  }

233  template<typename DerivedType, typename MatType, typename NodeType>
234  void XFunction<DerivedType, MatType, NodeType>::resort_postpone(
235  std::vector<NodeType*>& algnodes, std::vector<int>& lind) {
236  // Number of levels
237  int nlevels = lind.size()-1;
238 
239  // Set the counter to be the corresponding place in the algorithm
240  for (int i=0; i<algnodes.size(); ++i)
241  algnodes[i]->temp = i;
242 
243  // Save the level of each element
244  std::vector<int> level(algnodes.size());
245  for (int i=0; i<nlevels; ++i)
246  for (int j=lind[i]; j<lind[i+1]; ++j)
247  level[j] = i;
248 
249  // Count the number of times each node is referenced inside the algorithm
250  std::vector<int> numref(algnodes.size(), 0);
251  for (int i=0; i<algnodes.size(); ++i) {
252  for (int c=0; c<algnodes[i]->ndep(); ++c) { // for both children
253  NodeType* child = static_cast<NodeType*>(algnodes[i]->dep(c).get());
254  if (child && child->hasDep())
255  numref[child->temp]++;
256  }
257  }
258 
259  // Stacks of additional nodes at the current and previous level
260  std::stack<int> extra[2];
261 
262  // Loop over the levels in reverse order
263  for (int i=nlevels-1; i>=0; --i) {
264 
265  // The stack for the current level (we are removing elements from this stack)
266  std::stack<int>& extra_this = extra[i%2]; // i odd -> use extra[1]
267 
268  // The stack for the previous level (we are adding elements to this stack)
269  std::stack<int>& extra_prev = extra[1-i%2]; // i odd -> use extra[0]
270 
271  // Loop over the nodes of the level
272  for (int j=lind[i]; j<lind[i+1]; ++j) {
273  // element to be treated
274  int el = j;
275 
276  // elements in the stack have priority
277  if (!extra_this.empty()) {
278  // Replace the element with one from the stack
279  el = extra_this.top();
280  extra_this.pop();
281  --j; // redo the loop
282  }
283 
284  // Skip the element if belongs to a higher level (i.e. was already treated)
285  if (level[el] > i) continue;
286 
287  // for both children
288  for (int c=0; c<algnodes[el]->ndep(); ++c) {
289 
290  NodeType* child = static_cast<NodeType*>(algnodes[el]->dep(c).get());
291 
292  if (child && child->hasDep()) {
293  // Decrease the reference count of the children
294  numref[child->temp]--;
295 
296  // If this was the last time the child was referenced ...
297  // ... and it is not the previous level...
298  if (numref[child->temp]==0 && level[child->temp] != i-1) {
299 
300  // ... then assign a new level ...
301  level[child->temp] = i-1;
302 
303  // ... and add to stack
304  extra_prev.push(child->temp);
305 
306  } // if no more references
307  } // if binary
308  } // for c = ...
309  } // for j
310  } // for i
311 
312  // Count the number of elements on each level
313  for (std::vector<int>::iterator it=lind.begin(); it!=lind.end(); ++it)
314  *it = 0;
315  for (std::vector<int>::const_iterator it=level.begin(); it!=level.end(); ++it)
316  lind[*it + 1]++;
317 
318  // Cumsum to get the index corresponding to the first element of each level
319  for (int i=0; i<nlevels; ++i)
320  lind[i+1] += lind[i];
321 
322  // New index for each element
323  std::vector<int> runind = lind; // running index for each level
324  std::vector<int> newind(algnodes.size());
325  for (int i=0; i<algnodes.size(); ++i)
326  newind[i] = runind[level[algnodes[i]->temp]]++;
327 
328  // Resort the algorithm and reset the temporary
329  std::vector<NodeType*> oldalgnodes = algnodes;
330  for (int i=0; i<algnodes.size(); ++i) {
331  algnodes[newind[i]] = oldalgnodes[i];
332  oldalgnodes[i]->temp = 0;
333  }
334 
335  }

337  template<typename DerivedType, typename MatType, typename NodeType>
338  void XFunction<DerivedType, MatType, NodeType>::resort_breadth_first(
339  std::vector<NodeType*>& algnodes) {
340  // We shall assign a "level" to each element of the algorithm.
341  // A node which does not depend on other binary nodes are assigned
342  // level 0 and for nodes that depend on other nodes of the algorithm,
343  // the level will be the maximum level of any of the children plus 1.
344  // Note that all nodes of a level can be evaluated in parallel.
345  // The level will be saved in the temporary variable
346 
347  // Total number of levels
348  int nlevels = 0;
349 
350  // Get the earliest possible level
351  for (typename std::vector<NodeType*>::iterator it=algnodes.begin(); it!=algnodes.end(); ++it) {
352  // maximum level of any of the children
353  int maxlevel = -1;
354  for (int c=0; c<(*it)->ndep(); ++c) { // Loop over the children
355  NodeType* child = static_cast<NodeType*>((*it)->dep(c).get());
356  if (child->hasDep() && child->temp > maxlevel)
357  maxlevel = child->temp;
358  }
359 
360  // Save the level of this element
361  (*it)->temp = 1 + maxlevel;
362 
363  // Save if new maximum reached
364  if (1 + maxlevel > nlevels)
365  nlevels = 1 + maxlevel;

525  template<typename DerivedType, typename MatType, typename NodeType>
526  MatType XFunction<DerivedType, MatType, NodeType>::grad(int iind, int oind) {
527  casadi_assert_message(sparsity_out(oind).is_scalar(),
528  "Only gradients of scalar functions allowed. Use jacobian instead.");
529 
530  // Quick return if trivially empty
531  if (nnz_in(iind)==0 || nnz_out(oind)==0 ||
532  sparsity_jac(iind, oind, true, false).nnz()==0) {
533  return MatType(size_in(iind));
534  }
535 
536  // Adjoint seeds
537  typename std::vector<std::vector<MatType> > aseed(1, std::vector<MatType>(outputv_.size()));
538  for (int i=0; i<outputv_.size(); ++i) {
539  if (i==oind) {
540  aseed[0][i] = MatType::ones(outputv_[i].sparsity());
541  } else {
542  aseed[0][i] = MatType::zeros(outputv_[i].sparsity());
543  }
544  }
545 
546  // Adjoint sensitivities
547  std::vector<std::vector<MatType> > asens(1, std::vector<MatType>(inputv_.size()));
548  for (int i=0; i<inputv_.size(); ++i) {
549  asens[0][i] = MatType::zeros(inputv_[i].sparsity());
550  }
551 
552  // Calculate with adjoint mode AD
553  reverse_x(inputv_, outputv_, aseed, asens);
554 
555  int dir = 0;
556  for (int i=0; i<n_in(); ++i) { // Correct sparsities #1025
557  if (asens[dir][i].sparsity()!=inputv_[i].sparsity()) {
558  asens[dir][i] = project(asens[dir][i], inputv_[i].sparsity());
559  }
560  }
561 
562  // Return adjoint directional derivative
563  return asens[0].at(iind);
564  }

566  template<typename DerivedType, typename MatType, typename NodeType>
567  MatType XFunction<DerivedType, MatType, NodeType>::tang(int iind, int oind) {
568  casadi_assert_message(sparsity_in(iind).is_scalar(),
569  "Only tangent of scalar input functions allowed. Use jacobian instead.");
570 
571  // Forward seeds
572  typename std::vector<std::vector<MatType> > fseed(1, std::vector<MatType>(inputv_.size()));
573  for (int i=0; i<inputv_.size(); ++i) {
574  if (i==iind) {
575  fseed[0][i] = MatType::ones(inputv_[i].sparsity());
576  } else {
577  fseed[0][i] = MatType::zeros(inputv_[i].sparsity());
578  }
579  }
580 
581  // Forward sensitivities
582  std::vector<std::vector<MatType> > fsens(1, std::vector<MatType>(outputv_.size()));
583  for (int i=0; i<outputv_.size(); ++i) {
584  fsens[0][i] = MatType::zeros(outputv_[i].sparsity());
585  }
586 
587  // Calculate with forward mode AD
588  forward(inputv_, outputv_, fseed, fsens, true, false);
589 
590  // Return adjoint directional derivative
591  return fsens[0].at(oind);
592  }

594  template<typename DerivedType, typename MatType, typename NodeType>
595  MatType XFunction<DerivedType, MatType, NodeType>
596  ::jac(int iind, int oind, bool compact, bool symmetric, bool always_inline, bool never_inline) {
597  using namespace std;
598  if (verbose()) userOut() << "XFunction::jac begin" << std::endl;
599 
600  // Quick return if trivially empty
601  if (nnz_in(iind)==0 || nnz_out(oind)==0) {
602  std::pair<int, int> jac_shape;
603  jac_shape.first = compact ? nnz_out(oind) : numel_out(oind);
604  jac_shape.second = compact ? nnz_in(iind) : numel_in(iind);
605  return MatType(jac_shape);
606  }
607 
608  if (symmetric) {
609  casadi_assert(sparsity_out(oind).is_dense());
610  }
611 
612  // Create return object
613  MatType ret = MatType::zeros(sparsity_jac(iind, oind, compact, symmetric).T());
614  if (verbose()) userOut() << "XFunction::jac allocated return value" << std::endl;
615 
616  // Quick return if empty
617  if (ret.nnz()==0) {
618  return ret.T();
619  }
620 
621  // Get a bidirectional partition
622  Sparsity D1, D2;
623  getPartition(iind, oind, D1, D2, true, symmetric);
624  if (verbose()) userOut() << "XFunction::jac graph coloring completed" << std::endl;
625 
626  // Get the number of forward and adjoint sweeps
627  int nfdir = D1.is_null() ? 0 : D1.size2();
628  int nadir = D2.is_null() ? 0 : D2.size2();
629 
630  // Number of derivative directions supported by the function
631  int max_nfdir = optimized_num_dir;
632  int max_nadir = optimized_num_dir;
633 
634  // Current forward and adjoint direction
635  int offset_nfdir = 0, offset_nadir = 0;
636 
637  // Evaluation result (known)
638  std::vector<MatType> res(outputv_);
639 
640  // Forward and adjoint seeds and sensitivities
641  std::vector<std::vector<MatType> > fseed, aseed, fsens, asens;
642 
643  // Get the sparsity of the Jacobian block
644  Sparsity jsp = sparsity_jac(iind, oind, true, symmetric).T();
645  const int* jsp_colind = jsp.colind();
646  const int* jsp_row = jsp.row();
647 
648  // Input sparsity
649  std::vector<int> input_col = sparsity_in(iind).get_col();
650  const int* input_row = sparsity_in(iind).row();
651 
652  // Output sparsity
653  std::vector<int> output_col = sparsity_out(oind).get_col();
654  const int* output_row = sparsity_out(oind).row();
655 
656  // Get transposes and mappings for jacobian sparsity pattern if we are using forward mode
657  if (verbose()) userOut() << "XFunction::jac transposes and mapping" << std::endl;
658  std::vector<int> mapping;
659  Sparsity jsp_trans;
660  if (nfdir>0) {
661  jsp_trans = jsp.transpose(mapping);
662  }
663 
664  // The nonzeros of the sensitivity matrix
665  std::vector<int> nzmap, nzmap2;
666 
667  // Additions to the jacobian matrix
668  std::vector<int> adds, adds2;
669 
670  // Temporary vector
671  std::vector<int> tmp;
672 
673  // Progress
674  int progress = -10;
675 
676  // Number of sweeps
677  int nsweep_fwd = nfdir/max_nfdir; // Number of sweeps needed for the forward mode
678  if (nfdir%max_nfdir>0) nsweep_fwd++;
679  int nsweep_adj = nadir/max_nadir; // Number of sweeps needed for the adjoint mode
680  if (nadir%max_nadir>0) nsweep_adj++;
681  int nsweep = std::max(nsweep_fwd, nsweep_adj);
682  if (verbose()) userOut() << "XFunction::jac " << nsweep << " sweeps needed for "
683  << nfdir << " forward and " << nadir << " adjoint directions"
684  << std::endl;
685 
686  // Sparsity of the seeds
687  vector<int> seed_col, seed_row;
688 
689  // Evaluate until everything has been determined
690  for (int s=0; s<nsweep; ++s) {
691  // Print progress
692  if (verbose()) {
693  int progress_new = (s*100)/nsweep;
694  // Print when entering a new decade
695  if (progress_new / 10 > progress / 10) {
696  progress = progress_new;
697  userOut() << progress << " %" << std::endl;
698  }
699  }
700 
701  // Number of forward and adjoint directions in the current "batch"
702  int nfdir_batch = std::min(nfdir - offset_nfdir, max_nfdir);
703  int nadir_batch = std::min(nadir - offset_nadir, max_nadir);
704 
705  // Forward seeds
706  fseed.resize(nfdir_batch);
707  for (int d=0; d<nfdir_batch; ++d) {
708  // Nonzeros of the seed matrix
709  seed_col.clear();
710  seed_row.clear();
711 
712  // For all the directions
713  for (int el = D1.colind(offset_nfdir+d); el<D1.colind(offset_nfdir+d+1); ++el) {
714 
715  // Get the direction
716  int c = D1.row(el);
717 
718  // Give a seed in the direction
719  seed_col.push_back(input_col[c]);
720  seed_row.push_back(input_row[c]);
721  }
722 
723  // initialize to zero
724  fseed[d].resize(n_in());
725  for (int ind=0; ind<fseed[d].size(); ++ind) {
726  int nrow = size1_in(ind), ncol = size2_in(ind); // Input dimensions
727  if (ind==iind) {
728  fseed[d][ind] = MatType::ones(Sparsity::triplet(nrow, ncol, seed_row, seed_col));
729  } else {
730  fseed[d][ind] = MatType(nrow, ncol);
731  }
732  }
733  }
734 
735  // Adjoint seeds
736  aseed.resize(nadir_batch);
737  for (int d=0; d<nadir_batch; ++d) {
738  // Nonzeros of the seed matrix
739  seed_col.clear();
740  seed_row.clear();
741 
742  // For all the directions
743  for (int el = D2.colind(offset_nadir+d); el<D2.colind(offset_nadir+d+1); ++el) {
744 
745  // Get the direction
746  int c = D2.row(el);
747 
748  // Give a seed in the direction
749  seed_col.push_back(output_col[c]);
750  seed_row.push_back(output_row[c]);
751  }
752 
753  //initialize to zero
754  aseed[d].resize(n_out());
755  for (int ind=0; ind<aseed[d].size(); ++ind) {
756  int nrow = size1_out(ind), ncol = size2_out(ind); // Output dimensions
757  if (ind==oind) {
758  aseed[d][ind] = MatType::ones(Sparsity::triplet(nrow, ncol, seed_row, seed_col));
759  } else {
760  aseed[d][ind] = MatType(nrow, ncol);
761  }
762  }
763  }
764 
765  // Forward sensitivities
766  fsens.resize(nfdir_batch);
767  for (int d=0; d<nfdir_batch; ++d) {
768  // initialize to zero
769  fsens[d].resize(n_out());
770  for (int oind=0; oind<fsens[d].size(); ++oind) {
771  fsens[d][oind] = MatType::zeros(sparsity_out(oind));
772  }
773  }
774 
775  // Adjoint sensitivities
776  asens.resize(nadir_batch);
777  for (int d=0; d<nadir_batch; ++d) {
778  // initialize to zero
779  asens[d].resize(n_in());
780  for (int ind=0; ind<asens[d].size(); ++ind) {
781  asens[d][ind] = MatType::zeros(sparsity_in(ind));
782  }
783  }
784 
785  // Evaluate symbolically
786  if (verbose()) userOut() << "XFunction::jac making function call" << std::endl;
787  if (fseed.size()>0) {
788  casadi_assert(aseed.size()==0);
789  forward(inputv_, outputv_, fseed, fsens, always_inline, never_inline);
790  } else if (aseed.size()>0) {
791  casadi_assert(fseed.size()==0);
792  reverse(inputv_, outputv_, aseed, asens, always_inline, never_inline);
793  }
794 
795  // Carry out the forward sweeps
796  for (int d=0; d<nfdir_batch; ++d) {
797 
798  // If symmetric, see how many times each output appears
799  if (symmetric) {
800  // Initialize to zero
801  tmp.resize(nnz_out(oind));
802  fill(tmp.begin(), tmp.end(), 0);
803 
804  // "Multiply" Jacobian sparsity by seed vector
805  for (int el = D1.colind(offset_nfdir+d); el<D1.colind(offset_nfdir+d+1); ++el) {
806 
807  // Get the input nonzero
808  int c = D1.row(el);
809 
810  // Propagate dependencies
811  for (int el_jsp=jsp_colind[c]; el_jsp<jsp_colind[c+1]; ++el_jsp) {
812  tmp[jsp_row[el_jsp]]++;
813  }
814  }
815  }
816 
817  // Locate the nonzeros of the forward sensitivity matrix
818  sparsity_out(oind).find(nzmap);
819  fsens[d][oind].sparsity().get_nz(nzmap);
820 
821  if (symmetric) {
822  sparsity_in(iind).find(nzmap2);
823  fsens[d][oind].sparsity().get_nz(nzmap2);
824  }
825 
826  // Assignments to the Jacobian
827  adds.resize(fsens[d][oind].nnz());
828  fill(adds.begin(), adds.end(), -1);
829  if (symmetric) {
830  adds2.resize(adds.size());
831  fill(adds2.begin(), adds2.end(), -1);
832  }
833 
834  // For all the input nonzeros treated in the sweep
835  for (int el = D1.colind(offset_nfdir+d); el<D1.colind(offset_nfdir+d+1); ++el) {
836 
837  // Get the input nonzero
838  int c = D1.row(el);
839  //int f2_out;
840  //if (symmetric) {
841  // f2_out = nzmap2[c];
842  //}
843 
844  // Loop over the output nonzeros corresponding to this input nonzero
845  for (int el_out = jsp_trans.colind(c); el_out<jsp_trans.colind(c+1); ++el_out) {
846 
847  // Get the output nonzero
848  int r_out = jsp_trans.row(el_out);
849 
850  // Get the forward sensitivity nonzero
851  int f_out = nzmap[r_out];
852  if (f_out<0) continue; // Skip if structurally zero
853 
854  // The nonzero of the Jacobian now treated
855  int elJ = mapping[el_out];
856 
857  if (symmetric) {
858  if (tmp[r_out]==1) {
859  adds[f_out] = el_out;
860  adds2[f_out] = elJ;
861  }
862  } else {
863  // Get the output seed
864  adds[f_out] = elJ;
865  }
866  }
867  }
868 
869  // Get entries in fsens[d][oind] with nonnegative indices
870  tmp.resize(adds.size());
871  int sz = 0;
872  for (int i=0; i<adds.size(); ++i) {
873  if (adds[i]>=0) {
874  adds[sz] = adds[i];
875  tmp[sz++] = i;
876  }
877  }
878  adds.resize(sz);
879  tmp.resize(sz);
880 
881  // Add contribution to the Jacobian
882  ret[adds] = fsens[d][oind][tmp];
883 
884  if (symmetric) {
885  // Get entries in fsens[d][oind] with nonnegative indices
886  tmp.resize(adds2.size());
887  sz = 0;
888  for (int i=0; i<adds2.size(); ++i) {
889  if (adds2[i]>=0) {
890  adds2[sz] = adds2[i];
891  tmp[sz++] = i;
892  }
893  }
894  adds2.resize(sz);
895  tmp.resize(sz);
896 
897  // Add contribution to the Jacobian
898  ret[adds2] = fsens[d][oind][tmp];
899  }
900  }
901 
902  // Add elements to the Jacobian matrix
903  for (int d=0; d<nadir_batch; ++d) {
904 
905  // Locate the nonzeros of the adjoint sensitivity matrix
906  sparsity_in(iind).find(nzmap);
907  asens[d][iind].sparsity().get_nz(nzmap);
908 
909  // For all the output nonzeros treated in the sweep
910  for (int el = D2.colind(offset_nadir+d); el<D2.colind(offset_nadir+d+1); ++el) {
911 
912  // Get the output nonzero
913  int r = D2.row(el);
914 
915  // Loop over the input nonzeros that influences this output nonzero
916  for (int elJ = jsp.colind(r); elJ<jsp.colind(r+1); ++elJ) {
917 
918  // Get the input nonzero
919  int inz = jsp.row(elJ);
920 
921  // Get the corresponding adjoint sensitivity nonzero
922  int anz = nzmap[inz];
923  if (anz<0) continue;
924 
925  // Get the input seed
926  ret[elJ] = asens[d][iind][anz];
927  }
928  }
929  }
930 
931  // Update direction offsets
932  offset_nfdir += nfdir_batch;
933  offset_nadir += nadir_batch;
934  }
935 
936  // Return
937  if (verbose()) userOut() << "XFunction::jac end" << std::endl;
938  return ret.T();
939  }
