83  void ShellCompiler::init(const Dict& opts) {
84  // Base class
85  CompilerInternal::init(opts);
86 
87  // Default options
88  string compiler = "gcc";
89  string compiler_setup = "-fPIC -shared";
90  vector<string> flags;
91 
92  // Read options
93  for (auto&& op : opts) {
94  if (op.first=="compiler") {
95  compiler = op.second.to_string();
96  } else if (op.first=="compiler_setup") {
97  compiler_setup = op.second.to_string();
98  } else if (op.first=="flags") {
99  flags = op.second;
100  }
101  }
102 
103  // Construct the compiler command
104  stringstream cmd;
105  cmd << compiler << " " << compiler_setup;
106  for (vector<string>::const_iterator i=flags.begin(); i!=flags.end(); ++i) {
107  cmd << " " << *i;
108  }
109 
110  // C/C++ source file
111  cmd << " " << name_;
112 
113  // Name of temporary file
114 #ifdef HAVE_MKSTEMPS
115  // Preferred solution
116  char bin_name[] = "tmp_casadi_compiler_shell_XXXXXX.so";
117  if (mkstemps(bin_name, 3) == -1) {
118  casadi_error("Failed to create a temporary file name");
119  }
120  bin_name_ = bin_name;
121 #else
122  // Fallback, may result in deprecation warnings
123  char* bin_name = tempnam(0, "ca.so");
124  bin_name_ = bin_name;
125  free(bin_name);
126 #endif
127 
128  // Have relative paths start with ./
129  if (bin_name_.at(0)!='/') {
130  bin_name_ = "./" + bin_name_;
131  }
132 
133  // Temporary file
134  cmd << " -o " << bin_name_;
135 
136  // Compile into a shared library
137  if (system(cmd.str().c_str())) {
138  casadi_error("Compilation failed. Tried \"" + cmd.str() + "\"");
139  }
140 
141  // Load shared library
142  handle_ = dlopen(bin_name_.c_str(), RTLD_LAZY);
143  casadi_assert_message(handle_!=0, "CommonExternal: Cannot open function: "
144  << bin_name_ << ". error code: "<< dlerror());
145  // reset error
146  dlerror();
147  }
