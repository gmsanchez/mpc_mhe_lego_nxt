204  void MX::set(const MX& m, bool ind1, const Matrix<int>& rr, const Matrix<int>& cc) {
205  // Row vector rr (e.g. in MATLAB) is transposed to column vector
206  if (rr.size1()==1 && rr.size2()>1) {
207  return set(m, ind1, rr.T(), cc);
208  }
209 
210  // Row vector cc (e.g. in MATLAB) is transposed to column vector
211  if (cc.size1()==1 && cc.size2()>1) {
212  return set(m, ind1, rr, cc.T());
213  }
214 
215  // Make sure rr and cc are dense vectors
216  casadi_assert_message(rr.is_dense() && rr.is_column(),
217  "MX::set: First index not dense vector");
218  casadi_assert_message(cc.is_dense() && cc.is_column(),
219  "MX::set: Second index not dense vector");
220 
221  // Assert dimensions of assigning matrix
222  if (rr.size1() != m.size1() || cc.size1() != m.size2()) {
223  if (m.is_scalar()) {
224  // m scalar means "set all"
225  return set(repmat(m, rr.size1(), cc.size1()), ind1, rr, cc);
226  } else if (rr.size1() == m.size2() && cc.size1() == m.size1()
227  && std::min(m.size1(), m.size2()) == 1) {
228  // m is transposed if necessary
229  return set(m.T(), ind1, rr, cc);
230  } else {
231  // Error otherwise
232  casadi_error("Dimension mismatch." << "lhs is " << rr.size1() << "-by-"
233  << cc.size1() << ", while rhs is " << m.size());
234  }
235  }
236 
237  // Dimensions
238  int sz1 = size1(), sz2 = size2();
239 
240  // Report out-of-bounds
241  if (!inBounds(rr.nonzeros(), -sz1+ind1, sz1+ind1)) {
242  casadi_error("set[., rr, cc] out of bounds. Your rr contains "
243  << *std::min_element(rr->begin(), rr->end()) << " up to "
244  << *std::max_element(rr->begin(), rr->end())
245  << ", which is outside the range [" << -sz1+ind1 << ","<< sz1+ind1 << ").");
246  }
247  if (!inBounds(cc.nonzeros(), -sz2+ind1, sz2+ind1)) {
248  casadi_error("set [., rr, cc] out of bounds. Your cc contains "
249  << *std::min_element(cc->begin(), cc->end()) << " up to "
250  << *std::max_element(cc->begin(), cc->end())
251  << ", which is outside the range [" << -sz2+ind1 << ","<< sz2+ind1 << ").");
252  }
253 
254  // If we are assigning with something sparse, first remove existing entries
255  if (!m.is_dense()) {
256  erase(rr.nonzeros(), cc.nonzeros(), ind1);
257  }
258 
259  // Collect all assignments
260  IM el = IM::zeros(m.sparsity());
261  for (int j=0; j<el.size2(); ++j) { // Loop over columns of m
262  int this_j = cc->at(j) - ind1; // Corresponding column in this
263  if (this_j<0) this_j += sz2;
264  for (int k=el.colind(j); k<el.colind(j+1); ++k) { // Loop over rows of m
265  int i = m.row(k);
266  int this_i = rr->at(i) - ind1; // Corresponding row in this
267  if (this_i<0) this_i += sz1;
268  el->at(k) = this_i + this_j*sz1;
269  }
270  }
271  return set(m, false, el);
272  }

279  void MX::set(const MX& m, bool ind1, const Matrix<int>& rr) {
280  // Assert dimensions of assigning matrix
281  if (rr.sparsity() != m.sparsity()) {
282  if (rr.size() == m.size()) {
283  // Remove submatrix to be replaced
284  erase(rr.nonzeros(), ind1);
285 
286  // Find the intersection between rr's and m's sparsity patterns
287  Sparsity sp = rr.sparsity() * m.sparsity();
288 
289  // Project both matrices to this sparsity
290  return set(project(m, sp), ind1, Matrix<int>::project(rr, sp));
291  } else if (m.is_scalar()) {
292  // m scalar means "set all"
293  if (m.is_dense()) {
294  return set(MX(rr.sparsity(), m), ind1, rr);
295  } else {
296  return set(MX(rr.size()), ind1, rr);
297  }
298  } else if (rr.size1() == m.size2() && rr.size2() == m.size1()
299  && std::min(m.size1(), m.size2()) == 1) {
300  // m is transposed if necessary
301  return set(m.T(), ind1, rr);
302  } else {
303  // Error otherwise
304  casadi_error("Dimension mismatch." << "lhs is " << rr.size()
305  << ", while rhs is " << m.size());
306  }
307  }
308 
309  // Dimensions of this
310  int sz1 = size1(), sz2 = size2(), sz = nnz(), nel = numel(), rrsz = rr.nnz();
311 
312  // Quick return if nothing to set
313  if (rrsz==0) return;
314 
315  // Check bounds
316  if (!inBounds(rr.nonzeros(), -nel+ind1, nel+ind1)) {
317  casadi_error("set[rr] out of bounds. Your rr contains "
318  << *std::min_element(rr->begin(), rr->end()) << " up to "
319  << *std::max_element(rr->begin(), rr->end())
320  << ", which is outside the range [" << -nel+ind1 << ","<< nel+ind1 << ").");
321  }
322 
323  // Dense mode
324  if (is_dense() && m.is_dense()) {
325  return set_nz(m, ind1, rr);
326  }
327 
328  // Construct new sparsity pattern
329  std::vector<int> new_row=sparsity().get_row(), new_col=sparsity().get_col(), nz(rr.nonzeros());
330  new_row.reserve(sz+rrsz);
331  new_col.reserve(sz+rrsz);
332  nz.reserve(rrsz);
333  for (std::vector<int>::iterator i=nz.begin(); i!=nz.end(); ++i) {
334  if (ind1) (*i)--;
335  if (*i<0) *i += nel;
336  new_row.push_back(*i % sz1);
337  new_col.push_back(*i / sz1);
338  }
339  Sparsity sp = Sparsity::triplet(sz1, sz2, new_row, new_col);
340 
341  // If needed, update pattern
342  if (sp != sparsity()) *this = project(*this, sp);
343 
344  // Find the nonzeros corresponding to rr
345  sparsity().get_nz(nz);
346 
347  // Create a nonzero assignment node
348  *this = simplify(m->getSetNonzeros(*this, nz));
349  }

369  void MX::get_nz(MX& m, bool ind1, const Matrix<int>& kk) const {
370  // If indexed matrix was a row/column vector, make sure that the result is too
371  bool tr = (is_column() && kk.is_row()) || (is_row() && kk.is_column());
372 
373  // Quick return if no entries
374  if (kk.nnz()==0) {
375  m = MX::zeros(tr ? kk.sparsity().T() : kk.sparsity());
376  return;
377  }
378 
379  // Check bounds
380  int sz = nnz();
381  if (!inBounds(kk.nonzeros(), -sz+ind1, sz+ind1)) {
382  casadi_error("get_nz[kk] out of bounds. Your kk contains "
383  << *std::min_element(kk->begin(), kk->end()) << " up to "
384  << *std::max_element(kk->begin(), kk->end())
385  << ", which is outside the range [" << -sz+ind1 << "," << sz+ind1 << ").");
386  }
387 
388  // Handle index-1, negative indices
389  if (ind1 || *std::min_element(kk->begin(), kk->end())<0) {
390  Matrix<int> kk_mod = kk;
391  for (auto&& i : kk_mod.nonzeros()) {
392  casadi_assert_message(!(ind1 && i<=0), "Matlab is 1-based, but requested index " <<
393  i << ". Note that negative slices are" <<
394  " disabled in the Matlab interface. " <<
395  "Possibly you may want to use 'end'.");
396  if (ind1) i--;
397  if (i<0) i += sz;
398  }
399  get_nz(m, false, kk_mod); // Call recursively
400  return;
401  }
402 
403  // Return reference to the nonzeros
404  m = (*this)->getGetNonzeros(tr ? kk.sparsity().T() : kk.sparsity(), kk.nonzeros());
405  }

412  void MX::set_nz(const MX& m, bool ind1, const Matrix<int>& kk) {
413  casadi_assert_message(kk.nnz()==m.nnz() || m.nnz()==1,
414  "MX::set_nz: length of non-zero indices (" << kk.nnz() << ") " <<
415  "must match size of rhs (" << m.nnz() << ").");
416 
417  // Assert dimensions of assigning matrix
418  if (kk.sparsity() != m.sparsity()) {
419  if (m.is_scalar()) {
420  // m scalar means "set all"
421  if (!m.is_dense()) return; // Nothing to set
422  return set_nz(MX(kk.sparsity(), m), ind1, kk);
423  } else if (kk.size() == m.size()) {
424  // Project sparsity if needed
425  return set_nz(project(m, kk.sparsity()), ind1, kk);
426  } else if (kk.size1() == m.size2() && kk.size2() == m.size1()
427  && std::min(m.size1(), m.size2()) == 1) {
428  // m is transposed if necessary
429  return set_nz(m.T(), ind1, kk);
430  } else {
431  // Error otherwise
432  casadi_error("Dimension mismatch." << "lhs is " << kk.size()
433  << ", while rhs is " << m.size());
434  }
435  }
436 
437  // Call recursively if points both objects point to the same node
438  if (this==&m) {
439  MX m_copy = m;
440  return set_nz(m_copy, ind1, kk);
441  }
442 
443  // Check bounds
444  int sz = nnz();
445  if (!inBounds(kk.nonzeros(), -sz+ind1, sz+ind1)) {
446  casadi_error("set_nz[kk] out of bounds. Your kk contains "
447  << *std::min_element(kk->begin(), kk->end()) << " up to "
448  << *std::max_element(kk->begin(), kk->end())
449  << ", which is outside the range [" << -sz+ind1 << ","<< sz+ind1 << ").");
450  }
451 
452  // Quick return if no assignments to be made
453  if (kk.nnz()==0) return;
454 
455  // Handle index-1, negative indices
456  if (ind1 || *std::min_element(kk->begin(), kk->end())<0) {
457  Matrix<int> kk_mod = kk;
458  for (auto&& i : kk_mod.nonzeros()) {
459  casadi_assert_message(!(ind1 && i<=0), "Matlab is 1-based, but requested index " <<
460  i << ". Note that negative slices are" <<
461  " disabled in the Matlab interface. " <<
462  "Possibly you may want to use 'end'.");
463  if (ind1) i--;
464  if (i<0) i += sz;

466  return set_nz(m, false, kk_mod); // Call recursively

1205  "Inconsistent shape for variable " << k << ".");

1217  f_out.insert(f_out.end(), ex.begin(), ex.end());
1218

1221  auto *ff = dynamic_cast<MXFunction *>(f.get());

1234  case OP_CONST:

1236  break;
1237  case OP_OUTPUT:
1238  if (it->res.front()<vdef.size()) {

1413  const vector<MXAlgEl>& algorithm = ff->algorithm_;
1414  vector<MX> work(ff->workloc_.size()-1);
1415 
1416  // Count how many times an expression has been used
1417  vector<int> usecount(work.size(), 0);
1418 
1419  // Remember the origin of every calculation
1420  vector<pair<int, int> > origin(work.size(), make_pair(-1, -1));
1421 
1422  // Which evaluations to replace
1423  vector<pair<int, int> > replace;
1424 
1425  // Evaluate the algorithm to identify which evaluations to replace
1426  int k=0;
1427  for (vector<MXAlgEl>::const_iterator it=algorithm.begin(); it<algorithm.end(); ++it, ++k) {
1428  // Increase usage counters
1429  switch (it->op) {
1430  case OP_CONST:
1431  case OP_PARAMETER:
1432  break;
1433  default: // Unary operation, binary operation or output
1434  for (int c=0; c<it->arg.size(); ++c) {
1435  if (usecount[it->arg[c]]==0) {
1436  usecount[it->arg[c]]=1;
1437  } else if (usecount[it->arg[c]]==1) {
1438  replace.push_back(origin[it->arg[c]]);

1440  }

1443 
1444  // Perform the operation
1445  switch (it->op) {
1446  case OP_OUTPUT:
1447  break;
1448  case OP_CONST:
1449  case OP_PARAMETER:
1450  usecount[it->res.front()] = -1; // Never extract since it is a primitive type
1451  break;
1452  default:

1454  if (it->res[c]>=0) {
1455  work[it->res[c]] = it->data.getOutput(c);
1456  usecount[it->res[c]] = 0; // Not (yet) extracted
1457  origin[it->res[c]] = make_pair(k, c);
1458  }
1459  }
1460  break;

1465  v.clear();
1466  v.reserve(replace.size());

1468  vdef.reserve(replace.size());
1469 
1470  // Quick return
1471  if (replace.empty()) return;
1472 
1473  // Sort the elements to be replaced in the order of appearence in the algorithm
1474  sort(replace.begin(), replace.end());
1475  vector<pair<int, int> >::const_iterator replace_it=replace.begin();
1476 
1477  // Name of intermediate variables
1478  stringstream v_name;
1479 
1480  // Arguments for calling the atomic operations
1481  vector<MX> oarg, ores;
1482 
1483  // Evaluate the algorithm
1484  k=0;
1485  for (vector<MXAlgEl>::const_iterator it=algorithm.begin(); it<algorithm.end(); ++it, ++k) {
1486  switch (it->op) {
1487  case OP_OUTPUT: ex[it->res.front()] = work[it->arg.front()]; break;
1488  case OP_CONST:
1489  case OP_PARAMETER: work[it->res.front()] = it->data; break;
1490  default:
1491  {
1492  // Arguments of the operation
1493  oarg.resize(it->arg.size());
1494  for (int i=0; i<oarg.size(); ++i) {
1495  int el = it->arg[i];

1502 
1503  // Get the result
1504  for (int i=0; i<ores.size(); ++i) {
1505  int el = it->res[i];
1506  if (el>=0) work.at(el) = ores[i];
1507  }
1508 
1509  // Possibly replace results with new variables
1510  for (int c=0; c<it->res.size(); ++c) {
1511  int ind = it->res[c];
1512  if (ind>=0 && replace_it->first==k && replace_it->second==c) {
1513  // Store the result
1514  vdef.push_back(work[ind]);
