78  void CompilerInternal::get_meta(std::vector<std::string>& lines, int& offset) const {
79  // Open source file and search for meta information
80  ifstream file(name_);
81  std::string line;
82  offset = 0;
83  lines.clear();
84  while (getline(file, line)) {
85  // Update offset
86  offset++;
87 
88  // Try to find a /*CASADI delimiter
89  if (line.find("/*CASADIMETA") != string::npos) {
90  // Delimimiter found, find */ delimiter
91  while (getline(file, line)) {
92  if (line.find("*/") != string::npos) return; // Successful return
93  lines.push_back(line);
94  }
95  casadi_error("End-of-file reached while searching for \"*/\"");
96  }
97  }
98  }
