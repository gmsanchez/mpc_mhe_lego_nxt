47  std::vector<int> Slice::all(int len, bool ind1) const {
48  int start = this->start;
49  if (start==std::numeric_limits<int>::min()) {
50  start = (step < 0) ? len - 1 : 0;
51  } else if (start<0) {
52  start+=len;
53  }
54  int stop = this->stop;
55  if (stop==std::numeric_limits<int>::max()) {
56  stop = (step < 0) ? -1 : len;
57  } else if (stop<0) {
58  stop+=len;
59  }
60 
61  casadi_assert_message(stop<=len,
62  "Slice (start=" << start << ", stop=" << stop << ", step=" << step
63  << ") out of bounds with supplied length of " << len);
64  casadi_assert_message(start>=0,
65  "Slice (start=" << start << ", stop=" << stop << ", step=" << step
66  << ") out of bounds with start<0.");
67  if ((stop>=start && step<0) || (stop<=start && step>0)) return std::vector<int>();
68 
69  return range(start+ind1, stop+ind1, step, len+ind1);

114  Slice CASADI_EXPORT to_slice(const std::vector<int>& v, bool ind1) {
115  Slice r;
116  casadi_assert_message(is_slice(v, ind1), "Cannot be represented as a Slice");
117  if (v.size()==0) {
118  r.start=r.stop=0;
119  r.step = 1;
120  } else if (v.size()==1) {
121  r.start = v.front()-ind1;
122  r.stop = r.start + 1;
123  r.step = 1;
124  } else {
125  r.start = v[0]-ind1;
126  r.step = v[1]-v[0];
127  r.stop = r.start + r.step*v.size();
128  }
129  return r;
130  }

132  bool CASADI_EXPORT is_slice(const std::vector<int>& v, bool ind1) {
133  // Always false if negative numbers or non-increasing
134  int last_v = -1;
135  for (int i=0; i<v.size(); ++i) {
136  casadi_assert_message(!(ind1 && v[i]<=0), "Matlab is 1-based, but requested index " <<
137  v[i] << ". Note that negative slices are" <<
138  " disabled in the Matlab interface. " <<
139  "Possibly you may want to use 'end'.");
140  if (v[i]-ind1<=last_v) return false;
141  last_v = v[i]-ind1;
142  }
143 
144  // Always true if less than 2 elements
145  if (v.size()<2) return true;
146 
147  // If two elements, true if they are different
148  if (v.size()==2) return v[0]!=v[1];
149 
150  // We can now get the beginning, end and step
151  int start = v[0]-ind1;
152  int step = v[1]-v[0];
153  //int stop = start + step*v.size();
154 
155  // Consistency check
156  for (int i=2; i<v.size(); ++i) {
157  if (v[i]-ind1!=start+i*step) return false;
158  }
159 
160  // True if reached this point
161  return true;
162  }

164  bool CASADI_EXPORT is_slice2(const std::vector<int>& v) {
165  // Always true if 1D slice
166  if (is_slice(v)) return true;
167 
168  // Always false if negative numbers or non-increasing
169  int last_v = -1;
170  for (int i=0; i<v.size(); ++i) {
171  if (v[i]<=last_v) return false;
172  last_v = v[i];
173  }
174 
175  // Get the slices
176  int start_outer = 0;
177  int step_outer = -1;
178  int start_inner = v.front();
179  int step_inner = v[1]-v[0];
180  int stop_inner = -1;
181  for (int i=2; i<v.size(); ++i) {
182  int predicted_v = start_inner+i*step_inner;
183  if (v[i]!=predicted_v) {
184  stop_inner = predicted_v;
185  step_outer = v[i] - start_inner;
186  break;
187  }
188  }
189  casadi_assert(stop_inner>=0);
190 
191  // Get the end of the outer slice
192  int stop_outer = v.back();
193  do {
194  if (step_outer>0) stop_outer++;
195  else stop_outer--;
196  } while (stop_outer % step_outer!=0);
197 
198  // Check consistency
199  std::vector<int>::const_iterator it=v.begin();
200  for (int i=start_outer; i!=stop_outer; i+=step_outer) {
201  for (int j=i+start_inner; j!=i+stop_inner; j+=step_inner) {
202  // False if we've reached the end
203  if (it==v.end()) return false;
204 
205  // Check if value matches
206  if (*it++ != j) return false;
207  }
208  }
209 
210  // False if there are still elements not accounted for
211  if (it!=v.end()) return false;
212 
213  // True if reached this point
214  return true;
215  }

217  std::pair<Slice, Slice> CASADI_EXPORT to_slice2(const std::vector<int>& v) {
218  casadi_assert_message(is_slice2(v), "Cannot be represented as a nested Slice");
219  Slice inner, outer;
220 
221  // If simple slice
222  if (is_slice(v)) {
223  inner = to_slice(v);
224  outer.start = 0;
225  outer.step = outer.stop = inner.stop;
226  return make_pair(inner, outer);
227  }
228 
229  // Get the slices
230  outer.start = 0;
231  outer.step = -1;
232  inner.start = v.front();
233  inner.step = v[1]-v[0];
234  inner.stop = -1;
235  for (int i=2; i<v.size(); ++i) {
236  int predicted_v = inner.start+i*inner.step;
237  if (v[i]!=predicted_v) {
238  inner.stop = predicted_v;
239  outer.step = v[i] - inner.start;
240  break;
241  }
242  }
243 
244  // Get the end of the outer slice
245  outer.stop = v.back();
246  do {
247  if (outer.step>0) outer.stop++;
248  else outer.stop--;
249  } while (outer.stop % outer.step!=0);
250  return make_pair(inner, outer);
251  }
