72  void LapackQr::linsol_factorize(void* mem, const double* A) const {
73  auto m = static_cast<LapackQrMemory*>(mem);
74 
75  // Dimensions
76  //int nrow = this->nrow();
77  int ncol = this->ncol();
78 
79  // Get the elements of the matrix, dense format
80  casadi_densify(A, sparsity_, get_ptr(m->mat), false);
81 
82  // Factorize the matrix
83  int info = -100;
84  int lwork = m->work.size();
85  dgeqrf_(&ncol, &ncol, get_ptr(m->mat), &ncol, get_ptr(m->tau),
86  get_ptr(m->work), &lwork, &info);
87  if (info != 0) throw CasadiException("LapackQr::prepare: dgeqrf_ "
88  "failed to factorize the Jacobian");

91  void LapackQr::linsol_solve(void* mem, double* x, int nrhs, bool tr) const {
92  auto m = static_cast<LapackQrMemory*>(mem);
93 
94  // Dimensions
95  int nrow = this->nrow();
96  int ncol = this->ncol();
97 
98  // Properties of R
99  char uploR = 'U';
100  char diagR = 'N';
101  char sideR = 'L';
102  double alphaR = 1.;
103  char transR = tr ? 'T' : 'N';
104 
105  // Properties of Q
106  char transQ = tr ? 'N' : 'T';
107  char sideQ = 'L';
108  int k = m->tau.size(); // minimum of ncol and nrow
109  int lwork = m->work.size();
110 
111  if (tr) {
112 
113  // Solve for transpose(R)
114  dtrsm_(&sideR, &uploR, &transR, &diagR, &ncol, &nrhs, &alphaR,
115  get_ptr(m->mat), &ncol, x, &ncol);
116 
117  // Multiply by Q
118  int info = 100;
119  dormqr_(&sideQ, &transQ, &ncol, &nrhs, &k, get_ptr(m->mat), &ncol, get_ptr(m->tau), x,
120  &ncol, get_ptr(m->work), &lwork, &info);
121  if (info != 0) throw CasadiException("LapackQr::solve: dormqr_ failed "
122  "to solve the linear system");
123 
124  } else {
125 
126  // Multiply by transpose(Q)
127  int info = 100;
128  dormqr_(&sideQ, &transQ, &ncol, &nrhs, &k, get_ptr(m->mat), &ncol, get_ptr(m->tau), x,
129  &ncol, get_ptr(m->work), &lwork, &info);
130  if (info != 0) throw CasadiException("LapackQr::solve: dormqr_ failed to "
131  "solve the linear system");
132 
133  // Solve for R
134  dtrsm_(&sideR, &uploR, &transR, &diagR, &ncol, &nrhs, &alphaR,
135  get_ptr(m->mat), &ncol, x, &ncol);
136  }
137  }
