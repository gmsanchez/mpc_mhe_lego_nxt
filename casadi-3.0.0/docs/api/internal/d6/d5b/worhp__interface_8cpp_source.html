67  void WorhpInterface::init(const Dict& opts) {
68 
69  // Call the init method of the base class
70  Nlpsol::init(opts);
71 
72  // Default options
73  Dict worhp_opts;
74 
75  // Read user options
76  for (auto&& op : opts) {
77  if (op.first=="worhp") {
78  worhp_opts = op.second;
79  }
80  }
81 
82  // Sort Worhp options
83  int nopts = WorhpGetParamCount();
84  for (auto&& op : worhp_opts) {
85  // Get corresponding index using a linear search
86  int ind;
87  for (ind=1; ind<=nopts; ++ind) {
88  // Get name in WORHP
89  const char* name = WorhpGetParamName(ind);
90  // Break if matching name
91  if (op.first.compare(name)==0) break;
92  }
93  if (ind>nopts) casadi_error("No such Worhp option: " + op.first);
94 
95  // Add to the corresponding list
96  switch (WorhpGetParamType(ind)) {
97  case WORHP_BOOL_T:
98  bool_opts_[op.first] = op.second;
99  break;
100  case WORHP_DOUBLE_T:
101  double_opts_[op.first] = op.second;
102  break;
103  case WORHP_INT_T:
104  int_opts_[op.first] = op.second;
105  break;
106  default:
107  casadi_error("Cannot handle WORHP option \"" + op.first + "\": Unknown type");
108  break;
109  }
110  }
111 
112  // Setup NLP functions
113  f_fcn_ = create_function("nlp_f", {"x", "p"}, {"f"});
114  g_fcn_ = create_function("nlp_g", {"x", "p"}, {"g"});
115  grad_f_fcn_ = create_function("nlp_grad_f", {"x", "p"}, {"f", "grad_f_x"});
116  jac_g_fcn_ = create_function("nlp_jac_g", {"x", "p"}, {"g", "jac_g_x"});
117  jacg_sp_ = jac_g_fcn_.sparsity_out(1);
118  hess_l_fcn_ = create_function("nlp_jac_f", {"x", "p", "lam_f", "lam_g"},
119  {"transpose_hess_gamma_x_x"},
120  {{"gamma", {"f", "g"}}});
121  hesslag_sp_ = hess_l_fcn_.sparsity_out(0);
122 
123  // Temporary vectors
124  alloc_w(nx_); // for fetching diagonal entries form Hessian
125  }

154  void WorhpInterface::set_work(void* mem, const double**& arg, double**& res,
155  int*& iw, double*& w) const {
156  auto m = static_cast<WorhpMemory*>(mem);
157 
158  // Set work in base classes
159  Nlpsol::set_work(mem, arg, res, iw, w);
160 
161  // Number of (free) variables
162  m->worhp_o.n = nx_;
163 
164  // Number of constraints
165  m->worhp_o.m = ng_;
166 
167  // Free existing Worhp memory (except parameters)
168  bool p_init_backup = m->worhp_p.initialised;
169  m->worhp_p.initialised = false; // Avoid freeing the memory for parameters
170  if (m->worhp_o.initialised || m->worhp_w.initialised || m->worhp_c.initialised) {
171  WorhpFree(&m->worhp_o, &m->worhp_w, &m->worhp_p, &m->worhp_c);
172  }
173  m->worhp_p.initialised = p_init_backup;
174 
176  m->worhp_c.initialised = false;
177  m->worhp_w.initialised = false;
178  m->worhp_o.initialised = false;
179 
180  // Worhp uses the CS format internally, hence it is the preferred sparse matrix format.
181  m->worhp_w.DF.nnz = nx_;
182  if (m->worhp_o.m>0) {
183  m->worhp_w.DG.nnz = jacg_sp_.nnz(); // Jacobian of G
184  } else {
185  m->worhp_w.DG.nnz = 0;
186  }
187 
188  if (true /*m->worhp_w.HM.NeedStructure*/) { // not initialized
189  m->worhp_w.HM.nnz = nx_ + hesslag_sp_.nnz_lower(true);
190  } else {
191  m->worhp_w.HM.nnz = 0;
192  }
193 
194  /* Data structure initialisation. */
195  WorhpInit(&m->worhp_o, &m->worhp_w, &m->worhp_p, &m->worhp_c);
196  if (m->worhp_c.status != FirstCall) {
197  string msg = return_codes(m->worhp_c.status);
198  casadi_error("Main: Initialisation failed. Status: " + msg);
199  }
200 
201  if (m->worhp_w.DF.NeedStructure) {
202  for (int i=0; i<nx_; ++i) {
203  m->worhp_w.DF.row[i] = i + 1; // Index-1 based
204  }
205  }
206 
207  if (m->worhp_o.m>0 && m->worhp_w.DG.NeedStructure) {
208  int nz=0;
209  const int* colind = jacg_sp_.colind();
210  const int* row = jacg_sp_.row();
211  for (int c=0; c<nx_; ++c) {
212  for (int el=colind[c]; el<colind[c+1]; ++el) {
213  int r = row[el];
214  m->worhp_w.DG.col[nz] = c + 1; // Index-1 based
215  m->worhp_w.DG.row[nz] = r + 1;
216  nz++;
217  }
218  }
219  }
220 
221  if (m->worhp_w.HM.NeedStructure) {
222  // Get the sparsity pattern of the Hessian
223  const int* colind = hesslag_sp_.colind();
224  const int* row = hesslag_sp_.row();
225 
226  int nz=0;
227 
228  // Strictly lower triangular part of the Hessian (note CCS -> CRS format change)
229  for (int c=0; c<nx_; ++c) {
230  for (int el=colind[c]; el<colind[c+1]; ++el) {
231  if (row[el]>c) {
232  m->worhp_w.HM.row[nz] = row[el] + 1;
233  m->worhp_w.HM.col[nz] = c + 1;
234  nz++;
235  }
236  }
237  }
238 
239  // Diagonal always included
240  for (int r=0; r<nx_; ++r) {
241  m->worhp_w.HM.row[nz] = r + 1;
242  m->worhp_w.HM.col[nz] = r + 1;
243  nz++;
244  }
245  }

248  void WorhpInterface::solve(void* mem) const {
249  auto m = static_cast<WorhpMemory*>(mem);
250 
251  // Statistics
252  for (auto&& s : m->fstats) s.second.reset();
253 
254  // Check the provided inputs
255  checkInputs(mem);
256 
257  double inf = numeric_limits<double>::infinity();
258 
259  if (m->lbx && m->ubx) {
260  for (int i=0; i<nx_;++i) {
261  casadi_assert_message(m->lbx[i]!=m->ubx[i],
262  "WorhpInterface::evaluate: Worhp cannot handle the case when "
263  "LBX == UBX."
264  "You have that case at non-zero " << i << " , which has value " <<
265  m->ubx[i] << ". Reformulate your problem by using a parameter "
266  "for the corresponding variable.");
267  }
268  }
269 
270  if (m->lbg && m->ubg) {
271  for (int i=0; i<ng_; ++i) {
272  casadi_assert_message(!(m->lbg[i]==-inf && m->ubg[i] == inf),
273  "WorhpInterface::evaluate: Worhp cannot handle the case when both "
274  "LBG and UBG are infinite."
275  "You have that case at non-zero " << i << "."
276  "Reformulate your problem eliminating the corresponding constraint.");
277  }
278  }
279 
280  m->fstats.at("mainloop").tic();
281 
282  // Pass inputs to WORHP data structures
283  casadi_copy(m->x0, nx_, m->worhp_o.X);
284  casadi_copy(m->lbx, nx_, m->worhp_o.XL);
285  casadi_copy(m->ubx, nx_, m->worhp_o.XU);
286  casadi_copy(m->lam_x0, nx_, m->worhp_o.Lambda);
287  if (m->worhp_o.m>0) {
288  casadi_copy(m->lam_g0, ng_, m->worhp_o.Mu);
289  casadi_copy(m->lbg, ng_, m->worhp_o.GL);
290  casadi_copy(m->ubg, ng_, m->worhp_o.GU);
291  }
292 
293  // Replace infinite bounds with m->worhp_p.Infty
294  for (int i=0; i<nx_; ++i) if (m->worhp_o.XL[i]==-inf) m->worhp_o.XL[i] = -m->worhp_p.Infty;
295  for (int i=0; i<nx_; ++i) if (m->worhp_o.XU[i]== inf) m->worhp_o.XU[i] = m->worhp_p.Infty;
296  for (int i=0; i<ng_; ++i) if (m->worhp_o.GL[i]==-inf) m->worhp_o.GL[i] = -m->worhp_p.Infty;
297  for (int i=0; i<ng_; ++i) if (m->worhp_o.GU[i]== inf) m->worhp_o.GU[i] = m->worhp_p.Infty;
298 
299  log("WorhpInterface::starting iteration");
300 
301  bool firstIteration = true;
302 
303  // Reverse Communication loop
304  while (m->worhp_c.status < TerminateSuccess && m->worhp_c.status > TerminateError) {
305  if (GetUserAction(&m->worhp_c, callWorhp)) {
306  Worhp(&m->worhp_o, &m->worhp_w, &m->worhp_p, &m->worhp_c);
307  }
308 
309 
310  if (GetUserAction(&m->worhp_c, iterOutput)) {
311 
312  if (!firstIteration) {
313  firstIteration = true;
314 
315  if (!fcallback_.is_null()) {
316  m->fstats.at("callback_prep").tic();
317  m->iter = m->worhp_w.MajorIter;
318  m->iter_sqp = m->worhp_w.MinorIter;
319  m->inf_pr = m->worhp_w.NormMax_CV;
320  m->inf_du = m->worhp_w.ScaledKKT;
321  m->alpha_pr = m->worhp_w.ArmijoAlpha;
322 
323  // Inputs
324  fill_n(m->arg, fcallback_.n_in(), nullptr);
325  m->arg[NLPSOL_X] = m->worhp_o.X;
326  m->arg[NLPSOL_F] = &m->worhp_o.F;
327  m->arg[NLPSOL_G] = m->worhp_o.G;
328  m->arg[NLPSOL_LAM_P] = 0;
329  m->arg[NLPSOL_LAM_X] = m->worhp_o.Lambda;
330  m->arg[NLPSOL_LAM_G] = m->worhp_o.Mu;
331 
332  // Outputs
333  fill_n(m->res, fcallback_.n_out(), nullptr);
334  double ret_double;
335  m->res[0] = &ret_double;
336 
337  m->fstats.at("callback_prep").toc();
338  m->fstats.at("callback_fun").tic();
339  // Evaluate the callback function
340  fcallback_(m->arg, m->res, m->iw, m->w, 0);
341  m->fstats.at("callback_fun").toc();
342  int ret = static_cast<int>(ret_double);
343 
344  if (ret) m->worhp_c.status = TerminatedByUser;
345  }
346  }
347 
348 
349  IterationOutput(&m->worhp_o, &m->worhp_w, &m->worhp_p, &m->worhp_c);
350  DoneUserAction(&m->worhp_c, iterOutput);
351  }
352 
353  if (GetUserAction(&m->worhp_c, evalF)) {
354  calc_function(m, f_fcn_, {m->worhp_o.X, m->p}, {&m->worhp_o.F});
355  if (m->f) *m->f = m->worhp_o.F; // Store cost, before scaling
356  m->worhp_o.F *= m->worhp_w.ScaleObj;
357  DoneUserAction(&m->worhp_c, evalF);
358  }
359 
360  if (GetUserAction(&m->worhp_c, evalG)) {
361  calc_function(m, g_fcn_, {m->worhp_o.X, m->p}, {m->worhp_o.G});
362  DoneUserAction(&m->worhp_c, evalG);
363  }
364 
365  if (GetUserAction(&m->worhp_c, evalDF)) {
366  calc_function(m, grad_f_fcn_, {m->worhp_o.X, m->p}, {0, m->worhp_w.DF.val});
367  casadi_scal(nx_, m->worhp_w.ScaleObj, m->worhp_w.DF.val);
368  DoneUserAction(&m->worhp_c, evalDF);
369  }
370 
371  if (GetUserAction(&m->worhp_c, evalDG)) {
372  calc_function(m, jac_g_fcn_, {m->worhp_o.X, m->p}, {0, m->worhp_w.DG.val});
373  DoneUserAction(&m->worhp_c, evalDG);
374  }
375 
376  if (GetUserAction(&m->worhp_c, evalHM)) {
377  calc_function(m, hess_l_fcn_, {m->worhp_o.X, m->p, &m->worhp_w.ScaleObj, m->worhp_o.Mu},
378  {m->worhp_w.HM.val});
379  // Diagonal values
380  double *dval = m->w;
381  casadi_fill(dval, nx_, 0.);
382 
383  // Remove diagonal
384  const int* colind = hesslag_sp_.colind();
385  const int* row = hesslag_sp_.row();
386  int ind=0;
387  for (int c=0; c<nx_; ++c) {
388  for (int el=colind[c]; el<colind[c+1]; ++el) {
389  if (row[el]==c) {
390  dval[c] = m->worhp_w.HM.val[el];
391  } else {
392  m->worhp_w.HM.val[ind++] = m->worhp_w.HM.val[el];
393  }
394  }
395  }
396 
397  // Add diagonal entries at the end
398  casadi_copy(dval, nx_, m->worhp_w.HM.val+ind);
399  DoneUserAction(&m->worhp_c, evalHM);
400  }
401 
402  if (GetUserAction(&m->worhp_c, fidif)) {
403  WorhpFidif(&m->worhp_o, &m->worhp_w, &m->worhp_p, &m->worhp_c);
404  }
405  }
406 
407  m->fstats.at("mainloop").toc();
408 
409  // Copy outputs
410  casadi_copy(m->worhp_o.X, nx_, m->x);
411  casadi_copy(m->worhp_o.G, ng_, m->g);
412  casadi_copy(m->worhp_o.Lambda, nx_, m->lam_x);
413  casadi_copy(m->worhp_o.Mu, ng_, m->lam_g);
414 
415  StatusMsg(&m->worhp_o, &m->worhp_w, &m->worhp_p, &m->worhp_c);
416 
417  // Show statistics
418  if (print_time_) print_fstats(m);
419 
420  m->return_code = m->worhp_c.status;
421  m->return_status = return_codes(m->worhp_c.status);
422  }
