107  void Newton::eval(void* mem, const double** arg, double** res,
108  int* iw, double* w) const {
109  auto m = static_cast<NewtonMemory*>(mem);
110 
111  // IO buffers
112  const double** arg1 = arg + n_in();
113  double** res1 = res + n_out();
114 
115  // Work vectors
116  double* x = w; w += n_;
117  double* f = w; w += jac_.nnz_out(1+iout_);
118  double* jac = w; w += jac_.nnz_out(0);
119 
120  // Get the initial guess
121  if (arg[iin_]) {
122  copy_n(arg[iin_], n_, x);
123  } else {
124  fill_n(x, n_, 0);
125  }
126 
127  // Perform the Newton iterations
128  int iter=0;
129  bool success = true;
130  while (true) {
131  // Break if maximum number of iterations already reached
132  if (iter >= max_iter_) {
133  log("eval", "Max. iterations reached.");
134  m->return_status = "max_iteration_reached";
135  success = false;
136  break;
137  }
138 
139  // Start a new iteration
140  iter++;
141 
142  // Use x to evaluate J
143  copy_n(arg, n_in(), arg1);
144  arg1[iin_] = x;
145  res1[0] = jac;
146  copy_n(res, n_out(), res1+1);
147  res1[1+iout_] = f;
148  jac_(arg1, res1, iw, w, 0);
149 
150  // Check convergence
151  double abstol = 0;
152  if (abstol_ != numeric_limits<double>::infinity()) {
153  for (int i=0; i<n_; ++i) {
154  abstol = max(abstol, fabs(f[i]));
155  }
156  if (abstol <= abstol_) {
157  casadi_msg("Converged to acceptable tolerance - abstol: " << abstol_);
158  break;
159  }
160  }
161 
162  // Factorize the linear solver with J
163  linsol_.setup(arg1 + LINSOL_NUM_IN, res1 + LINSOL_NUM_OUT, iw, w);
164  linsol_.linsol_factorize(jac);
165  linsol_.linsol_solve(f, 1, false);
166 
167  // Check convergence again
168  double abstolStep=0;
169  if (numeric_limits<double>::infinity() != abstolStep_) {
170  for (int i=0; i<n_; ++i) {
171  abstolStep = max(abstolStep, fabs(f[i]));
172  }
173  if (abstolStep <= abstolStep_) {
174  casadi_msg("Converged to acceptable tolerance - abstolStep: " << abstolStep_);
175  break;
176  }
177  }
178 
179  if (print_iteration_) {
180  // Only print iteration header once in a while
181  if (iter % 10==0) {
182  printIteration(userOut());
183  }
184 
185  // Print iteration information
186  printIteration(userOut(), iter, abstol, abstolStep);
187  }
188 
189  // Update Xk+1 = Xk - J^(-1) F
190  casadi_axpy(n_, -1., f, x);
191  }
192 
193  // Get the solution
194  if (res[iout_]) {
195  copy_n(x, n_, res[iout_]);
196  }
197 
198  // Store the iteration count
199  if (gather_stats_) m->iter = iter;
200  if (success) m->return_status = "success";
201 
202  casadi_msg("Newton::solveNonLinear():end after " << iter << " steps");
203  }
