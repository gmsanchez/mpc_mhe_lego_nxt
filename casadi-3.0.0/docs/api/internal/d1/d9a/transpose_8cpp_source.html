54  void Transpose::evalGen(const T* const* arg, T* const* res,
55  int* iw, T* w) const {
56  // Get sparsity patterns
57  //const vector<int>& x_colind = input[0]->colind();
58  const int* x_row = dep(0).row();
59  int x_sz = dep(0).nnz();
60  const int* xT_colind = sparsity().colind();
61  int xT_ncol = sparsity().size2();
62 
63  const T* x = arg[0];
64  T* xT = res[0];
65 
66  // Transpose
67  copy(xT_colind, xT_colind+xT_ncol+1, iw);
68  for (int el=0; el<x_sz; ++el) {
69  xT[iw[x_row[el]]++] = x[el];
70  }
71  }
