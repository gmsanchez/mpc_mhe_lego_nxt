104  HorzRepsum::HorzRepsum(const MX& x, int n) : n_(n) {
105  casadi_assert(x.size2() % n == 0);
106  std::vector<Sparsity> sp = horzsplit(x.sparsity(), x.size2()/n);
107  Sparsity block = sp[0];
108  for (int i=1;i<sp.size();++i) {
109  block = block+sp[i];
110  }
111  Sparsity goal = repmat(block, 1, n);
112  setDependencies(project(x, goal));
113  setSparsity(block);
114  }
