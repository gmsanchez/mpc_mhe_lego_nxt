60  std::string Reshape::print(const std::vector<std::string>& arg) const {
61  // For vectors, reshape is also a transpose
62  if (dep().is_vector() && sparsity().is_vector()) {
63  // Print as transpose: X'
64  return arg.at(0) + "'";
