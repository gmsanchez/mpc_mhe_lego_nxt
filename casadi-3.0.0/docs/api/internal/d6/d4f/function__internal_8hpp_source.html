578  virtual int index_in(const std::string &name) const {
579  const std::vector<std::string>& v=ischeme_;
580  for (std::vector<std::string>::const_iterator i=v.begin(); i!=v.end(); ++i) {
581  size_t col = i->find(':');
582  if (i->compare(0, col, name)==0) return i-v.begin();
583  }
584  casadi_error("FunctionInternal::index_in: could not find entry \""
585  << name << "\". Available names are: " << v << ".");
586  return -1;
587  }

942  template<typename D>
943  void FunctionInternal::_call(const std::vector<Matrix<D> >& arg, std::vector<Matrix<D> >& res,
944  bool always_inline, bool never_inline) {
945  casadi_assert_message(!never_inline, "Call-nodes only possible in MX expressions");
946 
947  // Get the number of inputs and outputs
948  int n_in = this->n_in();
949  int n_out = this->n_out();
950 
951  // Check if matching input sparsity
952  bool matching_sparsity = true;
953  casadi_assert(arg.size()==n_in);
954  for (int i=0; matching_sparsity && i<n_in; ++i)
955  matching_sparsity = arg[i].sparsity()==sparsity_in(i);
956 
957  // Correct input sparsity if needed
958  if (!matching_sparsity) {
959  std::vector<Matrix<D> > arg2(arg);
960  for (int i=0; i<n_in; ++i)
961  if (arg2[i].sparsity()!=sparsity_in(i))
962  arg2[i] = project(arg2[i], sparsity_in(i));
963  return _call(arg2, res, always_inline, never_inline);
964  }
965 
966  // Allocate results
967  res.resize(n_out);
968  for (int i=0; i<n_out; ++i) {
969  if (res[i].sparsity()!=sparsity_out(i)) {
970  res[i] = Matrix<D>::zeros(sparsity_out(i));
971  }
972  }
973 
974  // Allocate temporary memory if needed
975  std::vector<int> iw_tmp(sz_iw());
976  std::vector<D> w_tmp(sz_w());
977 
978  // Get pointers to input arguments
979  std::vector<const D*> argp(sz_arg());
980  for (int i=0; i<n_in; ++i) argp[i]=get_ptr(arg[i]);
981 
982  // Get pointers to output arguments
983  std::vector<D*> resp(sz_res());
984  for (int i=0; i<n_out; ++i) resp[i]=get_ptr(res[i]);
985 
986  // Call memory-less
987  _eval(get_ptr(argp), get_ptr(resp), get_ptr(iw_tmp), get_ptr(w_tmp), 0);
988  }
