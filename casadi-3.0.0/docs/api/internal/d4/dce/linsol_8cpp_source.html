177  void Linsol::eval(void* mem, const double** arg, double** res,
178  int* iw, double* w) const {
179  // Get inputs and outputs
180  const double *A = arg[LINSOL_A];
181  const double *b = arg[LINSOL_B];
182  arg += LINSOL_NUM_IN;
183  double *x = res[LINSOL_X];
184  res += LINSOL_NUM_OUT;
185 
186  // If output not requested, nothing to do
187  if (!x) return;
188 
189  // A zero linear system would be singular
190  if (A==0) {
191  casadi_fill(x, neq_*nrhs_, numeric_limits<double>::quiet_NaN());
192  return;
193  }
194 
195  // If right hand side is zero, solution is trivially zero (if well-defined)
196  if (!b) {
197  casadi_fill(x, neq_*nrhs_, 0.);
198  return;
199  }
200 
201  // Setup memory object
202  setup(mem, arg, res, iw, w);
203 
204  // Factorize the linear system
205  linsol_factorize(mem, A);
206 
207  // Solve the factorized system
208  casadi_copy(b, neq_*nrhs_, x);
209  linsol_solve(mem, x, nrhs_, false);
210  }

212  void Linsol::
213  linsol_forward(const std::vector<MX>& arg, const std::vector<MX>& res,
214  const std::vector<std::vector<MX> >& fseed,
215  std::vector<std::vector<MX> >& fsens, bool tr) {
216  // Number of derivatives
217  int nfwd = fseed.size();
218  const MX& A = arg[1];
219  const MX& X = res[0];
220 
221  // Solve for all directions at once
222  std::vector<MX> rhs(nfwd);
223  std::vector<int> col_offset(nfwd+1, 0);
224  for (int d=0; d<nfwd; ++d) {
225  const MX& B_hat = fseed[d][0];
226  const MX& A_hat = fseed[d][1];
227  rhs[d] = tr ? B_hat - mtimes(A_hat.T(), X) : B_hat - mtimes(A_hat, X);
228  col_offset[d+1] = col_offset[d] + rhs[d].size2();
229  }
230  rhs = horzsplit(linsol_solve(A, horzcat(rhs), tr), col_offset);
231 
232  // Fetch result
233  fsens.resize(nfwd);
234  for (int d=0; d<nfwd; ++d) {
235  fsens[d].resize(1);
236  fsens[d][0] = rhs[d];
237  }
238  }

240  void Linsol::
241  linsol_reverse(const std::vector<MX>& arg, const std::vector<MX>& res,
242  const std::vector<std::vector<MX> >& aseed,
243  std::vector<std::vector<MX> >& asens, bool tr) {
244  // Number of derivatives
245  int nadj = aseed.size();
246  const MX& A = arg[1];
247  const MX& X = res[0];
248 
249  // Solve for all directions at once
250  std::vector<MX> rhs(nadj);
251  std::vector<int> col_offset(nadj+1, 0);
252  for (int d=0; d<nadj; ++d) {
253  rhs[d] = aseed[d][0];
254  col_offset[d+1] = col_offset[d] + rhs[d].size2();
255  }
256  rhs = horzsplit(linsol_solve(A, horzcat(rhs), !tr), col_offset);
257 
258  // Collect sensitivities
259  asens.resize(nadj);
260  for (int d=0; d<nadj; ++d) {
261  asens[d].resize(2);
262 
263  // Propagate to A
264  MX a;
265  if (!tr) {
266  a = -mac(rhs[d], X.T(), MX::zeros(A.sparsity()));
267  } else {
268  a = -mac(X, rhs[d].T(), MX::zeros(A.sparsity()));
269  }
270  if (asens[d][1].is_empty(true)) {
271  asens[d][1] = a;
272  } else {
273  asens[d][1] += a;
274  }
275 
276  // Propagate to B
277  if (asens[d][0].is_empty(true)) {
278  asens[d][0] = rhs[d];
279  } else {
280  asens[d][0] += rhs[d];
281  }
282  }
283  }

285  void Linsol::
286  linsol_spFwd(const bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem,
287  bool tr, int nrhs) {
288  // Sparsities
289  const Sparsity& A_sp = sparsity_in(LINSOL_A);
290  const int* A_colind = A_sp.colind();
291  const int* A_row = A_sp.row();
292  int n = A_sp.size1();
293 
294  // Get pointers to data
295  const bvec_t *B=arg[0], *A = arg[1];
296  bvec_t* X = res[0];
297  bvec_t* tmp = w;
298 
299  // For all right-hand-sides
300  for (int r=0; r<nrhs; ++r) {
301  // Copy B to a temporary vector
302  copy(B, B+n, tmp);
303 
304  // Add A_hat contribution to tmp
305  for (int cc=0; cc<n; ++cc) {
306  for (int k=A_colind[cc]; k<A_colind[cc+1]; ++k) {
307  int rr = A_row[k];
308  tmp[tr ? cc : rr] |= A[k];
309  }
310  }
311 
312  // Propagate to X
313  std::fill(X, X+n, 0);
314  linsol_spsolve(X, tmp, tr);
315 
316  // Continue to the next right-hand-side
317  B += n;
318  X += n;
319  }
320  }

322  void Linsol::
323  linsol_spAdj(bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem,
324  bool tr, int nrhs) {
325  // Sparsities
326  const Sparsity& A_sp = sparsity_in(LINSOL_A);
327  const int* A_colind = A_sp.colind();
328  const int* A_row = A_sp.row();
329  int n = A_sp.size1();
330 
331  // Get pointers to data
332  bvec_t *B=arg[0], *A=arg[1], *X=res[0];
333  bvec_t* tmp = w;
334 
335  // For all right-hand-sides
336  for (int r=0; r<nrhs; ++r) {
337  // Solve transposed
338  std::fill(tmp, tmp+n, 0);
339  linsol_spsolve(tmp, X, !tr);
340 
341  // Clear seeds
342  std::fill(X, X+n, 0);
343 
344  // Propagate to B
345  for (int i=0; i<n; ++i) B[i] |= tmp[i];
346 
347  // Propagate to A
348  for (int cc=0; cc<n; ++cc) {
349  for (int k=A_colind[cc]; k<A_colind[cc+1]; ++k) {
350  int rr = A_row[k];
351  A[k] |= tmp[tr ? cc : rr];
352  }
353  }
354 
355  // Continue to the next right-hand-side
356  B += n;
357  X += n;
358  }
359  }

368  void Linsol::linsol_spsolve(bvec_t* X, const bvec_t* B, bool tr) const {
369 
370  const Sparsity& A_sp = sparsity_in(LINSOL_A);
371  const int* A_colind = A_sp.colind();
372  const int* A_row = A_sp.row();
373  int nb = rowblock_.size()-1; // number of blocks
374 
375  if (!tr) {
376  for (int b=0; b<nb; ++b) { // loop over the blocks forward
377 
378  // Get dependencies from all right-hand-sides in the block ...
379  bvec_t block_dep = 0;
380  for (int el=rowblock_[b]; el<rowblock_[b+1]; ++el) {
381  int rr = rowperm_[el];
382  block_dep |= B[rr];
383  }
384 
385  // ... as well as all other variables in the block
386  for (int el=colblock_[b]; el<colblock_[b+1]; ++el) {
387  int cc = colperm_[el];
388  block_dep |= X[cc];
389  }
390 
391  // Propagate ...
392  for (int el=colblock_[b]; el<colblock_[b+1]; ++el) {
393  int cc = colperm_[el];
394 
395  // ... to all variables in the block ...
396  X[cc] |= block_dep;
397 
398  // ... as well as to other variables which depends on variables in the block
399  for (int k=A_colind[cc]; k<A_colind[cc+1]; ++k) {
400  int rr=A_row[k];
401  X[rr] |= block_dep;
402  }
403  }
404  }
405 
406  } else { // transpose
407  for (int b=nb-1; b>=0; --b) { // loop over the blocks backward
408 
409  // Get dependencies ...
410  bvec_t block_dep = 0;
411  for (int el=colblock_[b]; el<colblock_[b+1]; ++el) {
412  int cc = colperm_[el];
413 
414  // .. from all right-hand-sides in the block ...
415  block_dep |= B[cc];
416 
417  // ... as well as from all depending variables ...
418  for (int k=A_colind[cc]; k<A_colind[cc+1]; ++k) {
419  int rr=A_row[k];
420  block_dep |= X[rr];
421  }
422  }
423 
424  // Propagate to all variables in the block
425  for (int el=rowblock_[b]; el<rowblock_[b+1]; ++el) {
426  int rr = rowperm_[el];
427  X[rr] |= block_dep;
428  }
429  }
430  }
431  }
