34  class CASADI_EXPORT SparsityInternal : public SharedObjectNode {
35  private:
36  /* \brief Sparsity pattern in compressed column storage (CCS) format
37  The first two entries are the number of rows (nrow) and columns (ncol).
38  The next (ncol+1) entries are the column offsets (colind). This means that
39  the number of nonzeros (nnz) is given as sp_[sp_[1]+2].
40  The last nnz entries are the rows of the nonzeros (row). See public class
41  for more info about the CCS format used in CasADi. */
42  std::vector<int> sp_;
43 
44  public:
46  SparsityInternal(int nrow, int ncol, const int* colind, const int* row) :
47  sp_(2 + ncol+1 + colind[ncol]) {
48  sp_[0] = nrow;
49  sp_[1] = ncol;
50  std::copy(colind, colind+ncol+1, sp_.begin()+2);
51  std::copy(row, row+colind[ncol], sp_.begin()+2+ncol+1);
52  sanity_check(false);
53  }
