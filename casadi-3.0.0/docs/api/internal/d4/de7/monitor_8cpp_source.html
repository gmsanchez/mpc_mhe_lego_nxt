93  void Monitor::spAdj(bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem) {
94  bvec_t *a = arg[0];
95  bvec_t *r = res[0];
96  int n = nnz();
97  if (a != r) {
98  for (int i=0; i<n; ++i) {
99  *a++ |= *r;
100  *r++ = 0;
101  }
102  }
103  }
