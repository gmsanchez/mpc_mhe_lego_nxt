52  bool GenericType::can_cast_to(TypeID other) const {
53  switch (other) {
54  case OT_BOOL:
55  return is_bool() || is_int() || is_double();
56  case OT_BOOLVECTOR:
57  return is_int_vector() || is_double_vector();
58  case OT_INT:
59  case OT_DOUBLE:
60  return is_int() || is_double();
61  case OT_INTVECTOR:
62  case OT_DOUBLEVECTOR:
63  return is_double_vector() || is_int_vector();
64  case OT_STRINGVECTOR:
65  return is_string_vector() || is_string();
66  default:
67  return getType() == other;
68  }
69  }

376  bool GenericType::operator!=(const GenericType& op2) const {
377  if (is_string() && op2.is_string()) {
378  return to_string().compare(op2.to_string()) != 0;
379  }
380 
381  if (is_int() && op2.is_int()) {
382  return to_int() != op2.to_int();
383  }
384 
385  if (is_double() && op2.is_double()) {
386  return to_double() != op2.to_double();
387  }
388 
389  if (is_double_vector() && op2.is_double_vector()) {
390  const vector<double> &v1 = to_double_vector();
391  const vector<double> &v2 = op2.to_double_vector();
392  if (v1.size() != v2.size()) return true;
393  for (int i=0; i<v1.size(); ++i)
394  if (v1[i] != v2[i]) return true;
395  return false;
396  }
397 
398  if (is_int_vector() && op2.is_int_vector()) {
399  const vector<int> &v1 = to_int_vector();
400  const vector<int> &v2 = op2.to_int_vector();
401  if (v1.size() != v2.size()) return true;
402  for (int i=0; i<v1.size(); ++i)
403  if (v1[i] != v2[i]) return true;
404  return false;
405  }
406 
407  if (is_int_vector_vector() && op2.is_int_vector_vector()) {
408  const vector< vector<int> > &v1 = to_int_vector_vector();
409  const vector< vector<int> > &v2 = op2.to_int_vector_vector();
410  if (v1.size() != v2.size()) return true;
411  for (int i=0; i<v1.size(); ++i) {
412  if (v1[i].size() != v2[i].size()) return true;
413  for (int j=0; j<v1[i].size(); ++j) {
414  if (v1[i][j] != v2[i][j]) return true;
415  }
416  }
417  return false;
418  }
419 
420  // Different types
421  return true;
422  }
