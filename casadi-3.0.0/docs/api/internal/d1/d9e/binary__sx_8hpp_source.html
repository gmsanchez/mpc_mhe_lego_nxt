68  virtual ~BinarySX() {
69  // Start destruction method if any of the dependencies has dependencies
70  for (int c1=0; c1<2; ++c1) {
71  // Get the node of the dependency and remove it from the smart pointer
72  SXNode* n1 = dep(c1).assignNoDelete(casadi_limits<SXElem>::nan);
73 
74  // Check if this was the last reference
75  if (n1->count==0) {
76 
77  // Check if binary
78  if (!n1->hasDep()) { // n1 is not binary
79 
80  delete n1; // Delete straight away
81 
82  } else { // n1 is binary
83 
84  // Stack of expressions to be deleted
85  std::stack<SXNode*> deletion_stack;
86 
87  // Add the node to the deletion stack
88  deletion_stack.push(n1);
89 
90  // Process stack
91  while (!deletion_stack.empty()) {
92 
93  // Top element
94  SXNode *t = deletion_stack.top();
95 
96  // Check if the top element has dependencies with dependencies
97  bool added_to_stack = false;
98  for (int c2=0; c2<t->ndep(); ++c2) { // for all dependencies of the dependency
99 
100  // Get the node of the dependency of the top element
101  // and remove it from the smart pointer
102  SXNode *n2 = t->dep(c2).assignNoDelete(casadi_limits<SXElem>::nan);
103 
104  // Check if this is the only reference to the element
105  if (n2->count == 0) {
106 
107  // Check if binary
108  if (!n2->hasDep()) {
109 
110  // Delete straight away if not binary
111  delete n2;
112 
113  } else {
114 
115  // Add to deletion stack
116  deletion_stack.push(n2);
117  added_to_stack = true;
118  }
119  }
120  }
121 
122  // Delete and pop from stack if nothing added to the stack
123  if (!added_to_stack) {
124  delete deletion_stack.top();
125  deletion_stack.pop();
126  }
127  } // while
128  }
129  }
130  }
131  }

138  virtual bool is_equal(const SXNode* node, int depth) const {
139  const BinarySX* n = dynamic_cast<const BinarySX*>(node);
140  if (n==0) return false;
141  if (n->op_ != op_) return false;
142  if (SXElem::is_equal(n->dep0_, dep0_, depth-1)
143  && SXElem::is_equal(n->dep1_, dep1_, depth-1)) return true;
144  if (operation_checker<CommChecker>(op_)
145  && SXElem::is_equal(n->dep1_, dep0_, depth-1)
146  && SXElem::is_equal(n->dep0_, dep1_, depth-1)) return true;
147  return false;
148  }
