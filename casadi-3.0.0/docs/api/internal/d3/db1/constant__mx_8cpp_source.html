107  ConstantMX* ConstantMX::create(const Matrix<double>& val) {
108  if (val.nnz()==0) {
109  return create(val.sparsity(), 0);
110  } else if (val.is_scalar()) {
111  return create(val.sparsity(), val.scalar());
112  } else {
113  // Check if all values are the same
114  const vector<double> vdata = val.nonzeros();
115  double v = vdata[0];
116  for (vector<double>::const_iterator i=vdata.begin(); i!=vdata.end(); ++i) {
117  if (*i!=v) {
118  // Values not all the same
119  return new ConstantDM(val);
120  }
121  }
122 
123  // All values identical if reached this point
124  return create(val.sparsity(), v);
125  }
126  }
