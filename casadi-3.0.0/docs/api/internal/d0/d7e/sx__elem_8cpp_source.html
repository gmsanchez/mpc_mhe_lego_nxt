63  SXElem::SXElem(double val) {
64  int intval = static_cast<int>(val);
65  if (val-intval == 0) { // check if integer
66  if (intval == 0) node = casadi_limits<SXElem>::zero.node;
67  else if (intval == 1) node = casadi_limits<SXElem>::one.node;
68  else if (intval == 2) node = casadi_limits<SXElem>::two.node;
69  else if (intval == -1) node = casadi_limits<SXElem>::minus_one.node;
70  else node = IntegerSX::create(intval);
71  node->count++;
72  } else {
73  if (isnan(val)) node = casadi_limits<SXElem>::nan.node;
74  else if (isinf(val)) node = val > 0 ? casadi_limits<SXElem>::inf.node :
75  casadi_limits<SXElem>::minus_inf.node;
76  else node = RealtypeSX::create(val);
77  node->count++;
78  }
79  }

187  SXElem SXElem::binary(int op, const SXElem& x, const SXElem& y) {
188  // Simplifications
189  if (GlobalOptions::simplification_on_the_fly) {
190  switch (op) {
191  case OP_ADD:
192  if (x.is_zero())
193  return y;
194  else if (y->is_zero()) // term2 is zero
195  return x;
196  else if (y.is_op(OP_NEG)) // x + (-y) -> x - y
197  return x - (-y);
198  else if (x.is_op(OP_NEG)) // (-x) + y -> y - x
199  return y - x.dep();
200  else if (x.is_op(OP_MUL) && y.is_op(OP_MUL) &&
201  x.dep(0).is_constant() && static_cast<double>(x.dep(0))==0.5 &&
202  y.dep(0).is_constant() && static_cast<double>(y.dep(0))==0.5 &&
203  is_equal(y.dep(1), x.dep(1), SXNode::eq_depth_)) // 0.5x+0.5x = x
204  return x.dep(1);
205  else if (x.is_op(OP_DIV) && y.is_op(OP_DIV) &&
206  x.dep(1).is_constant() && static_cast<double>(x.dep(1))==2 &&
207  y.dep(1).is_constant() && static_cast<double>(y.dep(1))==2 &&
208  is_equal(y.dep(0), x.dep(0), SXNode::eq_depth_)) // x/2+x/2 = x
209  return x.dep(0);
210  else if (x.is_op(OP_SUB) && is_equal(x.dep(1), y, SXNode::eq_depth_))
211  return x.dep(0);
212  else if (y.is_op(OP_SUB) && is_equal(x, y.dep(1), SXNode::eq_depth_))
213  return y.dep(0);
214  else if (x.is_op(OP_SQ) && y.is_op(OP_SQ) &&
215  ((x.dep().is_op(OP_SIN) && y.dep().is_op(OP_COS))
216  || (x.dep().is_op(OP_COS) && y.dep().is_op(OP_SIN)))
217  && is_equal(x.dep().dep(), y.dep().dep(), SXNode::eq_depth_))
218  return 1; // sin^2 + cos^2 -> 1
219  break;
220  case OP_SUB:
221  if (y->is_zero()) // term2 is zero
222  return x;
223  if (x.is_zero()) // term1 is zero
224  return -y;
225  if (is_equal(x, y, SXNode::eq_depth_)) // the terms are equal
226  return 0;
227  else if (y.is_op(OP_NEG)) // x - (-y) -> x + y
228  return x + y.dep();
229  else if (x.is_op(OP_ADD) && is_equal(x.dep(1), y, SXNode::eq_depth_))
230  return x.dep(0);
231  else if (x.is_op(OP_ADD) && is_equal(x.dep(0), y, SXNode::eq_depth_))
232  return x.dep(1);
233  else if (y.is_op(OP_ADD) && is_equal(x, y.dep(1), SXNode::eq_depth_))
234  return -y.dep(0);
235  else if (y.is_op(OP_ADD) && is_equal(x, y.dep(0), SXNode::eq_depth_))
236  return -y.dep(1);
237  else if (x.is_op(OP_NEG))
238  return -(x.dep() + y);
239  break;
240  case OP_MUL:
241  if (is_equal(y, x, SXNode::eq_depth_))
242  return sq(x);
243  else if (!x.is_constant() && y.is_constant())
244  return y * x;
245  else if (x.is_zero() || y->is_zero()) // one of the terms is zero
246  return 0;
247  else if (x.is_one()) // term1 is one
248  return y;
249  else if (y->is_one()) // term2 is one
250  return x;
251  else if (y->is_minus_one())
252  return -x;
253  else if (x.is_minus_one())
254  return -y;
255  else if (y.is_op(OP_INV))
256  return x/y.inv();
257  else if (x.is_op(OP_INV))
258  return y / x.inv();
259  else if (x.is_constant() && y.is_op(OP_MUL) && y.dep(0).is_constant() &&
260  static_cast<double>(x)*static_cast<double>(y.dep(0))==1) // 5*(0.2*x) = x
261  return y.dep(1);
262  else if (x.is_constant() && y.is_op(OP_DIV) && y.dep(1).is_constant() &&
263  static_cast<double>(x)==static_cast<double>(y.dep(1))) // 5*(x/5) = x
264  return y.dep(0);
265  else if (x.is_op(OP_DIV) && is_equal(x.dep(1), y, SXNode::eq_depth_)) // ((2/x)*x)
266  return x.dep(0);
267  else if (y.is_op(OP_DIV) &&
268  is_equal(y.dep(1), x, SXNode::eq_depth_)) // ((2/x)*x)
269  return y.dep(0);
270  else if (x.is_op(OP_NEG))
271  return -(x.dep() * y);
272  else if (y.is_op(OP_NEG))
273  return -(x * y.dep());
274  break;
275  case OP_DIV:
276  if (y->is_zero()) // term2 is zero
277  return casadi_limits<SXElem>::nan;
278  else if (x.is_zero()) // term1 is zero
279  return 0;
280  else if (y->is_one()) // term2 is one
281  return x;
282  else if (y->is_minus_one())
283  return -x;
284  else if (is_equal(x, y, SXNode::eq_depth_)) // terms are equal
285  return 1;
286  else if (x.is_doubled() && is_equal(y, 2))
287  return x.dep(0);
288  else if (x.is_op(OP_MUL) && is_equal(y, x.dep(0), SXNode::eq_depth_))
289  return x.dep(1);
290  else if (x.is_op(OP_MUL) && is_equal(y, x.dep(1), SXNode::eq_depth_))
291  return x.dep(0);
292  else if (x.is_one())
293  return y.inv();
294  else if (y.is_op(OP_INV))
295  return x*y.inv();
296  else if (x.is_doubled() && y.is_doubled())
297  return x.dep(0) / y->dep(0);
298  else if (y.is_constant() && x.is_op(OP_DIV) && x.dep(1).is_constant() &&
299  static_cast<double>(y)*static_cast<double>(x.dep(1))==1) // (x/5)/0.2
300  return x.dep(0);
301  else if (y.is_op(OP_MUL) &&
302  is_equal(y.dep(1), x, SXNode::eq_depth_)) // x/(2*x) = 1/2
303  return BinarySX::create(OP_DIV, 1, y.dep(0));
304  else if (x.is_op(OP_NEG) &&
305  is_equal(x.dep(0), y, SXNode::eq_depth_)) // (-x)/x = -1
306  return -1;
307  else if (y.is_op(OP_NEG) &&
308  is_equal(y.dep(0), x, SXNode::eq_depth_)) // x/(-x) = 1
309  return -1;
310  else if (y.is_op(OP_NEG) && x.is_op(OP_NEG) &&
311  is_equal(x.dep(0), y.dep(0), SXNode::eq_depth_)) // (-x)/(-x) = 1
312  return 1;
313  else if (x.is_op(OP_DIV) && is_equal(y, x.dep(0), SXNode::eq_depth_))
314  return x.dep(1).inv();
315  else if (x.is_op(OP_NEG))
316  return -(x.dep() / y);
317  else if (y.is_op(OP_NEG))
318  return -(x / y.dep());
319  break;
320  case OP_POW:
321  if (y->is_constant()) {
322  if (y->is_integer()) {
323  int nn = y->to_int();
324  if (nn == 0) {
325  return 1;
326  } else if (nn>100 || nn<-100) { // maximum depth
327  return binary(OP_CONSTPOW, x, nn);
328  } else if (nn<0) { // negative power
329  return 1/pow(x, -nn);
330  } else if (nn%2 == 1) { // odd power
331  return x*pow(x, nn-1);
332  } else { // even power
333  SXElem rt = pow(x, nn/2);
334  return rt*rt;
335  }
336  } else if (y->to_double()==0.5) {
337  return sqrt(x);
338  } else {
339  return binary(OP_CONSTPOW, x, y);
340  }
341  }
342  break;
343  case OP_LE:
344  if ((y-x).isNonNegative())
345  return 1;
346  break;
347  case OP_LT:
348  if (((x)-y).isNonNegative())
349  return 0;
350  break;
351  case OP_EQ:
352  if (is_equal(x, y))
353  return 1;
354  break;
355  case OP_NE:
356  if (is_equal(x, y))
357  return 0;
358  break;
359  case OP_SINH:
360  case OP_TANH:
361  case OP_ATANH:
362  case OP_ACOSH:
363  case OP_ASINH:
364  if (x.is_zero())
365  return 0;
366  break;
367  case OP_COSH:
368  if (x.is_zero())
369  return 1;
370  break;
371  case OP_SQRT:
372  if (x.is_op(OP_SQ))
373  return fabs(x.dep());
374  break;
375  case OP_SQ:
376  if (x.is_op(OP_SQRT))
377  return x.dep();
378  else if (x.is_op(OP_NEG))
379  return sq(x.dep());
380  break;
381  case OP_FABS:
382  if (x.is_op(OP_FABS) || x.is_op(OP_SQ))
383  return x;
384  break;
385  case OP_NOT:
386  if (x.is_op(OP_NOT))
387  return x.dep();
388  break;
389  case OP_IF_ELSE_ZERO:
390  if (y->is_zero()) {
391  return y;
392  } else if (x.is_constant()) {
393  if (static_cast<double>(x)!=0) {
394  return y;
395  } else {
396  return 0;
397  }
398  }
399  }
400  }
401  return BinarySX::create(Operation(op), x, y);
402  }
