36  Function external(const string& name, Library li,
37  const Dict& opts) {
38  Function ret;
39  if (li.has(name + "_simple")) {

42  } else {

187  void External::init(const Dict& opts) {
188  // Call the initialization method of the base class
189  FunctionInternal::init(opts);
190 
191  // Reference counting?
192  has_refcount_ = li_.has(name_ + "_incref");
193  casadi_assert_message(has_refcount_==li_.has(name_ + "_decref"),
194  "External functions must provide functions for both increasing "
195  "and decreasing the reference count, or neither.");
196 
197  // Allocate work vectors
198  int sz_arg=0, sz_res=0, sz_iw=0, sz_w=0;
199  if (work_) {
200  int flag = work_(&sz_arg, &sz_res, &sz_iw, &sz_w);
201  casadi_assert_message(flag==0, "External: \"work\" failed");
202  } else if (li_.meta().has(name_ + "_WORK")) {
203  vector<int> v = li_.meta().to_vector<int>(name_ + "_WORK");
204  casadi_assert(v.size()==4);
205  sz_arg = v[0];
206  sz_res = v[1];
207  sz_iw = v[2];
208  sz_w = v[3];
209  }
210  alloc_arg(sz_arg);
211  alloc_res(sz_res);
212  alloc_iw(sz_iw);
213  alloc_w(sz_w);
214  }
