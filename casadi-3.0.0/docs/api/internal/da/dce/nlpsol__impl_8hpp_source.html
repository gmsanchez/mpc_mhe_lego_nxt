224  template<typename XType>
225  Oracle* Nlpsol::map2problem(const std::map<std::string, XType>& d) {
226  std::vector<XType> nl_in(NL_NUM_IN), nl_out(NL_NUM_OUT);
227  for (auto&& i : d) {
228  if (i.first=="x") {
229  nl_in[NL_X]=i.second;
230  } else if (i.first=="p") {
231  nl_in[NL_P]=i.second;
232  } else if (i.first=="f") {
233  nl_out[NL_F]=i.second;
234  } else if (i.first=="g") {
235  nl_out[NL_G]=i.second;
236  } else {
237  casadi_error("No such field: " + i.first);
238  }
239  }
240  return Oracle::construct(nl_in, nl_out, NL_INPUTS, NL_OUTPUTS);
241  }
