66  template<bool Add>
67  void SetNonzeros<Add>::eval_mx(const std::vector<MX>& arg, std::vector<MX>& res) {
68  // Get all the nonzeros
69  vector<int> nz = all();
70 
71  // Output sparsity
72  const Sparsity &osp = sparsity();
73  const int* orow = osp.row();
74  vector<int> ocol = osp.get_col();
75 
76  // Input sparsity (first input same as output)
77  const Sparsity &isp = dep(1).sparsity();
78  vector<int> icol = isp.get_col();
79 
80  // We next need to resort the assignment vector by outputs instead of inputs
81  // Start by counting the number of output nonzeros corresponding to each input nonzero
82  vector<int> onz_count(osp.nnz()+2, 0);
83  for (vector<int>::const_iterator it=nz.begin(); it!=nz.end(); ++it) {
84  onz_count[*it+2]++;
85  }
86 
87  // Cumsum to get index offset for output nonzero
88  for (int i=0; i<onz_count.size()-1; ++i) {
89  onz_count[i+1] += onz_count[i];
90  }
91 
92  // Get the order of assignments
93  vector<int> nz_order(nz.size());
94  for (int k=0; k<nz.size(); ++k) {
95  // Save the new index
96  nz_order[onz_count[1+nz[k]]++] = k;
97  }
98 
99  // Find out which elements are being set
100  vector<int>& with_duplicates = onz_count; // Reuse memory
101  onz_count.resize(nz.size());
102  for (int k=0; k<nz.size(); ++k) {
103  // Get output nonzero
104  int onz_k = nz[nz_order[k]];
105 
106  // Get element (note: may contain duplicates)
107  if (onz_k>=0) {
108  with_duplicates[k] = ocol[onz_k]*osp.size1() + orow[onz_k];
109  } else {
110  with_duplicates[k] = -1;
111  }
112  }
113 
114  // Get all output elements (this time without duplicates)
115  vector<int> el_output;
116  osp.find(el_output);
117 
118  // Sparsity pattern being formed and corresponding nonzero mapping
119  vector<int> r_colind, r_row, r_nz, r_ind;
120 
121  // Get references to arguments and results
122  res[0] = arg[0];
123 
124  // Entries in res with elements zero'ed out
125  if (!Add) {
126 
127  // Get the nz locations in res corresponding to the output sparsity pattern
128  r_nz.resize(with_duplicates.size());
129  copy(with_duplicates.begin(), with_duplicates.end(), r_nz.begin());
130  res[0].sparsity().get_nz(r_nz);
131 
132  // Zero out the corresponding entries
133  res[0] = MX::zeros(isp)->getSetNonzeros(res[0], r_nz);
134  }
135 
136  // Get the nz locations of the elements in arg corresponding to the argument sparsity pattern
137  arg[1].sparsity().find(r_nz);
138  isp.get_nz(r_nz);
139 
140  // Filter out ignored entries and check if there is anything to add at all
141  bool elements_to_add = false;
142  for (vector<int>::iterator k=r_nz.begin(); k!=r_nz.end(); ++k) {
143  if (*k>=0) {
144  if (nz[*k]>=0) {
145  elements_to_add = true;
146  } else {
147  *k = -1;
148  }
149  }
150  }
151 
152  // Quick continue of no elements to set/add
153  if (!elements_to_add) return;
154 
155  // Get the nz locations in the argument corresponding to the inputs
156  r_ind.resize(el_output.size());
157  copy(el_output.begin(), el_output.end(), r_ind.begin());
158  res[0].sparsity().get_nz(r_ind);
159 
160  // Enlarge the sparsity pattern of the arguments if not all assignments fit
161  for (vector<int>::iterator k=r_nz.begin(); k!=r_nz.end(); ++k) {
162  if (*k>=0 && nz[*k]>=0 && r_ind[nz[*k]]<0) {
163 
164  // Create a new pattern which includes both the the previous seed
165  // and the addition/assignment
166  Sparsity sp = res[0].sparsity().unite(osp);
167  res[0] = res[0]->getProject(sp);
168 
169  // Recalculate the nz locations in the arguments corresponding to the inputs
170  copy(el_output.begin(), el_output.end(), r_ind.begin());
171  res[0].sparsity().get_nz(r_ind);
172 
173  break;
174  }
175  }
176 
177  // Have r_nz point to locations in the result instead of the output
178  for (vector<int>::iterator k=r_nz.begin(); k!=r_nz.end(); ++k) {
179  if (*k>=0) {
180  *k = r_ind[nz[*k]];
181  }
182  }
183 
184  // Add to the element to the sensitivity, if any
185  res[0] = arg[1]->getAddNonzeros(res[0], r_nz);
186  }

321  template<bool Add>
322  void SetNonzeros<Add>::evalAdj(const std::vector<std::vector<MX> >& aseed,
323  std::vector<std::vector<MX> >& asens) {
324  // Get all the nonzeros
325  vector<int> nz = all();
326 
327  // Number of derivative directions
328  int nadj = aseed.size();
329 
330  // Output sparsity
331  const Sparsity &osp = sparsity();
332  const int* orow = osp.row();
333  vector<int> ocol = osp.get_col();
334 
335  // Input sparsity (first input same as output)
336  const Sparsity &isp = dep(1).sparsity();
337  const int* irow = isp.row();
338  vector<int> icol = isp.get_col();
339 
340  // We next need to resort the assignment vector by outputs instead of inputs
341  // Start by counting the number of output nonzeros corresponding to each input nonzero
342  vector<int> onz_count(osp.nnz()+2, 0);
343  for (vector<int>::const_iterator it=nz.begin(); it!=nz.end(); ++it) {
344  onz_count[*it+2]++;
345  }
346 
347  // Cumsum to get index offset for output nonzero
348  for (int i=0; i<onz_count.size()-1; ++i) {
349  onz_count[i+1] += onz_count[i];
350  }
351 
352  // Get the order of assignments
353  vector<int> nz_order(nz.size());
354  for (int k=0; k<nz.size(); ++k) {
355  // Save the new index
356  nz_order[onz_count[1+nz[k]]++] = k;
357  }
358 
359  // Find out which elements are being set
360  vector<int>& with_duplicates = onz_count; // Reuse memory
361  onz_count.resize(nz.size());
362  for (int k=0; k<nz.size(); ++k) {
363  // Get output nonzero
364  int onz_k = nz[nz_order[k]];
365 
366  // Get element (note: may contain duplicates)
367  if (onz_k>=0) {
368  with_duplicates[k] = ocol[onz_k]*osp.size1() + orow[onz_k];
369  } else {
370  with_duplicates[k] = -1;
371  }
372  }
373 
374  // Get all output elements (this time without duplicates)
375  vector<int> el_output;
376  osp.find(el_output);
377 
378  // Sparsity pattern being formed and corresponding nonzero mapping
379  vector<int> r_colind, r_row, r_nz, r_ind;
380 
381  for (int d=0; d<nadj; ++d) {
382  // Get the matching nonzeros
383  r_ind.resize(el_output.size());
384  copy(el_output.begin(), el_output.end(), r_ind.begin());
385  aseed[d][0].sparsity().get_nz(r_ind);
386 
387  // Sparsity pattern for the result
388  r_colind.resize(isp.size2()+1); // Col count
389  fill(r_colind.begin(), r_colind.end(), 0);
390  r_row.clear();
391 
392  // Perform the assignments
393  r_nz.clear();
394  for (int k=0; k<nz.size(); ++k) {
395 
396  // Get the corresponding nonzero for the input
397  int el = nz[k];
398 
399  // Skip if zero assignment
400  if (el==-1) continue;
401 
402  // Get the corresponding nonzero in the argument
403  int el_arg = r_ind[el];
404 
405  // Skip if no argument
406  if (el_arg==-1) continue;
407 
408  // Save the assignment
409  r_nz.push_back(el_arg);
410 
411  // Get the corresponding element
412  int i=icol[k], j=irow[k];
413 
414  // Add to sparsity pattern
415  r_row.push_back(j);
416  r_colind[1+i]++;
417  }
418 
419  // col count -> col offset
420  for (int i=1; i<r_colind.size(); ++i) r_colind[i] += r_colind[i-1];
421 
422  // If anything to set/add
423  if (!r_nz.empty()) {
424  // Create a sparsity pattern from vectors
425  Sparsity f_sp(isp.size1(), isp.size2(), r_colind, r_row);
426  asens[d][1] += aseed[d][0]->getGetNonzeros(f_sp, r_nz);
427  if (!Add) {
428  asens[d][0] += MX::zeros(f_sp)->getSetNonzeros(aseed[d][0], r_nz);
429  } else {
430  asens[d][0] += aseed[d][0];
431  }
432  } else {
433  asens[d][0] += aseed[d][0];
434  }
435  }
436  }
