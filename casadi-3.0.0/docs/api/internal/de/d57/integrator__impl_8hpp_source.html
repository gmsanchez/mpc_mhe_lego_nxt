240  template<typename XType>
241  Oracle* Integrator::map2problem(const std::map<std::string, XType>& d) {
242  std::vector<XType> de_in(DE_NUM_IN), de_out(DE_NUM_OUT);
243  for (auto&& i : d) {
244  if (i.first=="t") {
245  de_in[DE_T]=i.second;
246  } else if (i.first=="x") {
247  de_in[DE_X]=i.second;
248  } else if (i.first=="z") {
249  de_in[DE_Z]=i.second;
250  } else if (i.first=="p") {
251  de_in[DE_P]=i.second;
252  } else if (i.first=="rx") {
253  de_in[DE_RX]=i.second;
254  } else if (i.first=="rz") {
255  de_in[DE_RZ]=i.second;
256  } else if (i.first=="rp") {
257  de_in[DE_RP]=i.second;
258  } else if (i.first=="ode") {
259  de_out[DE_ODE]=i.second;
260  } else if (i.first=="alg") {
261  de_out[DE_ALG]=i.second;
262  } else if (i.first=="quad") {
263  de_out[DE_QUAD]=i.second;
264  } else if (i.first=="rode") {
265  de_out[DE_RODE]=i.second;
266  } else if (i.first=="ralg") {
267  de_out[DE_RALG]=i.second;
268  } else if (i.first=="rquad") {
269  de_out[DE_RQUAD]=i.second;
270  } else {
271  casadi_error("No such field: " + i.first);
272  }
273  }
274  return Oracle::construct(de_in, de_out, DE_INPUTS, DE_OUTPUTS);
275  }

277  template<typename XType>
278  Oracle* Integrator::fun2problem(Function f, Function g) {
279  std::vector<XType> dae_in(DE_NUM_IN), dae_out(DE_NUM_OUT);
280  std::vector<XType> v = XType::get_input(f), vf=v, vg=v;
281  dae_in[DE_T] = v[DAE_T];
282  dae_in[DE_X] = v[DAE_X];
283  dae_in[DE_Z] = v[DAE_Z];
284  dae_in[DE_P] = v[DAE_P];
285  v = f(v);
286  dae_out[DE_ODE] = v[DAE_ODE];
287  dae_out[DE_ALG] = v[DAE_ALG];
288  dae_out[DE_QUAD] = v[DAE_QUAD];
289  if (!g.is_null()) {
290  v = XType::get_input(g);
291  dae_in[DE_RX] = v[RDAE_RX];
292  dae_in[DE_RZ] = v[RDAE_RZ];
293  dae_in[DE_RP] = v[RDAE_RP];
294  vg[DAE_T] = v[RDAE_T];
295  vg[DAE_X] = v[RDAE_X];
296  vg[DAE_Z] = v[RDAE_Z];
297  vg[DAE_P] = v[RDAE_P];
298  v = substitute(g(v), vg, vf);
299  dae_out[DE_RODE] = v[RDAE_ODE];
300  dae_out[DE_RALG] = v[RDAE_ALG];
301  dae_out[DE_RQUAD] = v[RDAE_QUAD];
302  }
303  return Oracle::construct(dae_in, dae_out, DE_INPUTS, DE_OUTPUTS);
304  }
