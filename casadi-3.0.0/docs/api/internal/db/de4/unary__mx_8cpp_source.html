69  void UnaryMX::evalFwd(const std::vector<std::vector<MX> >& fseed,
70  std::vector<std::vector<MX> >& fsens) {
71  // Get partial derivatives
72  MX pd[2];
73  MX dummy; // Function value, dummy second argument
74  casadi_math<MX>::der(op_, dep(), dummy, shared_from_this<MX>(), pd);
75 
76  // Propagate forward seeds
77  for (int d=0; d<fsens.size(); ++d) {
78  fsens[d][0] = pd[0]*fseed[d][0];
79  }

103  void UnaryMX::generate(CodeGenerator& g, const std::string& mem,
104  const std::vector<int>& arg, const std::vector<int>& res) const {
105  string r, x;
106  g.body << " ";
107  if (nnz()==1) {
108  // Scalar assignment
109  r = g.workel(res[0]);
110  x = g.workel(arg[0]);
111  } else {
112  // Vector assignment
113  g.body << "for (i=0, rr=" << g.work(res[0], nnz()) << ", cs=" << g.work(arg[0], nnz())
114  << "; i<" << sparsity().nnz() << "; ++i) ";
115  r = "*rr++";
116  x = "*cs++";
117  }
118 
119  // Output the operation
120  g.body << r << " = ";
121  casadi_math<double>::printPre(op_, g.body);
122  g.body << " " << x << " ";
123  casadi_math<double>::printPost(op_, g.body);
124  g.body << ";" << endl;

127  MX UnaryMX::getUnary(int op) const {
128  if (!GlobalOptions::simplification_on_the_fly) return MXNode::getUnary(op);
129 
130  switch (op_) {
131  case OP_NEG:
132  if (op==OP_NEG) return dep();
133  else if (op==OP_SQ) return dep()->getUnary(OP_SQ);
134  else if (op==OP_FABS) return dep()->getUnary(OP_FABS);
135  else if (op==OP_COS) return dep()->getUnary(OP_COS);
136  break;
137  case OP_SQRT:
138  if (op==OP_SQ) return dep();
139  else if (op==OP_FABS) return shared_from_this<MX>();
140  break;
141  case OP_SQ:
142  if (op==OP_SQRT) return dep()->getUnary(OP_FABS);
143  else if (op==OP_FABS) return shared_from_this<MX>();
144  break;
145  case OP_EXP:
146  if (op==OP_LOG) return dep();
147  else if (op==OP_FABS) return shared_from_this<MX>();
148  break;
149  case OP_LOG:
150  if (op==OP_EXP) return dep();
151  break;
152  case OP_FABS:
153  if (op==OP_FABS) return shared_from_this<MX>();
154  else if (op==OP_SQ) return dep()->getUnary(OP_SQ);
155  else if (op==OP_COS) return dep()->getUnary(OP_COS);
156  break;
157  case OP_INV:
158  if (op==OP_INV) return dep();
159  break;
160  default: break; // no rule
161  }
162 
163  // Fallback to default implementation
164  return MXNode::getUnary(op);
165  }

167  MX UnaryMX::getBinary(int op, const MX& y, bool scX, bool scY) const {
168  switch (op_) {
169  case OP_NEG:
170  if (op==OP_ADD) return y->getBinary(OP_SUB, dep(), scY, scX);
171  else if (op==OP_MUL) return -dep()->getBinary(OP_MUL, y, scX, scY);
172  else if (op==OP_DIV) return -dep()->getBinary(OP_DIV, y, scX, scY);
173  break;
174  case OP_TWICE:
175  if (op==OP_SUB && MX::is_equal(y, dep(), maxDepth())) return dep();
176  break;
177  case OP_SQ:
178  if (op==OP_ADD && y.op()==OP_SQ) /*sum of squares:*/
179  if ((dep().op()==OP_SIN && y->dep().op()==OP_COS) ||
180  (dep().op()==OP_COS && y->dep()->op()==OP_SIN)) /* sin^2(x)+sin^2(y) */
181  if (MX::is_equal(dep()->dep(), y->dep()->dep(), maxDepth())) /*sin^2(x) + cos^2(x) */
182  return MX::ones(y.sparsity());
183  break;
184  default: break; // no rule
185  }
186 
187  // Fallback to default implementation
188  return MXNode::getBinary(op, y, scX, scY);
189  }
