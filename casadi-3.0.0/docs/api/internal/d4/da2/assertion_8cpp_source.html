85  void Assertion::spAdj(bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem) {
86  bvec_t *a = arg[0];
87  bvec_t *r = res[0];
88  int n = nnz();
89  if (a != r) {
90  for (int i=0; i<n; ++i) {
91  *a++ |= *r;
92  *r++ = 0;
93  }
94  }
