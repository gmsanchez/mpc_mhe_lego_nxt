761  template<typename DataType>
762  void Sparsity::set(DataType* data, const DataType* val_data, const Sparsity& val_sp) const {
763  // Get dimensions of this
764  const int sz = nnz();
765  const int sz1 = size1();
766  const int sz2 = size2();
767 
768  // Get dimensions of assigning matrix
769  const int val_sz = val_sp.nnz();
770  const int val_sz1 = val_sp.size1();
771  const int val_sz2 = val_sp.size2();
772  const int val_nel = val_sz1*val_sz2;
773 
774  // Check if sparsity matches
775  if (val_sp==*this) {
776  std::copy(val_data, val_data+sz, data);
777  } else if (this->is_empty()) {
778  // Quick return
779  return;
780  } else if (val_sp.is_empty()) {
781  // Quick return
782  return;
783  } else if (val_nel==1) { // if scalar
784  std::fill(data, data+sz, val_sz==0 ? DataType(0) : val_data[0]);
785  } else if (sz2==val_sz2 && sz1==val_sz1) {
786  // Matching dimensions
787  // Sparsity
788  const int* c = row();
789  const int* rind = colind();
790  const int* v_c = val_sp.row();
791  const int* v_rind = val_sp.colind();
792 
793  // For all columns
794  for (int i=0; i<sz2; ++i) {
795 
796  // Nonzero of the assigning matrix
797  int v_el = v_rind[i];
798 
799  // First nonzero of the following column
800  int v_el_end = v_rind[i+1];
801 
802  // Next row of the assigning matrix
803  int v_j = v_el<v_el_end ? v_c[v_el] : sz1;
804 
805  // Assign all nonzeros
806  for (int el=rind[i]; el!=rind[i+1]; ++el) {
807 
808  // Get row
809  int j=c[el];
810 
811  // Forward the assigning nonzero
812  while (v_j<j) {
813  v_el++;
814  v_j = v_el<v_el_end ? v_c[v_el] : sz1;
815  }
816 
817  // Assign nonzero
818  if (v_j==j) {
819  data[el] = val_data[v_el++];
820  v_j = v_el<v_el_end ? v_c[v_el] : sz1;
821  } else {
822  data[el] = 0;
823  }
824  }
825  }
826  } else if (sz1==val_sz2 && sz2==val_sz1 && sz2 == 1) {
827  // Assign transposed (this is column)
828  const int* v_cind = val_sp.colind();
829  const int* r = row();
830  for (int el=0; el<sz; ++el) {
831  int rr=r[el];
832  data[el] = v_cind[rr]==v_cind[rr+1] ? 0 : val_data[v_cind[rr]];
833  }
834  } else if (sz1==val_sz2 && sz2==val_sz1 && sz1 == 1) {
835  // Assign transposed (this is row)
836  for (int el=0; el<sz; ++el) data[el] = 0;
837  const int* cind = colind();
838  const int* v_r = val_sp.row();
839  for (int el=0; el<val_sz; ++el) {
840  int rr=v_r[el];
841  if (cind[rr]!=cind[rr+1]) {
842  data[cind[rr]] = val_data[el];
843  }
844  }
845  } else {
846  // Make sure that dimension matches
847  casadi_error("Sparsity::set<DataType>: shape mismatch. lhs is matrix of shape "
848  << dim() << ", while rhs is shape " << val_sp.dim() << ".");
849  }
850  }

913

916

919  v_el++;
920  v_j = v_el<v_el_end ? v_c[v_el] : sz1;
921  }
922 
923  // Assign nonzero
924  if (v_j==j) {
925  data[el] += val_data[v_el++];

927  }
928  }
929  }
