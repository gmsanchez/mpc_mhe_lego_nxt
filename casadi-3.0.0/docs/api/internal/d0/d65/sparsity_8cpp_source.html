183  int Sparsity::add_nz(int rr, int cc) {
184  // If negative index, count from the back
185  if (rr<0) rr += size1();
186  if (cc<0) cc += size2();
187 
188  // Check consistency
189  casadi_assert_message(rr>=0 && rr<size1(), "Row index out of bounds");
190  casadi_assert_message(cc>=0 && cc<size2(), "Column index out of bounds");
191 
192  // Quick return if matrix is dense
193  if (is_dense()) return rr+cc*size1();
194 
195  // Get sparsity pattern
196  int size1=this->size1(), size2=this->size2(), nnz=this->nnz();
197  const int *colind = this->colind(), *row = this->row();
198 
199  // Quick return if we are adding an element to the end
200  if (colind[cc]==nnz || (colind[cc+1]==nnz && row[nnz-1]<rr)) {
201  std::vector<int> rowv(nnz+1);
202  copy(row, row+nnz, rowv.begin());
203  rowv[nnz] = rr;
204  std::vector<int> colindv(colind, colind+size2+1);
205  for (int c=cc; c<size2; ++c) colindv[c+1]++;
206  assign_cached(size1, size2, colindv, rowv);
207  return rowv.size()-1;
208  }
209 
210  // go to the place where the element should be
211  int ind;
212  for (ind=colind[cc]; ind<colind[cc+1]; ++ind) { // better: loop from the back to the front
213  if (row[ind] == rr) {
214  return ind; // element exists
215  } else if (row[ind] > rr) {
216  break; // break at the place where the element should be added
217  }
218  }
219 
220  // insert the element
221  std::vector<int> rowv = get_row(), colindv = get_colind();
222  rowv.insert(rowv.begin()+ind, rr);
223  for (int c=cc+1; c<size2+1; ++c) colindv[c]++;
224 
225  // Return the location of the new element
226  assign_cached(size1, size2, colindv, rowv);
227  return ind;
228  }

419  void Sparsity::append(const Sparsity& sp) {
420  if (sp.size1()==0 && sp.size2()==0) {
421  // Appending pattern is empty
422  return;
423  } else if (size1()==0 && size2()==0) {
424  // This is empty
425  *this = sp;
426  } else {
427  casadi_assert_message(size2()==sp.size2(),
428  "Sparsity::append: Dimension mismatch. "
429  "You attempt to append a shape " << sp.dim()
430  << " to a shape " << dim()
431  << ". The number of columns must match.");
432  if (sp.size1()==0) {
433  // No rows to add
434  return;
435  } else if (size1()==0) {
436  // No rows before
437  *this = sp;
438  } else if (is_column()) {
439  // Append to vector (inefficient)
440  *this = (*this)->_appendVector(*sp);
441  } else {
442  // Append to matrix (inefficient)
443  *this = vertcat({*this, sp});
444  }
445  }

640  void Sparsity::assign_cached(int nrow, int ncol, const int* colind, const int* row) {
641  // Scalars and empty patterns are handled separately
642  if (ncol==0 && nrow==0) {
643  // If empty
644  *this = getEmpty();
645  return;
646  } else if (ncol==1 && nrow==1) {
647  if (colind[ncol]==0) {
648  // If sparse scalar
649  *this = getScalarSparse();
650  return;
651  } else {
652  // If dense scalar
653  *this = getScalar();
654  return;
655  }
656  }
657 
658  // Hash the pattern
659  std::size_t h = hash_sparsity(nrow, ncol, colind, row);
660 
661  // Get a reference to the cache
662  CachingMap& cache = getCache();
663 
664  // Record the current number of buckets (for garbage collection below)
665  int bucket_count_before = cache.bucket_count();
666 
667  // WORKAROUND, functions do not appear to work when bucket_count==0
668  if (bucket_count_before>0) {
669 
670  // Find the range of patterns equal to the key (normally only zero or one)
671  pair<CachingMap::iterator, CachingMap::iterator> eq = cache.equal_range(h);
672 
673  // Loop over maching patterns
674  for (CachingMap::iterator i=eq.first; i!=eq.second; ++i) {
675 
676  // Get a weak reference to the cached sparsity pattern
677  WeakRef& wref = i->second;
678 
679  // Check if the pattern still exists
680  if (wref.alive()) {
681 
682  // Get an owning reference to the cached pattern
683  Sparsity ref = shared_cast<Sparsity>(wref.shared());
684 
685  // Check if the pattern matches
686  if (ref.is_equal(nrow, ncol, colind, row)) {
687 
688  // Found match!
689  assignNode(ref.get());
690  return;
691 
692  } else { // There is a hash rowision (unlikely, but possible)
693  // Leave the pattern alone, continue to the next matching pattern
694  continue;
695  }
696  } else {
697 
698  // Check if one of the other cache entries indeed has a matching sparsity
699  CachingMap::iterator j=i;
700  j++; // Start at the next matching key
701  for (; j!=eq.second; ++j) {
702  if (j->second.alive()) {
703 
704  // Recover cached sparsity
705  Sparsity ref = shared_cast<Sparsity>(j->second.shared());
706 
707  // Match found if sparsity matches
708  if (ref.is_equal(nrow, ncol, colind, row)) {

710  return;
711  }
712  }
713  }
714 
715  // The cached entry has been deleted, create a new one
716  assignNode(new SparsityInternal(nrow, ncol, colind, row));
717 
718  // Cache this pattern
719  wref = *this;
720 
721  // Return
722  return;
723  }
724  }
725  }
726 
727  // No matching sparsity pattern could be found, create a new one
728  assignNode(new SparsityInternal(nrow, ncol, colind, row));
729 
730  // Cache this pattern
731  cache.insert(std::pair<std::size_t, WeakRef>(h, *this));
732

734  int bucket_count_after = cache.bucket_count();
735 
736  // We we increased the number of buckets, take time to garbage-collect deleted references
737  if (bucket_count_before!=bucket_count_after) {
738  CachingMap::const_iterator i=cache.begin();
739  while (i!=cache.end()) {
740  if (!i->second.alive()) {
741  i = cache.erase(i);
742  } else {
743  i++;
744  }
745  }
746  }

894  Sparsity Sparsity::triplet(int nrow, int ncol, const std::vector<int>& row,
895  const std::vector<int>& col, std::vector<int>& mapping,
896  bool invert_mapping) {
897  // Assert dimensions
898  casadi_assert_message(col.size()==row.size(), "inconsistent lengths");
899 
900  // Create the return sparsity pattern and access vectors
901  std::vector<int> r_colind(ncol+1, 0);
902  std::vector<int> r_row;
903  r_row.reserve(row.size());
904 
905  // Consistency check and check if elements are already perfectly ordered with no duplicates
906  int last_col=-1, last_row=-1;
907  bool perfectly_ordered=true;
908  for (int k=0; k<col.size(); ++k) {
909  // Consistency check
910  casadi_assert_message(col[k]>=0 && col[k]<ncol, "Col index out of bounds");
911  casadi_assert_message(row[k]>=0 && row[k]<nrow, "Row index out of bounds");
912 
913  // Check if ordering is already perfect
914  perfectly_ordered = perfectly_ordered && (col[k]<last_col ||
915  (col[k]==last_col && row[k]<=last_row));
916  last_col = col[k];
917  last_row = row[k];
918  }
919 
920  // Quick return if perfectly ordered
921  if (perfectly_ordered) {
922  // Save rows
923  r_row.resize(row.size());
924  copy(row.begin(), row.end(), r_row.begin());
925 
926  // Find offset index
927  int el=0;
928  for (int i=0; i<ncol; ++i) {
929  while (el<col.size() && col[el]==i) el++;
930  r_colind[i+1] = el;
931  }
932 
933  // Identity mapping
934  mapping.resize(row.size());
935  for (int k=0; k<row.size(); ++k)
936  mapping[k] = k;
937 
938  // Quick return
939  return Sparsity(nrow, ncol, r_colind, r_row);
940  }
941 
942  // Reuse data
943  std::vector<int>& mapping1 = invert_mapping ? r_row : mapping;
944  std::vector<int>& mapping2 = invert_mapping ? mapping : r_row;
945 
946  // Make sure that enough memory is allocated to use as a work vector
947  mapping1.reserve(std::max(nrow+1, static_cast<int>(col.size())));
948 
949  // Number of elements in each row
950  std::vector<int>& rowcount = mapping1; // reuse memory
951  rowcount.resize(nrow+1);
952  fill(rowcount.begin(), rowcount.end(), 0);
953  for (std::vector<int>::const_iterator it=row.begin(); it!=row.end(); ++it) {
954  rowcount[*it+1]++;
955  }
956 
957  // Cumsum to get index offset for each row
958  for (int i=0; i<nrow; ++i) {
959  rowcount[i+1] += rowcount[i];
960  }
961 
962  // New row for each old row
963  mapping2.resize(row.size());
964  for (int k=0; k<row.size(); ++k) {
965  mapping2[rowcount[row[k]]++] = k;
966  }
967 
968  // Number of elements in each col
969  std::vector<int>& colcount = r_colind; // reuse memory, r_colind is already the right size
970  // and is filled with zeros
971  for (std::vector<int>::const_iterator it=mapping2.begin(); it!=mapping2.end(); ++it) {
972  colcount[col[*it]+1]++;
973  }
974 
975  // Cumsum to get index offset for each col
976  for (int i=0; i<ncol; ++i) {
977  colcount[i+1] += colcount[i];
978  }
979 
980  // New col for each old col
981  mapping1.resize(col.size());
982  for (std::vector<int>::const_iterator it=mapping2.begin(); it!=mapping2.end(); ++it) {
983  mapping1[colcount[col[*it]]++] = *it;
984  }
985 
986  // Current element in the return matrix
987  int r_el = 0;
988  r_row.resize(col.size());
989 
990  // Current nonzero
991  std::vector<int>::const_iterator it=mapping1.begin();
992 
993  // Loop over columns
994  r_colind[0] = 0;
995  for (int i=0; i<ncol; ++i) {
996 
997  // Previous row (to detect duplicates)
998  int j_prev = -1;
999 
1000  // Loop over nonzero elements of the col
1001  while (it!=mapping1.end() && col[*it]==i) {
1002 
1003  // Get the element
1004  int el = *it;
1005  it++;
1006 
1007  // Get the row
1008  int j = row[el];
1009 
1010  // If not a duplicate, save to return matrix
1011  if (j!=j_prev)
1012  r_row[r_el++] = j;
1013 
1014  if (invert_mapping) {
1015  // Save to the inverse mapping
1016  mapping2[el] = r_el-1;
1017  } else {
1018  // If not a duplicate, save to the mapping vector
1019  if (j!=j_prev)
1020  mapping1[r_el-1] = el;
1021  }
1022 
1023  // Save row
1024  j_prev = j;
1025  }
1026 
1027  // Update col offset
1028  r_colind[i+1] = r_el;
1029  }
1030 
1031  // Resize the row vector
1032  r_row.resize(r_el);
1033 
1034  // Resize mapping matrix
1035  if (!invert_mapping) {
1036  mapping1.resize(r_el);
1037  }
1038 
1039  return Sparsity(nrow, ncol, r_colind, r_row);
1040  }

1358  std::fill(mask, mask+n_col, -1);

1369  mask[k] = i;
1370  row_nnz++;

1372  }
1373  }

1375  nnz = next_nnz;

1401  length++;
1402  }
1403  }
1404  }

1408  }
