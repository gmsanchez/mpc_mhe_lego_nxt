71  void CvodesInterface::init(const Dict& opts) {
72  log("CvodesInterface::init", "begin");
73 
74  // Initialize the base classes
75  SundialsInterface::init(opts);
76 
77  // Default options
78  string linear_multistep_method = "bdf";
79  string nonlinear_solver_iteration = "newton";
80 
81  // Read options
82  for (auto&& op : opts) {
83  if (op.first=="linear_multistep_method") {
84  linear_multistep_method = op.second.to_string();
85  } else if (op.first=="nonlinear_solver_iteration") {
86  nonlinear_solver_iteration = op.second.to_string();
87  }
88  }
89 
90  // Algebraic variables not supported
91  casadi_assert_message(nz_==0 && nrz_==0,
92  "CVODES does not support algebraic variables");
93 
94  // Read options
95  monitor_rhsB_ = monitored("resB");
96  monitor_rhs_ = monitored("res");
97  monitor_rhsQB_ = monitored("resQB");
98 
99  if (linear_multistep_method=="adams") {
100  lmm_ = CV_ADAMS;
101  } else if (linear_multistep_method=="bdf") {
102  lmm_ = CV_BDF;
103  } else {
104  casadi_error("Unknown linear multistep method: " + linear_multistep_method);
105  }
106 
107  if (nonlinear_solver_iteration=="newton") {
108  iter_ = CV_NEWTON;
109  } else if (nonlinear_solver_iteration=="functional") {
110  iter_ = CV_FUNCTIONAL;
111  } else {
112  casadi_error("Unknown nonlinear solver iteration: " + nonlinear_solver_iteration);
113  }
114 
115  // Attach functions for jacobian information
116  if (exact_jacobian_) {
117  switch (linsol_f_) {
118  case SD_ITERATIVE:
119  f_fwd_ = f_.derivative(1, 0);
120  alloc(f_fwd_);
121  break;
122  default: break;
123  }
124  }
125 
126  if (exact_jacobianB_) {
127  switch (linsol_g_) {
128  case SD_ITERATIVE:
129  g_fwd_ = g_.derivative(1, 0);
130  alloc(g_fwd_);
131  break;
132  default: break;
133  }
134  }

137  void CvodesInterface::initAdj(CvodesMemory* m) const {
138 
139  // Create backward problem (use the same lmm and iter)
140  int flag = CVodeCreateB(m->mem, lmm_, iter_, &m->whichB);
141  if (flag != CV_SUCCESS) cvodes_error("CVodeCreateB", flag);
142 
143  // Initialize the backward problem
144  double tB0 = grid_.back();
145  flag = CVodeInitB(m->mem, m->whichB, rhsB_wrapper, tB0, m->rxz);
146  if (flag != CV_SUCCESS) cvodes_error("CVodeInitB", flag);
147 
148  // Set tolerances
149  flag = CVodeSStolerancesB(m->mem, m->whichB, reltolB_, abstolB_);
150  if (flag!=CV_SUCCESS) cvodes_error("CVodeSStolerancesB", flag);
151 
152  // User data
153  flag = CVodeSetUserDataB(m->mem, m->whichB, m);
154  if (flag != CV_SUCCESS) cvodes_error("CVodeSetUserDataB", flag);
155 
156  // attach linear solver to backward problem
157  switch (linsol_g_) {
158  case SD_DENSE:
159  initDenseLinsolB(m);
160  break;
161  case SD_BANDED:
162  initBandedLinsolB(m);
163  break;
164  case SD_ITERATIVE:
165  initIterativeLinsolB(m);
166  break;
167  case SD_USER_DEFINED:
168  initUserDefinedLinsolB(m);
169  break;
170  }
171 
172  // Quadratures for the backward problem
173  flag = CVodeQuadInitB(m->mem, m->whichB, rhsQB_wrapper, m->rq);
174  if (flag!=CV_SUCCESS) cvodes_error("CVodeQuadInitB", flag);
175 
176  if (quad_err_con_) {
177  flag = CVodeSetQuadErrConB(m->mem, m->whichB, true);
178  if (flag != CV_SUCCESS) cvodes_error("CVodeSetQuadErrConB", flag);
179 
180  flag = CVodeQuadSStolerancesB(m->mem, m->whichB, reltolB_, abstolB_);
181  if (flag != CV_SUCCESS) cvodes_error("CVodeQuadSStolerancesB", flag);
182  }
183 
184  // Mark initialized
185  m->isInitAdj = true;
186  }

188  void CvodesInterface::init_memory(void* mem) const {
189  SundialsInterface::init_memory(mem);
190  auto m = static_cast<CvodesMemory*>(mem);
191 
192  // Create CVodes memory block
193  m->mem = CVodeCreate(lmm_, iter_);
194  casadi_assert_message(m->mem!=0, "CVodeCreate: Creation failed");
195 
196  // Set error handler function
197  int flag = CVodeSetErrHandlerFn(m->mem, ehfun_wrapper, &m);
198  if (flag != CV_SUCCESS) cvodes_error("CVodeSetErrHandlerFn", flag);
199 
200  // Initialize CVodes
201  double t0 = 0;
202  flag = CVodeInit(m->mem, rhs_wrapper, t0, m->xz);
203  if (flag!=CV_SUCCESS) cvodes_error("CVodeInit", flag);
204 
205  // Set tolerances
206  flag = CVodeSStolerances(m->mem, reltol_, abstol_);
207  if (flag!=CV_SUCCESS) cvodes_error("CVodeInit", flag);
208 
209  // Maximum number of steps
210  CVodeSetMaxNumSteps(m->mem, max_num_steps_);
211  if (flag != CV_SUCCESS) cvodes_error("CVodeSetMaxNumSteps", flag);
212 
213  // attach a linear solver
214  switch (linsol_f_) {
215  case SD_DENSE:
216  initDenseLinsol(m);
217  break;
218  case SD_BANDED:
219  initBandedLinsol(m);
220  break;
221  case SD_ITERATIVE:
222  initIterativeLinsol(m);
223  break;
224  case SD_USER_DEFINED:
225  initUserDefinedLinsol(m);
226  break;
227  }
228 
229  // Set user data
230  flag = CVodeSetUserData(m->mem, m);
231  if (flag!=CV_SUCCESS) cvodes_error("CVodeSetUserData", flag);
232 
233  // Quadrature equations
234  if (nq_>0) {
235  // Initialize quadratures in CVodes
236  flag = CVodeQuadInit(m->mem, rhsQ_wrapper, m->q);
237  if (flag != CV_SUCCESS) cvodes_error("CVodeQuadInit", flag);
238 
239  // Should the quadrature errors be used for step size control?
240  if (quad_err_con_) {
241  flag = CVodeSetQuadErrCon(m->mem, true);
242  if (flag != CV_SUCCESS) cvodes_error("CVodeSetQuadErrCon", flag);
243 
244  // Quadrature error tolerances
245  // TODO(Joel): vector absolute tolerances
246  flag = CVodeQuadSStolerances(m->mem, reltol_, abstol_);
247  if (flag != CV_SUCCESS) cvodes_error("CVodeQuadSStolerances", flag);
248  }
249  }
250 
251  // Adjoint sensitivity problem
252  if (!g_.is_null()) {
253  // Get the interpolation type
254  int interpType;
255  if (interpolation_type_=="hermite") {
256  interpType = CV_HERMITE;
257  } else if (interpolation_type_=="polynomial") {
258  interpType = CV_POLYNOMIAL;
259  } else {
260  casadi_error("\"interpolation_type\" must be \"hermite\" or \"polynomial\"");
261  }
262 
263  // Initialize adjoint sensitivities
264  flag = CVodeAdjInit(m->mem, steps_per_checkpoint_, interpType);
265  if (flag != CV_SUCCESS) cvodes_error("CVodeAdjInit", flag);
266  m->isInitAdj = false;
267  }
268  }

317  void CvodesInterface::reset(IntegratorMemory* mem, double t, const double* x,
318  const double* z, const double* _p) const {
319  casadi_msg("CvodesInterface::reset begin");
320  CvodesMemory* m = static_cast<CvodesMemory*>(mem);
321 
322  // Reset the base classes
323  SundialsInterface::reset(mem, t, x, z, _p);
324 
325  // Reset timers
326  m->t_res = m->t_fres = m->t_jac = m->t_lsolve = m->t_lsetup_jac = m->t_lsetup_fac = 0;
327 
328  // Re-initialize
329  int flag = CVodeReInit(m->mem, t, m->xz);
330  if (flag!=CV_SUCCESS) cvodes_error("CVodeReInit", flag);
331 
332  // Re-initialize quadratures
333  if (nq_>0) {
334  N_VConst(0.0, m->q);
335  flag = CVodeQuadReInit(m->mem, m->q);
336  if (flag != CV_SUCCESS) cvodes_error("CVodeQuadReInit", flag);
337  }
338 
339  // Turn off sensitivities
340  flag = CVodeSensToggleOff(m->mem);
341  if (flag != CV_SUCCESS) cvodes_error("CVodeSensToggleOff", flag);
342 
343  // Re-initialize backward integration
344  if (nrx_>0) {
345  flag = CVodeAdjReInit(m->mem);
346  if (flag != CV_SUCCESS) cvodes_error("CVodeAdjReInit", flag);
347  }
348 
349  // Set the stop time of the integration -- don't integrate past this point
350  if (stop_at_end_) setStopTime(m, grid_.back());
351  casadi_msg("CvodesInterface::reset end");
352  }

354  void CvodesInterface::advance(IntegratorMemory* mem, double t, double* x,
355  double* z, double* q) const {
356  auto m = static_cast<CvodesMemory*>(mem);
357 
358  casadi_assert_message(t>=grid_.front(),
359  "CvodesInterface::integrate(" << t << "): "
360  "Cannot integrate to a time earlier than t0 ("
361  << grid_.front() << ")");
362  casadi_assert_message(t<=grid_.back() || !stop_at_end_, "CvodesInterface::integrate("
363  << t << "):"
364  " Cannot integrate past a time later than tf (" << grid_.back() << ") "
365  "unless stop_at_end is set to False.");
366 
367  // Integrate, unless already at desired time
368  const double ttol = 1e-9;
369  if (fabs(m->t-t)>=ttol) {
370  // Integrate forward ...
371  if (nrx_>0) {
372  // ... with taping
373  int flag = CVodeF(m->mem, t, m->xz, &m->t, CV_NORMAL, &m->ncheck);
374  if (flag!=CV_SUCCESS && flag!=CV_TSTOP_RETURN) cvodes_error("CVodeF", flag);
375  } else {
376  // ... without taping
377  int flag = CVode(m->mem, t, m->xz, &m->t, CV_NORMAL);
378  if (flag!=CV_SUCCESS && flag!=CV_TSTOP_RETURN) cvodes_error("CVode", flag);
379  }
380 
381  // Get quadratures
382  if (nq_>0) {
383  double tret;
384  int flag = CVodeGetQuad(m->mem, &tret, m->q);
385  if (flag!=CV_SUCCESS) cvodes_error("CVodeGetQuad", flag);
386  }
387  }
388 
389  // Set function outputs
390  casadi_copy(NV_DATA_S(m->xz), nx_, x);
391  casadi_copy(NV_DATA_S(m->q), nq_, q);
392 
393  // Print statistics
394  if (print_stats_) printStats(m, userOut());
395 
396  if (gather_stats_) {
397  int flag = CVodeGetIntegratorStats(m->mem, &m->nsteps, &m->nfevals, &m->nlinsetups,
398  &m->netfails, &m->qlast, &m->qcur, &m->hinused,
399  &m->hlast, &m->hcur, &m->tcur);
400  if (flag!=CV_SUCCESS) cvodes_error("CVodeGetIntegratorStats", flag);
401  }
402 
403  casadi_msg("CvodesInterface::integrate(" << t << ") end");
404  }

465  void CvodesInterface::printStats(IntegratorMemory* mem, std::ostream &stream) const {
466  auto m = static_cast<CvodesMemory*>(mem);
467 
468  long nsteps, nfevals, nlinsetups, netfails;
469  int qlast, qcur;
470  double hinused, hlast, hcur, tcur;
471  int flag = CVodeGetIntegratorStats(m->mem, &nsteps, &nfevals, &nlinsetups, &netfails, &qlast,
472  &qcur, &hinused, &hlast, &hcur, &tcur);
473  if (flag!=CV_SUCCESS) cvodes_error("CVodeGetIntegratorStats", flag);
474 
475  // Get the number of right hand side evaluations in the linear solver
476  long nfevals_linsol=0;
477  switch (linsol_f_) {
478  case SD_DENSE:
479  case SD_BANDED:
480  flag = CVDlsGetNumRhsEvals(m->mem, &nfevals_linsol);
481  if (flag!=CV_SUCCESS) cvodes_error("CVDlsGetNumRhsEvals", flag);
482  break;
483  case SD_ITERATIVE:
484  flag = CVSpilsGetNumRhsEvals(m->mem, &nfevals_linsol);
485  if (flag!=CV_SUCCESS) cvodes_error("CVSpilsGetNumRhsEvals", flag);
486  break;
487  default:
488  nfevals_linsol = 0;
489  }
490 
491  stream << "number of steps taken by CVODES: " << nsteps << std::endl;
492  stream << "number of calls to the user's f function: " << (nfevals + nfevals_linsol)
493  << std::endl;
494  stream << " step calculation: " << nfevals << std::endl;
495  stream << " linear solver: " << nfevals_linsol << std::endl;
496  stream << "number of calls made to the linear solver setup function: " << nlinsetups
497  << std::endl;
498  stream << "number of error test failures: " << netfails << std::endl;
499  stream << "method order used on the last internal step: " << qlast << std::endl;
500  stream << "method order to be used on the next internal step: " << qcur << std::endl;
501  stream << "actual value of initial step size: " << hinused << std::endl;
502  stream << "step size taken on the last internal step: " << hlast << std::endl;
503  stream << "step size to be attempted on the next internal step: " << hcur << std::endl;
504  stream << "current internal time reached: " << tcur << std::endl;
505  stream << std::endl;
506 
507  stream << "number of checkpoints stored: " << m->ncheck << endl;
508  stream << std::endl;
509 
510  stream << "Time spent in the ODE residual: " << m->t_res << " s." << endl;
511  stream << "Time spent in the forward sensitivity residual: " << m->t_fres << " s." << endl;
512  stream << "Time spent in the jacobian function or jacobian times vector function: "
513  << m->t_jac << " s." << endl;
514  stream << "Time spent in the linear solver solve function: " << m->t_lsolve << " s." << endl;
515  stream << "Time spent to generate the jacobian in the linear solver setup function: "
516  << m->t_lsetup_jac << " s." << endl;
517  stream << "Time spent to factorize the jacobian in the linear solver setup function: "
518  << m->t_lsetup_fac << " s." << endl;
519  stream << std::endl;
520  }
