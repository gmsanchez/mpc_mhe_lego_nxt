68  void SnoptInterface::init(const Dict& opts) {
69  // Call the init method of the base class
70  Nlpsol::init(opts);
71 
72  // Read user options
73  for (auto&& op : opts) {
74  if (op.first=="snopt") {
75  opts_ = op.second;
76  }
77  }
78 
79  // Get/generate required functions
80  jac_f_fcn_ = create_function("nlp_jac_f", {"x", "p"}, {"f", "jac_f_x"});
81  jac_g_fcn_ = create_function("nlp_jac_g", {"x", "p"}, {"g", "jac_g_x"});
82  jacg_sp_ = jac_g_fcn_.sparsity_out(1);
83 
84  // prepare the mapping for constraints
85  nnJac_ = nx_;
86  nnObj_ = nx_;
87  nnCon_ = ng_;
88 
89  // Here follows the core of the mapping
90  // Two integer matrices are constructed:
91  // one with gradF sparsity, and one with jacG sparsity
92  // the integer values denote the nonzero locations into the original gradF/jacG
93  // but with a special encoding: entries of gradF are encoded "-1-i" and
94  // entries of jacG are encoded "1+i"
95  // "0" is to be interpreted not as an index but as a literal zero
96 
97  IM mapping_jacG = IM(0, nx_);
98  IM mapping_gradF = IM(jac_f_fcn_.sparsity_out(1),
99  range(-1, -1-jac_f_fcn_.nnz_out(1), -1));
100 
101  if (!jac_g_fcn_.is_null()) {
102  mapping_jacG = IM(jacg_sp_, range(1, jacg_sp_.nnz()+1));
103  }
104 
105  // First, remap jacG
106  A_structure_ = mapping_jacG;
107 
108  m_ = ng_;
109 
110  // Construct the linear objective row
111  IM d = mapping_gradF(Slice(0), Slice());
112 
113  std::vector<int> ii = mapping_gradF.sparsity().get_col();
114  for (int j = 0; j < nnObj_; ++j) {
115  if (d.colind(j) != d.colind(j+1)) {
116  int k = d.colind(j);
117  d[k] = 0;
118  }
119  }
120 
121  // Make it as sparse as you can
122  d = sparsify(d);
123 
124  jacF_row_ = d.nnz() != 0;
125  if (jacF_row_) { // We need an objective gradient row
126  A_structure_ = vertcat(A_structure_, d);
127  m_ +=1;
128  }
129  iObj_ = jacF_row_ ? (m_ - 1) : -1;
130 
131  // Is the A matrix completely empty?
132  dummyrow_ = A_structure_.nnz() == 0; // Then we need a dummy row
133  if (dummyrow_) {
134  IM dummyrow = IM(1, nx_);
135  dummyrow(0, 0) = 0;
136  A_structure_ = vertcat(A_structure_, dummyrow);
137  m_+=1;
138  }
139 
140  // We don't need a dummy row if a linear objective row is present
141  casadi_assert(!(dummyrow_ && jacF_row_));
142 
143  // Allocate temporary memory
144  alloc_w(nx_, true); // xk2_
145  alloc_w(ng_, true); // lam_gk_
146  alloc_w(nx_, true); // lam_xk_
147  alloc_w(ng_, true); // gk_
148  alloc_w(jac_f_fcn_.nnz_out(1), true); // jac_fk_
149  if (!jacg_sp_.is_null()) {
150  alloc_w(jacg_sp_.nnz(), true); // jac_gk_
151  }
152  }

175  auto m = static_cast<SnoptMemory*>(mem);
176 
177  // Set work in base classes
178  Nlpsol::set_work(mem, arg, res, iw, w);
179 
180  // Work vectors

182  m->lam_gk = w; w += ng_;
183  m->lam_xk = w; w += nx_;
184  m->gk = w; w += ng_;
185  m->jac_fk = w; w += jac_f_fcn_.nnz_out(1);

187  m->jac_gk = w; w += jacg_sp_.nnz();

191  void SnoptInterface::solve(void* mem) const {
192  auto m = static_cast<SnoptMemory*>(mem);
193 
194  // Check the provided inputs
195  checkInputs(mem);
196 
197  m->fstats.at("mainloop").tic();
198 
199  // Memory object
200  snProblem prob;
201 
202  // Evaluate gradF and jacG at initial value
203  if (!jac_g_fcn_.is_null()) {
204  std::fill_n(m->arg, jac_g_fcn_.n_in(), nullptr);
205  m->arg[0] = m->x0;
206  m->arg[1] = m->p;
207  std::fill_n(m->res, jac_g_fcn_.n_out(), nullptr);
208  m->res[0] = 0;
209  m->res[1] = m->jac_gk;
210  jac_g_fcn_(m->arg, m->res, m->iw, m->w, 0);
211  }
212  calc_function(m, jac_f_fcn_, {m->x0, m->p}, {0, m->jac_fk});
213 
214  // perform the mapping:
215  // populate A_data_ (the nonzeros of A)
216  // with numbers pulled from jacG and gradF
217  for (int k = 0; k < A_structure_.nnz(); ++k) {
218  int i = A_structure_.nonzeros()[k];
219  if (i == 0) {
220  m->A_data[k] = 0;
221  } else if (i > 0) {
222  m->A_data[k] = m->jac_gk[i-1];
223  } else {
224  m->A_data[k] = m->jac_fk[-i-1];
225  }
226  }
227 
228  int n = nx_;
229  int nea = A_structure_.nnz();
230  double ObjAdd = 0;
231 
232  casadi_assert(m_ > 0);
233  casadi_assert(n > 0);
234  casadi_assert(nea > 0);
235  casadi_assert(A_structure_.nnz() == nea);
236 
237  // Pointer magic, courtesy of Greg
238  casadi_assert_message(!jac_f_fcn_.is_null(), "blaasssshc");
239 
240  // Outputs
241  double Obj = 0; // TODO(Greg): get this from snopt
242 
243  // snInit must be called first.
244  // 9, 6 are print and summary unit numbers (for Fortran).
245  // 6 == standard out
246  int iprint = 9;
247  int isumm = 6;
248  std::string outname = name_ + ".out";
249  snInit(&prob, const_cast<char*>(name_.c_str()),
250  const_cast<char*>(outname.c_str()), iprint, isumm);
251 
252  // Set the problem size and other data.
253  // This will allocate arrays inside snProblem struct.
254  setProblemSize(&prob, m_, nx_, nea, nnCon_, nnJac_, nnObj_);
255  setObjective(&prob, iObj_, ObjAdd);
256  setUserfun(&prob, userfunPtr);
257 
258  // user data
259  prob.leniu = 1;
260  prob.iu = &m->memind;
261 
262  // Pass bounds
263  casadi_copy(m->lbx, nx_, prob.bl);
264  casadi_copy(m->ubx, nx_, prob.bu);
265  casadi_copy(m->lbg, ng_, prob.bl + nx_);
266  casadi_copy(m->ubg, ng_, prob.bu + nx_);
267 
268  // Initialize states and slack
269  casadi_fill(prob.hs, ng_ + nx_, 0);
270  casadi_copy(m->x0, nx_, prob.x);
271  casadi_fill(prob.x + nx_, ng_, 0.);
272 
273  // Initialize multipliers
274  casadi_copy(m->lam_g0, ng_, prob.pi);
275 
276  // Set up Jacobian matrix
277  casadi_copy(A_structure_.colind(), A_structure_.size2()+1, prob.locJ);
278  casadi_copy(A_structure_.row(), A_structure_.nnz(), prob.indJ);
279  casadi_copy(get_ptr(m->A_data), A_structure_.nnz(), prob.valJ);
280 
281  for (auto&& op : opts_) {
282  // Replace underscores with spaces
283  std::string opname = op.first;
284  std::replace(opname.begin(), opname.end(), '_', ' ');
285 
286  // Try integer
287  if (op.second.can_cast_to(OT_INT)) {
288  casadi_assert(opname.size() <= 55);
289  int flag = setIntParameter(&prob, const_cast<char*>(opname.c_str()),
290  op.second.to_int());
291  if (flag==0) continue;
292  }
293 
294  // Try double
295  if (op.second.can_cast_to(OT_DOUBLE)) {
296  casadi_assert(opname.size() <= 55);
297  int flag = setRealParameter(&prob, const_cast<char*>(opname.c_str()),
298  op.second.to_double());
299  if (flag==0) continue;
300  }
301 
302  // try string
303  if (op.second.can_cast_to(OT_STRING)) {
304  std::string buffer = opname + " " + op.second.to_string();
305  casadi_assert(buffer.size() <= 72);
306  int flag = setParameter(&prob, const_cast<char*>(buffer.c_str()));
307  if (flag==0) continue;
308  }
309 
310  // Error if reached this point
311  casadi_error("SNOPT error setting option \"" + opname + "\"");
312  }
313 
314  m->fstats.at("mainloop").toc();
315 
316  // Run SNOPT
317  int Cold = 0;
318  int info = solveC(&prob, Cold, &m->fk);
319  casadi_assert_message(99 != info, "snopt problem set up improperly");
320 
321  // Negate rc to match CasADi's definition
322  casadi_scal(nx_ + ng_, -1., prob.rc);
323 
324  // Get primal solution
325  casadi_copy(prob.x, nx_, m->x);
326 
327  // Get dual solution
328  casadi_copy(prob.rc, nx_, m->lam_x);
329  casadi_copy(prob.rc+nx_, ng_, m->lam_g);
330 
331  // Copy optimal cost to output
332  if (m->f) *m->f = m->fk;
333 
334  // Copy optimal constraint values to output
335  casadi_copy(m->gk, ng_, m->g);
336 
337  // Show statistics
338  if (print_time_) print_fstats(m);
339 
340  // Free memory
341  deleteSNOPT(&prob);
342  }

344  void SnoptInterface::
345  userfun(SnoptMemory* m, int* mode, int nnObj, int nnCon, int nnJac, int nnL, int neJac,
346  double* x, double* fObj, double*gObj, double* fCon, double* gCon,
347  int nState, char* cu, int lencu, int* iu, int leniu, double* ru,
348  int lenru) const {
349  try {
350 
351  casadi_assert_message(nnCon_ == nnCon, "Con " << nnCon_ << " <-> " << nnCon);
352  casadi_assert_message(nnObj_ == nnObj, "Obj " << nnObj_ << " <-> " << nnObj);
353  casadi_assert_message(nnJac_ == nnJac, "Jac " << nnJac_ << " <-> " << nnJac);
354 
355  // Get reduced decision variables
356  casadi_fill(m->xk2, nx_, 0.);
357  for (int k = 0; k < nnObj; ++k) m->xk2[k] = x[k];
358 
359  // Evaluate gradF with the linear variables put to zero
360  calc_function(m, jac_f_fcn_, {m->xk2, m->p}, {fObj, m->jac_fk});
361 
362  // provide nonlinear part of objective gradient to SNOPT
363  for (int k = 0; k < nnObj; ++k) {
364  int el = jac_f_fcn_.sparsity_out(1).colind(k);
365  if (jac_f_fcn_.sparsity_out(1).colind(k+1) > el) {
366  gObj[k] = m->jac_fk[el];
367  } else {
368  gObj[k] = 0;
369  }
370  }
371 
372  jac_f_fcn_.sparsity_out(1).sanity_check(true);
373  jac_f_fcn_.sparsity_out(1).sanity_check(false);
374 
375  if (!jac_g_fcn_.is_null()) {
376  // Get reduced decision variables
377  casadi_fill(m->xk2, nx_, 0.);
378  for (int k = 0; k < nnJac; ++k) {
379  m->xk2[k] = x[k];
380  }
381 
382  // Evaluate jacG with the linear variabes put to zero
383  std::fill_n(m->arg, jac_g_fcn_.n_in(), nullptr);
384  m->arg[0] = m->xk2;
385  m->arg[1] = m->p;
386  std::fill_n(m->res, jac_g_fcn_.n_out(), nullptr);
387  m->res[0] = m->gk;
388  m->res[1] = m->jac_gk;
389  jac_g_fcn_(m->arg, m->res, m->iw, m->w, 0);
390 
391  // provide nonlinear part of constraint jacobian to SNOPT
392  int kk = 0;
393  for (int j = 0; j < nnJac; ++j) {
394  for (int k = A_structure_.colind(j); k < A_structure_.sparsity().colind(j+1); ++k) {
395  if (A_structure_.row(k) >= nnCon) break;
396  int i = A_structure_.nonzeros()[k];
397  if (i > 0) {
398  gCon[kk++] = m->jac_gk[i-1];
399  }
400  }
401  }
402 
403  casadi_assert(kk == 0 || kk == neJac);
404 
405  // provide nonlinear part of objective to SNOPT
406  for (int k = 0; k < nnCon; ++k) {
407  fCon[k] = m->gk[k];
408  }
409  }
410 
411  } catch(std::exception& ex) {
412  userOut<true, PL_WARN>() << "eval_nlp failed: " << ex.what() << std::endl;
413  *mode = -1; // Reduce step size - we've got problems
414  return;
415  }
416  }
