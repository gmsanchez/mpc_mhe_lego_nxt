65  void QpToNlp::init(const Dict& opts) {
66  // Initialize the base classes
67  Qpsol::init(opts);
68 
69  // Default options
70  string nlpsol_plugin;
71  Dict nlpsol_options;
72 
73  // Read user options
74  for (auto&& op : opts) {
75  if (op.first=="nlpsol") {
76  nlpsol_plugin = op.second.to_string();
77  } else if (op.first=="nlpsol_options") {
78  nlpsol_options = op.second;
79  }
80  }
81 
82  // Create a symbolic matrix for the decision variables
83  SX X = SX::sym("X", n_, 1);
84 
85  // Parameters to the problem
86  SX H = SX::sym("H", sparsity_in(QPSOL_H));
87  SX G = SX::sym("G", sparsity_in(QPSOL_G));
88  SX A = SX::sym("A", sparsity_in(QPSOL_A));
89 
90  // Put parameters in a vector
91  std::vector<SX> par;
92  par.push_back(H.nonzeros());
93  par.push_back(G.nonzeros());
94  par.push_back(A.nonzeros());
95 
96  // The nlp looks exactly like a mathematical description of the NLP
97  SXDict nlp = {{"x", X}, {"p", vertcat(par)},
98  {"f", mtimes(G.T(), X) + 0.5*mtimes(mtimes(X.T(), H), X)},
99  {"g", mtimes(A, X)}};
100 
101  // Create an Nlpsol instance
102  casadi_assert_message(!nlpsol_plugin.empty(), "'nlpsol' option has not been set");
103  solver_ = nlpsol("nlpsol", nlpsol_plugin, nlp, nlpsol_options);
104  alloc(solver_);
105 
106  // Allocate storage for NLP solver parameters
107  alloc_w(solver_.nnz_in(NLPSOL_P), true);
108  }

110  void QpToNlp::
111  eval(void* mem, const double** arg, double** res, int* iw, double* w) const {
112  // Inputs
113  const double *h_, *g_, *a_, *lba_, *uba_, *lbx_, *ubx_, *x0_;
114  // Outputs
115  double *x_, *f_, *lam_a_, *lam_x_;
116 
117  // Get input pointers
118  h_ = arg[QPSOL_H];
119  g_ = arg[QPSOL_G];
120  a_ = arg[QPSOL_A];
121  lba_ = arg[QPSOL_LBA];
122  uba_ = arg[QPSOL_UBA];
123  lbx_ = arg[QPSOL_LBX];
124  ubx_ = arg[QPSOL_UBX];
125  x0_ = arg[QPSOL_X0];
126 
127  // Get output pointers
128  x_ = res[QPSOL_X];
129  f_ = res[QPSOL_COST];
130  lam_a_ = res[QPSOL_LAM_A];
131  lam_x_ = res[QPSOL_LAM_X];
132 
133  // Buffers for calling the NLP solver
134  const double** arg1 = arg + n_in();
135  double** res1 = res + n_out();
136  fill_n(arg1, static_cast<int>(NLPSOL_NUM_IN), nullptr);
137  fill_n(res1, static_cast<int>(NLPSOL_NUM_OUT), nullptr);
138 
139  // NLP inputs
140  arg1[NLPSOL_X0] = x0_;
141  arg1[NLPSOL_LBG] = lba_;
142  arg1[NLPSOL_UBG] = uba_;
143  arg1[NLPSOL_LBX] = lbx_;
144  arg1[NLPSOL_UBX] = ubx_;
145 
146  // NLP parameters
147  arg1[NLPSOL_P] = w;
148 
149  // Quadratic term
150  int nh = nnz_in(QPSOL_H);
151  if (h_) {
152  copy_n(h_, nh, w);
153  } else {
154  fill_n(w, nh, 0);
155  }
156  w += nh;
157 
158  // Linear objective term
159  int ng = nnz_in(QPSOL_G);
160  if (g_) {
161  copy_n(g_, ng, w);
162  } else {
163  fill_n(w, ng, 0);
164  }
165  w += ng;
166 
167  // Linear constraints
168  int na = nnz_in(QPSOL_A);
169  if (a_) {
170  copy_n(a_, na, w);
171  } else {
172  fill_n(w, na, 0);
173  }
174  w += na;
175 
176  // Solution
177  res1[NLPSOL_X] = x_;
178  res1[NLPSOL_F] = f_;
179  res1[NLPSOL_LAM_X] = lam_x_;
180  res1[NLPSOL_LAM_G] = lam_a_;
181 
182  // Solve the NLP
183  solver_(arg1, res1, iw, w, 0);
184  }
