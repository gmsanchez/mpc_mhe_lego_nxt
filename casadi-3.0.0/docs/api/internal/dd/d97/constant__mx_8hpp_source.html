331  MX Constant<Value>::getHorzcat(const std::vector<MX>& x) const {
332  // Check if all arguments have the same constant value
333  for (std::vector<MX>::const_iterator i=x.begin()+1; i!=x.end(); ++i) {
334  if (!(*i)->isValue(v_.value)) {
335  // Not all the same value, fall back to base class
336  return ConstantMX::getHorzcat(x);
337  }
338  }
339 
340  // Assemble the sparsity pattern
341  Sparsity sp = sparsity();
342  for (std::vector<MX>::const_iterator i=x.begin()+1; i!=x.end(); ++i) {
343  sp.appendColumns(i->sparsity());
344  }
345  return MX(sp, v_.value, false);
346  }

376  template<typename Value>
377  MX Constant<Value>::getUnary(int op) const {
378  // Constant folding
379  double ret(0);
380  casadi_math<double>::fun(op, v_.value, 0.0, ret);
381  if (operation_checker<F0XChecker>(op) || sparsity().is_dense()) {
382  return MX(sparsity(), ret);
383  } else {
384  if (v_.value==0) {
385  if (is_zero() && operation_checker<F0XChecker>(op)) {
386  return MX(sparsity(), ret, false);
387  } else {
388  return repmat(MX(ret), size1(), size2());
389  }
390  }
391  double ret2;
392  casadi_math<double>::fun(op, 0, 0.0, ret2);
393  return DM(sparsity(), ret, false)
394  + DM(sparsity().pattern_inverse(), ret2, false);
395  }
396  }

398  template<typename Value>
399  MX Constant<Value>::getBinary(int op, const MX& y, bool ScX, bool ScY) const {
400  casadi_assert(sparsity()==y.sparsity() || ScX || ScY);
401 
402  if (ScX && !operation_checker<FX0Checker>(op)) {
403  double ret;
404  casadi_math<double>::fun(op, nnz()> 0 ? v_.value: 0, 0, ret);
405 
406  if (ret!=0) {
407  Sparsity f = Sparsity::dense(y.size1(), y.size2());
408  MX yy = project(y, f);
409  return MX(f, shared_from_this<MX>())->getBinary(op, yy, false, false);
410  }
411  } else if (ScY && !operation_checker<F0XChecker>(op)) {
412  bool grow = true;
413  if (y->op()==OP_CONST && dynamic_cast<const ConstantDM*>(y.get())==0) {
414  double ret;
415  casadi_math<double>::fun(op, 0, y.nnz()>0 ? y->to_double() : 0, ret);
416  grow = ret!=0;
417  }
418  if (grow) {
419  Sparsity f = Sparsity::dense(size1(), size2());
420  MX xx = project(shared_from_this<MX>(), f);
421  return xx->getBinary(op, MX(f, y), false, false);
422  }
423  }
424 
425  switch (op) {
426  case OP_ADD:
427  if (v_.value==0) return ScY && !y->is_zero() ? repmat(y, size1(), size2()) : y;
428  break;
429  case OP_SUB:
430  if (v_.value==0) return ScY && !y->is_zero() ? repmat(-y, size1(), size2()) : -y;
431  break;
432  case OP_MUL:
433  if (v_.value==1) return y;
434  if (v_.value==-1) return -y;
435  if (v_.value==2) return y->getUnary(OP_TWICE);
436  break;
437  case OP_DIV:
438  if (v_.value==1) return y->getUnary(OP_INV);
439  if (v_.value==-1) return -y->getUnary(OP_INV);
440  break;
441  case OP_POW:
442  if (v_.value==0) return MX::zeros(y.sparsity());
443  if (v_.value==1) return MX::ones(y.sparsity());
444  if (v_.value==std::exp(1.0)) return y->getUnary(OP_EXP);
445  break;
446  default: break; //no rule
447  }
448 
449  // Constant folding
450  // NOTE: ugly, should use a function instead of a cast
451  if (y->op()==OP_CONST && dynamic_cast<const ConstantDM*>(y.get())==0) {
452  double y_value = y.nnz()>0 ? y->to_double() : 0;
453  double ret;
454  casadi_math<double>::fun(op, nnz()> 0 ? v_.value: 0, y_value, ret);
455 
456  return MX(y.sparsity(), ret, false);
457  }
458 
459  // Fallback
460  return MXNode::getBinary(op, y, ScX, ScY);
461  }
