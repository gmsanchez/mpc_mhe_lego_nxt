93  void Split::generate(CodeGenerator& g, const std::string& mem,
94  const std::vector<int>& arg, const std::vector<int>& res) const {
95  int nx = nout();
96  for (int i=0; i<nx; ++i) {
97  int nz_first = offset_[i];
98  int nz_last = offset_[i+1];
99  int nz = nz_last-nz_first;
100  if (res[i]>=0 && nz>0) { // if anything to assign
101  if (nz==1) { // assign scalar
102  g.body << " " << g.workel(res[i]) << " = ";
103  if (dep(0).nnz()==1) {
104  // rhs is also scalar
105  casadi_assert(nz_first==0);
106  g.body << g.workel(arg[0]) << ";" << endl;
107  } else {
108  // rhs is an element in a vector
109  g.body << g.work(arg[0], dep(0).nnz()) << "[" << nz_first << "];" << endl;
110  }
111  } else {
112  // assign vector
113  std::string r = g.work(arg[0], dep(0).nnz());
114  if (nz_first!=0) r = r + "+" + g.to_string(nz_first);
115  g.body << " " << g.copy(r, nz, g.work(res[i], nnz(i))) << endl;
116  }
117  }
118  }
