135  void KinsolInterface::init(const Dict& opts) {
136  // Initialize the base classes
137  Rootfinder::init(opts);
138 
139  // Default (temporary) options
140  string strategy = "none";
141  vector<double> u_scale;
142  vector<double> f_scale;
143  string linear_solver_type = "dense";
144  string iterative_solver = "gmres";
145 
146  // Read options
147  for (auto&& op : opts) {
148  if (op.first=="strategy") {
149  strategy = op.second.to_string();
150  } else if (op.first=="exact_jacobian") {
151  exact_jacobian_ = op.second;
152  } else if (op.first=="u_scale") {
153  u_scale = op.second;
154  } else if (op.first=="f_scale") {
155  f_scale = op.second;
156  } else if (op.first=="disable_internal_warnings") {
157  disable_internal_warnings_ = op.second;
158  } else if (op.first=="max_iter") {
159  max_iter_ = op.second;
160  } else if (op.first=="linear_solver_type") {
161  linear_solver_type = op.second.to_string();
162  } else if (op.first=="max_krylov") {
163  maxl_ = op.second;
164  } else if (op.first=="upper_bandwidth") {
165  upper_bandwidth_ = op.second;
166  } else if (op.first=="lower_bandwidth") {
167  lower_bandwidth_ = op.second;
168  } else if (op.first=="iterative_solver") {
169  iterative_solver = op.second.to_string();
170  } else if (op.first=="use_preconditioner") {
171  use_preconditioner_ = op.second;
172  } else if (op.first=="abstol") {
173  abstol_ = op.second;
174  }
175  }
176 
177  // Get globalization strategy
178  if (strategy=="linesearch") {
179  strategy_ = KIN_LINESEARCH;
180  } else {
181  casadi_assert(strategy=="none");
182  strategy_ = KIN_NONE;
183  }
184 
185  // Allocate N_Vectors
186  if (u_scale_) N_VDestroy_Serial(u_scale_);
187  if (f_scale_) N_VDestroy_Serial(f_scale_);
188  u_scale_ = N_VNew_Serial(n_);
189  f_scale_ = N_VNew_Serial(n_);
190 
191  // Set scaling factors on variables
192  if (!u_scale.empty()) {
193  casadi_assert(u_scale.size()==NV_LENGTH_S(u_scale_));
194  copy(u_scale.begin(), u_scale.end(), NV_DATA_S(u_scale_));
195  } else {
196  N_VConst(1.0, u_scale_);
197  }
198 
199  // Set scaling factors on equations
200  if (!f_scale.empty()) {
201  casadi_assert(f_scale.size()==NV_LENGTH_S(f_scale_));
202  copy(f_scale.begin(), f_scale.end(), NV_DATA_S(f_scale_));
203  } else {
204  N_VConst(1.0, f_scale_);
205  }
206 
207  // Type of linear solver
208  if (linear_solver_type=="dense") {
209  linear_solver_type_ = DENSE;
210  if (exact_jacobian_) {
211  // For storing Jacobian nonzeros
212  alloc_w(jac_.nnz_out(0), true);
213  }
214  } else if (linear_solver_type=="banded") {
215  linear_solver_type_ = BANDED;
216  casadi_assert(upper_bandwidth_>=0);
217  casadi_assert(lower_bandwidth_>=0);
218  if (exact_jacobian_) {
219  // For storing Jacobian nonzeros
220  alloc_w(jac_.nnz_out(0), true);
221  }
222  } else if (linear_solver_type=="iterative") {
223  linear_solver_type_ = ITERATIVE;
224  if (iterative_solver=="gmres") {
225  iterative_solver_ = GMRES;
226  } else if (iterative_solver=="bcgstab") {
227  iterative_solver_ = BCGSTAB;
228  } else if (iterative_solver=="tfqmr") {
229  iterative_solver_ = TFQMR;
230  } else {
231  casadi_error("KINSOL: Unknown sparse solver");
232  }
233  if (exact_jacobian_) {
234  // Form the Jacobian-times-vector function
235  f_fwd_ = f_.derivative(1, 0);
236  alloc(f_fwd_);
237  }
238  if (use_preconditioner_) {
239  // Make sure that a Jacobian has been provided
240  casadi_assert_message(!jac_.is_null(), "No Jacobian has been provided");
241 
242  // Make sure that a linear solver has been provided
243  casadi_assert_message(!linsol_.is_null(), "No linear solver has been provided.");
244  }
245  } else if (linear_solver_type=="user_defined") {
246  linear_solver_type_ = USER_DEFINED;
247  // Make sure that a Jacobian has been provided
248  casadi_assert(!jac_.is_null());
249 
250  // Make sure that a linear solver has been provided
251  casadi_assert(!linsol_.is_null());
252 
253  // Form the Jacobian-times-vector function
254  f_fwd_ = f_.derivative(1, 0);
255  alloc(f_fwd_);
256 
257  // Allocate space for Jacobian
258  alloc_w(jac_.nnz_out(0), true);
259  } else {
260  casadi_error("Unknown linear solver");
261  }
262  }

264  void KinsolInterface::eval(void* mem, const double** arg, double** res,
265  int* iw, double* w) const {
266  auto m = static_cast<KinsolMemory*>(mem);
267 
268  // Update IO references
269  m->arg = arg;
270  m->res = res;
271  m->iw = iw;
272  m->w = w;
273 
274  // Reset the counters
275  m->t_func = 0;
276  m->t_jac = 0;
277 
278  // Get the initial guess
279  if (arg[iin_]) {
280  copy(arg[iin_], arg[iin_]+nnz_in(iin_), NV_DATA_S(m->u));
281  } else {
282  N_VConst(0.0, m->u);
283  }
284 
285  // Solve the nonlinear system of equations
286  int flag = KINSol(m->mem, m->u, strategy_, u_scale_, f_scale_);
287  if (flag<KIN_SUCCESS) kinsol_error("KINSol", flag);
288 
289  // Warn if not successful return
290  if (verbose()) {
291  if (flag!=KIN_SUCCESS) kinsol_error("KINSol", flag, false);
292  }
293 
294  // Get the solution
295  if (res[iout_]) {
296  copy_n(NV_DATA_S(m->u), nnz_out(iout_), res[iout_]);
297  }
298 
299  // Evaluate auxiliary outputs
300  if (n_out()>0) {
301  // Temporary memory
302  const double** arg1 = arg + n_in();
303  double** res1 = res + n_out();
304 
305  // Evaluate f_
306  copy_n(arg, n_in(), arg1);
307  arg1[iin_] = NV_DATA_S(m->u);
308  copy_n(res, n_out(), res1);
309  res1[iout_] = 0;
310  f_(arg1, res1, iw, w, 0);
311  }
312  }

314  void KinsolInterface::func(KinsolMemory& m, N_Vector u, N_Vector fval) const {
315  // Get time
316  m.time1 = clock();
317 
318  // Temporary memory
319  const double** arg1 = m.arg + n_in();
320  double** res1 = m.res + n_out();
321 
322  // Evaluate f_
323  copy(m.arg, m.arg + n_in(), arg1);
324  arg1[iin_] = NV_DATA_S(u);
325  fill_n(res1, n_out(), static_cast<double*>(0));
326  res1[iout_] = NV_DATA_S(fval);
327  f_(arg1, res1, m.iw, m.w, 0);
328 
329  // Print it, if requested
330  if (monitored("eval_f")) {
331  userOut() << "f = ";
332  N_VPrint_Serial(fval);
333  }
334 
335  // Make sure that all entries of the linear system are valid
336  double *fdata = res1[iout_];
337  for (int k=0; k<n_; ++k) {
338  try {
339  casadi_assert_message(!isnan(fdata[k]), "Nonzero " << k << " is not-a-number");
340  casadi_assert_message(!isinf(fdata[k]), "Nonzero " << k << " is infinite");
341  } catch(exception& ex) {
342  stringstream ss;
343  ss << ex.what() << endl;
344  if (verbose()) {
345  userOut() << "u = ";
346  N_VPrint_Serial(u);
347  }
348 
349  throw CasadiException(ss.str());
350  }
351  }
352 
353  // Log time
354  m.time2 = clock();
355  m.t_func += static_cast<double>(m.time2-m.time1)/CLOCKS_PER_SEC;
356  }

383  void KinsolInterface::djac(KinsolMemory& m, long N, N_Vector u, N_Vector fu, DlsMat J,
384  N_Vector tmp1, N_Vector tmp2) const {
385  // Get time
386  m.time1 = clock();
387 
388  // Temporary memory
389  const double** arg1 = m.arg + n_in();
390  double** res1 = m.res + n_out();
391  double* jac = m.w + jac_.sz_w();
392 
393  // Evaluate jac_
394  copy(m.arg, m.arg + n_in(), arg1);
395  arg1[iin_] = NV_DATA_S(u);
396  fill_n(res1, jac_.n_out(), static_cast<double*>(0));
397  res1[0] = jac;
398  jac_(arg1, res1, m.iw, m.w, 0);
399 
400  // Get sparsity and non-zero elements
401  const int* colind = jac_.sparsity_out(0).colind();
402  int ncol = jac_.size2_out(0);
403  const int* row = jac_.sparsity_out(0).row();
404 
405  // Loop over columns
406  for (int cc=0; cc<ncol; ++cc) {
407  // Loop over non-zero entries
408  for (int el=colind[cc]; el<colind[cc+1]; ++el) {
409  // Get row
410  int rr = row[el];
411 
412  // Set the element
413  DENSE_ELEM(J, rr, cc) = jac[el];
414  }
415  }
416 
417  if (monitored("eval_djac")) {
418  userOut() << "djac = ";
419  PrintMat(J);
420  }
421 
422  // Log time duration
423  m.time2 = clock();
424  m.t_jac += static_cast<double>(m.time2 - m.time1)/CLOCKS_PER_SEC;
425  }

442  N_Vector fu, DlsMat J, N_Vector tmp1, N_Vector tmp2) const {
443  // Get time
444  m.time1 = clock();

446  // Temporary memory
447  const double** arg1 = m.arg + n_in();
448  double** res1 = m.res + n_out();
449  double* jac = m.w + jac_.sz_w();
450 
451  // Evaluate jac_
452  copy(m.arg, m.arg + jac_.n_in(), arg1);
453  arg1[iin_] = NV_DATA_S(u);
454  fill_n(res1, jac_.n_out(), static_cast<double*>(0));

459  const int* colind = jac_.sparsity_out(0).colind();

462 
463  // Loop over cols

628  N_Vector x, N_Vector b, double *res_norm) const {
629  // Get vectors
630  N_Vector u = kin_mem->kin_uu;
631  N_Vector uscale = kin_mem->kin_uscale;
632  N_Vector fval = kin_mem->kin_fval;
633  N_Vector fscale = kin_mem->kin_fscale;
634  N_Vector tmp1 = kin_mem->kin_vtemp1;
635  N_Vector tmp2 = kin_mem->kin_vtemp2;
636 
637  // Solve the linear system
638  N_VScale(1.0, b, x);
639  psolve(m, u, uscale, fval, fscale, x, tmp1);
640 
641  // Calculate residual
642  jtimes(m, x, tmp2, u, 0);
643 
644  // Calculate the error in residual norm
645  N_VLinearSum(1.0, b, -1.0, tmp2, tmp1);
646  *res_norm = sqrt(N_VDotProd(tmp1, tmp1));
647  }

797  void KinsolInterface::init_memory(void* mem) const {
798  auto m = static_cast<KinsolMemory*>(mem);
799 
800  // Current solution
801  m->u = N_VNew_Serial(n_);
802 
803  // Create KINSOL memory block
804  m->mem = KINCreate();
805 
806  // KINSOL bugfix
807  KINMem kin_mem = KINMem(m->mem);
808  kin_mem->kin_inexact_ls = FALSE;
809 
810  // Set optional inputs
811  int flag = KINSetUserData(m->mem, m);
812  casadi_assert_message(flag==KIN_SUCCESS, "KINSetUserData");
813 
814  // Set error handler function
815  flag = KINSetErrHandlerFn(m->mem, ehfun_wrapper, m);
816  casadi_assert_message(flag==KIN_SUCCESS, "KINSetErrHandlerFn");
817 
818  // Initialize KINSOL
819  flag = KINInit(m->mem, func_wrapper, m->u);
820  casadi_assert(flag==KIN_SUCCESS);
821 
822  // Setting maximum number of Newton iterations
823  flag = KINSetMaxNewtonStep(m->mem, max_iter_);
824  casadi_assert(flag==KIN_SUCCESS);
825 
826  // Set constraints
827  if (!u_c_.empty()) {
828  N_Vector domain = N_VNew_Serial(n_);
829  copy(u_c_.begin(), u_c_.end(), NV_DATA_S(domain));
830 
831  // Pass to KINSOL
832  flag = KINSetConstraints(m->mem, domain);
833  casadi_assert(flag==KIN_SUCCESS);
834 
835  // Free the temporary vector
836  N_VDestroy_Serial(domain);
837  }
838 
839  switch (linear_solver_type_) {
840  case KinsolInterface::DENSE:
841  // Dense Jacobian
842  flag = KINDense(m->mem, n_);
843  casadi_assert_message(flag==KIN_SUCCESS, "KINDense");
844 
845  if (exact_jacobian_) {
846  flag = KINDlsSetDenseJacFn(m->mem, djac_wrapper);
847  casadi_assert_message(flag==KIN_SUCCESS, "KINDlsSetDenseJacFn");
848  }
849  break;
850  case KinsolInterface::BANDED:
851  // Banded Jacobian
852  flag = KINBand(m->mem, n_, upper_bandwidth_, lower_bandwidth_);
853  casadi_assert_message(flag==KIN_SUCCESS, "KINBand");
854 
855  if (exact_jacobian_) {
856  flag = KINDlsSetBandJacFn(m->mem, bjac_wrapper);
857  casadi_assert_message(flag==KIN_SUCCESS, "KINDlsBandJacFn");
858  }
859  break;
860  case KinsolInterface::ITERATIVE:
861  // Attach the sparse solver
862  switch (iterative_solver_) {
863  case KinsolInterface::GMRES:
864  flag = KINSpgmr(m->mem, maxl_);
865  casadi_assert_message(flag==KIN_SUCCESS, "KINSpgmr");
866  break;
867  case KinsolInterface::BCGSTAB:
868  flag = KINSpbcg(m->mem, maxl_);
869  casadi_assert_message(flag==KIN_SUCCESS, "KINSpbcg");
870  break;
871  case KinsolInterface::TFQMR:
872  flag = KINSptfqmr(m->mem, maxl_);
873  casadi_assert_message(flag==KIN_SUCCESS, "KINSptfqmr");
874  break;
875  }
876 
877  // Attach functions for Jacobian information
878  if (exact_jacobian_) {
879  flag = KINSpilsSetJacTimesVecFn(m->mem, jtimes_wrapper);
880  casadi_assert_message(flag==KIN_SUCCESS, "KINSpilsSetJacTimesVecFn");
881  }
882 
883  // Add a preconditioner
884  if (use_preconditioner_) {
885  flag = KINSpilsSetPreconditioner(m->mem, psetup_wrapper, psolve_wrapper);
886  casadi_assert(flag==KIN_SUCCESS);
887  }
888  break;
889  case KinsolInterface::USER_DEFINED:
890  // Set fields in the IDA memory
891  KINMem kin_mem = KINMem(m->mem);
892  kin_mem->kin_lmem = m;
893  kin_mem->kin_lsetup = lsetup_wrapper;
894  kin_mem->kin_lsolve = lsolve_wrapper;
895  kin_mem->kin_setupNonNull = TRUE;
896  break;
897  }
898 
899  // Set stop criterion
900  flag = KINSetFuncNormTol(m->mem, abstol_);
901  casadi_assert(flag==KIN_SUCCESS);
902  }
