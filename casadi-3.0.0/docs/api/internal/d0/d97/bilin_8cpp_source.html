79  void Bilin::spFwd(const bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem) {
80  /* Get sparsities */
81  int ncol_A = sparsity().size2();
82  const int *colind_A = dep(0).colind(), *row_A = dep(0).row();
83 
84  /* Return value */
85  bvec_t r=0;
86 
87  /* Loop over the columns of A */
88  int cc, rr, el;
89  for (cc=0; cc<ncol_A; ++cc) {
90  /* Loop over the nonzeros of A */
91  for (el=colind_A[cc]; el<colind_A[cc+1]; ++el) {
92  /* Get the row */
93  rr=row_A[el];
94 
95  /* Add contribution */
96  r |= arg[1][rr] | arg[0][el] | arg[2][cc];
97  }
98  }
99  *res[0] = r;
100  }

102  void Bilin::spAdj(bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem) {
103  /* Get sparsities */
104  int ncol_A = sparsity().size2();
105  const int *colind_A = dep(0).colind(), *row_A = dep(0).row();
106 
107  /* Seed */
108  bvec_t s=*res[0];
109  *res[0] = 0;
110 
111  /* Loop over the columns of A */
112  int cc, rr, el;
113  for (cc=0; cc<ncol_A; ++cc) {
114  /* Loop over the nonzeros of A */
115  for (el=colind_A[cc]; el<colind_A[cc+1]; ++el) {
116  /* Get the row */
117  rr=row_A[el];
118 
119  /* Add contribution */
120  arg[0][el] |= s;
121  arg[1][rr] |= s;
122  arg[2][cc] |= s;
123  }
124  }
125  }
