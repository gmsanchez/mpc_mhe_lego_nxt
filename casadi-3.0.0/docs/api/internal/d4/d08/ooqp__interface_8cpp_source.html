145  void OoqpInterface::
146  eval(void* mem, const double** arg, double** res, int* iw, double* w) const {
147  if (inputs_check_) {
148  checkInputs(arg[QPSOL_LBX], arg[QPSOL_UBX], arg[QPSOL_LBA], arg[QPSOL_UBA]);
149  }
150 
151  // Get problem data
152  double* g=w; w += n_;
153  casadi_copy(arg[QPSOL_G], n_, g);
154  double* lbx=w; w += n_;
155  casadi_copy(arg[QPSOL_LBX], n_, lbx);
156  double* ubx=w; w += n_;
157  casadi_copy(arg[QPSOL_UBX], n_, ubx);
158  double* lba=w; w += nc_;
159  casadi_copy(arg[QPSOL_LBA], nc_, lba);
160  double* uba=w; w += nc_;
161  casadi_copy(arg[QPSOL_UBA], nc_, uba);
162  double* H=w; w += nnz_in(QPSOL_H);
163  casadi_copy(arg[QPSOL_H], nnz_in(QPSOL_H), H);
164  double* A=w; w += nnz_in(QPSOL_A);
165  casadi_copy(arg[QPSOL_A], nnz_in(QPSOL_A), A);
166 
167  // Temporary memory
168  double* c_ = w; w += n_;
169  double* bA_ = w; w += nc_;
170  double* xlow_ = w; w += n_;
171  double* xupp_ = w; w += n_;
172  double* clow_ = w; w += nc_;
173  double* cupp_ = w; w += nc_;
174  double* x_ = w; w += n_;
175  double* gamma_ = w; w += n_;
176  double* phi_ = w; w += n_;
177  double* y_ = w; w += nc_;
178  double* z_ = w; w += nc_;
179  double* lambda_ = w; w += nc_;
180  double* pi_ = w; w += nc_;
181  char* ixlow_ = reinterpret_cast<char*>(iw); iw += n_;
182  char* ixupp_ = reinterpret_cast<char*>(iw); iw += n_;
183  char* iclow_ = reinterpret_cast<char*>(iw); iw += nc_;
184  char* icupp_ = reinterpret_cast<char*>(iw); iw += nc_;
185  double* dQ_ = w; w += nQ_;
186  double* dA_ = w; w += nA_;
187  double* dC_ = w; w += nA_;
188  int* irowQ_ = iw; iw += nQ_;
189  int* jcolQ_ = iw; iw += nQ_;
190  int* irowA_ = iw; iw += nA_;
191  int* jcolA_ = iw; iw += nA_;
192  int* irowC_ = iw; iw += nA_;
193  int* jcolC_ = iw; iw += nA_;
194  int* x_index_ = iw; iw += n_;
195  int* c_index_ = iw; iw += nc_;
196  double* p_ = w; w += n_;
197  double* AT = w; w += nA_;
198 
199  // Parameter contribution to the objective
200  double objParam = 0;
201 
202  // Get the number of free variables and their types
203  int nx = 0, np=0;
204  for (int i=0; i<n_; ++i) {
205  if (lbx[i]==ubx[i]) {
206  // Save parameter
207  p_[np] = lbx[i];
208 
209  // Add contribution to objective
210  objParam += g[i]*p_[np];
211 
212  // Save index
213  x_index_[i] = -1-np++;
214 
215  } else {
216  // True free variable
217  if (lbx[i]==-numeric_limits<double>::infinity()) {
218  xlow_[nx] = 0;
219  ixlow_[nx] = 0;
220  } else {
221  xlow_[nx] = lbx[i];
222  ixlow_[nx] = 1;
223  }
224  if (ubx[i]==numeric_limits<double>::infinity()) {
225  xupp_[nx] = 0;
226  ixupp_[nx] = 0;
227  } else {
228  xupp_[nx] = ubx[i];
229  ixupp_[nx] = 1;
230  }
231  c_[nx] = g[i];
232  x_index_[i] = nx++;
233  }
234  }
235 
236  // Get quadratic term
237  const int* H_colind = sparsity_in(QPSOL_H).colind();
238  const int* H_row = sparsity_in(QPSOL_H).row();
239  int nnzQ = 0;
240  // Loop over the columns of the quadratic term
241  for (int cc=0; cc<n_; ++cc) {
242 
243  // Loop over nonzero elements of the column
244  for (int el=H_colind[cc]; el<H_colind[cc+1]; ++el) {
245 
246  // Only upper triangular part
247  int rr=H_row[el];
248  if (rr>cc) break;
249 
250  // Get variable types
251  int icc=x_index_[cc];
252  int irr=x_index_[rr];
253 
254  if (icc<0) {
255  if (irr<0) {
256  // Add contribution to objective
257  objParam += icc==irr ? H[el]*sq(p_[-1-icc])/2 : H[el]*p_[-1-irr]*p_[-1-icc];
258  } else {
259  // Add contribution to gradient term
260  c_[irr] += H[el]*p_[-1-icc];
261  }
262  } else {
263  if (irr<0) {
264  // Add contribution to gradient term
265  c_[icc] += H[el]*p_[-1-irr];
266  } else {
267  // Add to sparsity pattern
268  irowQ_[nnzQ] = icc; // row-major --> indices swapped
269  jcolQ_[nnzQ] = irr; // row-major --> indices swapped
270  dQ_[nnzQ++] = H[el];
271  }
272  }
273  }
274  }
275 
276  // Get the transpose of the sparsity pattern to be able to loop over the constraints
277  casadi_trans(A, sparsity_in(QPSOL_A), AT, spAT_, iw);
278 
279  // Loop over constraints
280  const int* A_colind = sparsity_in(QPSOL_A).colind();
281  const int* A_row = sparsity_in(QPSOL_A).row();
282  const int* AT_colind = spAT_.colind();
283  const int* AT_row = spAT_.row();
284  int nA=0, nC=0, /*mz=0, */ nnzA=0, nnzC=0;
285  for (int j=0; j<nc_; ++j) {
286  if (lba[j] == -numeric_limits<double>::infinity() &&
287  uba[j] == numeric_limits<double>::infinity()) {
288  // Redundant constraint
289  c_index_[j] = 0;
290  } else if (lba[j]==uba[j]) {
291  // Equality constraint
292  bA_[nA] = lba[j];
293 
294  // Add to A
295  for (int el=AT_colind[j]; el<AT_colind[j+1]; ++el) {
296  int i=AT_row[el];
297  if (x_index_[i]<0) {
298  // Parameter
299  bA_[nA] -= AT[el]*p_[-x_index_[i]-1];
300  } else {
301  // Free variable
302  irowA_[nnzA] = nA;
303  jcolA_[nnzA] = x_index_[i];
304  dA_[nnzA++] = AT[el];
305  }
306  }
307  c_index_[j] = -1-nA++;
308  } else {
309  // Inequality constraint
310  if (lba[j]==-numeric_limits<double>::infinity()) {
311  clow_[nC] = 0;
312  iclow_[nC] = 0;
313  } else {
314  clow_[nC] = lba[j];
315  iclow_[nC] = 1;
316  }
317  if (uba[j]==numeric_limits<double>::infinity()) {
318  cupp_[nC] = 0;
319  icupp_[nC] = 0;
320  } else {
321  cupp_[nC] = uba[j];
322  icupp_[nC] = 1;
323  }
324 
325  // Add to C
326  for (int el=AT_colind[j]; el<AT_colind[j+1]; ++el) {
327  int i=AT_row[el];
328  if (x_index_[i]<0) {
329  // Parameter
330  if (iclow_[nC]==1) clow_[nC] -= AT[el]*p_[-x_index_[i]-1];
331  if (icupp_[nC]==1) cupp_[nC] -= AT[el]*p_[-x_index_[i]-1];
332  } else {
333  // Free variable
334  irowC_[nnzC] = nC;
335  jcolC_[nnzC] = x_index_[i];
336  dC_[nnzC++] = AT[el];
337  }
338  }
339  c_index_[j] = 1+nC++;
340  }
341  }
342 
343  // Reset the solution
344  casadi_fill(x_, n_, 0.);
345  casadi_fill(gamma_, n_, 0.);
346  casadi_fill(phi_, n_, 0.);
347  casadi_fill(y_, nc_, 0.);
348  casadi_fill(z_, nc_, 0.);
349  casadi_fill(lambda_, nc_, 0.);
350  casadi_fill(pi_, nc_, 0.);
351 
352  // Solve the QP
353  double objectiveValue;
354 
355  int ierr;
356  if (false) { // Use C interface
357  // TODO(jgillis): Change to qpsolvehb, see OOQP users guide
358  qpsolvesp(c_, nx,
359  irowQ_, nnzQ, jcolQ_, dQ_,
360  xlow_, ixlow_,
361  xupp_, ixupp_,
362  irowA_, nnzA, jcolA_, dA_,
363  bA_, nA,
364  irowC_, nnzC, jcolC_, dC_,
365  clow_, nC, iclow_,
366  cupp_, icupp_,
367  x_, gamma_, phi_,
368  y_,
369  z_, lambda_, pi_,
370  &objectiveValue,
371  print_level_, &ierr);
372  } else { // Use C++ interface
373  ierr=0;
374  // All OOQP related allocations in evaluate
375 
376  std::vector<int> krowQ(nx+1);
377  std::vector<int> krowA(nA+1);
378  std::vector<int> krowC(nC+1);
379 
380  //int status_code = 0;
381  makehb(irowQ_, nnzQ, get_ptr(krowQ), nx, &ierr);
382  if (ierr == 0) makehb(irowA_, nnzA, get_ptr(krowA), nA, &ierr);
383  if (ierr == 0) makehb(irowC_, nnzC, get_ptr(krowC), nC, &ierr);
384 
385  if (ierr == 0) {
386  QpGenContext ctx;
387 
388  QpGenHbGondzioSetup(c_, nx, get_ptr(krowQ), jcolQ_, dQ_,
389  xlow_, ixlow_, xupp_, ixupp_,
390  get_ptr(krowA), nA, jcolA_, dA_, bA_,
391  get_ptr(krowC), nC, jcolC_, dC_,
392  clow_, iclow_, cupp_, icupp_, &ctx,
393  &ierr);
394  if (ierr == 0) {
395  Solver* solver = static_cast<Solver *>(ctx.solver);
396  gOoqpPrintLevel = print_level_;
397  solver->monitorSelf();
398  solver->setMuTol(mutol_);
399  solver->setMuTol(mutol_);
400 
401  QpGenFinish(&ctx, x_, gamma_, phi_,
402  y_, z_, lambda_, pi_,
403  &objectiveValue, &ierr);
404  }
405 
406  QpGenCleanup(&ctx);
407  }
408  }
409 
410  if (ierr>0) {
411  casadi_warning("Unable to solve problem: " << errFlag(ierr));
412  } else if (ierr<0) {
413  casadi_error("Fatal error: " << errFlag(ierr));
414  }
415 
416  // Retrieve eliminated decision variables
417  for (int i=n_-1; i>=0; --i) {
418  int ii = x_index_[i];
419  if (ii<0) {
420  x_[i] = p_[-1-ii];
421  } else {
422  x_[i] = x_[ii];
423  }
424  }
425 
426  // Retreive eliminated dual variables (linear bounds)
427  for (int j=nc_-1; j>=0; --j) {
428  int jj = c_index_[j];
429  if (jj==0) {
430  lambda_[j] = 0;
431  } else if (jj<0) {
432  lambda_[j] = -y_[-1-jj];
433  } else {
434  lambda_[j] = pi_[-1+jj]-lambda_[-1+jj];
435  }
436  }
437 
438  // Retreive eliminated dual variables (simple bounds)
439  for (int i=n_-1; i>=0; --i) {
440  int ii = x_index_[i];
441  if (ii<0) {
442  // The dual solution for the fixed parameters follows from the KKT conditions
443  gamma_[i] = -g[i];
444  for (int el=H_colind[i]; el<H_colind[i+1]; ++el) {
445  int j=H_row[el];
446  gamma_[i] -= H[el]*x_[j];
447  }
448  for (int el=A_colind[i]; el<A_colind[i+1]; ++el) {
449  int j=A_row[el];
450  gamma_[i] -= A[el]*lambda_[j];
451  }
452  } else {
453  gamma_[i] = phi_[ii]-gamma_[ii];
454  }
455  }
456 
457  // Save optimal cost
458  if (res[QPSOL_COST]) *res[QPSOL_COST] = objectiveValue + objParam;
459 
460  // Save primal solution
461  casadi_copy(x_, n_, res[QPSOL_X]);
462 
463  // Save dual solution (linear bounds)
464  casadi_copy(lambda_, nc_, res[QPSOL_LAM_A]);
465 
466  // Save dual solution (simple bounds)
467  casadi_copy(gamma_, n_, res[QPSOL_LAM_X]);
468  }
