84  void CollocationIntegrator::setupFG() {
85 
86  // All collocation time points
87  std::vector<double> tau_root = collocation_points(deg_, collocation_scheme_);
88  tau_root.insert(tau_root.begin(), 0);
89 
90  // Coefficients of the collocation equation
91  vector<vector<double> > C(deg_+1, vector<double>(deg_+1, 0));
92 
93  // Coefficients of the continuity equation
94  vector<double> D(deg_+1, 0);
95 
96  // Coefficients of the quadratures
97  vector<double> B(deg_+1, 0);
98 
99  // For all collocation points
100  for (int j=0; j<deg_+1; ++j) {
101 
102  // Construct Lagrange polynomials to get the polynomial basis at the collocation point
103  Polynomial p = 1;
104  for (int r=0; r<deg_+1; ++r) {
105  if (r!=j) {
106  p *= Polynomial(-tau_root[r], 1)/(tau_root[j]-tau_root[r]);
107  }
108  }
109 
110  // Evaluate the polynomial at the final time to get the
111  // coefficients of the continuity equation
112  if (collocation_scheme_=="radau") {
113  D[j] = j==deg_ ? 1 : 0;
114  } else {
115  D[j] = p(1.0);
116  }
117 
118  // Evaluate the time derivative of the polynomial at all collocation points to
119  // get the coefficients of the continuity equation
120  Polynomial dp = p.derivative();
121  for (int r=0; r<deg_+1; ++r) {
122  C[j][r] = dp(tau_root[r]);
123  }
124 
125  // Integrate polynomial to get the coefficients of the quadratures
126  Polynomial ip = p.anti_derivative();
127  B[j] = ip(1.0);
128  }
129 
130  // Symbolic inputs
131  MX x0 = MX::sym("x0", this->x());
132  MX p = MX::sym("p", this->p());
133  MX t = MX::sym("t", this->t());
134 
135  // Implicitly defined variables (z and x)
136  MX v = MX::sym("v", deg_*(nx_+nz_));
137  vector<int> v_offset(1, 0);
138  for (int d=0; d<deg_; ++d) {
139  v_offset.push_back(v_offset.back()+nx_);
140  v_offset.push_back(v_offset.back()+nz_);
141  }
142  vector<MX> vv = vertsplit(v, v_offset);
143  vector<MX>::const_iterator vv_it = vv.begin();
144 
145  // Collocated states
146  vector<MX> x(deg_+1), z(deg_+1);
147  for (int d=1; d<=deg_; ++d) {
148  x[d] = reshape(*vv_it++, size_in(INTEGRATOR_X0));
149  z[d] = reshape(*vv_it++, size_in(INTEGRATOR_Z0));
150  }
151  casadi_assert(vv_it==vv.end());
152 
153  // Collocation time points
154  vector<MX> tt(deg_+1);
155  for (int d=0; d<=deg_; ++d) {
156  tt[d] = t + h_*tau_root[d];
157  }
158 
159  // Equations that implicitly define v
160  vector<MX> eq;
161 
162  // Quadratures
163  MX qf = MX::zeros(this->q());
164 
165  // End state
166  MX xf = D[0]*x0;
167 
168  // For all collocation points
169  for (int j=1; j<deg_+1; ++j) {
170  //for (int j=deg_; j>=1; --j) {
171 
172  // Evaluate the DAE
173  vector<MX> f_arg(DAE_NUM_IN);
174  f_arg[DAE_T] = tt[j];
175  f_arg[DAE_P] = p;
176  f_arg[DAE_X] = x[j];
177  f_arg[DAE_Z] = z[j];
178  vector<MX> f_res = f_(f_arg);
179 
180  // Get an expression for the state derivative at the collocation point
181  MX xp_j = C[0][j] * x0;
182  for (int r=1; r<deg_+1; ++r) {
183  xp_j += C[r][j] * x[r];
184  }
185 
186  // Add collocation equation
187  eq.push_back(vec(h_*f_res[DAE_ODE] - xp_j));
188 
189  // Add the algebraic conditions
190  eq.push_back(vec(f_res[DAE_ALG]));
191 
192  // Add contribution to the final state
193  xf += D[j]*x[j];
194 
195  // Add contribution to quadratures
196  qf += (B[j]*h_)*f_res[DAE_QUAD];
197  }
198 
199  // Form forward discrete time dynamics
200  vector<MX> F_in(DAE_NUM_IN);
201  F_in[DAE_T] = t;
202  F_in[DAE_X] = x0;
203  F_in[DAE_P] = p;
204  F_in[DAE_Z] = v;
205  vector<MX> F_out(DAE_NUM_OUT);
206  F_out[DAE_ODE] = xf;
207  F_out[DAE_ALG] = vertcat(eq);
208  F_out[DAE_QUAD] = qf;
209  F_ = Function("dae", F_in, F_out);
210  alloc(F_);
211 
212  // Backwards dynamics
213  // NOTE: The following is derived so that it will give the exact adjoint
214  // sensitivities whenever g is the reverse mode derivative of f.
215  if (!g_.is_null()) {
216 
217  // Symbolic inputs
218  MX rx0 = MX::sym("rx0", this->rx());
219  MX rp = MX::sym("rp", this->rp());
220 
221  // Implicitly defined variables (rz and rx)
222  MX rv = MX::sym("v", deg_*(nrx_+nrz_));
223  vector<int> rv_offset(1, 0);
224  for (int d=0; d<deg_; ++d) {
225  rv_offset.push_back(rv_offset.back()+nrx_);
226  rv_offset.push_back(rv_offset.back()+nrz_);
227  }
228  vector<MX> rvv = vertsplit(rv, rv_offset);
229  vector<MX>::const_iterator rvv_it = rvv.begin();
230 
231  // Collocated states
232  vector<MX> rx(deg_+1), rz(deg_+1);
233  for (int d=1; d<=deg_; ++d) {
234  rx[d] = reshape(*rvv_it++, this->rx().size());
235  rz[d] = reshape(*rvv_it++, this->rz().size());
236  }
237  casadi_assert(rvv_it==rvv.end());
238 
239  // Equations that implicitly define v
240  eq.clear();
241 
242  // Quadratures
243  MX rqf = MX::zeros(this->rq());
244 
245  // End state
246  MX rxf = D[0]*rx0;
247 
248  // For all collocation points
249  for (int j=1; j<deg_+1; ++j) {
250 
251  // Evaluate the backward DAE
252  vector<MX> g_arg(RDAE_NUM_IN);
253  g_arg[RDAE_T] = tt[j];
254  g_arg[RDAE_P] = p;
255  g_arg[RDAE_X] = x[j];
256  g_arg[RDAE_Z] = z[j];
257  g_arg[RDAE_RX] = rx[j];
258  g_arg[RDAE_RZ] = rz[j];
259  g_arg[RDAE_RP] = rp;
260  vector<MX> g_res = g_(g_arg);
261 
262  // Get an expression for the state derivative at the collocation point
263  MX rxp_j = -D[j]*rx0;
264  for (int r=1; r<deg_+1; ++r) {
265  rxp_j += (B[r]*C[j][r]) * rx[r];
266  }
267 
268  // Add collocation equation
269  eq.push_back(vec(h_*B[j]*g_res[RDAE_ODE] - rxp_j));
270 
271  // Add the algebraic conditions
272  eq.push_back(vec(g_res[RDAE_ALG]));
273 
274  // Add contribution to the final state
275  rxf += -B[j]*C[0][j]*rx[j];
276 
277  // Add contribution to quadratures
278  rqf += h_*B[j]*g_res[RDAE_QUAD];
279  }
280 
281  // Form backward discrete time dynamics
282  vector<MX> G_in(RDAE_NUM_IN);
283  G_in[RDAE_T] = t;
284  G_in[RDAE_X] = x0;
285  G_in[RDAE_P] = p;
286  G_in[RDAE_Z] = v;
287  G_in[RDAE_RX] = rx0;
288  G_in[RDAE_RP] = rp;
289  G_in[RDAE_RZ] = rv;
290  vector<MX> G_out(RDAE_NUM_OUT);
291  G_out[RDAE_ODE] = rxf;
292  G_out[RDAE_ALG] = vertcat(eq);
293  G_out[RDAE_QUAD] = rqf;
294  G_ = Function("rdae", G_in, G_out);
295  alloc(G_);
296  }
297  }

299  void CollocationIntegrator::reset(IntegratorMemory* mem, double t, const double* x,
300  const double* z, const double* p) const {
301  auto m = static_cast<FixedStepMemory*>(mem);
302 
303  // Reset the base classes
304  ImplicitFixedStepIntegrator::reset(mem, t, x, z, p);
305 
306  // Initial guess for Z
307  double* Z = m->Z.ptr();
308  for (int d=0; d<deg_; ++d) {
309  casadi_copy(x, nx_, Z);
310  Z += nx_;
311  casadi_copy(z, nz_, Z);
312  Z += nz_;
313  }
