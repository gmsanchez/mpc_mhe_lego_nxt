33  template<typename XType>
34  Function XOracle<XType>::create(const std::string& fname,
35  const std::vector<std::string>& s_in,
36  const std::vector<std::string>& s_out,
37  const std::vector<LinComb>& lincomb,
38  const Dict& opts) const {
39  // Create maps for all inputs and outputs
40  map<string, XType> map_in, map_out;
41 
42  // Non-differentiated inputs
43  for (int i=0; i<in_.size(); ++i) {
44  map_in[ischeme_[i]] = in_[i];
45  }
46 
47  // Non-differentiated outputs
48  for (int i=0; i<out_.size(); ++i) {
49  map_out[oscheme_[i]] = out_[i];
50  }
51 
52  // Dual variables
53  for (int i=0; i<out_.size(); ++i) {
54  string dual_name = "lam_" + oscheme_[i];
55  map_in[dual_name] = XType::sym(dual_name, out_[i].sparsity());
56  }
57 
58  // Add linear combinations
59  for (auto i : lincomb) {
60  XType lc = 0;
61  for (auto j : i.second) {
62  lc += dot(map_in.at("lam_" + j), map_out.at(j));
63  }
64  map_out[i.first] = lc;
65  }
66 
67  // Inputs of function being assembled
68  vector<XType> ret_in(s_in.size());
69 
70  // Handle inputs
71  for (int i=0; i<ret_in.size(); ++i) {
72  ret_in[i] = map_in.at(s_in[i]);
73  }
74 
75  // Outputs of function being assembled
76  vector<XType> ret_out(s_out.size());
77 
78  // List of valid attributes
79  const vector<string> all_attr = {"transpose", "triu", "tril", "densify", "sym", "withdiag"};
80 
81  // Handle outputs
82  for (int i=0; i<ret_out.size(); ++i) {
83  XType& r = ret_out[i];
84 
85  // Output name
86  string s = s_out[i];
87 
88  // Separarate attributes
89  vector<string> attr;
90  while (true) {
91  // Find the first underscore separator
92  size_t pos = s.find('_');
93  if (pos>=s.size()) break; // No more underscore
94  string a = s.substr(0, pos);
95 
96  // Try to match with attributes
97  auto it = std::find(all_attr.begin(), all_attr.end(), a);
98  if (it==all_attr.end()) break; // No more attribute
99 
100  // Save attribute and strip from s
101  attr.push_back(*it);
102  s = s.substr(pos+1, string::npos);
103  }
104 
105  // Try to locate in list of outputs
106  auto it = map_out.find(s);
107  if (it!=map_out.end()) {
108  // Non-differentiated output
109  r = it->second;
110  } else {
111  // Must be an operator
112  size_t pos = s.find('_');
113  casadi_assert_message(pos<s.size(), s_out[i] + " is not an output or operator");
114  string op = s.substr(0, pos);
115  s = s.substr(pos+1);
116 
117  // Handle different types of operators
118  if (op=="grad" || op=="jac" || op=="hess") {
119  // Output
120  pos = s.find('_');
121  casadi_assert_message(pos<s.size(), s_out[i] + " is ill-posed");
122  string res = s.substr(0, pos);
123  auto res_i = map_out.find(res);
124  casadi_assert_message(res_i!=map_out.end(),
125  "Unrecognized output " + res + " in " + s_out[i]);
126  s = s.substr(pos+1);
127 
128  // Input
129  string arg;
130  if (op=="hess") {
131  pos = s.find('_');
132  casadi_assert_message(pos<s.size(), s_out[i] + " is ill-posed");
133  arg = s.substr(0, pos);
134  s = s.substr(pos+1);
135  casadi_assert_message(s==arg, "Mixed Hessian terms not supported");
136  } else {
137  arg = s;
138  }
139  auto arg_i = map_in.find(arg);
140  casadi_assert_message(arg_i!=map_in.end(),
141  "Unrecognized input " + arg + " in " + s_out[i]);
142 
143  // Calculate gradient or Jacobian
144  if (op=="jac") {
145  r = jacobian(res_i->second, arg_i->second);
146  } else if (op=="grad") {
147  r = project(gradient(res_i->second, arg_i->second), arg_i->second.sparsity());
148  } else {
149  casadi_assert(op=="hess");
150  r = triu(hessian(res_i->second, arg_i->second));
151  }
152  } else {
153  casadi_error("Unknown operator: " + op);
154  }
155  }
156 
157  // Apply attributes (starting from the right-most one)
158  for (auto a=attr.rbegin(); a!=attr.rend(); ++a) {
159  if (*a=="transpose") {
160  r = r = r.T();
161  } else if (*a=="triu") {
162  r = triu(r);
163  } else if (*a=="tril") {
164  r = tril(r);
165  } else if (*a=="densify") {
166  r = densify(r);
167  } else if (*a=="sym") {
168  r = triu2symm(r);
169  } else if (*a=="withdiag") {
170  r = project(r, r.sparsity() + Sparsity::diag(r.size1()));
171  } else {
172  casadi_assert(0);
173  }
174  }
175  }
176 
177  // Create function and return
178  Function ret(fname, ret_in, ret_out, s_in, s_out, opts);
179  if (ret.has_free()) {
180  stringstream ss;
181  ss << "Cannot generate " << fname << " as the expressions contain free variables: ";
182  if (ret.is_a("sxfunction")) {
183  ss << ret.free_sx();
184  } else {
185  casadi_assert(ret.is_a("mxfunction"));
186  ss << ret.free_mx();
187  }
188  casadi_error(ss.str());
189  }
190  return ret;
191  }

194  std::vector<bool>
195  XOracle<XType>::nl_var(const std::string& s_in,
196  const std::vector<std::string>& s_out) const {
197  // Input arguments
198  auto it = find(ischeme_.begin(), ischeme_.end(), s_in);
199  casadi_assert(it!=ischeme_.end());
200  XType arg = in_.at(it-ischeme_.begin());
201 
202  // Output arguments
203  vector<XType> res;
204  for (auto&& s : s_out) {
205  it = find(oscheme_.begin(), oscheme_.end(), s);
206  casadi_assert(it!=oscheme_.end());
207  res.push_back(out_.at(it-oscheme_.begin()));
208  }
209 
210  // Extract variables entering nonlinearly
211  return XType::nl_var(veccat(res), arg);
212  }
