86  void ClangCompiler::init(const Dict& opts) {
87  // Base class
88  CompilerInternal::init(opts);
89 
90  // Read options
91  for (auto&& op : opts) {
92  if (op.first=="include_path") {
93  include_path_ = op.second.to_string();
94  } else if (op.first=="flags") {
95  flags_ = op.second;
96  }
97  }
98 
99  // Arguments to pass to the clang frontend
100  vector<const char *> args(1, name_.c_str());
101  for (auto&& f : flags_) {
102  args.push_back(f.c_str());
103  }
104 
105  // Create the compiler instance
106  clang::CompilerInstance compInst;
107 
108  // A symbol in the DLL
109  void *addr = reinterpret_cast<void*>(&casadi_register_compiler_clang);
110 
111  // Get runtime include path
112  std::string jit_include, filesep;
113 #ifdef _WIN32
114  char buffer[MAX_PATH];
115  HMODULE hm = NULL;
116  if (!GetModuleHandleExA(GET_MODULE_HANDLE_EX_FLAG_FROM_ADDRESS |
117  GET_MODULE_HANDLE_EX_FLAG_UNCHANGED_REFCOUNT,
118  (LPCSTR)addr, &hm)) {
119  casadi_error("GetModuleHandle failed");
120  }
121  GetModuleFileNameA(hm, buffer, sizeof(buffer));
122  PathRemoveFileSpecA(buffer);
123  jit_include = buffer;
124  filesep = "\\";
125 #else // _WIN32
126  Dl_info dl_info;
127  if (!dladdr(addr, &dl_info)) {
128  casadi_error("dladdr failed");
129  }
130  jit_include = dl_info.dli_fname;
131  jit_include = jit_include.substr(0, jit_include.find_last_of('/'));
132  filesep = "/";
133 #endif // _WIN32
134  jit_include += filesep + "casadi" + filesep + "jit";
135 
136 #if 0
137  // Initialize target info with the default triple for our platform.
138  auto targetoptions = std::make_shared<clang::Taroptions>();
139  targetoptions->Triple = llvm::sys::getDefaultTargetTriple();
140  clang::TargetInfo *targetInfo =
141  clang::TargetInfo::CreateTargetInfo(compInst.get_diagnostics(), targetoptions);
142  compInst.setTarget(targetInfo);
143 #endif
144 
145  // The compiler invocation needs a DiagnosticsEngine so it can report problems
146  clang::DiagnosticOptions* diagOpts = new clang::DiagnosticOptions();
147  myerr_ = new llvm::raw_os_ostream(userOut<true>());
148  clang::TextDiagnosticPrinter *diagClient = new clang::TextDiagnosticPrinter(*myerr_, diagOpts);
149 
150  clang::DiagnosticIDs* diagID = new clang::DiagnosticIDs();
151  // This object takes ownerships of all three passed-in pointers
152  clang::DiagnosticsEngine diags(diagID, diagOpts, diagClient);
153 
154  // Create the compiler invocation
155  clang::CompilerInvocation* compInv = new clang::CompilerInvocation();
156  clang::CompilerInvocation::CreateFromArgs(*compInv, &args[0],
157  &args[0] + args.size(), diags);
158  compInst.setInvocation(compInv);
159 
160  // Get ready to report problems
161  compInst.createDiagnostics();
162  if (!compInst.hasDiagnostics())
163  casadi_error("Cannot create diagnostics");
164 
165  // Set resource directory
166  std::string resourcedir = jit_include + filesep + "clang" + filesep + CLANG_VERSION_STRING;
167  compInst.getHeaderSearchOpts().ResourceDir = resourcedir;
168 
169  // Read the system includes (C or C++)
170  vector<pair<string, bool> > system_include = getIncludes("system_includes.txt", jit_include);
171  for (auto i=system_include.begin(); i!=system_include.end(); ++i) {
172  compInst.getHeaderSearchOpts().AddPath(i->first,
173  clang::frontend::System, i->second, false);
174  }
175 
176  // Read the system includes (C only)
177  system_include = getIncludes("csystem_includes.txt", jit_include);
178  for (auto i=system_include.begin(); i!=system_include.end(); ++i) {
179  compInst.getHeaderSearchOpts().AddPath(i->first,
180  clang::frontend::CSystem, i->second, false);
181  }
182 
183  // Read the system includes (C++ only)
184  system_include = getIncludes("cxxsystem_includes.txt", jit_include);
185  for (auto i=system_include.begin(); i!=system_include.end(); ++i) {
186  compInst.getHeaderSearchOpts().AddPath(i->first,
187  clang::frontend::CXXSystem, i->second, false);
188  }
189 
190 #ifdef _WIN32
191  char pathsep = ';';
192 #else
193  char pathsep = ':';
194 #endif
195 
196  // Search path
197  std::stringstream paths;
198  paths << include_path_ << pathsep;
199  std::string path;
200  while (std::getline(paths, path, pathsep)) {
201  compInst.getHeaderSearchOpts().AddPath(path.c_str(), clang::frontend::System, false, false);
202  }
203 
204  // Create an LLVM context (NOTE: should use a static context instead?)
205  context_ = new llvm::LLVMContext();
206 
207  // Create an action and make the compiler instance carry it out
208  act_ = new clang::EmitLLVMOnlyAction(context_);
209  if (!compInst.ExecuteAction(*act_))
210  casadi_error("Cannot execute action");
211 
212  // Grab the module built by the EmitLLVMOnlyAction
213  #if LLVM_VERSION_MAJOR>=3 && LLVM_VERSION_MINOR>=5
214  std::unique_ptr<llvm::Module> module = act_->takeModule();
215  module_ = module.get();
216  #else
217  llvm::Module* module = act_->takeModule();
218  module_ = module;
219  #endif
220 
221  llvm::InitializeNativeTarget();
222  llvm::InitializeNativeTargetAsmPrinter();
223 
224  // Create the JIT. This takes ownership of the module.
225  std::string ErrStr;
226  executionEngine_ =
227  llvm::EngineBuilder(std::move(module)).setEngineKind(llvm::EngineKind::JIT)
228  .setErrorStr(&ErrStr).create();
229  if (!executionEngine_) {
230  casadi_error("Could not create ExecutionEngine: " << ErrStr);
231  }
232 
233  executionEngine_->finalizeObject();
234  }

245  std::vector<std::pair<std::string, bool> > ClangCompiler::
246  getIncludes(const std::string& file, const std::string& path) {
247  // File separator
248 #ifdef _WIN32
249  const char sep = '\\';
250 #else // _WIN32
251  const char sep = '/';
252 #endif // _WIN32
253 
254  // Return value
255  vector<pair<string, bool> > ret;
256 
257  // Read line-by-line
258  std::ifstream setup_file(path + sep + file);
259  std::string line;
260  while (std::getline(setup_file, line)) {
261  // Skip empty lines
262  if (line.empty()) continue;
263 
264  // Check if framework
265  size_t loc = line.find(" (framework directory)");
266  bool isframework = loc != string::npos;
267  if (isframework) {
268  // Truncate path
269  line = line.substr(0, loc);
270  }
271 
272  // Check if the path is absolute or relative
273 #ifdef _WIN32
274  bool relative = PathIsRelative(TEXT(line.c_str()));
275 #else // _WIN32
276  bool relative = line.at(0)!=sep;
277 #endif // _WIN32
278 
279  if (relative) {
280  // Relative path, make absolute
281  ret.push_back(make_pair(path + sep + line, isframework));
282  } else {
283  // Absolute path
284  ret.push_back(make_pair(line, isframework));
285  }
286  }
287 
288  return ret;
289  }
