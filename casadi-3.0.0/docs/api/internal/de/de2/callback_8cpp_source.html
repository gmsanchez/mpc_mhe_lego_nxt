62  void Callback::eval(const double** arg, double** res, int* iw, double* w, int mem) {
63  // Allocate input matrices
64  int n_in = this->n_in();
65  std::vector<DM> argv(n_in);
66  for (int i=0; i<n_in; ++i) {
67  argv[i] = DM(sparsity_in(i));
68  casadi_copy(arg[i], argv[i].nnz(), argv[i].ptr());
69  }
70 
71  // Evaluate
72  std::vector<DM> resv = eval(argv);
73 
74  // Get the outputs
75  int n_out = this->n_out();
76  for (int i=0; i<n_out; ++i) {
77  casadi_copy(resv[i].ptr(), resv[i].nnz(), res[i]);
78  }
79  }

137  void Callback::transfer_ownership() {
138  casadi_assert_message(!is_null(), "Null pointer.");
139  casadi_assert_message(!(*this)->own_, "Ownership has already been transferred.");
140  casadi_assert_message(getCount()>1, "There are no owning references");
141  // Decrease the reference counter to offset the effect of the owning reference
142  count_down();
143  // Mark internal class as owning
144  (*this)->own_ = true;
145  }
