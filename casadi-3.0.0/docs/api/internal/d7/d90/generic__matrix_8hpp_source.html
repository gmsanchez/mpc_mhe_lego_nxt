764  template<typename MatType>
765  MatType GenericMatrix<MatType>::linspace(const MatType& a, const MatType& b, int nsteps) {
766  std::vector<MatType> ret(nsteps);
767  ret[0] = a;
768  MatType step = (b-a)/(nsteps-1);
769 
770  for (int i=1; i<nsteps-1; ++i)
771  ret[i] = ret[i-1] + step;
772 
773  ret[nsteps-1] = b;
774  return vertcat(ret);
775  }

777  template<typename MatType>
778  MatType GenericMatrix<MatType>::cross(const MatType& a, const MatType& b, int dim) {
779  casadi_assert_message(a.size1()==b.size1() && a.size2()==b.size2(),
780  "cross(a, b): Inconsistent dimensions. Dimension of a ("
781  << a.dim() << " ) must equal that of b ("
782  << b.dim() << ").");
783 
784  casadi_assert_message(a.size1()==3 || a.size2()==3,
785  "cross(a, b): One of the dimensions of a should have length 3, but got "
786  << a.dim() << ".");
787  casadi_assert_message(dim==-1 || dim==1 || dim==2,
788  "cross(a, b, dim): Dim must be 1, 2 or -1 (automatic).");
789 
790  std::vector<MatType> ret(3);
791 
792  bool t = a.size1()==3;
793 
794  if (dim==1) t = true;
795  if (dim==2) t = false;
796 
797  MatType a1 = t ? a(0, Slice()) : a(Slice(), 0);
798  MatType a2 = t ? a(1, Slice()) : a(Slice(), 1);
799  MatType a3 = t ? a(2, Slice()) : a(Slice(), 2);
800 
801  MatType b1 = t ? b(0, Slice()) : b(Slice(), 0);
802  MatType b2 = t ? b(1, Slice()) : b(Slice(), 1);
803  MatType b3 = t ? b(2, Slice()) : b(Slice(), 2);
804 
805  ret[0] = a2*b3-a3*b2;
806  ret[1] = a3*b1-a1*b3;
807  ret[2] = a1*b2-a2*b1;
808 
809  return t ? vertcat(ret) : horzcat(ret);
810  }

843  template<typename MatType>
844  MatType GenericMatrix<MatType>::repsum(const MatType& x, int n, int m) {
845  casadi_assert(x.size1() % n==0);
846  casadi_assert(x.size2() % m==0);
847  std::vector< std::vector< MatType> > s =
848  blocksplit(x, x.size1()/n, x.size2()/m);
849  MatType sum = 0;
850  for (int i=0;i<s.size();++i) {
851  for (int j=0;j<s[i].size();++j) {
852  sum = sum + s[i][j];
853  }
854  }
855  return sum;
856  }
