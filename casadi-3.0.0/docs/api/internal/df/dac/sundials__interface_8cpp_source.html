169  void SundialsInterface::init(const Dict& opts) {
170  // Call the base class method
171  Integrator::init(opts);
172 
173  // Default options
174  abstol_ = 1e-8;
175  reltol_ = 1e-6;
176  exact_jacobian_ = true;
177  max_num_steps_ = 10000;
178  finite_difference_fsens_ = false;
179  stop_at_end_ = true;
180  use_preconditioner_ = false;
181  max_krylov_ = 10;
182  string linear_solver_type = "dense";
183  string iterative_solver = "gmres";
184  string pretype = "none";
185  string linear_solver;
186  Dict linear_solver_options;
187  upper_bandwidth_ = -1;
188  lower_bandwidth_ = -1;
189  upper_bandwidthB_ = -1;
190  lower_bandwidthB_ = -1;
191  quad_err_con_ = false;
192  interpolation_type_ = "hermite";
193  steps_per_checkpoint_ = 20;
194  disable_internal_warnings_ = false;
195  max_multistep_order_ = 5;
196 
197  // Read options
198  for (auto&& op : opts) {
199  if (op.first=="abstol") {
200  abstol_ = op.second;
201  } else if (op.first=="reltol") {
202  reltol_ = op.second;
203  } else if (op.first=="exact_jacobian") {
204  exact_jacobian_ = op.second;
205  } else if (op.first=="max_num_steps") {
206  max_num_steps_ = op.second;
207  } else if (op.first=="finite_difference_fsens") {
208  finite_difference_fsens_ = op.second;
209  } else if (op.first=="stop_at_end") {
210  stop_at_end_ = op.second;
211  } else if (op.first=="use_preconditioner") {
212  use_preconditioner_ = op.second;
213  } else if (op.first=="max_krylov") {
214  max_krylov_ = op.second;
215  } else if (op.first=="linear_solver_type") {
216  linear_solver_type = op.second.to_string();
217  } else if (op.first=="iterative_solver") {
218  iterative_solver = op.second.to_string();
219  } else if (op.first=="pretype") {
220  pretype = op.second.to_string();
221  } else if (op.first=="linear_solver") {
222  linear_solver = op.second.to_string();
223  } else if (op.first=="linear_solver_options") {
224  linear_solver_options = op.second;
225  } else if (op.first=="upper_bandwidth") {
226  upper_bandwidth_ = op.second;
227  } else if (op.first=="lower_bandwidth") {
228  lower_bandwidth_ = op.second;
229  } else if (op.first=="upper_bandwidthB") {
230  upper_bandwidthB_ = op.second;
231  } else if (op.first=="lower_bandwidthB") {
232  lower_bandwidthB_ = op.second;
233  } else if (op.first=="quad_err_con") {
234  quad_err_con_ = op.second;
235  } else if (op.first=="interpolation_type") {
236  interpolation_type_ = op.second.to_string();
237  } else if (op.first=="steps_per_checkpoint") {
238  steps_per_checkpoint_ = op.second;
239  } else if (op.first=="disable_internal_warnings") {
240  disable_internal_warnings_ = op.second;
241  } else if (op.first=="max_multistep_order") {
242  max_multistep_order_ = op.second;
243  }
244  }
245 
246  // Default dependent options
247  exact_jacobianB_ = exact_jacobian_;
248  fsens_abstol_ = abstol_;
249  fsens_reltol_ = reltol_;
250  abstolB_ = abstol_;
251  reltolB_ = reltol_;
252  use_preconditionerB_ = use_preconditioner_;
253  max_krylovB_ = max_krylov_;
254  std::string linear_solver_typeB = linear_solver_type;
255  std::string iterative_solverB = iterative_solver;
256  std::string pretypeB = pretype;
257  string linear_solverB = linear_solver;
258  Dict linear_solver_optionsB = linear_solver_options;
259 
260  // Read options again
261  for (auto&& op : opts) {
262  if (op.first=="exact_jacobianB") {
263  exact_jacobianB_ = op.second;
264  } else if (op.first=="fsens_abstol") {
265  fsens_abstol_ = op.second;
266  } else if (op.first=="fsens_reltol") {
267  fsens_reltol_ = op.second;
268  } else if (op.first=="abstolB") {
269  abstolB_ = op.second;
270  } else if (op.first=="reltolB") {
271  reltolB_ = op.second;
272  } else if (op.first=="use_preconditionerB") {
273  use_preconditionerB_ = op.second;
274  } else if (op.first=="max_krylovB") {
275  max_krylovB_ = op.second;
276  } else if (op.first=="linear_solver_typeB") {
277  linear_solver_typeB = op.second.to_string();
278  } else if (op.first=="iterative_solverB") {
279  iterative_solverB = op.second.to_string();
280  } else if (op.first=="pretypeB") {
281  pretypeB = op.second.to_string();
282  } else if (op.first=="linear_solverB") {
283  linear_solverB = op.second.to_string();
284  } else if (op.first=="linear_solver_optionsB") {
285  linear_solver_optionsB = op.second;
286  }
287  }
288 
289  // No Jacobian of g if g doesn't exist
290  if (g_.is_null()) {
291  exact_jacobianB_ = false;
292  }
293 
294  // Linear solver for forward integration
295  if (linear_solver_type=="dense") {
296  linsol_f_ = SD_DENSE;
297  } else if (linear_solver_type=="banded") {
298  linsol_f_ = SD_BANDED;
299  } else if (linear_solver_type=="iterative") {
300  linsol_f_ = SD_ITERATIVE;
301 
302  // Iterative solver
303  if (iterative_solver=="gmres") {
304  itsol_f_ = SD_GMRES;
305  } else if (iterative_solver=="bcgstab") {
306  itsol_f_ = SD_BCGSTAB;
307  } else if (iterative_solver=="tfqmr") {
308  itsol_f_ = SD_TFQMR;
309  } else {
310  casadi_error("Unknown iterative solver for forward integration: " + iterative_solver);
311  }
312 
313  // Preconditioning type
314  if (pretype=="none") {
315  pretype_f_ = PREC_NONE;
316  } else if (pretype=="left") {
317  pretype_f_ = PREC_LEFT;
318  } else if (pretype=="right") {
319  pretype_f_ = PREC_RIGHT;
320  } else if (pretype=="both") {
321  pretype_f_ = PREC_BOTH;
322  } else {
323  casadi_error("Unknown preconditioning type for forward integration: " + pretype);
324  }
325  } else if (linear_solver_type=="user_defined") {
326  linsol_f_ = SD_USER_DEFINED;
327  } else {
328  casadi_error("Unknown linear solver for forward integration: " + linear_solver_type);
329  }
330 
331  // Linear solver for backward integration
332  if (linear_solver_typeB=="dense") {
333  linsol_g_ = SD_DENSE;
334  } else if (linear_solver_typeB=="banded") {
335  linsol_g_ = SD_BANDED;
336  } else if (linear_solver_typeB=="iterative") {
337  linsol_g_ = SD_ITERATIVE;
338 
339  // Iterative solver
340  if (iterative_solverB=="gmres") {
341  itsol_g_ = SD_GMRES;
342  } else if (iterative_solverB=="bcgstab") {
343  itsol_g_ = SD_BCGSTAB;
344  } else if (iterative_solverB=="tfqmr") {
345  itsol_g_ = SD_TFQMR;
346  } else {
347  casadi_error("Unknown sparse solver for backward integration: " + iterative_solverB);
348  }
349 
350  // Preconditioning type
351  if (pretypeB=="none") {
352  pretype_g_ = PREC_NONE;
353  } else if (pretypeB=="left") {
354  pretype_g_ = PREC_LEFT;
355  } else if (pretypeB=="right") {
356  pretype_g_ = PREC_RIGHT;
357  } else if (pretypeB=="both") {
358  pretype_g_ = PREC_BOTH;
359  } else {
360  casadi_error("Unknown preconditioning type for backward integration: " + pretypeB);
361  }
362  } else if (linear_solver_typeB=="user_defined") {
363  linsol_g_ = SD_USER_DEFINED;
364  } else {
365  casadi_error("Unknown linear solver for backward integration: " + iterative_solverB);
366  }
367 
368  // Create a Jacobian if requested
369  if (exact_jacobian_) {
370  jac_ = getJac();
371  alloc(jac_);
372  alloc_w(jac_.nnz_out(0), true);
373  }
374 
375  if (!jac_.is_null()) {
376  casadi_assert_message(jac_.size2_out(0)==jac_.size1_out(0),
377  "SundialsInterface::init: the jacobian of the forward problem must "
378  "be square but got " << jac_.sparsity_out(0).dim());
379 
380  casadi_assert_message(!jac_.sparsity_out(0).is_singular(),
381  "SundialsInterface::init: singularity - the jacobian of the forward "
382  "problem is structurally rank-deficient. sprank(J)="
383  << sprank(jac_.sparsity_out(0)) << " (in stead of "<< jac_.size2_out(0)
384  << ")");
385  }
386 
387  // Create a backwards Jacobian if requested
388  if (exact_jacobianB_ && !g_.is_null()) jacB_ = getJacB();
389 
390  if (!jacB_.is_null()) {
391  alloc(jacB_);
392  alloc_w(jacB_.nnz_out(0), true);
393  casadi_assert_message(jacB_.size2_out(0)==jacB_.size1_out(0),
394  "SundialsInterface::init: the jacobian of the backward problem must be "
395  "square but got " << jacB_.sparsity_out(0).dim());
396 
397  casadi_assert_message(!jacB_.sparsity_out(0).is_singular(),
398  "SundialsInterface::init: singularity - the jacobian of the backward"
399  " problem is structurally rank-deficient. sprank(J)="
400  << sprank(jacB_.sparsity_out(0)) << " (instead of "
401  << jacB_.size2_out(0) << ")");
402  }
403 
404  // Create a linear solver
405  if (!linear_solver.empty() && !jac_.is_null()) {
406  linsol_ = linsol("linsol", linear_solver, jac_.sparsity_out(0),
407  1, linear_solver_options);
408  alloc(linsol_);
409  }
410 
411  // Create a linear solver
412  if (!linear_solverB.empty() && !jacB_.is_null()) {
413  linsolB_ = linsol("linsolB", linear_solverB, jacB_.sparsity_out(0),
414  1, linear_solver_optionsB);
415  alloc(linsolB_);
416  }

438  void SundialsInterface::reset(IntegratorMemory* mem, double t, const double* x,
439  const double* z, const double* p) const {
440  auto m = static_cast<SundialsMemory*>(mem);
441 
442  // Update time
443  m->t = t;
444 
445  // Set parameters
446  casadi_copy(p, np_, get_ptr(m->p));
447 
448  // Set the state
449  casadi_copy(x, nx_, NV_DATA_S(m->xz));
450  casadi_copy(z, nz_, NV_DATA_S(m->xz)+nx_);
451 
452  // Reset summation states
453  N_VConst(0., m->q);
454  }

473  std::pair<int, int> SundialsInterface::getBandwidth() const {
474  std::pair<int, int> bw;
475 
476  // Get upper bandwidth
477  if (upper_bandwidth_>=0) {
478  bw.first = upper_bandwidth_;
479  } else {
480  casadi_assert_message(!jac_.is_null(),
481  "\"upper_bandwidth\" has not been set and cannot be "
482  "detected since exact Jacobian is not available.");
483  bw.first = jac_.sparsity_out(0).bw_upper();
484  }
485 
486  // Get lower bandwidth
487  if (lower_bandwidth_>=0) {
488  bw.second = lower_bandwidth_;
489  } else {
490  casadi_assert_message(!jac_.is_null(),
491  "\"lower_bandwidth\" has not been set and cannot be "
492  "detected since exact Jacobian is not available.");
493  bw.second = jac_.sparsity_out(0).bw_lower();
494  }
495 
496  return bw;
497  }
