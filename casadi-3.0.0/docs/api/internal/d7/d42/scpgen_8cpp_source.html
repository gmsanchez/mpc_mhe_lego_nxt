131  Nlpsol::init(opts);
132 
133  // Default options
134  max_iter_ = 50;
135  max_iter_ls_ = 1;
136  c1_ = 1e-4;
137  beta_ = 0.8;
138  lbfgs_memory_ = 10;
139  tol_pr_ = 1e-6;
140  tol_du_ = 1e-6;
141  tol_reg_ = 1e-11;
142  regularize_ = false;
143  codegen_ = false;
144  reg_threshold_ = 1e-8;
145  tol_pr_step_ = 1e-6;
146  merit_memsize_ = 4;
147  merit_start_ = 1e-8;
148  string hessian_approximation = "exact";
149  string qpsol_plugin;
150  Dict qpsol_options;
151  print_header_ = true;
152 
153  // Read user options
154  for (auto&& op : opts) {
155  if (op.first=="max_iter") {
156  max_iter_ = op.second;
157  } else if (op.first=="max_iter_ls") {
158  max_iter_ls_ = op.second;
159  } else if (op.first=="c1") {
160  c1_ = op.second;
161  } else if (op.first=="beta") {
162  beta_ = op.second;
163  } else if (op.first=="lbfgs_memory") {
164  lbfgs_memory_ = op.second;
165  } else if (op.first=="tol_pr") {
166  tol_pr_ = op.second;
167  } else if (op.first=="tol_du") {
168  tol_du_ = op.second;
169  } else if (op.first=="tol_reg") {
170  tol_reg_ = op.second;
171  } else if (op.first=="regularize") {
172  regularize_ = op.second;
173  } else if (op.first=="codegen") {
174  codegen_ = op.second;
175  } else if (op.first=="reg_threshold") {
176  reg_threshold_ = op.second;
177  } else if (op.first=="tol_pr_step") {
178  tol_pr_step_ = op.second;
179  } else if (op.first=="merit_memsize") {
180  merit_memsize_ = op.second;
181  } else if (op.first=="merit_start") {
182  merit_start_ = op.second;
183  } else if (op.first=="hessian_approximation") {
184  hessian_approximation = op.second.to_string();
185  } else if (op.first=="name_x") {
186  name_x_ = op.second;
187  } else if (op.first=="print_x") {
188  print_x_ = op.second;
189  } else if (op.first=="qpsol") {
190  qpsol_plugin = op.second.to_string();
191  } else if (op.first=="qpsol_options") {
192  qpsol_options = op.second;
193  } else if (op.first=="print_header") {
194  print_header_ = op.second;
195  }
196  }
197 
198  // Gauss-Newton Hessian?
199  gauss_newton_ = hessian_approximation == "gauss-newton";
200 
201  // Name the components
202  if (name_x_.empty()) {
203  name_x_.resize(nx_);
204  for (int i=0; i<nx_; ++i) {
205  stringstream ss;
206  ss << "x" << i;
207  name_x_[i] = ss.str();
208  }
209  } else {
210  casadi_assert(name_x_.size()==nx_);
211  }
212 
213  // Generate lifting functions
214  Function fg = create_function("fg", {"x", "p"}, {"f", "g"}, {}, Dict(), false);
215  Function vdef_fcn, vinit_fcn;
216  fg.generate_lifted(vdef_fcn, vinit_fcn);
217  vinit_fcn_ = vinit_fcn;
218  alloc(vinit_fcn_);
219 
220  // Extract the expressions
221  vector<MX> vdef_in = vdef_fcn.mx_in();
222  vector<MX> vdef_out = vdef_fcn(vdef_in);
223 
224  // Get the dimensions
225  MX x = vdef_in.at(0);
226  MX p = vdef_in.at(1);
227  v_.resize(vdef_in.size()-2);
228  for (int i=0; i<v_.size(); ++i) {
229  v_[i].v = vdef_in.at(i+2);
230  v_[i].v_def = vdef_out.at(i+2);
231  v_[i].n = v_[i].v.nnz();
232  }
233 
234  // Scalar objective function
235  MX f;
236 
237  // Multipliers
238  MX g_lam;
239 
240  // Definition of the lifted dual variables
241  MX p_defL, gL_defL;
242 
243  if (gauss_newton_) {
244  // Least square objective
245  f = dot(vdef_out[0], vdef_out[0])/2;
246  gL_defL = vdef_out[0];
247  ngn_ = gL_defL.nnz();
248  } else {
249  // Scalar objective function
250  f = vdef_out[0];
251 
252  // Lagrange multipliers corresponding to the definition of the dependent variables
253  stringstream ss;
254  int i=0;
255  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
256  ss.str(string());
257  ss << "lam_x" << i++;
258  it->v_lam = MX::sym(ss.str(), it->v.sparsity());
259  }
260 
261  // Lagrange multipliers for the nonlinear constraints
262  g_lam = MX::sym("g_lam", ng_);
263 
264  if (verbose_) {
265  userOut() << "Allocated intermediate variables." << endl;
266  }
267 
268  // Adjoint sweep to get the definitions of the lifted dual variables
269  // (Equation 3.8 in Albersmeyer2010)
270  vector<vector<MX> > fseed, fsens, aseed(1), asens(1);
271  aseed[0].push_back(1.0);
272  aseed[0].push_back(g_lam);
273  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
274  aseed[0].push_back(it->v_lam);
275  }
276  vdef_fcn.derivative(vdef_in, vdef_out, fseed, fsens, aseed, asens, true);
277  i=0;
278 
279  gL_defL = asens[0].at(i++);
280  if (gL_defL.is_null()) gL_defL = MX::zeros(x.sparsity()); // Needed?
281 
282  p_defL = asens[0].at(i++);
283  if (p_defL.is_null()) p_defL = MX::zeros(p.sparsity()); // Needed?
284 
285  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
286  it->v_defL = asens[0].at(i++);
287  if (it->v_defL.is_null()) {
288  it->v_defL = MX::zeros(it->v.sparsity());
289  }
290  }
291 
292  if (verbose_) {
293  userOut() << "Generated the gradient of the Lagrangian." << endl;
294  }
295  }
296 
297  // Residual function
298 
299  // Inputs
300  vector<MX> res_fcn_in;
301  int n=0;
302  res_fcn_in.push_back(x); res_x_ = n++;
303  res_fcn_in.push_back(p); res_p_ = n++;
304  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
305  res_fcn_in.push_back(it->v); it->res_var = n++;
306  }
307  if (!gauss_newton_) {
308  res_fcn_in.push_back(g_lam); res_g_lam_ = n++;
309  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
310  res_fcn_in.push_back(it->v_lam); it->res_lam = n++;
311  }
312  }
313 
314  // Outputs
315  vector<MX> res_fcn_out;
316  n=0;
317  res_fcn_out.push_back(f); res_f_ = n++;
318  res_fcn_out.push_back(gL_defL); res_gl_ = n++;
319  res_fcn_out.push_back(vdef_out[1]); res_g_ = n++;
320  res_fcn_out.push_back(p_defL); res_p_d_ = n++;
321  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
322  res_fcn_out.push_back(it->v_def - it->v); it->res_d = n++;
323  }
324 
325  if (!gauss_newton_) {
326  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
327  res_fcn_out.push_back(it->v_defL - it->v_lam); it->res_lam_d = n++;
328  }
329  }
330 
331  // Generate function
332  Function res_fcn("res_fcn", res_fcn_in, res_fcn_out);
333  if (verbose_) {
334  userOut() << "Generated residual function ( " << res_fcn.n_nodes() << " nodes)." << endl;
335  }
336 
337  // Declare difference vector d and substitute out p and v
338  stringstream ss;
339  int i=0;
340  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
341  ss.str(string());
342  ss << "d" << i++;
343  it->d = MX::sym(ss.str(), it->v.sparsity());
344  it->d_def = it->v_def - it->d;
345  }
346 
347  // Declare difference vector lam_d and substitute out lam
348  if (!gauss_newton_) {
349  int i=0;
350  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
351  ss.str(string());
352  ss << "d_lam" << i++;
353  it->d_lam = MX::sym(ss.str(), it->v.sparsity());
354  it->d_defL = it->v_defL - it->d_lam;
355  }
356  }
357 
358  // Variables to be substituted and their definitions
359  vector<MX> svar, sdef;
360  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
361  svar.push_back(it->v);
362  sdef.push_back(it->d_def);
363  }
364  if (!gauss_newton_) {
365  for (vector<Var>::reverse_iterator it=v_.rbegin(); it!=v_.rend(); ++it) {
366  svar.push_back(it->v_lam);
367  sdef.push_back(it->d_defL);
368  }
369  }
370 
371  vector<MX> ex(4);
372  ex[0] = f;
373  ex[1] = vdef_out[1];
374  ex[2] = gL_defL;
375  ex[3] = p_defL;
376 
377  substitute_inplace(svar, sdef, ex, false);
378  i=0;
379  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
380  it->d_def = sdef[i++];
381  }
382  if (!gauss_newton_) {
383  for (vector<Var>::reverse_iterator it=v_.rbegin(); it!=v_.rend(); ++it) {
384  it->d_defL = sdef[i++];
385  }
386  }
387 
388  MX f_z = ex[0];
389  MX g_z = ex[1];
390  MX gL_z = ex[2];
391  MX p_z = ex[3];
392 
393  // Modified function inputs
394  vector<MX> mfcn_in;
395  n=0;
396  mfcn_in.push_back(p); mod_p_ = n++;
397  mfcn_in.push_back(x); mod_x_ = n++;
398  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
399  mfcn_in.push_back(it->d); it->mod_var = n++;
400  }
401 
402  // Modified function outputs
403  n=0;
404  vector<MX> mfcn_out;
405  mfcn_out.push_back(g_z); mod_g_ = n++;
406 
407  // Add multipliers to function inputs
408  if (!gauss_newton_) {
409  n = mfcn_in.size();
410  mfcn_in.push_back(g_lam); mod_g_lam_ = n++;
411  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
412  mfcn_in.push_back(it->d_lam); it->mod_lam = n++;
413  }
414  }
415 
416  // Add gradient of the Lagrangian
417  n = mfcn_out.size();
418  mfcn_out.push_back(f_z); mod_f_ = n++;
419  mfcn_out.push_back(gL_z); mod_gl_ = n++;
420 
421  // Lagrangian gradient function
422  Function lgrad("lgrad", mfcn_in, mfcn_out);
423 
424  // Jacobian of the constraints
425  MX jac = MX::jac(lgrad, mod_x_, mod_g_);
426  log("Formed Jacobian of the constraints.");
427 
428  // Hessian of the Lagrangian
429  MX hes = MX::jac(lgrad, mod_x_, mod_gl_, false, !gauss_newton_);
430  if (gauss_newton_) {
431  log("Formed square root of Gauss-Newton Hessian.");
432  } else {
433  log("Formed Hessian of the Lagrangian.");
434  }
435 
436  // Matrices in the reduced QP
437  n=0;
438  vector<MX> mat_out;
439  mat_out.push_back(jac); mat_jac_ = n++;
440  mat_out.push_back(hes); mat_hes_ = n++;
441  Function mat_fcn("mat_fcn", mfcn_in, mat_out);
442 
443  // Definition of intermediate variables
444  n = mfcn_out.size();
445  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
446  mfcn_out.push_back(it->d_def); it->mod_def = n++;
447  if (!gauss_newton_) {
448  mfcn_out.push_back(it->d_defL); it->mod_defL = n++;
449  }
450  }
451 
452  // Modifier function
453  Function mfcn("mfcn", mfcn_in, mfcn_out);
454 
455  // Directional derivative of Z
456  vector<vector<MX> > mfcn_fwdSeed(1, mfcn_in), mfcn_fwdSens(1, mfcn_out);
457  vector<vector<MX> > mfcn_adjSeed, mfcn_adjSens;
458 
459  // Linearization in the d-direction (see Equation (2.12) in Alberspeyer2010)
460  fill(mfcn_fwdSeed[0].begin(), mfcn_fwdSeed[0].end(), MX());
461  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
462  mfcn_fwdSeed[0][it->mod_var] = it->d;
463  if (!gauss_newton_) {
464  mfcn_fwdSeed[0][it->mod_lam] = it->d_lam;
465  }
466  }
467  mfcn.derivative(mfcn_in, mfcn_out, mfcn_fwdSeed, mfcn_fwdSens,
468  mfcn_adjSeed, mfcn_adjSens, true);
469 
470  // Vector(s) b in Lifted Newton
471  MX b_gf = densify(mfcn_fwdSens[0][mod_gl_]);
472  MX b_g = densify(mfcn_fwdSens[0][mod_g_]);
473 
474  // Tangent function
475  vector<MX> vec_fcn_out;
476  n=0;
477  vec_fcn_out.push_back(b_gf); vec_gf_ = n++;
478  vec_fcn_out.push_back(b_g); vec_g_ = n++;
479  casadi_assert(n==vec_fcn_out.size());
480 
481  Function vec_fcn("vec_fcn", mfcn_in, vec_fcn_out);
482  if (verbose_) {
483  userOut() << "Generated linearization function ( " << vec_fcn.n_nodes()
484  << " nodes)." << endl;
485  }
486 
487  // Expression a + A*du in Lifted Newton (Section 2.1 in Alberspeyer2010)
488  MX du = MX::sym("du", nx_); // Step in u
489  MX g_dlam; // Step lambda_g
490  if (!gauss_newton_) {
491  g_dlam = MX::sym("g_dlam", g_lam.sparsity());
492  }
493 
494  // Interpret the Jacobian-vector multiplication as a forward directional derivative
495  fill(mfcn_fwdSeed[0].begin(), mfcn_fwdSeed[0].end(), MX());
496  mfcn_fwdSeed[0][mod_x_] = du;
497  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
498  mfcn_fwdSeed[0][it->mod_var] = -it->d;
499  }
500  if (!gauss_newton_) {
501  mfcn_fwdSeed[0][mod_g_lam_] = g_dlam;
502  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
503  mfcn_fwdSeed[0][it->mod_lam] = -it->d_lam;
504  }
505  }
506 
507  mfcn.derivative(mfcn_in, mfcn_out, mfcn_fwdSeed, mfcn_fwdSens,
508  mfcn_adjSeed, mfcn_adjSens, true);
509 
510  // Step expansion function inputs
511  n = mfcn_in.size();
512  mfcn_in.push_back(du); mod_du_ = n++;
513  if (!gauss_newton_) {
514  mfcn_in.push_back(g_dlam); mod_dlam_g_ = n++;
515  }
516 
517  // Step expansion function outputs
518  vector<MX> exp_fcn_out;
519  n=0;
520  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
521  exp_fcn_out.push_back(mfcn_fwdSens[0][it->mod_def]); it->exp_def = n++;
522  }
523 
524  if (!gauss_newton_) {
525  for (vector<Var>::iterator it=v_.begin(); it!=v_.end(); ++it) {
526  exp_fcn_out.push_back(mfcn_fwdSens[0][it->mod_defL]); it->exp_defL = n++;
527  }
528  }
529 
530  // Step expansion function
531  Function exp_fcn("exp_fcn", mfcn_in, exp_fcn_out);
532  if (verbose_) {
533  userOut() << "Generated step expansion function ( " << exp_fcn.n_nodes() << " nodes)."
534  << endl;
535  }
536 
537  // Generate c code and load as DLL
538  if (codegen_) {
539  // Codegen the functions
540  CodeGenerator gen;
541  gen.add(res_fcn);
542  gen.add(mat_fcn);
543  gen.add(vec_fcn);
544  gen.add(exp_fcn);
545 
546  // Name of temporary file
547  string cname;
548 #ifdef HAVE_MKSTEMPS
549  // Preferred solution
550  char cname_array[] = "tmp_casadi_scpgen_XXXXXX.c";
551  if (mkstemps(cname_array, 3) == -1) {
552  casadi_error("Failed to create a temporary file name");
553  }
554  cname = cname_array;
555 #else
556  // Fallback, may result in deprecation warnings
557  char* cname_array = tempnam(0, "scp.c");
558  cname = cname_array;
559  free(cname_array);
560 #endif
561 
562  // Generate code
563  if (verbose_) {
564  userOut() << "Generating \"" << cname << "\"" << endl;
565  }
566  string name = cname.substr(0, cname.find_first_of('.'));
567  gen.generate(name);
568 
569  // Complile and run
570  if (verbose_) {
571  userOut() << "Starting compilation" << endl;
572  }
573  time_t time1 = time(0);
574  compiler_ = Compiler(cname, compilerplugin_, jit_options_);
575  time_t time2 = time(0);
576  double comp_time = difftime(time2, time1);
577  if (verbose_) {
578  userOut() << "Compilation completed after " << comp_time << " s." << endl;
579  }
580 
581  // Load the generated code
582  res_fcn_ = external("res_fcn", compiler_);
583  mat_fcn_ = external("mat_fcn", compiler_);
584  vec_fcn_ = external("vec_fcn", compiler_);
585  exp_fcn_ = external("exp_fcn", compiler_);
586  } else {
587  mat_fcn_ = mat_fcn;
588  res_fcn_ = res_fcn;
589  vec_fcn_ = vec_fcn;
590  exp_fcn_ = exp_fcn;
591  }

783  void Scpgen::solve(void* mem) const {
784  auto m = static_cast<ScpgenMemory*>(mem);
785 
786  // Statistics
787  for (auto&& s : m->fstats) s.second.reset();
788 
789  if (v_.size()>0) {
790  // Initialize lifted variables using the generated function
791  fill_n(m->arg, vinit_fcn_.n_in(), nullptr);
792  m->arg[0] = m->x0;
793  m->arg[1] = m->p;
794  fill_n(m->res, vinit_fcn_.n_out(), nullptr);
795  for (int i=0; i<v_.size(); ++i) {
796  m->res[i] = m->lifted_mem[i].x0;
797  }
798  vinit_fcn_(m->arg, m->res, m->iw, m->w, 0);
799  }
800  if (verbose_) {
801  userOut() << "Passed initial guess" << endl;
802  }
803 
804  // Reset dual guess
805  casadi_fill(m->lam_gk, ng_, 0.);
806  casadi_fill(m->dlam_gk, ng_, 0.);
807  casadi_fill(m->lam_xk, nx_, 0.);
808  casadi_fill(m->dlam_xk, nx_, 0.);
809  if (!gauss_newton_) {
810  for (auto&& v : m->lifted_mem) {
811  casadi_fill(v.lam, v.n, 0.);
812  casadi_fill(v.dlam, v.n, 0.);
813  }
814  }
815 
816  // Objective value
817  m->fk = numeric_limits<double>::quiet_NaN();
818 
819  // Reset line-search
820  fill(m->merit_mem.begin(), m->merit_mem.end(), 0.0);
821  m->merit_ind = 0;
822 
823  // Current guess for the primal solution
824  casadi_copy(m->x0, nx_, m->xk);
825  for (auto&& v : m->lifted_mem) {
826  casadi_copy(v.x0, v.n, v.x);
827  }
828 
829  // Get current time and reset timers
830  double time1 = clock();
831  m->t_eval_mat = m->t_eval_res = m->t_eval_vec = m->t_eval_exp = m->t_solve_qp = 0;
832 
833  // Initial evaluation of the residual function
834  eval_res(m);
835 
836  // Number of SQP iterations
837  m->iter_count = 0;
838 
839  // Reset last step-size
840  m->pr_step = 0;
841  m->du_step = 0;
842 
843  // Reset line-search
844  int ls_iter = 0;
845  bool ls_success = true;
846 
847  // Reset regularization
848  m->reg = 0;
849 
850  // Reset iteration message
851  m->iteration_note = 0;
852 
853  // MAIN OPTIMZATION LOOP
854  while (true) {
855 
856  // Evaluate the vectors in the condensed QP
857  eval_vec(m);
858 
859  // Evaluate the matrices in the condensed QP
860  eval_mat(m);
861 
862  // 1-norm of the primal infeasibility
863  double pr_inf = primalInfeasibility(m);
864 
865  // 1-norm of the dual infeasibility
866  double du_inf = dualInfeasibility(m);
867 
868  // Print header occasionally
869  if (m->iter_count % 10 == 0) printIteration(m, userOut());
870 
871  // Printing information about the actual iterate
872  printIteration(m, userOut(), m->iter_count, m->fk, pr_inf, du_inf, m->reg,
873  ls_iter, ls_success);
874 
875  // Checking convergence criteria
876  bool converged = pr_inf <= tol_pr_ && m->pr_step <= tol_pr_step_ && m->reg <= tol_reg_;
877  converged = converged && du_inf <= tol_du_;
878  if (converged) {
879  userOut() << endl << "casadi::SCPgen: Convergence achieved after "
880  << m->iter_count << " iterations." << endl;
881  break;
882  }
883 
884  if (m->iter_count >= max_iter_) {
885  userOut() << endl;
886  userOut() << "casadi::SCPgen: Maximum number of iterations reached." << endl;
887  break;
888  }
889 
890  // Check if not-a-number
891  if (m->fk!=m->fk || m->pr_step != m->pr_step || pr_inf != pr_inf) {
892  userOut() << "casadi::SCPgen: Aborted, nan detected" << endl;
893  break;
894  }
895 
896  // Start a new iteration
897  m->iter_count++;
898 
899  // Regularize the QP
900  if (regularize_) {
901  regularize(m);
902  }
903 
904  // Solve the condensed QP
905  solve_qp(m);
906 
907  // Expand the step
908  eval_exp(m);
909 
910  // Line-search to take the step
911  line_search(m, ls_iter, ls_success);
912  }
913 
914  double time2 = clock();
915  m->t_mainloop = (time2-time1)/CLOCKS_PER_SEC;
916 
917  // Store optimal value
918  userOut() << "optimal cost = " << m->fk << endl;
919 
920  // Save results to outputs
921  casadi_copy(&m->fk, 1, m->f);
922  casadi_copy(m->xk, nx_, m->x);
923  casadi_copy(m->lam_gk, ng_, m->lam_g);
924  casadi_copy(m->lam_xk, nx_, m->lam_x);
925  casadi_copy(m->gk, ng_, m->g);
926 
927  // Write timers
928  if (print_time_) {
929  userOut() << endl;
930  userOut() << "time spent in eval_mat: " << setw(9) << m->t_eval_mat << " s." << endl;
931  userOut() << "time spent in eval_res: " << setw(9) << m->t_eval_res << " s." << endl;
932  userOut() << "time spent in eval_vec: " << setw(9) << m->t_eval_vec << " s." << endl;
933  userOut() << "time spent in eval_exp: " << setw(9) << m->t_eval_exp << " s." << endl;
934  userOut() << "time spent in solve_qp: " << setw(9) << m->t_solve_qp << " s." << endl;
935  userOut() << "time spent in main loop: " << setw(9) << m->t_mainloop << " s." << endl;
936  }
937 
938  userOut() << endl;
939  }

941  double Scpgen::primalInfeasibility(ScpgenMemory* m) const {
942  // L1-norm of the primal infeasibility
943  double pr_inf = 0;
944 
945  // Simple bounds
946  for (int i=0; i<nx_; ++i) {
947  double lbx = m->lbx ? m->lbx[i] : 0, ubx = m->ubx ? m->ubx[i] : 0;
948  pr_inf += std::max(m->xk[i]-ubx, 0.);
949  pr_inf += std::max(lbx-m->xk[i], 0.);
950  }
951 
952  // Lifted variables
953  for (auto&& v : m->lifted_mem) pr_inf += casadi_asum(v.n, v.res);
954 
955  // Nonlinear bounds
956  for (int i=0; i<ng_; ++i) {
957  double lbg = m->lbg ? m->lbg[i] : 0, ubg = m->ubg ? m->ubg[i] : 0;
958  pr_inf += std::max(m->gk[i]-ubg, 0.);
959  pr_inf += std::max(lbg-m->gk[i], 0.);
960  }
961 
962  return pr_inf;
963  }

1032  void Scpgen::eval_mat(ScpgenMemory* m) const {

1034  double time1 = clock();

1036  // Inputs
1037  fill_n(m->arg, mat_fcn_.n_in(), nullptr);
1038  m->arg[mod_p_] = m->p; // Parameters
1039  m->arg[mod_x_] = m->xk; // Primal step/variables
1040  for (size_t i=0; i<v_.size(); ++i) {
1041  m->arg[v_[i].mod_var] = m->lifted_mem[i].res;
1042  }
1043  if (!gauss_newton_) { // Dual steps/variables
1044  m->arg[mod_g_lam_] = m->lam_gk;
1045  for (size_t i=0; i<v_.size(); ++i) {

1053  m->res[mat_hes_] = gauss_newton_ ? m->qpL : m->qpH; // Condensed Hessian
1054

1056  mat_fcn_(m->arg, m->res, m->iw, m->w, 0);
1057 
1058  if (gauss_newton_) {
1059  // Gauss-Newton Hessian
1060  casadi_fill(m->qpH, spH_.nnz(), 0.);

1062

1064  casadi_fill(m->gfk, nx_, 0.);

1066  }
1067 
1068  // Calculate the gradient of the lagrangian
1069  casadi_copy(m->gfk, nx_, m->gL);
1070  casadi_axpy(nx_, 1., m->lam_xk, m->gL);
1071  casadi_mv(m->qpA, spA_, m->lam_gk, m->gL, true);
1072 
1073  double time2 = clock();
1074  m->t_eval_mat += (time2-time1)/CLOCKS_PER_SEC;

1157  casadi_assert(nx_==2 && spH_.is_dense());
1158

1161 
1162  // Check the smallest eigenvalue of the Hessian

1166  double d = m->qpH[3];

1232  void Scpgen::line_search(ScpgenMemory* m, int& ls_iter, bool& ls_success) const {
1233  // Make sure that we have a decent direction
1234  if (!gauss_newton_) {
1235  // Get the curvature in the step direction
1236  double gain = casadi_bilin(m->qpH, spH_, m->dxk, m->dxk);
1237  if (gain < 0) {
1238  m->iteration_note = "Hessian indefinite in the search direction";
1239  }
1240  }
1241 
1242  // Calculate L1-merit function in the actual iterate
1243  double l1_infeas = primalInfeasibility(m);
1244 
1245  // Right-hand side of Armijo condition
1246  double F_sens = 0;
1247  for (int i=0; i<nx_; ++i) F_sens += m->dxk[i] * m->gfk[i];
1248  double L1dir = F_sens - m->sigma * l1_infeas;
1249  double L1merit = m->fk + m->sigma * l1_infeas;
1250 
1251  // Storing the actual merit function value in a list
1252  m->merit_mem[m->merit_ind] = L1merit;
1253  ++m->merit_ind %= merit_memsize_;
1254 
1255  // Stepsize
1256  double t = 1.0, t_prev = 0.0;
1257 
1258  // Merit function value in candidate
1259  double L1merit_cand = 0;
1260 
1261  // Reset line-search counter, success marker
1262  ls_iter = 0;
1263  ls_success = false;
1264 
1265  // Line-search
1266  //log("Starting line-search");
1267 
1268  // Line-search loop
1269  while (true) {
1270  // Take the primal step
1271  double dt = t-t_prev;
1272  casadi_axpy(nx_, dt, m->dxk, m->xk);
1273  for (auto&& v : m->lifted_mem) casadi_axpy(v.n, dt, v.dx, v.x);
1274 
1275  // Take the dual step
1276  casadi_axpy(ng_, dt, m->dlam_gk, m->lam_gk);
1277  casadi_axpy(nx_, dt, m->dlam_xk, m->lam_xk);
1278  if (!gauss_newton_) {
1279  for (auto&& v : m->lifted_mem) casadi_axpy(v.n, dt, v.dlam, v.lam);
1280  }
1281 
1282  // Evaluate residual function to get objective and constraints
1283  // (and residuals for the next iteration)
1284  eval_res(m);
1285  ls_iter++;
1286 
1287  // Calculating merit-function in candidate
1288  l1_infeas = primalInfeasibility(m);
1289  L1merit_cand = m->fk + m->sigma * l1_infeas;
1290 
1291  // Calculating maximal merit function value so far
1292  double meritmax = *max_element(m->merit_mem.begin(), m->merit_mem.end());
1293  if (L1merit_cand <= meritmax + t * c1_ * L1dir) {
1294 
1295  // Accepting candidate
1296  ls_success = true;
1297  //log("Line-search completed, candidate accepted");
1298  break;
1299  }
1300 
1301  // Line-search not successful, but we accept it.
1302  if (ls_iter == max_iter_ls_) {
1303  //log("Line-search completed, maximum number of iterations");
1304  break;
1305  }
1306 
1307  // Backtracking
1308  t_prev = t;
1309  t = beta_ * t;
1310  }
1311 
1312  // Calculate primal step-size
1313  m->pr_step = casadi_asum(nx_, m->dxk);
1314  for (auto&& v : m->lifted_mem) m->pr_step += casadi_asum(v.n, v.dx);
1315  m->pr_step *= t;
1316 
1317  // Calculate the dual step-size
1318  m->du_step = casadi_asum(ng_, m->dlam_gk) + casadi_asum(nx_, m->dlam_xk);
1319  for (auto&& v : m->lifted_mem) m->du_step += casadi_asum(v.n, v.dlam);
1320  m->du_step *= t;
1321  }
