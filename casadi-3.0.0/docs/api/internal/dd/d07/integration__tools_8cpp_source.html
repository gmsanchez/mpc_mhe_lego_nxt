130  Function simpleRK(Function f, int N, int order) {
131  // Consistency check
132  casadi_assert_message(N>=1, "Parameter N (number of steps) must be at least 1, but got "
133  << N << ".");
134  casadi_assert_message(order==4, "Only RK order 4 is supported now.");
135  casadi_assert_message(f.n_in()==2, "Function must have two inputs: x and p");
136  casadi_assert_message(f.n_out()==1, "Function must have one outputs: dot(x)");
137 
138  MX x0 = MX::sym("x0", f.sparsity_in(0));
139  MX p = MX::sym("p", f.sparsity_in(1));
140  MX h = MX::sym("h");
141 
142  std::vector<double> b(order);
143  b[0]=1.0/6;
144  b[1]=1.0/3;
145  b[2]=1.0/3;
146  b[3]=1.0/6;
147 
148  std::vector<double> c(order);
149  c[0]=0;
150  c[1]=1.0/2;
151  c[2]=1.0/2;
152  c[3]=1;
153 
154  std::vector< std::vector<double> > A(order-1);
155  A[0].resize(1);
156  A[0][0]=1.0/2;
157  A[1].resize(2);
158  A[1][0]=0;A[1][1]=1.0/2;
159  A[2].resize(3);
160  A[2][0]=0;
161  A[2][1]=0;A[2][2]=1;
162 
163  // Time step
164  MX dt = h/N;
165 
166  std::vector<MX> k(order);
167  vector<MX> f_arg(2);
168 
169  // Integrate
170  MX xf = x0;
171  for (int i=0; i<N; ++i) {
172  for (int j=0; j<order; ++j) {
173  MX xL = 0;
174  for (int jj=0; jj<j; ++jj) {
175  xL += k.at(jj)*A.at(j-1).at(jj);
176  }
177  f_arg[0] = xf+xL;
178  f_arg[1] = p;
179  k[j] = dt*f(f_arg).at(0);
180  }
181 
182  for (int j=0; j<order; ++j) {
183  xf += b.at(j)*k.at(j);
184  }
185  }
186 
187  // Form discrete-time dynamics
188  return Function("F", {x0, p, h}, {xf}, {"x0", "p", "h"}, {"xf"});
189  }

191  void collocationInterpolators(const std::vector<double> & tau_root,
192  std::vector< std::vector<double> > &C, std::vector< double > &D) {
193 
194  // Find the degree of the interpolation
195  int deg = tau_root.size()-1;
196 
197 
198  // Allocate storage space for resulting coefficients
199  C.resize(deg+1);
200  for (int i=0;i<deg+1;++i) {
201  C[i].resize(deg+1);
202  }
203  D.resize(deg+1);
204 
205  // Collocation point
206  SX tau = SX::sym("tau");
207 
208  // For all collocation points
209  for (int j=0; j<deg+1; ++j) {
210  // Construct Lagrange polynomials to get the polynomial basis at the collocation point
211  SX L = 1;
212  for (int j2=0; j2<deg+1; ++j2) {
213  if (j2 != j) {
214  L *= (tau-tau_root[j2])/(tau_root[j]-tau_root[j2]);
215  }
216  }
217 
218  Function lfcn("lfcn", {tau}, {L});
219 
220  // Evaluate the polynomial at the final time to get the
221  // coefficients of the continuity equation
222  D[j] = lfcn(vector<DM>{1.}).at(0)->front();
223 
224  // Evaluate the time derivative of the polynomial at all collocation points to
225  // get the coefficients of the continuity equation
226  Function tfcn = lfcn.tangent();
227  for (int j2=0; j2<deg+1; ++j2) {
228  C[j2][j] = tfcn(vector<DM>{tau_root[j2]}).at(0)->front();
229  }
230  }
231  }

233  Function simpleIRK(Function f, int N, int order, const std::string& scheme,
234  const std::string& solver,
235  const Dict& solver_options) {
236  // Consistency check
237  casadi_assert_message(N>=1, "Parameter N (number of steps) must be at least 1, but got "
238  << N << ".");
239  casadi_assert_message(f.n_in()==2, "Function must have two inputs: x and p");
240  casadi_assert_message(f.n_out()==1, "Function must have one outputs: dot(x)");
241 
242  // Obtain collocation points
243  std::vector<double> tau_root = collocation_points(order, scheme);
244  tau_root.insert(tau_root.begin(), 0);
245 
246  // Retrieve collocation interpolating matrices
247  std::vector < std::vector <double> > C;
248  std::vector < double > D;
249  collocationInterpolators(tau_root, C, D);
250 
251  // Inputs of constructed function
252  MX x0 = MX::sym("x0", f.sparsity_in(0));
253  MX p = MX::sym("p", f.sparsity_in(1));
254  MX h = MX::sym("h");
255 
256  // Time step
257  MX dt = h/N;
258 
259  // Implicitly defined variables
260  MX v = MX::sym("v", repmat(x0.sparsity(), order));
261  std::vector<MX> x = vertsplit(v, x0.size1());
262  x.insert(x.begin(), x0);
263 
264  // Collect the equations that implicitly define v
265  std::vector<MX> V_eq, f_in(2), f_out;
266  for (int j=1; j<order+1; ++j) {
267  // Expression for the state derivative at the collocation point
268  MX xp_j = 0;
269  for (int r=0; r<=order; ++r) xp_j+= C[j][r]*x[r];
270 
271  // Collocation equations
272  f_in[0] = x[j];
273  f_in[1] = p;
274  f_out = f(f_in);
275  V_eq.push_back(dt*f_out.at(0)-xp_j);
276  }
277 
278  // Root-finding function
279  Function rfp("rfp", {v, x0, p, h}, {vertcat(V_eq)});
280 
281  // Create a implicit function instance to solve the system of equations
282  Function ifcn = rootfinder("ifcn", solver, rfp, solver_options);
283 
284  // Get state at end time
285  MX xf = x0;
286  for (int k=0; k<N; ++k) {
287  std::vector<MX> ifcn_out = ifcn({repmat(xf, order), xf, p, h});
288  x = vertsplit(ifcn_out[0], x0.size1());
289 
290  // State at end of step
291  xf = D[0]*x0;
292  for (int i=1; i<=order; ++i) {
293  xf += D[i]*x[i-1];
294  }
295  }
296 
297  // Form discrete-time dynamics
298  return Function("F", {x0, p, h}, {xf}, {"x0", "p", "h"}, {"xf"});
299  }

301  Function simpleIntegrator(Function f, const std::string& plugin,
302  const Dict& plugin_options) {
303  // Consistency check
304  casadi_assert_message(f.n_in()==2, "Function must have two inputs: x and p");
305  casadi_assert_message(f.n_out()==1, "Function must have one outputs: dot(x)");
306 
307  // Sparsities
308  Sparsity x_sp = f.sparsity_in(0);
309  Sparsity p_sp = f.sparsity_in(1);
310 
311  // Wrapper function inputs
312  MX x = MX::sym("x", x_sp);
313  MX u = MX::sym("u", vertcat(Sparsity::scalar(), vec(p_sp))); // augment p with t
314 
315  // Normalized xdot
316  int u_offset[] = {0, 1, 1+p_sp.size1()};
317  vector<MX> pp = vertsplit(u, vector<int>(u_offset, u_offset+3));
318  MX h = pp[0];
319  MX p = reshape(pp[1], p_sp.size());
320  MX f_in[] = {x, p};
321  MX xdot = f(vector<MX>(f_in, f_in+2)).at(0);
322  xdot *= h;
323 
324  // Form DAE function
325  MXDict dae = {{"x", x}, {"p", u}, {"ode", xdot}};
326 
327  // Create integrator function
328  Dict plugin_options2 = plugin_options;
329  plugin_options2["t0"] = 0; // Normalized time
330  plugin_options2["tf"] = 1; // Normalized time
331  Function ifcn = integrator("integrator", plugin, dae, plugin_options2);
332 
333  // Inputs of constructed function
334  MX x0 = MX::sym("x0", x_sp);
335  p = MX::sym("p", p_sp);
336  h = MX::sym("h");
337 
338  // State at end
339  MX xf = ifcn(MXDict{{"x0", x0}, {"p", vertcat(h, vec(p))}}).at("xf");
340 
341  // Form discrete-time dynamics
342  return Function("F", {x0, p, h}, {xf}, {"x0", "p", "h"}, {"xf"});
343  }
