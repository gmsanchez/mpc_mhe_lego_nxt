58  XmlNode TinyXmlInterface::addNode(TiXmlNode* n) {
59  if (!n) throw CasadiException("Error in TinyXmlInterface::addNode: Node is 0");
60  XmlNode ret;
61 
62  // Save name
63  ret.setName(n->Value());
64 
65  // Save attributes
66  int type = n->Type();
67  if (type == TiXmlNode::TINYXML_ELEMENT) {
68  if (n->ToElement()!=0) {
69  for (TiXmlAttribute* pAttrib=n->ToElement()->FirstAttribute();
70  pAttrib;
71  pAttrib=pAttrib->Next()) {
72  ret.set_attribute(pAttrib->Name(), pAttrib->Value());
73  }
74  }
75  } else if (type == TiXmlNode::TINYXML_DOCUMENT) {
76  // do nothing
77  } else {
78  throw CasadiException("TinyXmlInterface::addNode");
79  }
80 
81  // Count the number of children
82  int num_children = 0;
83  for (TiXmlNode* child = n->FirstChild(); child != 0; child= child->NextSibling()) {
84  num_children++;
85  }
86  ret.children_.reserve(num_children);
87 
88  // add children
89  int ch = 0;
90  for (TiXmlNode* child = n->FirstChild(); child != 0; child= child->NextSibling(), ++ch) {
91  int childtype = child->Type();
92 
93  if (childtype == TiXmlNode::TINYXML_ELEMENT) {
94  XmlNode newnode = addNode(child);
95  ret.children_.push_back(newnode);
96  ret.child_indices_[newnode.name()] = ch;
97  } else if (childtype == TiXmlNode::TINYXML_COMMENT) {
98  ret.comment_ = child->Value();
99  } else if (childtype == TiXmlNode::TINYXML_TEXT) {
100  ret.text_ = child->ToText()->Value();
101  } else if (childtype == TiXmlNode::TINYXML_DECLARATION) {
102  userOut() << "Warning: Skipped TiXmlNode::TINYXML_DECLARATION" << endl;
103  } else {
104  throw CasadiException("Error in TinyXmlInterface::addNode: Unknown node type");
105  }
106  }
107 
108  // Note: Return value optimization
109  return ret;
110  }
