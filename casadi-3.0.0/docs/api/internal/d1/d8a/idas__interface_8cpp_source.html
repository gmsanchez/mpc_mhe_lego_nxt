96  void IdasInterface::init(const Dict& opts) {
97  log("IdasInterface::init", "begin");
98 
99  // Call the base class init
100  SundialsInterface::init(opts);
101 
102  // Default options
103  cj_scaling_ = false;
104  calc_ic_ = true;
105  suppress_algebraic_ = false;
106  max_step_size_ = 0;
107 
108  // Read options
109  for (auto&& op : opts) {
110  if (op.first=="init_xdot") {
111  init_xdot_ = op.second;
112  } else if (op.first=="cj_scaling") {
113  cj_scaling_ = op.second;
114  } else if (op.first=="calc_ic") {
115  calc_ic_ = op.second;
116  } else if (op.first=="suppress_algebraic") {
117  suppress_algebraic_ = op.second;
118  } else if (op.first=="max_step_size") {
119  max_step_size_ = op.second;
120  } else if (op.first=="abstolv") {
121  abstolv_ = op.second;
122  } else if (op.first=="fsens_abstolv") {
123  fsens_abstolv_ = op.second;
124  }
125  }
126 
127  // Default dependent options
128  calc_icB_ = calc_ic_;
129  first_time_ = grid_.back();
130 
131  // Read dependent options
132  for (auto&& op : opts) {
133  if (op.first=="calc_icB") {
134  calc_icB_ = op.second;
135  } else if (op.first=="first_time") {
136  first_time_ = op.second;
137  }
138  }
139 
140  // Get initial conditions for the state derivatives
141  if (init_xdot_.empty()) {
142  init_xdot_.resize(nx_, 0);
143  } else {
144  casadi_assert_message(
145  init_xdot_.size()==nx_,
146  "Option \"init_xdot\" has incorrect length. Expecting " << nx_
147  << ", but got " << init_xdot_.size()
148  << ". Note that this message may actually be generated by the augmented"
149  " integrator. In that case, make use of the 'augmented_options' options"
150  " to correct 'init_xdot' for the augmented integrator.");
151  }
152 
153  // Attach functions for jacobian information
154  if (exact_jacobian_) {
155  switch (linsol_f_) {
156  case SD_ITERATIVE:
157  f_fwd_ = f_.derivative(1, 0);
158  alloc(f_fwd_);
159  break;
160  default: break;
161  }
162  }
163 
164  if (exact_jacobianB_) {
165  switch (linsol_g_) {
166  case SD_ITERATIVE:
167  g_fwd_ = g_.derivative(1, 0);
168  alloc(g_fwd_);
169  break;
170  default: break;
171  }
172  }
173 
174  log("IdasInterface::init", "end");
175  }

197  void IdasInterface::initAdj(IdasMemory* m) const {
198  log("IdasInterface::initAdj", "start");
199 
200  casadi_assert(!m->isInitAdj);
201  int flag;
202 
203  // Create backward problem
204  flag = IDACreateB(m->mem, &m->whichB);
205  if (flag != IDA_SUCCESS) idas_error("IDACreateB", flag);
206 
207  // Initialize the backward problem
208  double tB0 = grid_.back();
209  flag = IDAInitB(m->mem, m->whichB, resB_wrapper, tB0, m->rxz, m->rxzdot);
210  if (flag != IDA_SUCCESS) idas_error("IDAInitB", flag);
211 
212  // Set tolerances
213  flag = IDASStolerancesB(m->mem, m->whichB, reltolB_, abstolB_);
214  if (flag!=IDA_SUCCESS) idas_error("IDASStolerancesB", flag);
215 
216  // User data
217  flag = IDASetUserDataB(m->mem, m->whichB, m);
218  if (flag != IDA_SUCCESS) idas_error("IDASetUserDataB", flag);
219 
220  // Maximum number of steps
221  IDASetMaxNumStepsB(m->mem, m->whichB, max_num_steps_);
222  if (flag != IDA_SUCCESS) idas_error("IDASetMaxNumStepsB", flag);
223 
224  // Set algebraic components
225  N_Vector id = N_VNew_Serial(nrx_+nrz_);
226  fill_n(NV_DATA_S(id), nrx_, 1);
227  fill_n(NV_DATA_S(id)+nrx_, nrz_, 0);
228 
229  // Pass this information to IDAS
230  flag = IDASetIdB(m->mem, m->whichB, id);
231  if (flag != IDA_SUCCESS) idas_error("IDASetIdB", flag);
232 
233  // Delete the allocated memory
234  N_VDestroy_Serial(id);
235 
236  // attach linear solver
237  switch (linsol_g_) {
238  case SD_DENSE:
239  initDenseLinsolB(m);
240  break;
241  case SD_BANDED:
242  initBandedLinsolB(m);
243  break;
244  case SD_ITERATIVE:
245  initIterativeLinsolB(m);
246  break;
247  case SD_USER_DEFINED:
248  initUserDefinedLinsolB(m);
249  break;
250  default: casadi_error("Uncaught switch");
251  }
252 
253  // Quadratures for the adjoint problem
254  flag = IDAQuadInitB(m->mem, m->whichB, rhsQB_wrapper, m->rq);
255  if (flag!=IDA_SUCCESS) idas_error("IDAQuadInitB", flag);
256 
257  // Quadrature error control
258  if (quad_err_con_) {
259  flag = IDASetQuadErrConB(m->mem, m->whichB, true);
260  if (flag != IDA_SUCCESS) idas_error("IDASetQuadErrConB", flag);
261 
262  flag = IDAQuadSStolerancesB(m->mem, m->whichB, reltolB_, abstolB_);
263  if (flag != IDA_SUCCESS) idas_error("IDAQuadSStolerancesB", flag);
264  }
265 
266  // Mark initialized
267  m->isInitAdj = true;
268 
269  log("IdasInterface::initAdj", "end");
270  }

273  res(IdasMemory* m, double t, N_Vector xz, N_Vector xzdot, N_Vector rr) const {
274  log("IdasInterface::res", "begin");
275 
276  // Get time
277  m->time1 = clock();
278 
279  // Debug output
280  if (monitored("res")) {
281  printvar("t", t);
282  printvar("xz", xz);
283  printvar("xzdot", xzdot);
284  }
285 
286  // Evaluate f_
287  m->arg[DAE_T] = &t;
288  m->arg[DAE_X] = NV_DATA_S(xz);
289  m->arg[DAE_Z] = NV_DATA_S(xz)+nx_;
290  m->arg[DAE_P] = get_ptr(m->p);
291  m->res[DAE_ODE] = NV_DATA_S(rr);
292  m->res[DAE_ALG] = NV_DATA_S(rr)+nx_;
293  m->res[DAE_QUAD] = 0;
294  f_(m->arg, m->res, m->iw, m->w, 0);
295 
296  // Subtract state derivative to get residual
297  casadi_axpy(nx_, -1., NV_DATA_S(xzdot), NV_DATA_S(rr));
298 
299  // Debug output
300  if (monitored("res")) {
301  printvar("res", rr);
302  }
303 
304  // Regularity check
305  casadi_assert_message(!regularity_check_ || is_regular(rr),
306  "IdasInterface::res: not regular.");
307 
308  m->time2 = clock();
309  m->t_res += static_cast<double>(m->time2-m->time1)/CLOCKS_PER_SEC;
310  log("IdasInterface::res", "end");
311  }

496  void IdasInterface::init_memory(void* mem) const {
497  SundialsInterface::init_memory(mem);
498  auto m = static_cast<IdasMemory*>(mem);
499 
500  // Sundials return flag
501  int flag;
502 
503  // Create IDAS memory block
504  m->mem = IDACreate();
505  if (m->mem==0) throw CasadiException("IDACreate(): Creation failed");
506 
507  // Allocate n-vectors for ivp
508  m->xzdot = N_VNew_Serial(nx_+nz_);
509 
510  // Initialize Idas
511  double t0 = 0;
512  N_VConst(0.0, m->xz);
513  N_VConst(0.0, m->xzdot);
514  IDAInit(m->mem, res_wrapper, t0, m->xz, m->xzdot);
515  log("IdasInterface::init", "IDA initialized");
516 
517  // Set error handler function
518  flag = IDASetErrHandlerFn(m->mem, ehfun_wrapper, &m);
519  casadi_assert_message(flag == IDA_SUCCESS, "IDASetErrHandlerFn");
520 
521  // Include algebraic variables in error testing
522  flag = IDASetSuppressAlg(m->mem, suppress_algebraic_);
523  casadi_assert_message(flag == IDA_SUCCESS, "IDASetSuppressAlg");
524 
525  // Maxinum order for the multistep method
526  flag = IDASetMaxOrd(m->mem, max_multistep_order_);
527  casadi_assert_message(flag == IDA_SUCCESS, "IDASetMaxOrd");
528 
529  // Set user data
530  flag = IDASetUserData(m->mem, m);
531  casadi_assert_message(flag == IDA_SUCCESS, "IDASetUserData");
532 
533  // Set maximum step size
534  flag = IDASetMaxStep(m->mem, max_step_size_);
535  casadi_assert_message(flag == IDA_SUCCESS, "IDASetMaxStep");
536 
537  if (!abstolv_.empty()) {
538  // Vector absolute tolerances
539  N_Vector nv_abstol = N_VNew_Serial(abstolv_.size());
540  copy(abstolv_.begin(), abstolv_.end(), NV_DATA_S(nv_abstol));
541  flag = IDASVtolerances(m->mem, reltol_, nv_abstol);
542  casadi_assert_message(flag == IDA_SUCCESS, "IDASVtolerances");
543  N_VDestroy_Serial(nv_abstol);
544  } else {
545  // Scalar absolute tolerances
546  flag = IDASStolerances(m->mem, reltol_, abstol_);
547  casadi_assert_message(flag == IDA_SUCCESS, "IDASStolerances");
548  }
549 
550  // Maximum number of steps
551  IDASetMaxNumSteps(m->mem, max_num_steps_);
552  if (flag != IDA_SUCCESS) idas_error("IDASetMaxNumSteps", flag);
553 
554  // Set algebraic components
555  N_Vector id = N_VNew_Serial(nx_+nz_);
556  fill_n(NV_DATA_S(id), nx_, 1);
557  fill_n(NV_DATA_S(id)+nx_, nz_, 0);
558 
559  // Pass this information to IDAS
560  flag = IDASetId(m->mem, id);
561  if (flag != IDA_SUCCESS) idas_error("IDASetId", flag);
562 
563  // Delete the allocated memory
564  N_VDestroy_Serial(id);
565 
566  // attach a linear solver
567  switch (linsol_f_) {
568  case SD_DENSE:
569  initDenseLinsol(m);
570  break;
571  case SD_BANDED:
572  initBandedLinsol(m);
573  break;
574  case SD_ITERATIVE:
