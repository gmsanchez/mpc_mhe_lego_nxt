64  void GurobiInterface::init(const Dict& opts) {
65  // Initialize the base classes
66  Qpsol::init(opts);
67 
68  // Default options
69  std::vector<std::string> vtype;
70 
71  // Read options
72  for (auto&& op : opts) {
73  if (op.first=="vtype") {
74  vtype = op.second;
75  }
76  }
77 
78  // Variable types
79  if (!vtype.empty()) {
80  casadi_assert_message(vtype.size()==n_, "Option 'vtype' has wrong length");
81  vtype_.resize(n_);
82  for (int i=0; i<n_; ++i) {
83  if (vtype[i]=="continuous") {
84  vtype_[i] = GRB_CONTINUOUS;
85  } else if (vtype[i]=="binary") {
86  vtype_[i] = GRB_BINARY;
87  } else if (vtype[i]=="integer") {
88  vtype_[i] = GRB_INTEGER;
89  } else if (vtype[i]=="semicont") {
90  vtype_[i] = GRB_SEMICONT;
91  } else if (vtype[i]=="semiint") {
92  vtype_[i] = GRB_SEMIINT;
93  } else {
94  casadi_error("No such variable type: " + vtype[i]);
95  }
96  }
97  }
98 
99  // Temporary memory
100  alloc_w(n_, true); // val
101  alloc_iw(n_, true); // ind
102  alloc_iw(n_, true); // ind2
103  alloc_iw(n_, true); // tr_ind
104  }

114  void GurobiInterface::
115  eval(void* mem, const double** arg, double** res, int* iw, double* w) const {
116  auto m = static_cast<GurobiMemory*>(mem);
117 
118  // Inputs
119  const double *h=arg[QPSOL_H],
120  *g=arg[QPSOL_G],
121  *a=arg[QPSOL_A],
122  *lba=arg[QPSOL_LBA],
123  *uba=arg[QPSOL_UBA],
124  *lbx=arg[QPSOL_LBX],
125  *ubx=arg[QPSOL_UBX],
126  *x0=arg[QPSOL_X0],
127  *lam_x0=arg[QPSOL_LAM_X0];
128 
129  // Outputs
130  double *x=res[QPSOL_X],
131  *cost=res[QPSOL_COST],
132  *lam_a=res[QPSOL_LAM_A],
133  *lam_x=res[QPSOL_LAM_X];
134 
135  // Temporary memory
136  double *val=w; w+=n_;
137  int *ind=iw; iw+=n_;
138  int *ind2=iw; iw+=n_;
139  int *tr_ind=iw; iw+=n_;
140 
141  // Greate an empty model
142  GRBmodel *model = 0;
143  try {
144  int flag = GRBnewmodel(m->env, &model, name_.c_str(), 0, 0, 0, 0, 0, 0);
145  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
146 
147  // Add variables
148  for (int i=0; i<n_; ++i) {
149  // Get bounds
150  double lb = lbx ? lbx[i] : 0., ub = ubx ? ubx[i] : 0.;
151  if (isinf(lb)) lb = -GRB_INFINITY;
152  if (isinf(ub)) ub = GRB_INFINITY;
153 
154  // Get variable type
155  char vtype;
156  if (!vtype_.empty()) {
157  // Explicitly set 'vtype' takes precedence
158  vtype = vtype_.at(i);
159  } else if (!discrete_.empty() && discrete_.at(i)) {
160  // Variable marked as discrete (integer or binary)
161  vtype = lb==0 && ub==1 ? GRB_BINARY : GRB_INTEGER;
162  } else {
163  // Continious variable
164  vtype = GRB_CONTINUOUS;
165  }
166 
167  // Pass to model
168  flag = GRBaddvar(model, 0, 0, 0, g ? g[i] : 0., lb, ub, vtype, 0);
169  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
170  }
171  flag = GRBupdatemodel(model);
172  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
173 
174  // Add quadratic terms
175  const int *H_colind=sparsity_in(QPSOL_H).colind(), *H_row=sparsity_in(QPSOL_H).row();
176  for (int i=0; i<n_; ++i) {
177 
178  // Quadratic term nonzero indices
179  int numqnz = H_colind[1]-H_colind[0];
180  casadi_copy(H_row, numqnz, ind);
181  H_colind++;
182  H_row += numqnz;
183 
184  // Corresponding column
185  casadi_fill(ind2, numqnz, i);
186 
187  // Quadratic term nonzeros
188  if (h) {
189  casadi_copy(h, numqnz, val);
190  casadi_scal(numqnz, 0.5, val);
191  h += numqnz;
192  } else {
193  casadi_fill(val, numqnz, 0.);
194  }
195 
196  // Pass to model
197  flag = GRBaddqpterms(model, numqnz, ind, ind2, val);
198  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
199  }
200 
201  // Add constraints
202  const int *A_colind=sparsity_in(QPSOL_A).colind(), *A_row=sparsity_in(QPSOL_A).row();
203  casadi_copy(A_colind, n_, tr_ind);
204  for (int i=0; i<nc_; ++i) {
205  // Get bounds
206  double lb = lba ? lba[i] : 0., ub = uba ? uba[i] : 0.;
207 // if (isinf(lb)) lb = -GRB_INFINITY;
208 // if (isinf(ub)) ub = GRB_INFINITY;
209 
210  // Constraint nonzeros
211  int numnz = 0;
212  for (int j=0; j<n_; ++j) {
213  if (tr_ind[j]<A_colind[j+1] && A_row[tr_ind[j]]==i) {
214  ind[numnz] = j;
215  val[numnz] = a ? a[tr_ind[j]] : 0;
216  numnz++;
217  tr_ind[j]++;
218  }
219  }
220 
221  // Pass to model
222  if (isinf(lb)) {
223  if (isinf(ub)) {
224  // Neither upper or lower bounds, skip
225  } else {
226  // Only upper bound
227  flag = GRBaddconstr(model, numnz, ind, val, GRB_LESS_EQUAL, ub, 0);
228  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
229  }
230  } else {
231  if (isinf(ub)) {
232  // Only lower bound
233  flag = GRBaddconstr(model, numnz, ind, val, GRB_GREATER_EQUAL, lb, 0);
234  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
235  } else if (lb==ub) {
236  // Upper and lower bounds equal
237  flag = GRBaddconstr(model, numnz, ind, val, GRB_EQUAL, lb, 0);
238  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
239  } else {
240  // Both upper and lower bounds
241  flag = GRBaddrangeconstr(model, numnz, ind, val, lb, ub, 0);
242  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
243  }
244  }
245  }
246 
247  // Solve the optimization problem
248  flag = GRBoptimize(model);
249  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
250  int optimstatus;
251  flag = GRBgetintattr(model, GRB_INT_ATTR_STATUS, &optimstatus);
252  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
253 
254  // Get the objective value, if requested
255  if (cost) {
256  flag = GRBgetdblattr(model, GRB_DBL_ATTR_OBJVAL, cost);
257  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
258  }
259 
260  // Get the optimal solution, if requested
261  if (x) {
262  flag = GRBgetdblattrarray(model, GRB_DBL_ATTR_X, 0, n_, x);
263  casadi_assert_message(!flag, GRBgeterrormsg(m->env));
264  }
265 
266  // Free memory
267  GRBfreemodel(model);
268 
269  } catch (...) {
270  // Free memory
271  if (model) GRBfreemodel(model);
272  throw;
273  }
274  }
