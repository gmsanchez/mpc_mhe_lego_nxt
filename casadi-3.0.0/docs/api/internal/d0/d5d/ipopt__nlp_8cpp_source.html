101  // return the structure or values of the jacobian
102  bool IpoptUserClass::eval_jac_g(Index n, const Number* x, bool new_x,
103  Index m, Index nele_jac, Index* iRow, Index *jCol,
104  Number* values) {
105  if (values) {
106  // Evaluate Jacobian
107  return solver_.calc_function(mem_, solver_.jac_g_fcn_, {x, mem_->p}, {0, values})==0;
108  } else {
109  // Get the sparsity pattern
110  int ncol = solver_.jacg_sp_.size2();
111  const int* colind = solver_.jacg_sp_.colind();
112  const int* row = solver_.jacg_sp_.row();
113  if (nele_jac!=colind[ncol]) return false; // consistency check
114 
115  // Pass to IPOPT
116  for (int cc=0; cc<ncol; ++cc) {
117  for (int el=colind[cc]; el<colind[cc+1]; ++el) {
118  *iRow++ = row[el];
119  *jCol++ = cc;
120  }
121  }
122  return true;
123  }
124  }

128  Number obj_factor, Index m, const Number* lambda,
129  bool new_lambda, Index nele_hess, Index* iRow,
130  Index* jCol, Number* values) {

132  // Evaluate Hessian

145  *iRow++ = row[el];
146  *jCol++ = cc;
147  }
148  }
149  return true;

165  bool IpoptUserClass::intermediate_callback(AlgorithmMode mode, Index iter, Number obj_value,
166  Number inf_pr, Number inf_du,
167  Number mu, Number d_norm,
168  Number regularization_size,
169  Number alpha_du, Number alpha_pr,
170  Index ls_trials,
171  const IpoptData* ip_data,
172  IpoptCalculatedQuantities* ip_cq) {
173 
174  // Only do the callback every few iterations
175  if (iter % solver_.callback_step_!=0) return true;
176 
179  // http://list.coin-or.org/pipermail/ipopt/2010-April/001965.html
180 
181  bool full_callback = false;
182 
183 #ifdef WITH_IPOPT_CALLBACK
184  mem_->fstats.at("callback_prep").tic();
185  OrigIpoptNLP* orignlp = dynamic_cast<OrigIpoptNLP*>(GetRawPtr(ip_cq->GetIpoptNLP()));
186  if (!orignlp) return true;
187  TNLPAdapter* tnlp_adapter = dynamic_cast<TNLPAdapter*>(GetRawPtr(orignlp->nlp()));
188  if (!tnlp_adapter) return true;
189 
190  const Vector& x = *ip_data->curr()->x();
191  const Vector& z_L = *ip_data->curr()->z_L();
192  const Vector& z_U = *ip_data->curr()->z_U();
193  const Vector& c = *ip_cq->curr_c();
194  const Vector& d = *ip_cq->curr_d();
195  const Vector& y_c = *ip_data->curr()->y_c();
196  const Vector& y_d = *ip_data->curr()->y_d();
197 
198  std::fill_n(x_, n_, 0);
199  std::fill_n(g_, m_, 0);
200  std::fill_n(z_L_, n_, 0);
201  std::fill_n(z_U_, n_, 0);
202  std::fill_n(lambda_, m_, 0);
203 
204  tnlp_adapter->ResortX(x, x_); // no further steps needed
205  tnlp_adapter->ResortG(y_c, y_d, lambda_); // no further steps needed
206  tnlp_adapter->ResortG(c, d, g_);
207  // Copied from Ipopt source: To Ipopt, the equality constraints are presented with right
208  // hand side zero, so we correct for the original right hand side.
209  const Index* c_pos = tnlp_adapter->P_c_g_->ExpandedPosIndices();
210  Index n_c_no_fixed = tnlp_adapter->P_c_g_->NCols();
211  for (Index i=0; i<n_c_no_fixed; i++) {
212  g_[c_pos[i]] += tnlp_adapter->c_rhs_[i];
213  }
214 
215  tnlp_adapter->ResortBnds(z_L, z_L_, z_U, z_U_);
216  // Copied from Ipopt source: Hopefully the following is correct to recover the bound
217  // multipliers for fixed variables (sign ok?)
218  if (tnlp_adapter->fixed_variable_treatment_==TNLPAdapter::MAKE_CONSTRAINT &&
219  tnlp_adapter->n_x_fixed_>0) {
220  const DenseVector* dy_c = static_cast<const DenseVector*>(&y_c);
221  Index n_c_no_fixed = y_c.Dim() - tnlp_adapter->n_x_fixed_;
222  if (!dy_c->IsHomogeneous()) {
223  const Number* values = dy_c->Values();
224  for (Index i=0; i<tnlp_adapter->n_x_fixed_; i++) {
225  z_L_[tnlp_adapter->x_fixed_map_[i]] = Max(0., -values[n_c_no_fixed+i]);
226  z_U_[tnlp_adapter->x_fixed_map_[i]] = Max(0., values[n_c_no_fixed+i]);
227  }
228  } else {
229  double value = dy_c->Scalar();
230  for (Index i=0; i<tnlp_adapter->n_x_fixed_; i++) {
231  z_L_[tnlp_adapter->x_fixed_map_[i]] = Max(0., -value);
232  z_U_[tnlp_adapter->x_fixed_map_[i]] = Max(0., value);
233  }
234  }
235  }
236  mem_->fstats.at("callback_prep").toc();
237  full_callback = true;
238 #endif // WITH_IPOPT_CALLBACK
239 
240  return solver_.intermediate_callback(mem_, x_, z_L_, z_U_, g_, lambda_, obj_value, iter,
241  inf_pr, inf_du, mu, d_norm, regularization_size,
242  alpha_du, alpha_pr, ls_trials, full_callback);
243  }
