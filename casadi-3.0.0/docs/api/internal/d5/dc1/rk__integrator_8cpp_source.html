62  void RkIntegrator::setupFG() {
63 
64  // Symbolic inputs
65  MX x0 = MX::sym("x0", this->x());
66  MX p = MX::sym("p", this->p());
67  MX t = MX::sym("t", this->t());
68 
69  // Intermediate variables (does not enter in F_, only in G_)
70  MX v = MX::sym("v", x0.size1(), x0.size2()*3);
71  vector<MX> x = horzsplit(v, x0.size2());
72  casadi_assert(x.size()==3);
73 
74  // Definitions of x
75  vector<MX> x_def(3);
76 
77  // Time points
78  vector<MX> tt(3);
79 
80  // Forward integration
81  {
82  // Arguments when calling f
83  vector<MX> f_arg(DAE_NUM_IN);
84  vector<MX> f_res;
85  f_arg[DAE_P] = p;
86 
87  // k1
88  f_arg[DAE_T] = t;
89  f_arg[DAE_X] = x0;
90  f_res = f_(f_arg);
91  MX k1 = f_res[DAE_ODE];
92  MX k1q = f_res[DAE_QUAD];
93 
94  // k2
95  tt[0] = f_arg[DAE_T] = t + h_/2;
96  x_def[0] = f_arg[DAE_X] = x0 + (h_/2) * k1;
97  f_res = f_(f_arg);
98  MX k2 = f_res[DAE_ODE];
99  MX k2q = f_res[DAE_QUAD];
100 
101  // k3
102  tt[1] = tt[0];
103  x_def[1] = f_arg[DAE_X] = x0 + (h_/2) * k2;
104  f_res = f_(f_arg);
105  MX k3 = f_res[DAE_ODE];
106  MX k3q = f_res[DAE_QUAD];
107 
108  // k4
109  tt[2] = f_arg[DAE_T] = t + h_;
110  x_def[2] = f_arg[DAE_X] = x0 + h_ * k3;
111  f_res = f_(f_arg);
112  MX k4 = f_res[DAE_ODE];
113  MX k4q = f_res[DAE_QUAD];
114 
115  // Take step
116  MX xf = x0 + (h_/6)*(k1 + 2*k2 + 2*k3 + k4);
117  MX qf = (h_/6)*(k1q + 2*k2q + 2*k3q + k4q);
118 
119  // Define discrete time dynamics
120  // TODO(Joel): Change this and make x_k1, x_k2, x_k3 and x_k4 algebraic outputs
121  f_arg[DAE_T] = t;
122  f_arg[DAE_X] = x0;
123  f_arg[DAE_P] = p;
124  f_arg[DAE_Z] = v;
125  f_res[DAE_ODE] = xf;
126  f_res[DAE_QUAD] = qf;
127  f_res[DAE_ALG] = horzcat(x_def);
128  F_ = Function("dae", f_arg, f_res);
129  alloc(F_);
130  }
131 
132  // Backward integration
133  if (!g_.is_null()) {
134  // Symbolic inputs
135  MX rx0 = MX::sym("rx0", this->rx());
136  MX rp = MX::sym("rp", this->rp());
137 
138  // Intermediate variables (do not enter in G_)
139  MX rv = MX::sym("rv", rx0.size1(), 3*rx0.size2());
140  vector<MX> rx_def(3);
141 
142  // Arguments when calling g
143  vector<MX> g_arg(RDAE_NUM_IN);
144  vector<MX> g_res;
145  g_arg[RDAE_P] = p;
146  g_arg[RDAE_RP] = rp;
147 
148  // k1
149  g_arg[RDAE_T] = tt[2];
150  g_arg[RDAE_X] = x[2];
151  g_arg[RDAE_RX] = rx0;
152  g_res = g_(g_arg);
153  MX k1 = g_res[RDAE_ODE];
154  MX k1q = g_res[RDAE_QUAD];
155 
156  // k2
157  g_arg[RDAE_T] = tt[1];
158  g_arg[RDAE_X] = x[1];
159  g_arg[RDAE_RX] = rx_def[2] = rx0 + (h_/2) * k1;
160  g_res = g_(g_arg);
161  MX k2 = g_res[RDAE_ODE];
162  MX k2q = g_res[RDAE_QUAD];
163 
164  // k3
165  g_arg[RDAE_T] = tt[0];
166  g_arg[RDAE_X] = x[0];
167  g_arg[RDAE_RX] = rx_def[1] = rx0 + (h_/2) * k2;
168  g_res = g_(g_arg);
169  MX k3 = g_res[RDAE_ODE];
170  MX k3q = g_res[RDAE_QUAD];
171 
172  // k4
173  g_arg[RDAE_T] = t;
174  g_arg[RDAE_X] = x0;
175  g_arg[RDAE_RX] = rx_def[0] = rx0 + h_ * k3;
176  g_res = g_(g_arg);
177  MX k4 = g_res[RDAE_ODE];
178  MX k4q = g_res[RDAE_QUAD];
179 
180  // Take step
181  MX rxf = rx0 + (h_/6)*(k1 + 2*k2 + 2*k3 + k4);
182  MX rqf = (h_/6)*(k1q + 2*k2q + 2*k3q + k4q);
183 
184  // Define discrete time dynamics
185  g_arg[RDAE_T] = t;
186  g_arg[RDAE_X] = x0;
187  g_arg[RDAE_P] = p;
188  g_arg[RDAE_Z] = v;
189  g_arg[RDAE_RX] = rx0;
190  g_arg[RDAE_RP] = rp;
191  g_arg[RDAE_RZ] = rv;
192  g_res[RDAE_ODE] = rxf;
193  g_res[RDAE_QUAD] = rqf;
194  g_res[RDAE_ALG] = horzcat(rx_def);
195  G_ = Function("rdae", g_arg, g_res);
196  alloc(G_);
197  }
198  }
