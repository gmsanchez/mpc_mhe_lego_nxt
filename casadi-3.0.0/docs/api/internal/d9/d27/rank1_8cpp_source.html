58  void Rank1::evalAdj(const std::vector<std::vector<MX> >& aseed,

82  void Rank1::spFwd(const bvec_t** arg, bvec_t** res, int* iw, bvec_t* w, int mem) {
83  /* If not inline, copy to result */
84  if (arg[0]!=res[0]) copy(arg[0], arg[0]+dep(0).nnz(), res[0]);
85 
86  /* Get sparsities */
87  int ncol_A = sparsity().size2();
88  const int *colind_A = sparsity().colind(), *row_A = sparsity().row();
89 
90  /* Loop over the columns of A */
91  int cc, rr, el;
92  for (cc=0; cc<ncol_A; ++cc) {
93  /* Loop over the nonzeros of A */
94  for (el=colind_A[cc]; el<colind_A[cc+1]; ++el) {
95  /* Get row */
96  rr = row_A[el];
97 
98  /* Add the multiple */
99  res[0][el] |= *arg[1] | arg[2][rr] | arg[3][cc];
100  }
101  }
102  }
