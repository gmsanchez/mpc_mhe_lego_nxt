165  void QpoasesInterface::init(const Dict& opts) {
166  Qpsol::init(opts);
167 
168  // Default options
169  sparse_ = false;
170  max_nWSR_ = 5 *(n_ + nc_);
171  max_cputime_ = -1;
172  ops_.setToDefault();
173 
174  // Read options
175  for (auto&& op : opts) {
176  if (op.first=="sparse") {
177  sparse_ = op.second;
178  } else if (op.first=="nWSR") {
179  max_nWSR_ = op.second;
180  } else if (op.first=="CPUtime") {
181  max_cputime_ = op.second;
182  } else if (op.first=="printLevel") {
183  ops_.printLevel = to_PrintLevel(op.second);
184  } else if (op.first=="enableRamping") {
185  ops_.enableRamping = to_BooleanType(op.second);
186  } else if (op.first=="enableFarBounds") {
187  ops_.enableFarBounds = to_BooleanType(op.second);
188  } else if (op.first=="enableFlippingBounds") {
189  ops_.enableFlippingBounds = to_BooleanType(op.second);
190  } else if (op.first=="enableRegularisation") {
191  ops_.enableRegularisation = to_BooleanType(op.second);
192  } else if (op.first=="enableFullLITests") {
193  ops_.enableFullLITests = to_BooleanType(op.second);
194  } else if (op.first=="enableNZCTests") {
195  ops_.enableNZCTests = to_BooleanType(op.second);
196  } else if (op.first=="enableDriftCorrection") {
197  ops_.enableRegularisation = to_BooleanType(op.second);
198  } else if (op.first=="enableCholeskyRefactorisation") {
199  ops_.enableCholeskyRefactorisation = op.second;
200  } else if (op.first=="terminationTolerance") {
201  ops_.terminationTolerance = op.second;
202  } else if (op.first=="boundTolerance") {
203  ops_.boundTolerance = op.second;
204  } else if (op.first=="boundRelaxation") {
205  ops_.boundRelaxation = op.second;
206  } else if (op.first=="epsNum") {
207  ops_.epsNum = op.second;
208  } else if (op.first=="epsDen") {
209  ops_.epsDen = op.second;
210  } else if (op.first=="maxPrimalJump") {
211  ops_.maxPrimalJump = op.second;
212  } else if (op.first=="maxDualJump") {
213  ops_.maxDualJump = op.second;
214  } else if (op.first=="initialRamping") {
215  ops_.initialRamping = op.second;
216  } else if (op.first=="finalRamping") {
217  ops_.finalRamping = op.second;
218  } else if (op.first=="initialFarBounds") {
219  ops_.initialFarBounds = op.second;
220  } else if (op.first=="growFarBounds") {
221  ops_.growFarBounds = op.second;
222  } else if (op.first=="initialStatusBounds") {
223  ops_.initialStatusBounds = to_SubjectToStatus(op.second);
224  } else if (op.first=="epsFlipping") {
225  ops_.epsFlipping = op.second;
226  } else if (op.first=="numRegularisationSteps") {
227  ops_.numRegularisationSteps = op.second;
228  } else if (op.first=="epsRegularisation") {
229  ops_.epsRegularisation = op.second;
230  } else if (op.first=="numRefinementSteps") {
231  ops_.numRefinementSteps = op.second;
232  } else if (op.first=="epsIterRef") {
233  ops_.epsIterRef = op.second;
234  } else if (op.first=="epsLITests") {
235  ops_.epsLITests = op.second;
236  } else if (op.first=="epsNZCTests") {
237  ops_.epsNZCTests = op.second;
238  }
239  }
240 
241  // Allocate work vectors
242  if (sparse_) {
243  alloc_w(nnz_in(QPSOL_H), true); // h
244  alloc_w(nnz_in(QPSOL_A), true); // a
245  } else {
246  alloc_w(n_*n_, true); // h
247  alloc_w(n_*nc_, true); // a
248  }
249  alloc_w(n_, true); // g
250  alloc_w(n_, true); // lbx
251  alloc_w(n_, true); // ubx
252  alloc_w(nc_, true); // lba
253  alloc_w(nc_, true); // uba
254  alloc_w(n_+nc_, true); // dual
255  }

257  void QpoasesInterface::init_memory(void* mem) const {
258  auto m = static_cast<QpoasesMemory*>(mem);
259  m->called_once = false;
260 
261  // Create qpOASES instance
262  if (m->qp) delete m->qp;
263  if (nc_==0) {
264  m->qp = new qpOASES::QProblemB(n_);
265  } else {
266  m->sqp = new qpOASES::SQProblem(n_, nc_);
267  }
268 
269  // Pass to qpOASES
270  m->qp->setOptions(ops_);
271  }

273  void QpoasesInterface::
274  eval(void* mem, const double** arg, double** res, int* iw, double* w) const {
275  auto m = static_cast<QpoasesMemory*>(mem);
276 
277  if (inputs_check_) {
278  checkInputs(arg[QPSOL_LBX], arg[QPSOL_UBX], arg[QPSOL_LBA], arg[QPSOL_UBA]);
279  }
280 
281  // Maxiumum number of working set changes
282  int nWSR = max_nWSR_;
283  double cputime = max_cputime_;
284  double *cputime_ptr = cputime<=0 ? 0 : &cputime;
285 
286  // Get the arguments to call qpOASES with
287  double* g=w; w += n_;
288  casadi_copy(arg[QPSOL_G], n_, g);
289  double* lb=w; w += n_;
290  casadi_copy(arg[QPSOL_LBX], n_, lb);
291  double* ub=w; w += n_;
292  casadi_copy(arg[QPSOL_UBX], n_, ub);
293  double* lbA=w; w += nc_;
294  casadi_copy(arg[QPSOL_LBA], nc_, lbA);
295  double* ubA=w; w += nc_;
296  casadi_copy(arg[QPSOL_UBA], nc_, ubA);
297 
298  // Return flag
299  int flag;
300 
301  // QP matrices sparsities
302  const Sparsity& hsp = sparsity_in(QPSOL_H);
303  const Sparsity& asp = sparsity_in(QPSOL_A);
304 
305  // Sparse or dense mode?
306  if (sparse_) {
307  // Get quadratic term
308  int* h_colind = const_cast<int*>(hsp.colind());
309  int* h_row = const_cast<int*>(hsp.row());
310  double* h=w; w += hsp.nnz();
311  casadi_copy(arg[QPSOL_H], hsp.nnz(), h);
312  if (m->h) delete m->h;
313  m->h = new qpOASES::SymSparseMat(hsp.size1(), hsp.size2(), h_row, h_colind, h);
314  m->h->createDiagInfo();
315 
316  // Get linear term
317  int* a_colind = const_cast<int*>(asp.colind());
318  int* a_row = const_cast<int*>(asp.row());
319  double* a=w; w += asp.nnz();
320  casadi_copy(arg[QPSOL_A], asp.nnz(), a);
321  if (m->a) delete m->a;
322  m->a = new qpOASES::SparseMatrix(asp.size1(), asp.size2(), a_row, a_colind, a);
323 
324  // Solve sparse
325  if (m->called_once) {
326  flag = m->sqp->hotstart(m->h, g, m->a, lb, ub, lbA, ubA, nWSR, cputime_ptr);
327  } else {
328  flag = m->sqp->init(m->h, g, m->a, lb, ub, lbA, ubA, nWSR, cputime_ptr);
329  }
330 
331  } else {
332  // Get quadratic term
333  double* h=w; w += n_*n_;
334  casadi_densify(arg[QPSOL_H], hsp, h, false);
335 
336  // Get linear term
337  double* a = w; w += n_*nc_;
338  casadi_densify(arg[QPSOL_A], asp, a, true);
339 
340  // Solve dense
341  if (nc_==0) {
342  if (m->called_once) {
343  // Broken?
344  //flag = m->qp->hotstart(g, lb, ub, nWSR, cputime_ptr);
345  m->qp->reset();
346  flag = m->qp->init(h, g, lb, ub, nWSR, cputime_ptr);
347  } else {
348  flag = m->qp->init(h, g, lb, ub, nWSR, cputime_ptr);
349  }
350  } else {
351  if (m->called_once) {
352  flag = m->sqp->hotstart(h, g, a, lb, ub, lbA, ubA, nWSR, cputime_ptr);
353  } else {
354  flag = m->sqp->init(h, g, a, lb, ub, lbA, ubA, nWSR, cputime_ptr);
355  }
356  }
357  }
358 
359  // Solver is "warm" now
360  m->called_once = true;
361 
362  if (flag!=qpOASES::SUCCESSFUL_RETURN && flag!=qpOASES::RET_MAX_NWSR_REACHED) {
363  throw CasadiException("qpOASES failed: " + getErrorMessage(flag));
364  }
365 
366  // Get optimal cost
367  if (res[QPSOL_COST]) *res[QPSOL_COST] = m->qp->getObjVal();
368 
369  // Get the primal solution
370  if (res[QPSOL_X]) m->qp->getPrimalSolution(res[QPSOL_X]);
371 
372  // Get the dual solution
373  if (res[QPSOL_LAM_X] || res[QPSOL_LAM_A]) {
374  double* dual=w; w += n_+nc_;
375  m->qp->getDualSolution(dual);
376  casadi_scal(n_+nc_, -1., dual);
377  casadi_copy(dual, n_, res[QPSOL_LAM_X]);
378  casadi_copy(dual+n_, nc_, res[QPSOL_LAM_A]);
379  }
380  }
